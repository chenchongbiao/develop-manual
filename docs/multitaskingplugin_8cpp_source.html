34 MultitaskingPlugin::MultitaskingPlugin(QObject *parent)
35  : QObject(parent)
36  , m_pluginLoaded(false)
37  , m_tipsLabel(new TipsWidget)
38 {
39  m_tipsLabel->setVisible(false);
40  m_tipsLabel->setObjectName("multitasking");
41 
42  connect(DWindowManagerHelper::instance(), &DWindowManagerHelper::hasCompositeChanged, this, [ = ] {
43  if (!m_proxyInter || !m_pluginLoaded)
44  return;
45 
46  if (DWindowManagerHelper::instance()->hasComposite()) {
47  m_proxyInter->itemAdded(this, PLUGIN_KEY);
48  } else {
49  m_proxyInter->itemRemoved(this, PLUGIN_KEY);
50  }
51  });
52 }

109 const QString MultitaskingPlugin::itemContextMenu(const QString &itemKey)
110 {
111  if (itemKey != PLUGIN_KEY) {
112  return QString();
113  }
114 
115  QList<QVariant> items;
116  items.reserve(6);
117 
118  QMap<QString, QVariant> desktop;
119  desktop["itemId"] = "multitasking";
120  desktop["itemText"] = tr("Multitasking View");
121  desktop["isActive"] = true;
122  items.push_back(desktop);
123 
124  QMap<QString, QVariant> power;
125  power["itemId"] = "remove";
126  power["itemText"] = tr("Undock");
127  power["isActive"] = true;
128  items.push_back(power);
129 
130  QMap<QString, QVariant> menu;
131  menu["items"] = items;
132  menu["checkableMenu"] = false;
133  menu["singleCheck"] = false;
134 
135  return QJsonDocument::fromVariant(menu).toJson();
136 }

138 void MultitaskingPlugin::invokedMenuItem(const QString &itemKey, const QString &menuId, const bool checked)
139 {
140  Q_UNUSED(itemKey)
141  Q_UNUSED(checked)
142 
143  if (menuId == "multitasking") {
144  DDBusSender()
145  .service("com.deepin.wm")
146  .interface("com.deepin.wm")
147  .path("/com/deepin/wm")
148  .method(QString("PerformAction"))
149  .arg(1)
150  .call();
151  } else if (menuId == "remove") {
152  pluginStateSwitched();
153  }

211 void MultitaskingPlugin::refreshPluginItemsVisible()
212 {
