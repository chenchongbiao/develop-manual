120 void DefaultAppsWidget::onCategoryClicked(const QModelIndex &index) {
121  if (m_lastIndex == index) return;
122 
123  m_lastIndex = index;
124  qDebug() << "DefaultAppsWidget clicked row " << index.row();
125 
126  if(m_itemList[index.row()].pulgin) {
127  m_itemList[index.row()].itemSignal.invoke(m_itemList[index.row()].pulgin);
128  } else {
129  m_itemList[index.row()].itemSignal.invoke(this, Qt::ConnectionType::DirectConnection, Q_ARG(int, index.row()));
130  }
131 
132  m_defAppCatView->resetStatus(index);
133 }
