83 void OverlayWarningPlugin::pluginStateSwitched()
84 {
85  m_proxyInter->saveValue(this, PLUGIN_STATE_KEY, !m_proxyInter->getValue(this, PLUGIN_STATE_KEY, true).toBool());
86 
87  if (pluginIsDisable())
88  {
89  m_proxyInter->itemRemoved(this, pluginName());
90  } else {
91  if (!m_pluginLoaded) {
92  loadPlugin();
93  return;
94  }
95  m_proxyInter->itemAdded(this, pluginName());
96  }
97 }

134 void OverlayWarningPlugin::loadPlugin()
135 {
136  if (m_pluginLoaded) {
137  qDebug() << "overlay-warning plugin has been loaded! return";
138  return;
139  }
140 
141  m_pluginLoaded = true;
142 
143  m_warningWidget = new OverlayWarningWidget;
144 
145  if (!isOverlayRoot()) {
146  return;
147  }
148 
149  m_proxyInter->itemAdded(this, pluginName());
150  displayModeChanged(displayMode());
151 
152  QTimer::singleShot(0, m_showDisableOverlayDialogTimer, static_cast<void (QTimer::*) ()>(&QTimer::start));
153 }

155 bool OverlayWarningPlugin::isOverlayRoot()
156 {
157  // ignore live/recovery mode
158  QFile cmdline("/proc/cmdline");
159  cmdline.open(QFile::ReadOnly);
160  QString content(cmdline.readAll());
161  cmdline.close();
162  if (content.contains("boot=live")) {
163  return false;
164  }
165 
166  return QString(QStorageInfo::root().fileSystemType()) == OverlayFileSystemType;
167 }

169 void OverlayWarningPlugin::showCloseOverlayDialogPre()
170 {
171  Q_ASSERT(sender() == m_showDisableOverlayDialogTimer);
172 
173  if (QDBusConnection::sessionBus().interface()->isServiceRegistered(AuthAgentDbusService)) {
174  m_showDisableOverlayDialogTimer->stop();
175  WaitingAuthAgentTimes = 0;
176  showCloseOverlayDialog();
177  return;
178  }
179  WaitingAuthAgentTimes++;
180  qDebug() << "Waiting for AuthAgent service" << WaitingAuthAgentTimes << "times";
181  if (WaitingAuthAgentTimes > 10) {
182  qDebug() << "AuthAgent service timeout...";
183  m_showDisableOverlayDialogTimer->stop();
184  return;
185  }
186 }

191 void OverlayWarningPlugin::showCloseOverlayDialog()
192 {
193  qDebug() << "start disable overlayroot process";
194  const int result = QProcess::execute("/usr/bin/pkexec /usr/sbin/overlayroot-disable");
195  if (result == 0) {
196  QProcess::startDetached("reboot");
197  } else {
198  qDebug() << "disable overlayroot failed, the return code is" << result;
199  }
