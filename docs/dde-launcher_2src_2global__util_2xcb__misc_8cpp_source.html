114 void XcbMisc::set_deepin_override(xcb_window_t winId)
115 {
116  QString deepinOverride("_DEEPIN_OVERRIDE");
117 
118  xcb_connection_t * connection = QX11Info::connection();
119  xcb_intern_atom_cookie_t cookie = xcb_intern_atom(connection,
120  0,
121  deepinOverride.length(),
122  deepinOverride.toLatin1());
123 
124  xcb_intern_atom_reply_t *reply = xcb_intern_atom_reply(connection,
125  cookie,
126  nullptr);
127  if (reply) {
128  xcb_atom_t atom = reply->atom;
129  free (reply);
130 
131  int value = 1;
132  xcb_change_property(connection,
133  XCB_PROP_MODE_REPLACE,
134  winId,
135  atom,
136  XCB_ATOM_CARDINAL,
137  32,
138  1,
139  &value);
140  xcb_flush(connection);
141  }
142 }
