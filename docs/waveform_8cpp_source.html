337 void Waveform::onProgressChanged(qint64 value, qint64 duration, qint64 coefficient)
338 {
339  if (IsShowwaveformScale)
340  waveformScale->show();
341  auto length = maximum() - minimum();
342  Q_ASSERT(length != 0);
343 
344  /*------curCoefficient-------*/
345  curCoefficient = coefficient;
346  auto progress = 0;
347  if (0 != duration) {
348  progress = static_cast<int>(length * value / duration);
349  }
350 
351  if (signalsBlocked()) {
352  return;
353  }
354 
355  curValue = value;
356  allDuration = duration;
357 
358  blockSignals(true);
359  setValue(progress);
360 
361  blockSignals(false);
362  update();
363  updateScaleSize();
364 }

370  curValue = static_cast<qint64>(allDuration * (value() * 1.0) / (maximum() - minimum()));

479 void Waveform::spline(QVector<float> &x, QVector<float> &y, QVector<float> &vx, QVector<float> &vy, int pnt)
480 {
481  vx.clear();
482  vy.clear();
483 
484  QVector<float> tx = x;
485  QVector<float> ty = y;
486 
487  for (int i = 1; i < x.size();) {
488  if (fabs(x[i] - x[i - 1]) < 0.01f) {
489  x.erase(x.begin() + i);
490  y.erase(y.begin() + i);
491  continue;
492  }
493 
494  i ++;
495  }
496 
497  int N = x.size();
498  if (N != y.size()) return;
499 
500  if (N == 1) {
501  vx = x;
502  vy = y;
503  return;
504  }
505 
506  if (N == 2) {
507  x.insert(x.begin() + 1, (x[0] + x[1]) / 2);
508  y.insert(y.begin() + 1, (y[0] + y[1]) / 2);
509  }
510 
511  QVector<float> h;
512  h.resize(N - 1);
513  for (int i = 0; i < N - 1; i ++) h[i] = x[i + 1] - x[i];
514 
515  QVector<float> M;
516  M.resize(N);
517  M[0] = 0;
518  M[N - 1] = 0;
519  for (int i = 1; i < N - 1; i ++) M[i] = 6 * ((y[i + 1] - y[i]) / h[i] - (y[i] - y[ i - 1]) / h[i - 1]);
520 
521  QVector<QVector<float> > A;
522  A.resize(N);
523  for (int i = 0; i < N; i ++) A[i].resize(3);
524 
525  A[0][0] = 1;
526  A[N - 1][2] = 1;
527 
528  for (int i = 1; i < N - 1; i ++) {
529  A[i][0] = h[i - 1];
530  A[i][1] = 2 * (h[i - 1] + h[i]);
531  A[i][2] = h[i];
532  }
533 
534  QVector<float> C;
535  C.resize(N);
536  C[0] = A[0][1] / A[0][0];
537  for (int i = 1; i < N; i ++) C[i] = A[i][2] / (A[i][1] - C[i - 1] * A[i][0]);
538 
539  QVector<float> D;
540  D.resize(N);
541  D[0] = M[0] / A[0][0];
542  for (int i = 1; i < N; i ++) D[i] = (M[i] - D[i - 1] * A[i][0]) / (A[i][1] - C[i - 1] * A[i][0]);
543 
544  QVector<float> m;
545  m.resize(N);
546  m[N - 1] = 0;
547 
548  for (int i = N - 2; i >= 0; i --) m[i] = D[i] - C[i] * m[i + 1];
549 
550  QVector<float> a, b, c, d;
551  a.resize(N - 1);
552  b.resize(N - 1);
553  c.resize(N - 1);
554  d.resize(N - 1);
555  for (int i = 0; i < N - 1; i ++) {
556  a[i] = y[i];
557  b[i] = (y[ i + 1] - y[i]) / h[i] - h[i] * m[i] / 2 - h[i] * (m[i + 1] - m[i]) / 6;
558  c[i] = m[i] / 2;
559  d[i] = (m[i + 1] - m[i]) / (6 * h[i]);
560  }
561 
562  vx.resize(pnt);
563  vy.resize(pnt);
564 
565  int index = 0;
566  for (int i = 0; i < (pnt - 1); i ++) {
567  vx[i] = i * (x[N - 1] - x[0]) / (pnt - 1) + x[0];
568 
569  while (vx[i] > x[index + 1]) {
570  index ++;
571  }
572  float fx = vx[i] - x[index];
573 
574  vy[i] = a[index] + b[index] * fx + c[index] * fx * fx + d[index] * fx * fx * fx;
575  }
576  x = tx;
577  y = ty;
578 }
