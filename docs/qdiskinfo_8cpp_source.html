192 void QDiskInfo::updateGvfsFileSystemInfo(int retryTimes/*=3*/)
193 {
194  if (m_mounted_root_uri.isEmpty()) {
195  return;
196  }
197  std::string file_uri = m_mounted_root_uri.toStdString();
198  GFile *file;
199  GFileInfo *systemInfo;
200  GFileInfo *info;
201  GError *error;
202  file = g_file_new_for_uri(file_uri.data());
203  if (file == nullptr) {
204  return;
205  }
206  error = nullptr;
207  systemInfo = g_file_query_filesystem_info(file, "*", nullptr, &error);
208  if (error) {
209  g_error_free(error);
210  error = nullptr;
211  }
212  info = g_file_query_info(file, "*", G_FILE_QUERY_INFO_NONE, nullptr, &error);
213  if (info == nullptr && error) {
214  g_object_unref(systemInfo);
215  g_object_unref(file);
216  qWarning() << "g_file_query_filesystem_info" << error->message << error->code;
217  if (error->code == 0 && retryTimes > 0) {
218  g_error_free(error);
219  return updateGvfsFileSystemInfo(--retryTimes);
220  }
221 
222  g_error_free(error);
223  return;
224  }
225 
226  m_total = g_file_info_get_attribute_uint64(systemInfo, G_FILE_ATTRIBUTE_FILESYSTEM_SIZE);
227  m_free = g_file_info_get_attribute_uint64(systemInfo, G_FILE_ATTRIBUTE_FILESYSTEM_FREE);
228  m_used = g_file_info_get_attribute_uint64(systemInfo, G_FILE_ATTRIBUTE_FILESYSTEM_USED);
229  m_read_only = g_file_info_get_attribute_boolean(systemInfo, G_FILE_ATTRIBUTE_FILESYSTEM_READONLY);
230  m_id_filesystem = QString(g_file_info_get_attribute_as_string(info, G_FILE_ATTRIBUTE_ID_FILESYSTEM));
231 
232  if (m_used == 0) {
233  m_used = m_total - m_free;
234  }
235 
236  g_object_unref(systemInfo);
237  g_object_unref(info);
238  g_object_unref(file);
239 
240 }

320 QDiskInfo QDiskInfo::getDiskInfo(const DAbstractFileInfo &fileInfo)
321 {
322  QDiskInfo info;
323  qDebug() << fileInfo.filePath();
324  UDiskDeviceInfoPointer uDiskDeviceInfoPointer = deviceListener->getDeviceByPath(fileInfo.filePath());
325  if (!uDiskDeviceInfoPointer) {
326  uDiskDeviceInfoPointer = deviceListener->getDeviceByFilePath(fileInfo.filePath());
327  }
328  if (uDiskDeviceInfoPointer) {
329  info = uDiskDeviceInfoPointer->getDiskInfo();
330  }
331  return info;
332 }
