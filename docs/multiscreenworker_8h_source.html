87  const QList<Monitor *> validMonitor()
88  {
89  QList<Monitor *> list;
90  QMapIterator<Monitor *, MonitorInter *>it(m_monitorInfo);
91  while (it.hasNext()) {
92  it.next();
93  // 仅显示在主屏的情况下，可用屏幕信息只提供主屏幕(m_primary有可能不是主屏幕的名字，数据还没来得及切换过来)
94  // 问题场景：连接双屏，设置任务栏仅显示在主屏（gsettings），然后拔掉主屏幕显示器，任务栏崩溃
95  if (it.key()->enable()) {
96  if (m_showInPrimary) {
97  if (it.key()->name() == m_primary) {
98  list.clear();
99  list.append(it.key());
100  return list;
101  }
102 
103  if (!list.isEmpty()) {
104  list.removeAt(0);
105  list.push_front(it.key());
106  } else
107  list << it.key();
108  } else
109  list << it.key();
110  }
111  }
112 
113  return list;
114  }
120  void insert(Monitor *mon, MonitorInter *monInter)
121  {
122  m_monitorInfo.insert(mon, monInter);
123 
124  Q_EMIT monitorChanged();
125  }
130  void remove(Monitor *mon)
131  {
132  m_monitorInfo.value(mon)->deleteLater();
133  m_monitorInfo.remove(mon);
134  mon->deleteLater();
135 
136  Q_EMIT monitorChanged();
137  }

144  if (m_showInPrimary != showIn)
145  m_showInPrimary = showIn;
