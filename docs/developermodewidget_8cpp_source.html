48 DeveloperModeWidget::DeveloperModeWidget(QWidget *parent)
49  : QWidget(parent)
50  , m_model(nullptr)
51 {
52  m_inter = new QDBusInterface("com.deepin.sync.Helper",
53  "/com/deepin/sync/Helper",
54  "com.deepin.sync.Helper",
55  QDBusConnection::systemBus(), this);
56  QVBoxLayout *vBoxLayout = new QVBoxLayout;
57  m_devBtn = new QPushButton(tr("Request Root Access"));
58  m_dtip = new DTipLabel(tr("Developer mode enables you to get root privileges, install and run unsigned apps not listed in app store, but your system integrity may also be damaged, please use it carefully."));
59  m_dtip->setAlignment(Qt::AlignLeft | Qt::AlignTop);
60  m_dtip->setWordWrap(true);
61 
62  //绑定选择激活开发模式窗口
63  connect(m_devBtn, &QPushButton::clicked, [this]{
64  auto devDlg = new DeveloperModeDialog;
65  devDlg->setModel(m_model);
66 
67  connect(devDlg, &DeveloperModeDialog::requestDeveloperMode, this, &DeveloperModeWidget::enableDeveloperMode);
68  connect(this, &DeveloperModeWidget::enableDeveloperMode, devDlg, &DeveloperModeDialog::requestSetNextBtnStatus);
69  connect(devDlg, &DeveloperModeDialog::requestLogin, this, &DeveloperModeWidget::requestLogin);
70  connect(devDlg, &DeveloperModeDialog::requestCommit, [this](QString filePathName){
71  //读取机器信息证书
72  QFile fFile(filePathName);
73  if(!fFile.open(QIODevice::ReadOnly)){
74  qDebug()<<"Can't open file for writing";
75  }
76  QByteArray data = fFile.readAll();
77  QDBusMessage msg = m_inter->call("EnableDeveloperMode", data);
78 
79  //当返回信息为错误接口信息才处理
80  if (msg.type() == QDBusMessage::MessageType::ErrorMessage) {
81  //系统通知弹窗qdbus 接口
82  QDBusInterface tInterNotify("com.deepin.dde.Notification",
83  "/com/deepin/dde/Notification",
84  "com.deepin.dde.Notification",
85  QDBusConnection::sessionBus());
86 
87  //初始化Notify 七个参数
88  QString in0("dde-control-center");
89  uint in1 = 101;
90  QString in2("preferences-system");
91  QString in3("");
92  QString in4("");
93  QStringList in5;
94  QVariantMap in6;
95  int in7 = 5000;
96 
97  //截取error接口 1001:未导入证书 1002:未登录 1003:无法获取硬件信息 1004:网络异常 1005:证书加载失败 1006:签名验证失败 1007:文件保存失败
98  QString msgcode = msg.errorMessage();
99  msgcode = msgcode.split(":").at(0);
100  if (msgcode == "1001") {
101  in3 = tr("Failed to get root access");
102  } else if (msgcode == "1002") {
103  in3 = tr("Please sign in to your Union ID first");
104  } else if (msgcode == "1003") {
105  in3 = tr("Cannot read your PC information");
106  } else if (msgcode == "1004") {
107  in3 = tr("No network connection");
108  } else if (msgcode == "1005") {
109  in3 = tr("Certificate loading failed, unable to get root access");
110  } else if (msgcode == "1006") {
111  in3 = tr("Signature verification failed, unable to get root access");
112  } else if (msgcode == "1007") {
113  in3 = tr("Failed to get root access");
114  }
115 
116  //系统通知 认证失败 无法进入开发模式
117  tInterNotify.call("Notify", in0, in1, in2, in3, in4, in5, in6, in7);
118  }
119 
120  });
121  devDlg->exec();
122  devDlg->deleteLater();
123  });
124  m_lab = new DLabel(tr("The feature is not available at present, please activate your system first"));
125  m_lab->setWordWrap(true);
126  m_lab->setVisible(false);
127 
128  vBoxLayout->setMargin(0);
129  vBoxLayout->setSpacing(10);
130  vBoxLayout->setContentsMargins(ThirdPageContentsMargins);
131  vBoxLayout->addWidget(m_devBtn);
132  vBoxLayout->addWidget(m_lab);
133  vBoxLayout->addWidget(m_dtip);
134  vBoxLayout->addStretch();
135  setLayout(vBoxLayout);
136

139 void DeveloperModeWidget::setModel(CommonInfoModel *model)
140 {
141  m_model = model;
142  onLoginChanged();
143  if (!model->developerModeState()) {
144  m_devBtn->setEnabled(model->isActivate());
145  m_lab->setVisible(!model->isActivate());
146  m_dtip->setVisible(model->isActivate());
147  }
148  updateDeveloperModeState(model->developerModeState());
149  connect(model, &CommonInfoModel::developerModeStateChanged, this, [this](const bool state){
150  //更新界面
151  updateDeveloperModeState(state);
152 
153  if (!state)
154  return;
155 
156  //弹窗提示重启
157  DDialog dlg("", tr("To make some features effective, a restart is required. Restart now?"));
158  dlg.addButtons({tr("Cancel"), tr("Restart Now")});
159  connect(&dlg, &DDialog::buttonClicked, this, [](int idx, QString str){
160  Q_UNUSED(str);
161  if (idx == 1) {
162  DDBusSender()
163  .service("com.deepin.SessionManager")
164  .interface("com.deepin.SessionManager")
165  .path("/com/deepin/SessionManager")
166  .method("RequestReboot")
167  .call();
168  }
169  });
170  dlg.exec();
171  });
172  connect(model, &CommonInfoModel::isLoginChenged, this, &DeveloperModeWidget::onLoginChanged);
173  if (!model->developerModeState()) {
174  connect(model, &CommonInfoModel::LicenseStateChanged, this, [ = ] (const bool &value) {
175  m_devBtn->setEnabled(value);
176  m_lab->setVisible(!value);
177  m_dtip->setVisible(value);
178  });
179  }
180 }
