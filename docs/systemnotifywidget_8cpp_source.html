113 void SystemNotifyWidget::initConnect()
114 {
115  connect(m_model, &SysItemModel::disturbModeChanged, this, [this](bool state) {
116  m_btnDisturbMode->setChecked(state);
117  m_settingsGrp->setVisible(state);
118  });
119  m_btnDisturbMode->setChecked(m_model->isDisturbMode());
120  m_settingsGrp->setVisible(m_model->isDisturbMode());
121  connect(m_model, &SysItemModel::showInDockChanged, this, [this](bool state) {
122  m_btnShowInDock->setChecked(state);
123  });
124  m_btnShowInDock->setChecked(m_model->isShowInDock());
125  connect(m_model, &SysItemModel::timeSlotChanged, this, [this](bool state) {
126  m_itemTimeSlot->setState(state);
127  });
128  m_itemTimeSlot->setState(m_model->isTimeSlot());
129  connect(m_model, &SysItemModel::timeStartChanged, this, [this](QString time) {
130  m_itemTimeSlot->setTimeStart(QTime::fromString(time, "hh:mm"));
131  });
132  m_itemTimeSlot->setTimeStart(QTime::fromString(m_model->timeStart(), "hh:mm"));
133  connect(m_model, &SysItemModel::timeEndChanged, this, [this](QString time) {
134  m_itemTimeSlot->setTimeEnd(QTime::fromString(time, "hh:mm"));
135  });
136  m_itemTimeSlot->setTimeEnd(QTime::fromString(m_model->timeEnd(), "hh:mm"));
137  connect(m_model, &SysItemModel::lockScreenChanged, this, [this](bool state) {
138  m_itemLockScreen->setState(state);
139  });
140  m_itemLockScreen->setState(m_model->isLockScreen());
141 
142  //set connects: this to module
143  connect(m_btnDisturbMode, &DSwitchButton::checkedChanged, this, [ = ](bool state) {
144  m_settingsGrp->setVisible(state);
145  Q_EMIT requestSetSysSetting(SysItemModel::DNDMODE, state);
146  });
147  connect(m_btnShowInDock, &SwitchWidget::checkedChanged, this, [ = ](bool state) {
148  Q_EMIT requestSetSysSetting(SysItemModel::SHOWICON, state);
149  });
150  connect(m_itemTimeSlot, &TimeSlotItem::stateChanged, this, [ = ](bool state) {
151  Q_EMIT requestSetSysSetting(SysItemModel::OPENBYTIMEINTERVAL, state);
152  });
153  connect(m_itemTimeSlot, &TimeSlotItem::timeStartChanged, this, [ = ](QTime time) {
154  Q_EMIT requestSetSysSetting(SysItemModel::STARTTIME, time.toString("hh:mm"));
155  });
156  connect(m_itemTimeSlot, &TimeSlotItem::timeEndChanged, this, [ = ](QTime time) {
157  Q_EMIT requestSetSysSetting(SysItemModel::ENDTIME, time.toString("hh:mm"));
158  });
159  connect(m_itemLockScreen, &NotificationItem::stateChanged, this, [ = ](bool state) {
160  Q_EMIT requestSetSysSetting(SysItemModel::LOCKSCREENOPENDNDMODE, state);
161  });
162 }
