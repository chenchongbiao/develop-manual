40 Parser::~Parser()
41 {
42  // In case we're still using the default handlers we have
43  // to clean them up. Messy, but we shouldn't use some smart
44  // pointer here, as it's very restrictive for the user of
45  // this library (std::auto_ptr would make it impossible to
46  // use MI, SharedPtr would potentially lead to circular references).
47  if ( m_ourInlineHandler )
48  delete m_inlineHandler;
49  if ( m_ourSubDocumentHandler )
50  delete m_subDocumentHandler;
51  if ( m_ourTableHandler )
52  delete m_tableHandler;
53  if ( m_ourTextHandler )
54  delete m_textHandler;
55 
56  // Don't forget to close everything properly here
57  delete m_wordDocument;
58  m_storage->close();
59  delete m_storage;
60 }
