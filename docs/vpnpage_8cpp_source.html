64 QString vpnConfigType(const QString &path)
65 {
66  QFile f(path);
67  f.open(QIODevice::ReadOnly);
68  const QString content = f.readAll();
69 
70  if (content.contains("openconnect"))
71  return "openconnect";
72  if (content.contains("l2tp"))
73  return "l2tp";
74  if (content.startsWith("[main]"))
75  return "vpnc";
76 
77  return "openvpn";
78 }

208 void VpnPage::onActiveConnsInfoChanged(const QList<QJsonObject> &infos)
209 {
210  QMap<QString, int> activeVpnStates;
211 
212  for (const auto &info : infos) {
213  const QString &type = info.value("ConnectionType").toString();
214  if (!type.startsWith("vpn"))
215  continue;
216 
217  const QString &uuid = info.value("ConnectionUuid").toString();
218  const int state = m_model->activeConnObjectByUuid(uuid).value("State").toInt();
219 
220  activeVpnStates.insert(uuid, state);
221  }
222 
223  for (int i = 0; i < m_modelprofiles->rowCount(); ++i) {
224  DStandardItem *it = dynamic_cast<DStandardItem *>(m_modelprofiles->item(i));
225  const QString &uuid = it->data(VpnInfoRole).value<QJsonObject>().value("Uuid").toString();
226  const bool exist = activeVpnStates.contains(uuid);
227 
228  if (it->actionList(Qt::RightEdge).size() > 1) {
229  it->actionList(Qt::RightEdge).first()->widget()->deleteLater();
230  it->actionList(Qt::RightEdge).last()->deleteLater();
231  }
232  it->actionList(Qt::RightEdge).first()->deleteLater();
233 
234  DViewItemActionList dActionList;
235  DViewItemAction *editaction = new DViewItemAction(Qt::AlignCenter, QSize(), QSize(), true);
236  QStyleOption opt;
237  editaction->setIcon(DStyleHelper(style()).standardIcon(DStyle::SP_ArrowEnter, &opt, nullptr));
238  editaction->setClickAreaMargins(ArrowEnterClickMargin);
239 
240  connect(editaction, &QAction::triggered, [this, uuid] {
241  this->onVpnDetailClicked(uuid);
242  });
243 
244  if (exist && activeVpnStates[uuid] == 2) {
245  it->setCheckState(Qt::Checked);
246  } else if (exist && activeVpnStates[uuid] == 1) {
247  it->setCheckState(Qt::Unchecked);
248 
249  DViewItemAction *loadingAction = new DViewItemAction(Qt::AlignCenter | Qt::AlignRight, QSize(), QSize(), false);
250  DSpinner *loadingIndicator = new DSpinner(m_lvprofiles->viewport());
251  loadingIndicator->setFixedSize(20, 20);
252  loadingAction->setWidget(loadingIndicator);
253  loadingIndicator->start();
254  loadingIndicator->show();
255  dActionList.append(loadingAction);
256  editaction->setVisible(false);
257  } else {
258  it->setCheckState(Qt::Unchecked);
259  }
260 
261  dActionList.append(editaction);
262  it->setActionList(Qt::RightEdge, dActionList);
263  }

266 void VpnPage::changeVpnId()
267 {
268  NetworkManager::Connection::List connList = NetworkManager::listConnections();
269  QString importName = "";
270  for (const auto &conn : connList) {
271  if (conn->settings()->connectionType() == NetworkManager::ConnectionSettings::Vpn) {
272  if (m_editingConnUuid == conn->uuid()) {
273  importName = conn->name();
274  break;
275  }
276  }
277  }
278  if (importName.isEmpty()) {
279  QTimer::singleShot(10, this, &VpnPage::changeVpnId);
280  return;
281  }
282 
283  QString changeName = "";
284  bool hasSameName = false;
285  for (const auto &conn : connList) {
286  const QString vpnName = conn->name();
287  const QString vpnUuid = conn->uuid();
288  if ((vpnName == importName) && (vpnUuid != m_editingConnUuid)) {
289  changeName = importName + "(1)";
290  hasSameName = true;
291  break;
292  }
293  }
294  if (!hasSameName) {
295  return;
296  }
297 
298  for (int index = 1; ; index++) {
299  hasSameName = false;
300  for (const auto &conn : connList) {
301  QString vpnName = conn->name();
302  if (vpnName == changeName) {
303  changeName = importName + "(%1)";
304  changeName = changeName.arg(index);
305  hasSameName = true;
306  break;
307  }
308  }
309  if (!hasSameName) {
310  break;
311  }
312  }
313  NetworkManager::Connection::Ptr uuidConn = NetworkManager::findConnectionByUuid(m_editingConnUuid);
314  if (uuidConn) {
315  NetworkManager::ConnectionSettings::Ptr connSettings = uuidConn->settings();
316  connSettings->setId(changeName);
317  // update function saves the settings on the hard disk
318  QDBusPendingReply<> reply = uuidConn->update(connSettings->toMap());
319  reply.waitForFinished();
320  if (reply.isError()) {
321  qDebug() << "error occurred while updating the connection" << reply.error();
322  return;
323  }
324  qDebug() << "find Connection By Uuid is success";
325  return;
326  }
327 }

329 void VpnPage::importVPN()
330 {
331  Q_EMIT requestFrameKeepAutoHide(false);
332  const auto file = QFileDialog::getOpenFileUrl(nullptr, "", QUrl::fromLocalFile(QDir::homePath()), "*.conf");
333  Q_EMIT requestFrameKeepAutoHide(true);
334  if (file.isEmpty())
335  return;
336 
337  const auto args = QStringList { "connection", "import", "type", vpnConfigType(file.path()), "file", file.path() };
338 
339  QProcess p;
340  p.start("nmcli", args);
341  p.waitForFinished();
342  const auto stat = p.exitCode();
343  const QString output = p.readAllStandardOutput();
344  QString error = p.readAllStandardError();
345 
346  qDebug() << stat << ",output:" << output << ",err:" << error;
347 
348  if (stat) {
349  const auto ratio = devicePixelRatioF();
350  QPixmap icon = QIcon::fromTheme("dialog-error").pixmap(QSize(48, 48) * ratio);
351  icon.setDevicePixelRatio(ratio);
352 
353  DDialog dialog;
354  dialog.setTitle(tr("Import Error"));
355  dialog.setMessage(tr("File error"));
356  dialog.addButton(tr("OK"));
357  dialog.setIcon(icon);
358  dialog.exec();
359 
360  return;
361  }
362 
363  const QRegularExpression regexp("\\(\\w{8}(-\\w{4}){3}-\\w{12}\\)");
364  const auto match = regexp.match(output);
365 
366  if (match.hasMatch()) {
367  m_editingConnUuid = match.captured();
368  m_editingConnUuid.replace("(", "");
369  m_editingConnUuid.replace(")", "");
370  qDebug() << "editing connection Uuid";
371  QTimer::singleShot(10, this, &VpnPage::changeVpnId);
372  }
373 }
