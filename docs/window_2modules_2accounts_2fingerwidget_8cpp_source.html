126 void FingerWidget::onThumbsListChanged(const QStringList &thumbs)
127 {
128  m_vecItem.clear();
129  m_listGrp->clear();
130 
131  for (int n = 0; n < 10 && n < thumbs.size(); ++n) {
132  QString finger = thumbs.at(n);
133  auto item = new AccounntFingeItem(this);
134  item->setTitle(finger);
135  item->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
136  DFontSizeManager::instance()->bind(item, DFontSizeManager::T6);
137  m_listGrp->appendItem(item);
138  connect(item, &AccounntFingeItem::removeClicked, this, [this, finger] {
139  Q_EMIT requestDeleteFingerItem(m_curUser->name(), finger);
140  });
141  connect(item, &AccounntFingeItem::editTextFinished, this, [this, finger, item, thumbs, n](QString newName) {
142  // 没有改名，直接返回
143  if (item->getTitle() == newName) {
144  return;
145  }
146  for (int i = 0; i < thumbs.size(); ++i) {
147  if (newName == thumbs.at(i) && i != n) {
148  QString errMsg = tr("The name already exists");
149  item->showAlertMessage(errMsg);
150  return;
151  }
152  }
153  item->setTitle(newName);
154  Q_EMIT requestRenameFingerItem(m_curUser->name(), finger, newName);
155  });
156 
157  connect(item, &AccounntFingeItem::editClicked, this, [this, item, thumbs]() {
158  for (int k = 0; k < thumbs.size(); ++k) {
159  static_cast<AccounntFingeItem *>(m_listGrp->getItem(k))->setEditTitle(item == m_listGrp->getItem(k));
160  }
161  });
162 
163  if(m_clearBtn->isChecked())
164  item->setShowIcon(true);
165 
166  m_vecItem.append(item);
167  }
168 
169  m_clearBtn->setVisible(m_listGrp->itemCount());
170  // 如果指纹数量大于等于10,则不显示添加指纹按钮
171  if (thumbs.size() >= 10) {
172  return;
173  }
174  // 找到最小的指纹名以便作为缺省名添加
175  for (int i = 0; i < m_model->getPredefineThumbsName().size(); ++i) {
176  bool findNotUsedThumb = false;
177  QString newFingerName = m_model->getPredefineThumbsName().at(i);
178  for (int n = 0; n < 10 && n < thumbs.size(); ++n) {
179  if (newFingerName == thumbs.at(n)) {
180  findNotUsedThumb = true;
181  break;
182  }
183  }
184  if (!findNotUsedThumb) {
185  addFingerButton(newFingerName);
186  break;
187  }
188  }
189 }

191 void FingerWidget::addFingerButton(const QString &newFingerName)
192 {
193  SettingsItem* addfingerItem = new SettingsItem(this);
194  QString strAddFinger = tr("Add Fingerprint");
195  DCommandLinkButton *addBtn = new DCommandLinkButton(strAddFinger);
196  QHBoxLayout *fingerLayout = new QHBoxLayout(this);
197  fingerLayout->addWidget(addBtn, 0, Qt::AlignLeft);
198  addfingerItem->setLayout(fingerLayout);
199  m_listGrp->insertItem(m_listGrp->itemCount(), addfingerItem);
200  addfingerItem->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
201 
202  DFontSizeManager::instance()->bind(addBtn, DFontSizeManager::T7);
203  QFontMetrics fontMetrics(font());
204  int nFontWidth = fontMetrics.width(strAddFinger);
205  addBtn->setMinimumWidth(nFontWidth);
206  connect(addBtn, &DCommandLinkButton::clicked, this, [ = ] {
207  Q_EMIT requestAddThumbs(m_curUser->name(), newFingerName);
208  });
209 }
