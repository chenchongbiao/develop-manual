106 void ShortcutItem::setShortcut(const QString &shortcut)
107 {
108  m_shortcutEdit->hide();
109  m_key->show();
110 
111  QString accels = shortcut;
112  accels = accels.replace("<", "");
113  accels = accels.replace(">", "-");
114  accels = accels.replace("_L", "");
115  accels = accels.replace("_R", "");
116  accels = accels.replace("Control", "Ctrl");
117 
118  m_key->setTextList(accels.split("-"));
119 }

145 void ShortcutItem::updateTitleSize()

148

150 
151  if (m_title->fontMetrics().width(m_title->text()) > v) {
152  QFontMetrics fontWidth(m_title->font());
153  QString elideNote = fontWidth.elidedText(m_title->text(), Qt::ElideRight, v / 2);
154  m_title->setText(elideNote);
155  }

159 void ShortcutItem::mouseReleaseEvent(QMouseEvent *e)
160 {
161  if (m_delBtn->isVisible())
162  return;
163 
164  if (!m_shortcutEdit->isVisible() && m_key->rect().contains(m_key->mapFromParent(e->pos()))) {
165  m_key->hide();
166  m_shortcutEdit->show();
167  m_info->item = this;
168 
169  Q_EMIT requestUpdateKey(m_info);
170  } else {
171  m_shortcutEdit->hide();
172  m_key->show();
173  }
174 }
