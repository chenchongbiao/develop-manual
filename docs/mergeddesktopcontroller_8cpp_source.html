102 void MergedDesktopWatcher::addWatcher(const DUrl &url)
103 {
104  Q_D(MergedDesktopWatcher);
105 
106  if (!url.isValid() || d->urlToWatcherMap.contains(url)) {
107  return;
108  }
109 
110  DUrl real_url = MergedDesktopController::convertToRealPath(url);
111 
112  DAbstractFileWatcher *watcher = DFileService::instance()->createFileWatcher(this, real_url);
113 
114  if (!watcher) {
115  return;
116  }
117 
118  watcher->moveToThread(this->thread());
119  watcher->setParent(this);
120 
121  connect(watcher, &DAbstractFileWatcher::fileAttributeChanged, this, &MergedDesktopWatcher::onFileAttributeChanged);
122  connect(watcher, &DAbstractFileWatcher::fileModified, this, &MergedDesktopWatcher::onFileModified);
123 
124  d->urlToWatcherMap[url] = watcher;
125 
126  if (d->started) {
127  watcher->startWatcher();
128  }
129 }

169 const QList<DAbstractFileInfoPointer> MergedDesktopController::getChildren(const QSharedPointer<DFMGetChildrensEvent> &event) const
170 {
171 // if (!dataInitialized) {
172 // initData();
173 // dataInitialized = true;
174 // }
175  // blumia: 文件监听占用完了的时候有可能桌面会监听不到文件变动,此时即便 F5 也不会刷新该 Controller 存储的整理桌面数据,故改为每次都重新初始化整理数据
176 
177  //加锁失败，说明有线程在跑后面的算法
178  if (!m_runMtx.tryLock(0)){
179  m_childrenLock.lock();
180  m_cv.wait(&m_childrenLock);
181  QList<DAbstractFileInfoPointer> infoList = m_childrenList;
182  m_childrenLock.unlock();
183  return infoList;
184  }
185 
186  QMutexLocker aful(&m_arrangedFileUrlsMtx); //禁止其他线程修改arrangedFileUrls
187 
188  arrangedFileUrls = initData(event->filters());
189  currentUrl = event->url();
190  QString path { currentUrl.path() };
191  QList<DAbstractFileInfoPointer> infoList;
192 
193  auto appendEntryFiles = [this](QList<DAbstractFileInfoPointer> &infoList, const DMD_TYPES &entryType){
194  for (const DUrl & url : arrangedFileUrls[entryType]) {
195  DAbstractFileInfoPointer info {
196  new MergedDesktopFileInfo(convertToDFMMDPath(url), currentUrl)
197  };
198  infoList.append(info);
199  }
200  };
201 
202  auto appendVirtualEntries = [this, &infoList,appendEntryFiles](bool displayEmptyEntry = false, const QStringList & expandedEntries = {}) {
203  for (unsigned int i = DMD_FIRST_TYPE; i <= DMD_ALL_ENTRY; i++) {
204  DMD_TYPES oneType = static_cast<DMD_TYPES>(i);
205  if (!displayEmptyEntry && arrangedFileUrls[oneType].isEmpty()) {
206  continue;
207  }
208  //屏蔽掉2个以下文件的分类图标
209  if(arrangedFileUrls[oneType].size() < 2){
210  appendEntryFiles(infoList, oneType);
211  continue;
212  }
213  QString entryName = entryNameByEnum(oneType);
214  DUrl url(DFMMD_ROOT VIRTUALENTRY_FOLDER + entryNameByEnum(oneType));
215  DAbstractFileInfoPointer infoPtr {
216  new MergedDesktopFileInfo(url, currentUrl)
217  };
218  infoList.push_back(infoPtr);
219  if (expandedEntries.contains(entryName)) {
220  appendEntryFiles(infoList, oneType);
221  }
222  }
223  };
224 
225  auto appendFolders = [this, &infoList]() {
226  for (const DUrl & url : arrangedFileUrls[DMD_FOLDER]) {
227  DAbstractFileInfoPointer info {
228  new MergedDesktopFileInfo(convertToDFMMDPath(url), currentUrl)
229  };
230  infoList.append(info);
231  }
232  };
233 
234  auto makeAndInsertInfo = [this, &infoList](QString urlStr) {
235  DUrl entryUrl(urlStr);
236  DAbstractFileInfoPointer adeEntryInfoPtr {
237  new MergedDesktopFileInfo(convertToDFMMDPath(entryUrl), currentUrl)
238  };
239  infoList.push_back(adeEntryInfoPtr);
240  };
241 
242  if(currentUrl.scheme() == DFMMD_SCHEME) {
243  if (path == QStringLiteral("/")) {
244  makeAndInsertInfo(DFMMD_ROOT VIRTUALENTRY_FOLDER);
245  makeAndInsertInfo(DFMMD_ROOT VIRTUALFOLDER_FOLDER);
246  makeAndInsertInfo(DFMMD_ROOT MERGEDDESKTOP_FOLDER);
247  } else if (path.startsWith(QStringLiteral(VIRTUALENTRY_PATH))) {
248  if (path == QStringLiteral(VIRTUALENTRY_PATH)) {
249  appendVirtualEntries();
250  } else {
251  QString entryName = path.split('/', QString::SkipEmptyParts).last();
252  DMD_TYPES entryType = entryTypeByName(entryName);
253  appendEntryFiles(infoList, entryType);
254  }
255  } else if (path == QStringLiteral(VIRTUALFOLDER_PATH)) {
256  appendFolders();
257  } else if (path == QStringLiteral(MERGEDDESKTOP_PATH)) {
258  QString expandedFolder = currentUrl.fragment();
259  QStringList expandedFolders;
260  if (!expandedFolder.isEmpty()) {
261  expandedFolders = expandedFolder.split(',', QString::SkipEmptyParts);
262  }
263  appendVirtualEntries(false, expandedFolders);
264  appendFolders();
265  }
266  }
267 
268  m_runMtx.unlock();
269  m_childrenLock.lock();
270  m_childrenList = infoList;
271  m_childrenLock.unlock();
272  m_cv.wakeAll();
273  return infoList;
274 }

473 DMD_TYPES MergedDesktopController::checkUrlArrangedType(const DUrl url)
474 {
475  DAbstractFileInfoPointer info = DFileService::instance()->createFileInfo(nullptr, url);
476  if (info) {

479  return DMD_APPLICATION;

481  return DMD_PICTURE;
482  } else if (mimetypeDisplayName.startsWith(qApp->translate("MimeTypeDisplayManager", "Audio"))) {
483  return DMD_MUSIC;
484  } else if (mimetypeDisplayName.startsWith(qApp->translate("MimeTypeDisplayManager", "Video"))) {
485  return DMD_VIDEO;
486  } else if (mimetypeDisplayName.startsWith(qApp->translate("MimeTypeDisplayManager", "Text"))) {
487  return DMD_DOCUMENT;
488  } else if (mimetypeDisplayName.startsWith(qApp->translate("MimeTypeDisplayManager", "Directory"))) {
489  return DMD_FOLDER;

491  return DMD_OTHER;
492  }
493  }

518 void MergedDesktopController::desktopFilesRemoved(const DUrl &url)

521 
522  for (unsigned int i = DMD_FIRST_TYPE; i <= DMD_ALL_TYPE; i++) {

525  aful.unlock();
526 
527  DUrl vUrl = convertToDFMMDPath(url, typeInfo);
528 
529  DUrl parentUrl = getVirtualEntryPath(typeInfo);
530  DAbstractFileWatcher::ghostSignal(parentUrl, &DAbstractFileWatcher::fileDeleted, vUrl);
531  DUrl parentUrl2(DFMMD_ROOT MERGEDDESKTOP_FOLDER);

533  return;

538 void
539 MergedDesktopController::desktopFilesRenamed(const DUrl &oriUrl, const DUrl &dstUrl)
540 {
541  QMutexLocker aful(&m_arrangedFileUrlsMtx); //禁止其他线程修改arrangedFileUrls
542 
543  //获取原文件的类型
544  DMD_TYPES orgTypeInfo = checkUrlArrangedType(oriUrl);
545  for (unsigned int i = DMD_FIRST_TYPE; i <= DMD_ALL_TYPE; i++) {
546  DMD_TYPES typeInfo = static_cast<DMD_TYPES>(i);
547  if (arrangedFileUrls[typeInfo].removeOne(oriUrl)) {
548  orgTypeInfo = typeInfo;
549  break;
550  }
551  }
552 
553  DMD_TYPES typeInfo = checkUrlArrangedType(dstUrl);
554  arrangedFileUrls[typeInfo].append(dstUrl);
555  aful.unlock();
556 
557  DUrl vOriUrl = convertToDFMMDPath(oriUrl, orgTypeInfo);
558  DUrl vDstUrl = convertToDFMMDPath(dstUrl, typeInfo);
559 
560  DUrl parentUrl = getVirtualEntryPath(typeInfo);
561  DAbstractFileWatcher::ghostSignal(parentUrl, &DAbstractFileWatcher::fileMoved, vOriUrl, vDstUrl);
562  DUrl parentUrl2(DFMMD_ROOT MERGEDDESKTOP_FOLDER);
563  DAbstractFileWatcher::ghostSignal(parentUrl2, &DAbstractFileWatcher::fileMoved, vOriUrl, vDstUrl);
564 }

566 QMap<DMD_TYPES, QList<DUrl> > MergedDesktopController::initData(QDir::Filters ftrs)
567 {
568  QMap<DMD_TYPES, QList<DUrl> > tArrangedFileUrls;
569 
570  QDir desktopDir(QStandardPaths::standardLocations(QStandardPaths::DesktopLocation).first());
571 #if 0
572  const QStringList &fileList = desktopDir.entryList(QDir::AllEntries | QDir::NoDotAndDotDot);
573 #else
574  //自动整理下需要显示隐藏文件,故新加QDir::Filters ftrs
575  QStringList fileList = desktopDir.entryList(ftrs, QDir::Name);
576 #endif
577  //文件名排序
578  auto compateFunc = [](const QString &str1, const QString &str2) -> bool{
579  return FileSortFunction::compareByString(str1,str2);
580  };
581  qSort(fileList.begin(),fileList.end(),compateFunc);
582  //end
583
