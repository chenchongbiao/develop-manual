141 void CGraphicsPenItem::drawComplete(bool doBz)
142 {
143  if (m_isShiftPress) {
144  m_isShiftPress = false;
145  m_path.lineTo(m_straightLine.p2());
146  }
147 
148  if (m_path.elementCount() > 5 && doBz) {
149  QPainterPath vout;
150  for (int i = 0; i < m_path.elementCount() - 5; i += 5) {
151  QPainterPath::Element p0 = m_path.elementAt(i);
152  QPainterPath::Element p1 = m_path.elementAt(i + 1);
153  QPainterPath::Element p2 = m_path.elementAt(i + 2);
154  QPainterPath::Element p3 = m_path.elementAt(i + 3);
155  QPainterPath::Element p4 = m_path.elementAt(i + 4);
156  QPainterPath::Element p5 = m_path.elementAt(i + 5);
157 
158  if (0 == i) {
159  QPointF dot1 = GetBezierValue(p0, p1, p2, p3, p4, p5, 0.);
160  vout.moveTo(dot1);
161  }
162  QPointF dot2 = GetBezierValue(p0, p1, p2, p3, p4, p5, 1 / 5.0);
163  QPointF dot3 = GetBezierValue(p0, p1, p2, p3, p4, p5, 2 / 5.0);
164  QPointF dot4 = GetBezierValue(p0, p1, p2, p3, p4, p5, 3 / 5.0);
165  QPointF dot5 = GetBezierValue(p0, p1, p2, p3, p4, p5, 4 / 5.0);
166  QPointF dot6 = GetBezierValue(p0, p1, p2, p3, p4, p5, 1);
167 
168  vout.lineTo(dot2);
169  vout.lineTo(dot3);
170  vout.lineTo(dot4);
171  vout.lineTo(dot5);
172  vout.lineTo(dot6);
173  }
174 
175  //保证未被优化的点也加入到最终的绘制路径中
176  if (vout.elementCount() < m_path.elementCount()) {
177  for (int i = vout.elementCount() - 1; i < m_path.elementCount(); ++i) {
178  QPointF psF(m_path.elementAt(i).x, m_path.elementAt(i).y);
179  vout.lineTo(psF);
180  }
181  }
182  prepareGeometryChange();
183  m_path = vout;
184  }
185 
186  updateShape();
187 
188  m_isStartWithLine = false;
189  m_isEndWithLine = false;
190 }

218 qreal CGraphicsPenItem::GetBezierValue(qreal p0, qreal p1, qreal p2, qreal p3, qreal p4, qreal p5, qreal t)
219 {
220 
221  return pow(1 - t, 5.0) * p0 + 5 * pow(1 - t, 4.0) * t * p1 + 10 * pow(1 - t, 3.0) * pow(t, 2.0) * p2 + 10 * pow(1 - t, 2.0) * pow(t, 3.0) * p3 + 5 * pow(1 - t, 1.0) * pow(t, 4.0) * p4 + pow(t, 5.0) * p5;
222 }

224 QPointF CGraphicsPenItem::GetBezierValue(QPainterPath::Element p0, QPainterPath::Element p1, QPainterPath::Element p2, QPainterPath::Element p3, QPainterPath::Element p4, QPainterPath::Element p5, qreal t)
225 {
226  QPointF dot;
227  dot = QPointF(GetBezierValue(p0.x, p1.x, p2.x, p3.x, p4.x, p5.x, t), GetBezierValue(p0.y, p1.y, p2.y, p3.y, p4.y, p5.y, t));
228  return dot;

231 void CGraphicsPenItem::updateStartPathStyle()
232 {
233  if (m_straightLine.isNull()) {
234  if (m_path.boundingRect().width() < minRectSize.width() && m_path.boundingRect().height() < minRectSize.height()) {
235  m_startPath = QPainterPath();
236  return;
237  }
238  }
239 
240  // 判断当前是否是以画直线开始绘制
241  if (!m_isStartWithLine && !m_straightLine.isNull() && m_path.isEmpty() && m_isShiftPress) {
242  m_isStartWithLine = true;
243  } else if (m_isStartWithLine && !m_isShiftPress) {
244  m_isStartWithLine = false;
245  }
246 
247  QLineF line;
248  if (!m_isStartWithLine && m_path.elementCount() > 2) { // 开始画曲线
249  line = QLineF(m_path.elementAt(0), m_path.elementAt(2));
250  } else if (!m_isStartWithLine && m_path.elementCount() > 1) {
251  line = QLineF(m_path.elementAt(0), m_path.elementAt(1));
252  } else { // 开始画直线
253  line = m_straightLine;
254  }
255 
256  QLineF templine(line.p2(), line.p1());
257 
258  QLineF v = templine.unitVector();
259  v.setLength(10 + pen().width() * 3); //改变单位向量的大小，实际就是改变箭头长度
260  v.translate(QPointF(templine.dx(), templine.dy()));
261 
262  QLineF n = v.normalVector(); //法向量
263  n.setLength(n.length() * 0.5); //这里设定箭头的宽度
264  QLineF n2 = n.normalVector().normalVector(); //两次法向量运算以后，就得到一个反向的法向量
265 
266  QPointF p1 = v.p2();
267  QPointF p2 = n.p2();
268  QPointF p3 = n2.p2();
269 
270  //减去一个箭头的宽度
271  QPointF diffV = p1 - line.p1();
272  p1 -= diffV;
273  p2 -= diffV;
274  p3 -= diffV;
275 
276  switch (m_penStartType) {
277  case noneLine: {
278  m_startPath = QPainterPath(p1);
279  break;
280  }
281  case normalArrow: {
282  p1 += diffV;
283  p2 += diffV;
284  p3 += diffV;
285  m_startPath = QPainterPath(p1);
286  m_startPath.lineTo(p2);
287  m_startPath.moveTo(p1);
288  m_startPath.lineTo(p3);
289  m_startPath.moveTo(p1);
290  QPointF center = (p2 + p3) / 2;
291  m_startPath.lineTo(center);
292  m_startPath.moveTo(p1);
293  break;
294  }
295  case soildArrow: {
296  p1 += diffV;
297  p2 += diffV;
298  p3 += diffV;
299  m_startPath = QPainterPath(p1);
300  m_startPath.lineTo(p3);
301  m_startPath.lineTo(p2);
302  m_startPath.lineTo(p1);
303  break;
304  }
305  case normalRing: {
306  //qreal radioWidth = this->pen().width() * 2;
307  qreal penWMin = qMax(pen().width(), 1);
308  qreal radioWidth = penWMin * 2;
309  QPointF center;
310  qreal yOff = qSin(line.angle() / 180 * M_PI) * radioWidth;
311  qreal xOff = qCos(line.angle() / 180 * M_PI) * radioWidth;
312  center = line.p1() + QPointF(-xOff, yOff);
313  QRectF ecliRect(center + QPointF(-radioWidth, -radioWidth), QSizeF(2 * radioWidth, 2 * radioWidth));
314  m_startPath = QPainterPath(center + QPointF(radioWidth, 0));
315  m_startPath.arcTo(ecliRect, 0, 360);
316  break;
317  }
318  case soildRing: {
319  //qreal radioWidth = this->pen().width() * 2;
320  qreal penWMin = qMax(pen().width(), 1);
321  qreal radioWidth = penWMin * 2;
322  QPointF center;
323  qreal yOff = qSin(line.angle() / 180 * M_PI) * radioWidth;
324  qreal xOff = qCos(line.angle() / 180 * M_PI) * radioWidth;
325  center = line.p1() + QPointF(-xOff, yOff);
326  QRectF ecliRect(center + QPointF(-radioWidth, -radioWidth), QSizeF(2 * radioWidth, 2 * radioWidth));
327  m_startPath = QPainterPath(center + QPointF(radioWidth, 0));
328  m_startPath.arcTo(ecliRect, 0, 360);
329  break;
330  }
331  }
332 }

334 void CGraphicsPenItem::updateEndPathStyle()
335 {
336  if (m_straightLine.isNull()) {
337  if (m_path.boundingRect().width() < minRectSize.width() && m_path.boundingRect().height() < minRectSize.height()) {
338  m_endPath = QPainterPath();
339  return;
340  }
341  }
342 
343  // 判断当前是否是以画直线结束绘制
344  if (!m_isEndWithLine && m_isShiftPress) {
345  m_isEndWithLine = true;
346  } else if (m_isEndWithLine && !m_isShiftPress) {
347  m_isEndWithLine = false;
348  }
349 
350  QLineF line;
351  if (m_isEndWithLine) {
352  line = m_straightLine;
353  } else {
354  int count = m_path.elementCount();
355  line = QLineF(m_path.elementAt(qMax(count - 10, 0)), m_path.elementAt(count - 1));
356  }
357 
358 
359 
360  QPointF prePoint = line.p1();
361  QPointF currentPoint = line.p2();
362  if (prePoint == currentPoint) {
363  return;
364  }
365 
366  QLineF v = line.unitVector();
367  v.setLength(10 + pen().width() * 3); //改变单位向量的大小，实际就是改变箭头长度
368  v.translate(QPointF(line.dx(), line.dy()));
369 
370  QLineF n = v.normalVector(); //法向量
371  n.setLength(n.length() * 0.5); //这里设定箭头的宽度
372  QLineF n2 = n.normalVector().normalVector(); //两次法向量运算以后，就得到一个反向的法向量
373 
374  QPointF p1 = v.p2();
375  QPointF p2 = n.p2();
376  QPointF p3 = n2.p2();
377 
378  //减去一个箭头的宽度
379  QPointF diffV = p1 - line.p2();
380  // QPointF diffV = p1 - m_line.p1();
381  p1 -= diffV;
382  p2 -= diffV;
383  p3 -= diffV;
384  // 绘制终点
385  switch (m_penEndType) {
386  case noneLine: {
387  m_endPath = QPainterPath(p1);
388  break;
389  }
390  case normalArrow: {
391  p1 += diffV;
392  p2 += diffV;
393  p3 += diffV;
394  m_endPath = QPainterPath(p1);
395  m_endPath.lineTo(p2);
396  m_endPath.moveTo(p1);
397  m_endPath.lineTo(p3);
398  m_endPath.moveTo(p1);
399  QPointF center = (p2 + p3) / 2;
400  m_endPath.lineTo(center);
401  m_endPath.moveTo(p1);
402  break;
403  }
404  case soildArrow: {
405  p1 += diffV;
406  p2 += diffV;
407  p3 += diffV;
408  m_endPath = QPainterPath(p1);
409  m_endPath.lineTo(p3);
410  m_endPath.lineTo(p2);
411  m_endPath.lineTo(p1);
412  break;
413  }
414  case normalRing: {
415  qreal penWMin = qMax(pen().width(), 1);
416  qreal radioWidth = penWMin * 2;
417  QPointF center;
418  qreal yOff = qSin(line.angle() / 180 * M_PI) * radioWidth;
419  qreal xOff = qCos(line.angle() / 180 * M_PI) * radioWidth;
420  center = line.p2() + QPointF(xOff, -yOff);
421  QRectF ecliRect(center + QPointF(-radioWidth, -radioWidth), QSizeF(2 * radioWidth, 2 * radioWidth));
422  m_endPath = QPainterPath(center + QPointF(radioWidth, 0));
423  m_endPath.arcTo(ecliRect, 0, 360);
424  break;
425  }
426  case soildRing: {
427  qreal penWMin = qMax(pen().width(), 1);
428  qreal radioWidth = penWMin * 2;
429  //qreal radioWidth = this->pen().width() * 2;
430  QPointF center;
431  qreal yOff = qSin(line.angle() / 180 * M_PI) * radioWidth;
432  qreal xOff = qCos(line.angle() / 180 * M_PI) * radioWidth;
433  center = line.p2() + QPointF(xOff, -yOff);
434  QRectF ecliRect(center + QPointF(-radioWidth, -radioWidth), QSizeF(2 * radioWidth, 2 * radioWidth));
435  m_endPath = QPainterPath(center + QPointF(radioWidth, 0));
436  m_endPath.arcTo(ecliRect, 0, 360);
437  break;
438  }
439  }

442 void CGraphicsPenItem::updateHandlesGeometry()
443 {
444  const QRectF &geom = this->boundingRect();
445 
446  const Handles::iterator hend = m_handles.end();
447  for (Handles::iterator it = m_handles.begin(); it != hend; ++it) {
448  CSizeHandleRect *hndl = *it;
449  qreal w = hndl->boundingRect().width();
450  qreal h = hndl->boundingRect().height();
451  switch (hndl->dir()) {
452  case CSizeHandleRect::LeftTop:
453  hndl->move(geom.x() - w / 2, geom.y() - h / 2);
454  break;
455  case CSizeHandleRect::Top:
456  hndl->move(geom.x() + geom.width() / 2 - w / 2, geom.y() - h / 2);
457  break;
458  case CSizeHandleRect::RightTop:
459  hndl->move(geom.x() + geom.width() - w / 2, geom.y() - h / 2);
460  break;
461  case CSizeHandleRect::Right:
462  hndl->move(geom.x() + geom.width() - w / 2, geom.y() + geom.height() / 2 - h / 2);
463  break;
464  case CSizeHandleRect::RightBottom:
465  hndl->move(geom.x() + geom.width() - w / 2, geom.y() + geom.height() - h / 2);
466  break;
467  case CSizeHandleRect::Bottom:
468  hndl->move(geom.x() + geom.width() / 2 - w / 2, geom.y() + geom.height() - h / 2);
469  break;
470  case CSizeHandleRect::LeftBottom:
471  hndl->move(geom.x() - w / 2, geom.y() + geom.height() - h / 2);
472  break;
473  case CSizeHandleRect::Left:
474  hndl->move(geom.x() - w / 2, geom.y() + geom.height() / 2 - h / 2);
475  break;
476  case CSizeHandleRect::Rotation:
477  hndl->move(geom.x() + geom.width() / 2 - w / 2, geom.y() - h - h / 2);
478  break;
479  default:
480  break;
481  }
482  }

594 void CGraphicsPenItem::calcVertexes(const QPointF &prePoint, const QPointF &currentPoint)
595 {
596  if (prePoint == currentPoint) {
597  return;
598  }
599 
600  prepareGeometryChange();
601 
602  updateStartPathStyle();
603  updateEndPathStyle();
604
