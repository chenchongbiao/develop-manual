58 {

139 void IDrawTool::toolDoStart(IDrawTool::CDrawToolEvent *event)
140 {
141  event->scene()->clearHighlight();
142  if (event->mouseButtons() == Qt::LeftButton || event->eventType() == CDrawToolEvent::ETouchEvent) {
143 
144  //m_bMousePress = true;
145  if (dueTouchDoubleClickedStart(event)) {
146  return;
147  }
148  int incW = drawApp->touchFeelingEnhanceValue();
149  ITERecordInfo info;
150 
151  info._prePos = event->pos();
152  info._startPos = event->pos();
153  info.startPosItems = event->eventType() == CDrawToolEvent::ETouchEvent ?
154  event->scene()->items(QRectF(event->pos() - QPoint(incW, incW), QSizeF(2 * incW, 2 * incW))) :
155  event->scene()->items(event->pos());
156  info.startPosTopBzItem = event->scene()->topBzItem(event->pos(), true,
157  event->eventType() == CDrawToolEvent::ETouchEvent ? drawApp->touchFeelingEnhanceValue() : 0);
158  info._isvaild = true;
159  info._curEvent = *event;
160  info._startEvent = *event;
161  info._scene = event->scene();
162  info.getTimeHandle()->restart();
163 
164  if (getCurVaildActivedPointCount() >= allowedMaxTouchPointCount()) {
165  event->setAccepted(true);
166  //超过可支持的点数了
167  info.eventLife = EDoNotthing;
168  qDebug() << "toolDoStart rInfo.eventLife = EDoNotthing id " << event->uuid();
169  _allITERecordInfo.insert(event->uuid(), info);
170  } else if (isPressEventHandledByQt(event, &info)) {
171  event->setAccepted(false);
172  info.eventLife = EDoQtCoversion;
173  qDebug() << "toolDoStart rInfo.eventLife = EDoQtCoversion id " << event->uuid();
174  //只有真实的鼠标事件的调用才需要备份该事件信息,之后会调用toolDoUpdate,toolDoFinished，从而在其中通过UUID找到该事件信息以决定再将该事件传递给mouseEvent
175  //触控操作在这里会被转成qt的鼠标事件直接去调用mouseEvent，不会再进入toolDoStart，toolDoUpdate,toolDoFinished，所以不需要备份保存信息
176  if (event->eventType() == CDrawToolEvent::EMouseEvent)
177  _allITERecordInfo.insert(event->uuid(), info);
178  } else {
179  info.businessItem = creatItem(event, &info);
180  info.eventLife = ENormal;
181  qDebug() << "toolDoStart rInfo.eventLife = ENormal id " << event->uuid();
182 
183  _allITERecordInfo.insert(event->uuid(), info);
184 
185  if (info.businessItem != nullptr) {
186 
187  //工具开始创建图元应该清理当前选中情况
188  event->scene()->clearSelection();
189 
190  toolCreatItemStart(event, &info);
191 
192  } else {
193  toolStart(event, &info);
194  }
195  }
196  }

199 void IDrawTool::toolDoUpdate(IDrawTool::CDrawToolEvent *event)
200 {
201  if (!_allITERecordInfo.isEmpty()) {
202  auto it = _allITERecordInfo.find(event->uuid());
203  if (it != _allITERecordInfo.end()) {
204  ITERecordInfo &rInfo = it.value();
205  rInfo._preEvent = rInfo._curEvent;
206  rInfo._curEvent = *event;
207 
208  //判定是否移动过(根据工具不同移动的最小距离值不同可重载minMoveUpdateDistance)
209  if (!rInfo.hasMoved()) {
210  int constDis = minMoveUpdateDistance();
211 
212  QPointF offset = event->pos(CDrawToolEvent::EViewportPos) -
213  rInfo._startEvent.pos(CDrawToolEvent::EViewportPos)/*event->view()->mapFromScene(rInfo._startPos)*/;
214  int curDis = qRound(offset.manhattanLength());
215  rInfo._moved = (curDis >= constDis);
216  }
217 
218  if (rInfo.eventLife == EDoNotthing) {
219  event->setAccepted(true);
220  } else if (rInfo.eventLife == EDoQtCoversion) {
221  //0.如果开始事件是不被接受的（传递给了Qt框架），那么Update也应该不被接受（也应该传递给Qt框架）
222  event->setAccepted(false);
223  } else {
224  if (!rInfo.haveDecidedOperateType) {
225  //判定应该做什么
226  if (rInfo.businessItem != nullptr) {
227  if (rInfo.hasMoved()) {
228  rInfo._opeTpUpdate = EToolCreatItemMove;
229  rInfo.haveDecidedOperateType = true;
230  }
231  } else {
232  QRectF rectf(event->view()->mapFromScene(rInfo._startPos) - QPointF(10, 10), QSizeF(20, 20));
233  if (!rectf.contains(event->pos(CDrawToolEvent::EViewportPos))) {
234  QTime *elTi = rInfo.getTimeHandle();
235  rInfo._elapsedToUpdate = (elTi == nullptr ? -1 : elTi->elapsed());
236  rInfo._opeTpUpdate = decideUpdate(event, &rInfo);
237 
238  if (rInfo._opeTpUpdate > 0) {
239  sendToolEventToItem(event, &rInfo, EChangedBegin);
240  }
241 
242  rInfo.haveDecidedOperateType = true;
243  }
244  }
245  }
246  //2.执行操作
247  if (rInfo.haveDecidedOperateType) {
248  if (rInfo._opeTpUpdate == EToolCreatItemMove) {
249  toolCreatItemUpdate(event, &rInfo);
250  } else if (rInfo._opeTpUpdate > EToolDoNothing) {
251  toolUpdate(event, &rInfo);
252  }
253  }
254  }
255  //qDebug() << "event->isPosXAccepted() = " << event->isPosXAccepted() << "event->isPosYAccepted() = " << event->isPosYAccepted();
256  if (event->isPosXAccepted())
257  rInfo._prePos.setX(event->pos().x());
258  if (event->isPosYAccepted())
259  rInfo._prePos.setY(event->pos().y());
260  }
261  } else {
262  mouseHoverEvent(event);
263  }
264 }

266 void IDrawTool::toolDoFinish(IDrawTool::CDrawToolEvent *event)
267 {
268  bool updateCursor = true;
269  qDebug() << "toolDoFinish ==== " << event->uuid();
270  if (!_allITERecordInfo.isEmpty()) {
271  auto it = _allITERecordInfo.find(event->uuid());
272  if (it != _allITERecordInfo.end()) {
273  ITERecordInfo &rInfo = it.value();
274  rInfo._prePos = event->pos();
275  rInfo._preEvent = rInfo._curEvent;
276  rInfo._curEvent = *event;
277  qDebug() << "toolDoFinish rInfo.eventLife = " << rInfo.eventLife;
278 
279  CGraphicsItem *pCreatedItem = nullptr;
280  if (rInfo.eventLife == EDoNotthing) {
281  event->setAccepted(true);
282  } else if (rInfo.eventLife == EDoQtCoversion) {
283  event->setAccepted(false);
284  } else {
285  //1.根据操作类型决定要做的事情
286  if (rInfo.businessItem != nullptr) {
287  toolCreatItemFinish(event, &rInfo);
288  pCreatedItem = rInfo.businessItem;
289  if (rInfo.businessItem != nullptr) {
290  if (rInfo.businessItem->scene() == event->scene()) {
291  //qWarning() << "zzzzzzzzz = " << rInfo.businessItem->zValue();
292  if (autoSupUndoForCreatItem())
293  CCmdBlock block(event->scene(), CSceneUndoRedoCommand::EItemAdded, rInfo.businessItem);
294  //event->scene()->selectItem(rInfo.businessItem);
295  }
296  }
297  } else { /*if (rInfo._opeTpUpdate > EToolDoNothing)*/
298  if (rInfo._opeTpUpdate > EToolDoNothing)
299  sendToolEventToItem(event, &rInfo, EChangedFinished);
300  toolFinish(event, &rInfo);
301  }
302  // 保证恢复到正常绘制
303  if (event->view()->isCacheEnabled()) {
304  event->view()->setCacheEnable(false);
305  }
306  }
307 
308  qDebug() << "finished uuid ===== " << event->uuid() << "allITERecordInfo count = " << _allITERecordInfo.count();
309 
310  //2.是否要回到select工具模式下去(多个触控时即将为空才判断)
311  if (_allITERecordInfo.count() == 1 && returnToSelectTool(event, &rInfo)) {
312  if (pCreatedItem != nullptr) {
313  event->scene()->selectItem(pCreatedItem);
314  event->scene()->setFocusItem(pCreatedItem);
315  }
316  setViewToSelectionTool();
317  } else {
318  //不回到select工具时不用刷新鼠标样式（维持工具的鼠标样式方便提示用户可继续绘制）
319  updateCursor = false;
320  }
321  _allITERecordInfo.erase(it);
322  }
323  } else {
324  event->setAccepted(false);
325  }
326  if (event->eventType() == CDrawToolEvent::EMouseEvent && updateCursor)
327  event->scene()->refreshLook(event->pos());
328 
329  event->view()->setFocus();
330 }

338 bool IDrawTool::dueTouchDoubleClickedStart(IDrawTool::CDrawToolEvent *event)
339 {
340  static const int intervalMs = 250;
341  if (event->eventType() == CDrawToolEvent::ETouchEvent) {
342  static QPointF prePos;
343  //touch触控判定是否是双击
344  if (!getTimerForDoubleCliked()->isActive()) {
345  prePos = event->pos();
346  getTimerForDoubleCliked()->start(intervalMs);
347  } else {
348  //判定移动的幅度很小
349  QRectF rectf(prePos - QPointF(10, 10), QSizeF(20, 20));
350  if (rectf.contains(event->pos())) {
351  m_bMousePress = true;
352 
353  ITERecordInfo info;
354 
355  info._prePos = event->pos();
356  info._startPos = event->pos();
357  info.startPosItems = event->scene()->items(event->pos());
358  info.startPosTopBzItem = event->scene()->topBzItem(event->pos());
359  info._isvaild = true;
360  info._curEvent = *event;
361  info._scene = event->scene();
362 
363  _allITERecordInfo.insert(event->uuid(), info);
364 
365  if (isPressEventHandledByQt(event, &info)) {
366  //触控事件的双击如果需要传递给Qt去处理,那么需要清理掉事件
367  event->setAccepted(false);
368  interrupt();
369  } else {
370  toolDoubleClikedEvent(event, &info);
371  }
372  getTimerForDoubleCliked()->stop();
373 
374  return true;
375  }
376  }
377  } else if (event->eventType() == CDrawToolEvent::EMouseEvent) {
378  m_bMousePress = true;
379  QEvent::Type tp = event->orgQtEvent()->type();
380  if ((tp == QEvent::MouseButtonDblClick || tp == QEvent::GraphicsSceneMouseDoubleClick) && event->mouseButtons() == Qt::LeftButton) {
381  ITERecordInfo info;
382 
383  info._prePos = event->pos();
384  info._startPos = event->pos();
385  info.startPosItems = event->scene()->items(event->pos());
386  info.startPosTopBzItem = event->scene()->topBzItem(event->pos());
387  info._isvaild = true;
388  info._curEvent = *event;
389  info._scene = event->scene();
390 
391  _allITERecordInfo.insert(event->uuid(), info);
392 
393  if (isPressEventHandledByQt(event, &info)) {
394  event->setAccepted(false);
395  _allITERecordInfo[event->uuid()].eventLife = EDoQtCoversion;
396  } else {
397  toolDoubleClikedEvent(event, &info);
398  }
399  return true;
400  }
401  }
402  return false;
403 }

776 IDrawTool::CDrawToolEvent::CDrawToolEvents IDrawTool::CDrawToolEvent::fromQEvent(QEvent *event, CDrawScene *scene)
777 {
778  CDrawToolEvents eList;
779  CDrawToolEvent e;
780  switch (event->type()) {
781  case QEvent::MouseButtonPress:
782  case QEvent::MouseButtonRelease:
783  case QEvent::MouseMove:
784  case QEvent::MouseButtonDblClick: {
785  QMouseEvent *msEvent = dynamic_cast<QMouseEvent *>(event);
786  //e._pos[EViewportPos] = msEvent->pos();
787  e._pos[EViewportPos] = /*msEvent->pos()*/scene->drawView()->viewport()->mapFromGlobal(msEvent->globalPos());
788  e._pos[EGlobelPos] = msEvent->globalPos();
789  e._msBtns = msEvent->button() | msEvent->buttons();
790  e._kbMods = msEvent->modifiers();
791  e._orgEvent = event;
792  e._scene = scene;
793  eList.insert(0, e);
794  break;
795  }
796  case QEvent::GraphicsSceneMouseMove:
797  case QEvent::GraphicsSceneMousePress:
798  case QEvent::GraphicsSceneMouseRelease:
799  case QEvent::GraphicsSceneMouseDoubleClick: {
800  QGraphicsSceneMouseEvent *msEvent = dynamic_cast<QGraphicsSceneMouseEvent *>(event);
801  e._pos[EViewportPos] = /*msEvent->pos()*/scene->drawView()->viewport()->mapFromGlobal(msEvent->screenPos());
802  e._pos[EScenePos] = msEvent->scenePos();
803  e._pos[EGlobelPos] = msEvent->screenPos();
804  e._msBtns = msEvent->button() | msEvent->buttons();
805  e._kbMods = msEvent->modifiers();
806  e._orgEvent = event;
807  e._scene = scene;
808  eList.insert(0, e);
809  break;
810  }
811  case QEvent::TouchBegin:
812  case QEvent::TouchUpdate:
813  case QEvent::TouchEnd: {
814  QTouchEvent *touchEvent = dynamic_cast<QTouchEvent *>(event);
815  const QList<QTouchEvent::TouchPoint> lists = touchEvent->touchPoints();
816  for (auto tPos : lists) {
817  e = fromTouchPoint(tPos, scene);
818  e._orgEvent = event;
819  eList.insert(e.uuid(), e);
820  }
821  break;
822  }
823  default:
824  break;
825  }
826  return eList;
827 }
