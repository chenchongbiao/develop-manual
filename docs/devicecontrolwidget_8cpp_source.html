41 DeviceControlWidget::DeviceControlWidget(QWidget *parent)
42  : QWidget(parent)
43 {
44  m_deviceName = new QLabel(this);
45  m_deviceName->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Preferred);
46  initFontColor(m_deviceName);
47 
48  m_switchBtn = new DSwitchButton;
49 
50  const QPixmap pixmap = DHiDPIHelper::loadNxPixmap(":/wireless/resources/wireless/refresh.svg");
51 
52  m_loadingIndicator = new DLoadingIndicator;
53  m_loadingIndicator->setLoading(false);
54  m_loadingIndicator->setSmooth(true);
55  m_loadingIndicator->setAniDuration(1000);
56  m_loadingIndicator->setAniEasingCurve(QEasingCurve::InOutCirc);
57  m_loadingIndicator->installEventFilter(this);
58  m_loadingIndicator->setFixedSize(pixmap.size() / devicePixelRatioF());
59  m_loadingIndicator->viewport()->setAutoFillBackground(false);
60  m_loadingIndicator->setFrameShape(QFrame::NoFrame);
61  refreshIcon();
62 
63  QHBoxLayout *infoLayout = new QHBoxLayout;
64  infoLayout->setMargin(0);
65  infoLayout->setSpacing(0);
66  infoLayout->addSpacing(3);
67  infoLayout->addWidget(m_deviceName);
68  infoLayout->addStretch();
69  infoLayout->addWidget(m_loadingIndicator);
70  infoLayout->addSpacing(10);
71  infoLayout->addWidget(m_switchBtn);
72  infoLayout->addSpacing(3);
73 
74  QVBoxLayout *centralLayout = new QVBoxLayout;
75  centralLayout->addStretch();
76  centralLayout->addLayout(infoLayout);

101 bool DeviceControlWidget::eventFilter(QObject *watched, QEvent *event)
102 {
103  if (watched == m_loadingIndicator) {
104  if (event->type() == QEvent::MouseButtonPress) {
105  if (!m_loadingIndicator->loading()) {
106  refreshNetwork();
107  }
108  }
109  }
110 
111  return QWidget::eventFilter(watched, event);
112 }

114 void DeviceControlWidget::refreshNetwork()
115 {
116  emit requestRefresh();
117 
118  m_loadingIndicator->setLoading(true);
119 
120  QTimer::singleShot(1000, this, [ = ] {
121  m_loadingIndicator->setLoading(false);
122  });
123 }

132 
133  m_loadingIndicator->setImageSource(pixmap);
