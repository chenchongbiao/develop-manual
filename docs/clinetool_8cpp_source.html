37 void CLineTool::toolCreatItemUpdate(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *pInfo)
38 {
39  Q_UNUSED(event)
40  if (pInfo != nullptr) {
41  CGraphicsLineItem *pItem = dynamic_cast<CGraphicsLineItem *>(pInfo->businessItem);
42  if (nullptr != pItem) {
43  bool shiftKeyPress = event->keyboardModifiers() & Qt::ShiftModifier;
44  if (!shiftKeyPress) {
45  pItem->setLine(QLineF(pInfo->_startPos, event->pos()));
46  } else {
47  QPointF local = pItem->mapFromScene(event->pos());
48  QPointF p1;
49  QPointF p2;
50 
51  p1 = pItem->line().p1();
52  p2 = local;
53 
54  QLineF v = QLineF(p1, p2);
55  if (fabs(v.dx()) - fabs(v.dy()) > 0.0001) {
56  p2.setY(p1.y());
57  } else {
58  p2.setX(p1.x());
59  }
60 
61  p1 = pItem->mapToScene(p1);
62  p2 = pItem->mapToScene(p2);
63  pItem->setRotation(0);
64  pItem->setPos(0, 0);
65  pItem->setLine(p1, p2);
66  }
67  }
68  }
69 }

71 void CLineTool::toolCreatItemFinish(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *pInfo)
72 {
73  if (pInfo != nullptr) {
74  CGraphicsLineItem *m_pItem = dynamic_cast<CGraphicsLineItem *>(pInfo->businessItem);
75  if (nullptr != m_pItem) {
76  if (!pInfo->hasMoved()) {
77  event->scene()->removeCItem(m_pItem);
78  delete m_pItem;
79  pInfo->businessItem = nullptr;
80  } else {
81  if (m_pItem->scene() == nullptr) {
82  m_pItem->drawScene()->addCItem(m_pItem);
83  }
84  m_pItem->setSelected(true);
85  }
86  }
87  }
88 
89  IDrawTool::toolCreatItemFinish(event, pInfo);
90 }

92 CGraphicsItem *CLineTool::creatItem(IDrawTool::CDrawToolEvent *event, ITERecordInfo *pInfo)
93 {
94  Q_UNUSED(pInfo)
95  if ((event->eventType() == CDrawToolEvent::EMouseEvent && event->mouseButtons() == Qt::LeftButton)
96  || event->eventType() == CDrawToolEvent::ETouchEvent) {
97 
98  CGraphicsView *pView = event->scene()->drawView();
99  CGraphicsLineItem *m_pItem = new CGraphicsLineItem(event->pos().x(), event->pos().y(), event->pos().x(), event->pos().y());
100  m_pItem->setPen(pView->getDrawParam()->getPen());
101  m_pItem->setBrush(pView->getDrawParam()->getBrush());
102  m_pItem->setLineStartType(event->view()->getDrawParam()->getLineStartType());
103  m_pItem->setLineEndType(event->view()->getDrawParam()->getLineEndType());
104  event->scene()->addCItem(m_pItem);
105  return m_pItem;
106  }
107  return nullptr;
108 }
