119 static bool cp_to_stream_offset(const Parser *parser, std::ostream &log_stream, U32 cp, U32 &stream_offset, bool *unicode_detected = NULL)
120 {
121  const Parser9x *parser9 = dynamic_cast<const Parser9x *>(parser);
122  if (parser9 == NULL) {
123  log_stream << "This is not a 9x parser.\n";
124  return false;
125  }
126  U32 piece = 0;
127  U32 offset = cp;
128  if (parser9->m_plcfpcd == NULL) {
129  log_stream << "No pieces table found.\n";
130  return false;
131  }
132  // Copied from wv2 code.
133  PLCFIterator<Word97::PCD> it( *parser9->m_plcfpcd );
134  for ( ; it.current(); ++it, ++piece ) {
135  if ( it.currentLim() > cp && it.currentStart() <= cp )
136  break;
137  offset -= it.currentRun();
138  }
139  log_stream << "Piece: " << piece << ", offset: " << offset << "\n";
140  PLCFIterator<Word97::PCD> it2(parser9->m_plcfpcd->at( piece ) );
141  if (!it2.current()) {
142  log_stream << "Specified piece not found.\n";
143  return false;
144  }
145  U32 fc = it2.current()->fc; // Start FC of this piece
146  log_stream << "Piece start at FC " << fc << ".\n";
147  bool unicode;
148  parser9->realFC(fc, unicode);
149  log_stream << "After unicode transition piece start at FC " << fc << ".\n";
150  if (offset != 0 )
151  fc += unicode ? offset * 2 : offset;
152  log_stream << "Stream offset is " << fc << ".\n";
153  stream_offset = fc;
154  if (unicode_detected != NULL)
155  *unicode_detected = unicode;
156  return true;
157 }

159 static bool parse_comments(const Parser *parser, std::ostream &log_stream, std::vector<Comment> &comments)
160 {
161  if (parser->fib().lcbPlcfandTxt == 0) {
162  log_stream << "No annotations.\n";
163  return true;
164  }
165 
166  U32 atn_part_cp = parser->fib().ccpText + parser->fib().ccpFtn + parser->fib().ccpHdd + parser->fib().ccpMcr;
167  log_stream << "Annotations part at CP " << atn_part_cp << ".\n";
168  AbstractOLEStreamReader *reader = parser->m_storage->createStreamReader("WordDocument");
169  if (!reader) {
170  log_stream << "Error opening WordDocument stream.\n";
171  return false;
172  }
173  const Parser9x *parser9 = dynamic_cast<const Parser9x *>(parser);
174  AbstractOLEStreamReader *table_reader = parser->m_storage->createStreamReader(parser9->tableStream());
175  if (!table_reader) {
176  log_stream << "Error opening table stream..\n";
177  delete reader;
178  return false;
179  }
180 
181  U32 annotation_txts_offset = parser->fib().fcPlcfandTxt;
182  log_stream << "Annotation texts table at offset " << annotation_txts_offset << ".\n";
183  table_reader->seek(annotation_txts_offset);
184  U32 annotation_begin_cp = table_reader->readU32();
185  std::vector<std::string> annotations;
186  for (;;) {
187  U32 annotation_end_cp = table_reader->readU32();
188  log_stream << "Annotation text position (CP) from " << annotation_begin_cp << " to " << annotation_end_cp << ".\n";
189  U32 stream_begin_offset;
190  bool unicode;
191  if (!cp_to_stream_offset(parser, log_stream, atn_part_cp + annotation_begin_cp, stream_begin_offset, &unicode)) {
192  log_stream << "Converting annotation start position to stream offset failed.\n";
193  delete reader;
194  delete table_reader;
195  return false;
196  }
197  U32 stream_end_offset;
198  if (!cp_to_stream_offset(parser, log_stream, atn_part_cp + annotation_end_cp, stream_end_offset)) {
199  log_stream << "Converting annotation end position to stream offset failed.\n";
200  delete reader;
201  delete table_reader;
202  return false;
203  }
204  log_stream << "Annotation text stream position from " << stream_begin_offset << " to " << stream_end_offset << ".\n";
205  reader->seek(stream_begin_offset);
206  U8 annotation_mark = reader->readU8();
207  if (!annotation_mark == 0x05) {
208  log_stream << "Incorrect annotation mark.\n";
209  delete reader;
210  delete table_reader;
211  return false;
212  }
213  std::string annotation;
214  while (reader->tell() < stream_end_offset - 1) {
215  log_stream << "Stream pos " << reader->tell() << "\n";
216 #warning TODO: Unicode support in comments
217  if (unicode)
218  reader->seek(1, SEEK_CUR); // skip unicode byte
219  S8 ch = reader->readS8();
220  if (ch >= 32 || (ch >= 8 && ch <= 13)) {
221  if (ch == '\r')
222  annotation += "\n";
223  else
224  annotation += ch;
225  }
226  }
227  log_stream << "Annotation text: \"" << annotation << "\"\n";
228  annotations.push_back(annotation);
229  if (annotation_end_cp >= parser->fib().ccpAtn - 1)
230  break;
231  annotation_begin_cp = annotation_end_cp;
232  }
233 
234  U32 annotation_owners_offset = parser->fib().fcGrpXstAtnOwners;
235  U32 annotation_owners_len = parser->fib().lcbGrpXstAtnOwners;
236  log_stream << "Annotation owners table at offset " << annotation_owners_offset << " has length " << annotation_owners_len << ".\n";
237  table_reader->seek(annotation_owners_offset);
238  std::vector<std::string> owners;
239  for (int total_len = 0; total_len < annotation_owners_len;) {
240  U16 len = table_reader->readU16();
241  total_len += 2;
242  std::string owner;
243  if (len * 2 + total_len > annotation_owners_len) {
244  log_stream << "Something is wrong with XST table. Authors of comments could not be parsed.\n";
245  owners.clear();
246  break;
247  }
248  for (int i = 0; i < len; i++) {
249  S16 ch = table_reader->readS16();
250  owner += (char)ch;
251  total_len += 2;
252  }
253  log_stream << "Owner \"" << owner << "\" found.\n";
254  owners.push_back(owner);
255  }
256 
257  U32 annotation_refs_offset = parser->fib().fcPlcfandRef;
258  log_stream << "Annotation refs table at offset " << annotation_refs_offset << ".\n";
259  table_reader->seek(annotation_refs_offset);
260  for (int i = 0;; i++) {
261  U32 annotation_ref_cp = table_reader->readU32();
262  if (annotation_ref_cp >= parser->fib().ccpText)
263  break;
264  log_stream << "Annotation " << i << " references text at CP " << annotation_ref_cp << ".\n";
265  U32 stream_offset;
266  if (!cp_to_stream_offset(parser, log_stream, annotation_ref_cp, stream_offset)) {
267  log_stream << "Converting annotation reference position to stream offset failed.\n";
268  delete reader;
269  delete table_reader;
270  return false;
271  }
272  if (i < annotations.size()) {
273  log_stream << "Annotation " << i << " references text at stream offset " << stream_offset << ".\n";
274  Comment c;
275  c.fc = stream_offset;
276  c.text = annotations[i];
277  comments.push_back(c);
278  }
279  }
280  table_reader->seek(annotation_refs_offset + 4 * (comments.size() + 1));
281  for (int i = 0; i < comments.size(); i++) {
282  Word97::ATRD atrd(table_reader);
283  if (atrd.ibst < owners.size()) {
284  log_stream << "Annotation owner is \"" << owners[atrd.ibst] << "\", index " << atrd.ibst << ".\n";
285  comments[i].author = owners[atrd.ibst];
286  }
287  }
288  delete table_reader;
289  delete reader;
290  return true;
291 }

293 static std::string format_comment(const std::string &author, const std::string &text)
294 {
295  std::string comment = "\n[[[COMMENT BY " + author + "]]]\n" + text;

297  comment += "\n";
298  comment += "[[[---]]]\n";

434  void fieldEnd(const FLD *fld,
435  SharedPtr<const Word97::CHP> chp)
436  {
437  UString params = m_curr_state->field_params;
438  int i = 0;
439  while (i < params.length() && params[i] == ' ') i++;
440  UString field_type;
441  while (i < params.length() && params[i] != ' ') {
442  field_type += UString(params[i]);
443  i++;
444  }
445  while (i < params.length() && params[i] == ' ') i++;
446  params = params.substr(i);
447  UString res_text;
448  switch (m_curr_state->field_type) {
449  case FLT_FILLIN:
450  res_text = params + UString(" ") + m_curr_state->field_value;
451  break;
452  case FLT_EMBED:
453  break;
454  case FLT_HYPERLINK:
455  if (params[0] == '"') {
456  UString hyperlink_url;
457  for (i = 1; i < params.length() && params[i] != '"'; i++)
458  hyperlink_url += UString(params[i]);
459  res_text = UString(formatUrl(ustring_to_string(hyperlink_url), ustring_to_string(m_curr_state->field_value), m_formatting).c_str());
460  } else
461  res_text = params + UString(" ") + m_curr_state->field_value;
462  break;
463  default:
464  res_text = m_curr_state->field_value;
465  }
466  m_curr_state->field_type = FLT_NONE;
467  m_curr_state->field_part = FIELD_PART_NONE;
468  if (m_curr_state->table.in_table)
469  m_curr_state->table.curr_cell_text += ustring_to_string(res_text);
470  else if (m_curr_state->header_footer.in_header)
471  m_curr_state->header_footer.header += res_text;
472  else if (m_curr_state->header_footer.in_footer)
473  m_curr_state->header_footer.footer += res_text;
474  else
475  (*Text) += res_text;
476  }

621 bool DOCParser::isDOC()
622 {
623  impl->m_error = false;
624  FILE *f = NULL;
625  if (!impl->m_buffer)
626  f = fopen(impl->m_file_name.c_str(), "r");
627  if (f == NULL && !impl->m_buffer) {
628  *impl->m_log_stream << "Error opening file " << impl->m_file_name << ".\n";
629  impl->m_error = true;
630  return false;
631  }
632  if (f)
633  fclose(f);
634  impl->modifyCerr();
635  ThreadSafeOLEStorage *storage = NULL;
636  //storage will be deleted inside parser from wv2 library
637  if (impl->m_buffer)
638  storage = new ThreadSafeOLEStorage(impl->m_buffer, impl->m_buffer_size);

702  }
