89 void FingerModel::onEnrollStatusChanged(int code, const QString& msg)
90 {
91  qDebug() << "onEnrollStatusChanged,code is: " << code << ",msg is: " << msg;
92  QJsonDocument jsonDocument;
93  QJsonObject jsonObject;
94 
95  if (!msg.isEmpty()) {
96  jsonDocument = QJsonDocument::fromJson(msg.toLocal8Bit().data());
97  jsonObject = jsonDocument.object();
98  }
99 
100  switch(code) {
101  case ET_Completed:
102  m_progress = 0;
103  Q_EMIT enrollCompleted();
104  break;
105  case ET_Failed: {
106  m_progress = 0;
107  QString title = tr("Scan failed");
108  QString msg = "";
109  do {
110  QStringList keys = jsonObject.keys();
111  if (!keys.contains("subcode")) {
112  break;
113  }
114  auto errCode = jsonObject.value("subcode").toInt();
115  switch(errCode) {
116  case FC_RepeatTemplet:
117  title = tr("The fingerprint already exists");
118  msg = tr("Please scan other fingers");
119  break;
120  case FC_UnkownError:
121  title = tr("Unknown error");
122  msg = tr("Scan suspended");
123  }
124  break;
125  } while(0);
126  Q_EMIT enrollFailed(title, msg);
127  break;
128  }
129  case ET_StagePass: {
130  if (msg.isEmpty()) {
131  // 厂商未给出进度值的情况,直接让进度值递减增加,无限趋近于100,直到录入完成
132  m_progress += (100 - m_progress)/3;
133  Q_EMIT enrollStagePass(m_progress);
134  break;
135  }
136  QStringList keys = jsonObject.keys();
137  if (!keys.contains("progress")) {
138  // 厂商未给出进度值的情况,直接让进度值递减增加,无限趋近于100,直到录入完成
139  m_progress += (100 - m_progress)/3;
140  Q_EMIT enrollStagePass(m_progress);
141  break;
142  }
143  auto pro = jsonObject.value("progress").toInt();
144  Q_EMIT enrollStagePass(pro);
145  break;
146  }
147  case ET_Retry: {
148  QString title = tr("Cannot recognize");
149  QString msg = tr("Cannot recognize");
150  do {
151  QStringList keys = jsonObject.keys();
152  if (!keys.contains("subcode")) {
153  break;
154  }
155  auto errCode = jsonObject.value("subcode").toInt();
156  switch(errCode) {
157  case RC_TouchTooShort: //接触时间过短
158  title = tr("Moved too fast");
159  msg = tr("Finger moved too fast, please do not lift until prompted");
160  break;
161  case RC_ErrorFigure: //图像不可用
162  title = tr("Unclear fingerprint");
163  msg = tr("Clean your finger or adjust the finger position, and try again");
164  break;
165  case RC_RepeatTouchData: //重复率过高
166  title = tr("Already scanned");
167  msg = tr("Adjust the finger position to scan your fingerprint fully");
168  break;
169  case RC_RepeatFingerData: //重复手指
170  title = tr("The fingerprint already exists");
171  msg = tr("Please scan other fingers");
172  break;
173  case RC_SwipeTooShort: //按压时间短
174  title = tr("Moved too fast");
175  msg = tr("Finger moved too fast. Please do not lift until prompted");
176  break;
177  case RC_FingerNotCenter: //手指不在中间
178  msg = tr("Adjust the finger position to scan your fingerprint fully");
179  break;
180  case RC_RemoveAndRetry: // 拿开手指从新扫描
181  msg = tr("Clean your finger or adjust the finger position, and try again");
182  break;
183  case RC_CannotRecognize:
184  title = tr("Cannot recognize");
185  msg = tr("Lift your finger and place it on the sensor again");
186  break;
187  }
188  break;
189  } while(0);
190  Q_EMIT enrollRetry(title, msg);
191  break;
192  }
193  case ET_Disconnect:
194  Q_EMIT enrollDisconnected();
195  break;
196  default:
197  break;
198  }
199 }
