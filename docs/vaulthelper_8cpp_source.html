46 bool VaultHelper::topVaultTasks()
47 {
48  // 如果正在有保险箱的移动、粘贴、删除操作，置顶弹出任务框
49  DTaskDialog *pTaskDlg = dialogManager->taskDialog();
50  if(pTaskDlg){
51  if(pTaskDlg->bHaveNotCompletedVaultTask()){
52  // Flashing alert
53  pTaskDlg->hide();
54  pTaskDlg->showDialogOnTop();
55  return true;
56  }
57  }
58  // 如果当前有保险箱的压缩或解压缩任务，激活任务对话框进程
59  QString strCmd = GET_COMPRESSOR_PID_SHELL(VAULT_BASE_PATH);
60  QStringList lstShellOutput;
61  // 执行shell命令，获得压缩进程PID
62  int res = InterfaceActiveVault::executionShellCommand(strCmd, lstShellOutput);
63  if(res == 0){ // shell命令执行成功
64  QStringList::const_iterator itr = lstShellOutput.begin();
65  QSet<QString> setResult;
66  for(; itr != lstShellOutput.end(); ++itr){
67  setResult.insert(*itr);
68  }
69  if(setResult.count() > 0){ // 有压缩任务
70  // 遍历桌面窗口
71  bool bFlag = false;
72  for(auto window: DWindowManagerHelper::instance()->currentWorkspaceWindows()) {
73  QString strWid = QString("%1").arg(window->pid());
74  // 如果当前窗口的进程PID属于压缩进程，则将窗口置顶
75  if(setResult.contains(strWid)){
76  window->raise();
77  bFlag = true;
78  }
79  }
80  if(bFlag){
81  return true;
82  }
83  }
84  }else{
85  qDebug() << "Failed to execute PID search command!";
86  }

90  "/org/freedesktop/FileManager1",
91  "org.freedesktop.FileManager1",
92  "topTaskDialog");
93  // 修复BUG-44055 设置超时等待为1000毫米，提高用户操作流畅度
94  QDBusMessage response = QDBusConnection::sessionBus().call(message, QDBus::Block, 1000);
