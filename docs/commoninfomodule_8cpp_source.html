118 int CommonInfoModule::load(const QString &path)
119 {
120  if (!m_commonWidget) {
121  active();
122  }
123 
124  QListView *list = m_commonWidget->getCommonListView();
125  if (!list) {
126  return 0;
127  }
128 
129  int indexRow = 0;
130  if (path == "Boot Menu") {
131  indexRow = 0;
132  }
133 
134  if (!IsServerSystem) {
135  if (!IsCommunitySystem && (path == "Developer Mode")) {
136  // 为开发者模式的search预留
137  indexRow = 1;
138  } else if (!IsCommunitySystem && (path == "User Experience Program")) {
139  // 为用户体验计划的search预留
140  indexRow = 2;
141  } else if (path == "Tablet Mode") {
142  // 为平板模式的search预留
143  //indexRow = 3;
144  }
145  }
146 
147 #ifdef DCC_DISABLE_GRUB
148  indexRow -= 1;
149  Q_ASSERT(indexRow >= 0);
150 #endif
151 
152  QModelIndex idx = list->model()->index(indexRow, 0);
153  list->setCurrentIndex(idx);
154  list->clicked(idx);
155 
156  return 0;
157 }
