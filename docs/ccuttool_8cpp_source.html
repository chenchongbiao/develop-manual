70 int CCutTool::decideUpdate(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *pInfo)
71 {
72  EOperateType ret = ENothing;
73 
74  QGraphicsItem *pStartPosTopQtItem = event->scene()->firstItem(pInfo->_startPos,
75  pInfo->startPosItems, true, true);
76  if (event->scene()->isBussizeHandleNodeItem(pStartPosTopQtItem)) {
77  CSizeHandleRect *pHandle = dynamic_cast<CSizeHandleRect *>(pStartPosTopQtItem);
78  pInfo->_etcopeTpUpdate = pHandle->dir();
79  pInfo->etcItems.clear();
80 
81  pInfo->etcItems.append(m_pCutItem);
82 
83  ret = EResizeMove;
84  } else if (pStartPosTopQtItem == m_pCutItem) {
85  ret = EDragMove;
86  }
87 
88  pInfo->_opeTpUpdate = ret;
89 
90  return pInfo->_opeTpUpdate;
91 }

93 void CCutTool::toolUpdate(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *pInfo)
94 {
95  if (pInfo->_opeTpUpdate == EDragMove) {
96  m_pCutItem->move(pInfo->_prePos, event->pos());
97  } else if (pInfo->_opeTpUpdate == EResizeMove) {
98  CSizeHandleRect::EDirection direction = CSizeHandleRect::EDirection(pInfo->_etcopeTpUpdate);
99  m_pCutItem->resizeCutSize(direction, pInfo->_prePos, event->pos(), &pInfo->_prePos);
100  drawApp->topToolbar()->attributWidget()->getCutWidget()->setCutSize(m_pCutItem->rect().size().toSize(), false);
101  }
102 }

111 void CCutTool::mouseHoverEvent(IDrawTool::CDrawToolEvent *event)
112 {
113  QPointF scenePos = event->pos();
114 
115  QList<QGraphicsItem *> items = event->scene()->items(scenePos);
116 
117  QGraphicsItem *pItem = event->scene()->firstItem(scenePos, items, true, true, false, false);
118 
119  if (event->scene()->isBussizeHandleNodeItem(pItem)) {
120  CSizeHandleRect *pHandle = dynamic_cast<CSizeHandleRect *>(pItem);
121  drawApp->setApplicationCursor(pHandle->getCursor());
122  } else if (pItem != nullptr && pItem->type() == CutType) {
123  drawApp->setApplicationCursor(Qt::OpenHandCursor);
124  } else {
125  drawApp->setApplicationCursor(Qt::ArrowCursor);
126  }
127 }

129 void CCutTool::createCutItem(CDrawScene *scene)
130 {
131  if (m_cutItems.find(scene) == m_cutItems.end()) {
132  deleteCutItem(scene);
133 
134  scene->clearSelection();
135 
136  m_pCutItem = new CGraphicsCutItem(scene->sceneRect());
137  scene->addCItem(m_pCutItem);
138 
139  m_dragHandle = CSizeHandleRect::None;
140 
141  m_pCutItem->setIsFreeMode(true);
142  m_pCutItem->setSelected(true);
143  m_bModify = false;
144 
145  m_cutItems.insert(scene, m_pCutItem);
146  }
147 
148 }

150 void CCutTool::deleteCutItem(CDrawScene *scene)
151 {
152  drawApp->setApplicationCursor(Qt::ArrowCursor);
153 
154  auto itf = m_cutItems.find(scene);
155  if (itf != m_cutItems.end()) {
156  qDebug() << "deleteCutItem scene tag name = " << scene->getDrawParam()->viewName();
157  CGraphicsCutItem *pCutItem = itf.value();
158  scene->removeCItem(pCutItem);
159 
160  if (pCutItem == m_pCutItem) {
161  m_pCutItem = nullptr;
162  }
163 
164  delete pCutItem;
165  m_cutItems.erase(itf);
166  }
167 }
