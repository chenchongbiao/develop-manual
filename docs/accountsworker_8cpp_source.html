287 {

295  first = false;
296  break; // 第一次打开用户列表只加载51个用户信息
297  } else if (count > 2 && !first) {
298  break; // 滚动条每滑动一次，加载3个用户
299  }
300  }

304 void AccountsWorker::setPassword(User *user, const QString &oldpwd, const QString &passwd)
305 {
306  QProcess process;
307  QProcessEnvironment env;
308  env.insert("LC_ALL", "C");
309  process.setProcessEnvironment(env);
310  process.start("/bin/bash", QStringList() << "-c" << QString("passwd"));
311  if (user->passwordStatus() == NO_PASSWORD) {
312  process.write(QString("%1\n%2\n").arg(passwd).arg(passwd).toLatin1());
313  } else {
314  process.write(QString("%1\n%2\n%3").arg(oldpwd).arg(passwd).arg(passwd).toLatin1());
315  }
316  process.closeWriteChannel();
317  process.waitForFinished();
318 
319  // process.exitCode() = 0 表示密码修改成功
320  int exitCode = process.exitCode();
321  Q_EMIT user->passwordModifyFinished(exitCode);
322 }

332 void AccountsWorker::addUser(const QString &userPath)
333 {
334  if (userPath.contains("User0", Qt::CaseInsensitive))
335  return;
336  AccountsUser *userInter = new AccountsUser(AccountsService, userPath, QDBusConnection::systemBus(), this);
337  userInter->setSync(false);

341  connect(userInter, &AccountsUser::UserNameChanged, user, [=](const QString &name) {
342  user->setName(name);
343  user->setOnline(m_onlineUsers.contains(name));
344  user->setIsCurrentUser(name == m_currentUserName);
345 #ifdef DCC_ENABLE_ADDOMAIN
346  checkADUser();
347 #endif
348  });
349 
350  connect(userInter, &AccountsUser::AutomaticLoginChanged, user, &User::setAutoLogin);
351  connect(userInter, &AccountsUser::IconListChanged, user, &User::setAvatars);
352  connect(userInter, &AccountsUser::IconFileChanged, user, &User::setCurrentAvatar);

355  connect(userInter, &AccountsUser::PasswordStatusChanged, user, &User::setPasswordStatus);
356  connect(userInter, &AccountsUser::CreatedTimeChanged, user, &User::setCreatedTime);
357  connect(userInter, &AccountsUser::GroupsChanged, user, &User::setGroups);
358  connect(userInter, &AccountsUser::AccountTypeChanged, user, &User::setUserType);
359  connect(userInter, &AccountsUser::MaxPasswordAgeChanged, user, &User::setPasswordAge);
360 
361  // 这里直接赋值的话, 由于请求是异步的, 所以一开始会被初始化成乱码,

365  userInter->fullName();
366  userInter->automaticLogin();
367  userInter->iconList();
368  userInter->groups();
369  userInter->iconFile();
370  userInter->noPasswdLogin();
371  userInter->passwordStatus();
372  userInter->createdTime();
373  userInter->accountType();
374  userInter->maxPasswordAge();
375  userInter->IsPasswordExpired();
376 
377  m_userInters[user] = userInter;
378  m_userModel->addUser(userPath, user);
379 }

549 CreationResult *AccountsWorker::createAccountInternal(const User *user)
550 {
551  CreationResult *result = new CreationResult;
552 
553  // validate username
554  QDBusPendingReply<bool, QString, int> reply = m_accountsInter->IsUsernameValid(user->name());
555  reply.waitForFinished();
556  if (reply.isError()) {
557  result->setType(CreationResult::UserNameError);
558  result->setMessage(reply.error().message());
559 
560  return result;
561  }
562  bool validation = reply.argumentAt(0).toBool();
563  if (!validation) {
564  result->setType(CreationResult::UserNameError);
565  result->setMessage(dgettext("dde-daemon", reply.argumentAt(1).toString().toUtf8().data()));
566  return result;
567  }
568 
569  // validate password
570  if (user->password() != user->repeatPassword()) {
571  result->setType(CreationResult::PasswordMatchError);
572  result->setMessage(tr("Password not match"));
573  return result;
574  }
575 
576  Authority::Result authenticationResult;
577  authenticationResult = Authority::instance()->checkAuthorizationSync("com.deepin.daemon.accounts.user-administration", UnixProcessSubject(getpid()),
578  Authority::AllowUserInteraction);
579 
580  if (Authority::Result::Yes != authenticationResult) {
581  result->setType(CreationResult::Canceled);
582  return result;
583  }
584 
585  // default FullName is empty string
586  QDBusObjectPath path;
587  QDBusPendingReply<QDBusObjectPath> createReply = m_accountsInter->CreateUser(user->name(), user->fullname(), user->userType());
588  createReply.waitForFinished();
589  if (createReply.isError()) {
590  /* 这里由后端保证出错时一定有错误信息返回，如果没有错误信息，就默认用户在认证时点了取消 */
591  result->setType(createReply.error().message().isEmpty() ? CreationResult::Canceled : CreationResult::UnknownError);
592  result->setMessage(createReply.error().message());
593  return result;
594  } else {
595  path = createReply.argumentAt<0>();
596  }
597  const QString userPath = path.path();
598 
599  AccountsUser *userDBus = new AccountsUser("com.deepin.daemon.Accounts", userPath, QDBusConnection::systemBus(), this);
600  if (!userDBus->isValid()) {
