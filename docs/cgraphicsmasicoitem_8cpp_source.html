43 void CGraphicsMasicoItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
44 {
45  Q_UNUSED(option)
46  Q_UNUSED(widget)
47  updateHandlesGeometry();
48  QGraphicsScene *scene = this->scene();
49  //绘制滤镜
50  if (scene != nullptr) {
51  //计算交叉矩形的区域
52  QPixmap tmpPixmap = m_pixmap;
53  painter->save();
54  painter->setClipPath(m_blurPath, Qt::IntersectClip);
55  //判断和他交叉的元素，裁剪出下层的像素
56  //下层有图元才显示
57  int imgWidth = tmpPixmap.width();
58  int imgHeigth = tmpPixmap.height();
59  const int radius = 10;
60  if (!tmpPixmap.isNull()) {
61  tmpPixmap = tmpPixmap.scaled(imgWidth / radius, imgHeigth / radius, Qt::IgnoreAspectRatio, Qt::SmoothTransformation);
62  if (m_nBlurEffect == BlurEffect) {
63  tmpPixmap = tmpPixmap.scaled(imgWidth, imgHeigth, Qt::IgnoreAspectRatio, Qt::SmoothTransformation);
64  } else {
65  tmpPixmap = tmpPixmap.scaled(imgWidth, imgHeigth);
66  }
67  }
68  painter->drawPixmap(boundingRect().topLeft(), tmpPixmap);
69  painter->restore();
70  }
71  paintMutBoundingLine(painter, option);
72 }

225 void CGraphicsMasicoItem::freeBlurUpdate()

245  if (!pItem->childItems().isEmpty()) {
246  QGraphicsItem *pChild = pItem->childItems().first();
247  pChild->setFlag(ItemHasNoContents, true);
248  }

261

279  for (int i = 0; i != filterItems.size(); i++) {
