68 int countFileCount(const char *name, bool isloop = false)
69 {
70  DIR *dir;
71  struct dirent *entry;
72  int fileCount = 0;
73 
74  if (!(dir = opendir(name)))
75  return fileCount;
76  if (!(entry = readdir(dir)))
77  return fileCount;
78 
79  do {
80  char path[1024];
81  int len = snprintf(path, sizeof(path)-1, "%s/%s", name, entry->d_name);
82  path[len] = 0;
83  if (strcmp(entry->d_name, ".") == 0 || strcmp(entry->d_name, "..") == 0)
84  continue;
85  if (entry->d_type == DT_DIR) {
86  if (isloop) {
87  fileCount += countFileCount(path, isloop);
88  }
89  else {
90  fileCount++;
91  }
92  }
93  else {
94  fileCount++;
95  }
96  } while ((entry = readdir(dir)));
97  closedir(dir);
98  return fileCount;
99 }

408 void DGvfsFileInfo::refreshCachesByStat()
409 {
410  Q_D(DGvfsFileInfo);
411 
412  struct stat statinfo;
413  int filestat = stat(d->fileInfo.absoluteFilePath().toStdString().c_str(), &statinfo);
414  if (filestat != 0) {
415  d->ownerid = static_cast<int>(getuid());
416  return;
417  }
418  d->cacheCanWrite = access(d->fileInfo.absoluteFilePath().toStdString().c_str(), W_OK) == 0 ? 1 : 0;
419  d->cacheIsSymLink = S_ISLNK(statinfo.st_mode);
420  d->cacheFileSize = statinfo.st_size;
421  d->inode = statinfo.st_ino;
422  d->ownerid = static_cast<int>(statinfo.st_uid);
423  d->cacheIsDir = S_ISDIR(statinfo.st_mode);
424  d->cacheModifyTime = statinfo.st_mtim.tv_sec;
425  d->cacheReadTime = statinfo.st_atim.tv_sec;
426 }
