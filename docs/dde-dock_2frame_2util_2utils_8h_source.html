33 inline QPixmap renderSVG(const QString &path, const QSize &size, const qreal devicePixelRatio) {
34  QImageReader reader;
35  QPixmap pixmap;
36  reader.setFileName(path);
37  if (reader.canRead()) {
38  reader.setScaledSize(size * devicePixelRatio);
39  pixmap = QPixmap::fromImage(reader.read());
40  pixmap.setDevicePixelRatio(devicePixelRatio);
41  }
42  else {
43  pixmap.load(path);
44  }
45 
46  return pixmap;
47 }

92 inline int comparePluginApi(const QString &pluginApi1, const QString &pluginApi2) {
93  // 版本号相同
94  if (pluginApi1 == pluginApi2)
95  return 0;
96 
97  // 拆分版本号
98  QStringList subPluginApis1 = pluginApi1.split(".", QString::SkipEmptyParts, Qt::CaseSensitive);
99  QStringList subPluginApis2 = pluginApi2.split(".", QString::SkipEmptyParts, Qt::CaseSensitive);
100  for (int i = 0; i < subPluginApis1.size(); ++i) {
101  auto subPluginApi1 = subPluginApis1[i];
102  if (subPluginApis2.size() > i) {
103  auto subPluginApi2 = subPluginApis2[i];
104 
105  // 相等判断下一个子版本号
106  if (subPluginApi1 == subPluginApi2)
107  continue;
108 
109  // 转成整形比较
110  if (subPluginApi1.toInt() > subPluginApi2.toInt()) {
111  return 1;
112  } else {
113  return -1;
114  }
115  }
116  }
117 
118  // 循环结束但是没有返回,说明子版本号个数不同,且前面的子版本号都相同
119  // 子版本号多的版本号大
120  if (subPluginApis1.size() > subPluginApis2.size()) {
121  return 1;
122  } else {
123  return -1;
124  }
125 }
126 }
