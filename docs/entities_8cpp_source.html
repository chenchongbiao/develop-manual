271 static int cmp(const void *key, const void *value)
272 {
273  return strncmp((const char *)key, *(const char **)value,
274  strlen(*(const char **)value));
275 }

277 static const char *get_named_entity(const char *name)
278 {
279  const char *const *entity = (const char *const *)bsearch(name,
280  NAMED_ENTITIES, sizeof NAMED_ENTITIES / sizeof *NAMED_ENTITIES,
281  sizeof *NAMED_ENTITIES, cmp);
282 
283  return entity ? entity[1] : NULL;
284 }

286 static size_t putc_utf8(unsigned long cp, char *buffer)
287 {
288  unsigned char *bytes = (unsigned char *)buffer;
289 
290  if(cp <= 0x007Ful)
291  {
292  bytes[0] = (unsigned char)cp;
293  return 1;
294  }
295 
296  if(cp <= 0x07FFul)
297  {
298  bytes[1] = (unsigned char)((2 << 6) | (cp & 0x3F));
299  bytes[0] = (unsigned char)((6 << 5) | (cp >> 6));
300  return 2;
301  }
302 
303  if(cp <= 0xFFFFul)
304  {
305  bytes[2] = (unsigned char)(( 2 << 6) | ( cp & 0x3F));
306  bytes[1] = (unsigned char)(( 2 << 6) | ((cp >> 6) & 0x3F));
307  bytes[0] = (unsigned char)((14 << 4) | (cp >> 12));
308  return 3;
309  }
310 
311  if(cp <= 0x10FFFFul)
312  {
313  bytes[3] = (unsigned char)(( 2 << 6) | ( cp & 0x3F));
314  bytes[2] = (unsigned char)(( 2 << 6) | ((cp >> 6) & 0x3F));
315  bytes[1] = (unsigned char)(( 2 << 6) | ((cp >> 12) & 0x3F));
316  bytes[0] = (unsigned char)((30 << 3) | (cp >> 18));
317  return 4;
318  }
319 
320  return 0;
321 }

323 static bool parse_entity(const char *current, char **to, const char **from)
324 {
325  const char *end = strchr(current, ';');
326  if(!end) return 0;
327 
328  if(current[1] == '#')
329  {
330  char *tail = NULL;
331  int errno_save = errno;
332  bool hex = current[2] == 'x' || current[2] == 'X';
333 
334  errno = 0;
335  unsigned long cp = strtoul(
336  current + (hex ? 3 : 2), &tail, hex ? 16 : 10);
337 
338  bool fail = errno || tail != end || cp > UNICODE_MAX;
339  errno = errno_save;
340  if(fail) return 0;
341 
342  *to += putc_utf8(cp, *to);
343  *from = end + 1;
344 
345  return 1;
346  }
347  else
348  {
349  const char *entity = get_named_entity(&current[1]);
350  if(!entity) return 0;
351 
352  size_t len = strlen(entity);
353  memcpy(*to, entity, len);
354 
355  *to += len;
356  *from = end + 1;
357 
358  return 1;
359  }
360 }

362 size_t decode_html_entities_utf8(char *dest, const char *src)
363 {
364  if(!src) src = dest;
365 
366  char *to = dest;
367  const char *from = src;
368 
369  for(const char *current; (current = strchr(from, '&'));)
370  {
371  memmove(to, from, (size_t)(current - from));
372  to += current - from;
373 
374  if(parse_entity(current, &to, &from))
375  continue;
376 
377  from = current;
378  *to++ = *from++;
379  }
380 
381  size_t remaining = strlen(from);
382 
383  memmove(to, from, remaining);
384  to += remaining;
385  *to = 0;
386 
387  return (size_t)(to - dest);
388 }
