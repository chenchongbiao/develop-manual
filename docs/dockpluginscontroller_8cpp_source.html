36 void DockPluginsController::itemAdded(PluginsItemInterface *const itemInter, const QString &itemKey)
37 {
38  QMap<PluginsItemInterface *, QMap<QString, QObject *>> &mPluginsMap = pluginsMap();
39 
40  // check if same item added
41  if (mPluginsMap.contains(itemInter))
42  if (mPluginsMap[itemInter].contains(itemKey))
43  return;
44 
45  // 取 plugin api
46  QPluginLoader *pluginLoader = qobject_cast<QPluginLoader*>(mPluginsMap[itemInter].value("pluginloader"));
47  const QJsonObject &meta = pluginLoader->metaData().value("MetaData").toObject();
48  const QString &pluginApi = meta.value("api").toString();
49 
50  PluginsItem *item = nullptr;
51  if (itemInter->pluginName() == "tray") {
52  item = new TrayPluginItem(itemInter, itemKey, pluginApi);
53  if (item->graphicsEffect()) {
54  item->graphicsEffect()->setEnabled(false);
55  }
56  connect(static_cast<TrayPluginItem *>(item), &TrayPluginItem::trayVisableCountChanged,
57  this, &DockPluginsController::trayVisableCountChanged, Qt::UniqueConnection);
58  } else {
59  item = new PluginsItem(itemInter, itemKey, pluginApi);
60  }
61 
62  mPluginsMap[itemInter][itemKey] = item;
63 
64  emit pluginItemInserted(item);
65 }

78 void DockPluginsController::itemRemoved(PluginsItemInterface *const itemInter, const QString &itemKey)
79 {
80  PluginsItem *item = static_cast<PluginsItem *>(pluginItemAt(itemInter, itemKey));
81  if (!item)
82  return;
83 
84  item->detachPluginWidget();
85 
86  emit pluginItemRemoved(item);
87 
88  QMap<PluginsItemInterface *, QMap<QString, QObject *>> &mPluginsMap = pluginsMap();
89  mPluginsMap[itemInter].remove(itemKey);
90 
91  // do not delete the itemWidget object(specified in the plugin interface)
92  item->centralWidget()->setParent(nullptr);
93 
94  if (item->isDragging()) {
95  QDrag::cancel();
96  }
97 
98  // just delete our wrapper object(PluginsItem)
99  item->deleteLater();
100 }
