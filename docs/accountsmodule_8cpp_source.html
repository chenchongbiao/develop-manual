115 int AccountsModule::load(const QString &path)
116 {
117  if (!m_accountsWidget) {
118  active();
119  }
120 
121  QStringList searchList;
122  searchList << "Accounts Detail"
123  << "New Account";
124 
125  User *pUser = nullptr;
126  for (auto &user : m_userModel->userList()) {
127  if (user->isCurrentUser()) {
128  pUser = user;
129  break;
130  }
131  }
132 
133  if (pUser == nullptr) {
134  return -1;
135  }
136 
137  if (path == searchList[0]) {
138  onShowAccountsDetailWidget(pUser);
139  return 0;
140  } else if (path == searchList[1]) {
141  onShowCreateAccountPage();
142  return 0;
143  }
144 
145  return -1;
146 }

222 //添加指纹界面
223 void AccountsModule::onShowAddThumb(const QString &name, const QString &thumb)
224 {
225  AddFingeDialog *dlg = new AddFingeDialog(thumb);
226  connect(dlg, &AddFingeDialog::requestEnrollThumb, m_fingerWorker, [=] {
227  m_fingerWorker->tryEnroll(name, thumb);
228  });
229  connect(dlg, &AddFingeDialog::requestStopEnroll, m_fingerWorker, &FingerWorker::stopEnroll);
230  connect(dlg, &AddFingeDialog::requesetCloseDlg, dlg, [=](const QString &userName) {
231  m_fingerWorker->refreshUserEnrollList(userName);
232  onSetMainWindowEnabled(true);
233  dlg->deleteLater();
234  });
235 
236  m_fingerWorker->tryEnroll(name, thumb);
237  connect(m_fingerWorker, &FingerWorker::tryEnrollResult, dlg, [=] (FingerWorker::EnrollResult res) {
238  // 第一次tryEnroll进入时显示添加指纹对话框
239  if (m_pMainWindow->isEnabled()) {
240  if (res == FingerWorker::Enroll_Success) {
241  onSetMainWindowEnabled(false);
242  m_fingerModel->resetProgress();
243  dlg->setFingerModel(m_fingerModel);
244  dlg->setWindowFlags(Qt::Dialog | Qt::Popup | Qt::WindowStaysOnTopHint);
245  dlg->setUsername(name);
246  dlg->show();
247  dlg->setFocus();
248  dlg->activateWindow();
249  } else {
250  m_fingerWorker->stopEnroll(name);
251  dlg->deleteLater();
252  }
253  } else {
254  //　已经在添加指纹对话框中的Enroll处理
255  if (res == FingerWorker::Enroll_AuthFailed) {
256  onSetMainWindowEnabled(true);
257  dlg->deleteLater();
258  } else {
259  dlg->setInitStatus();
260  }
261  }
262  });
263 }

265 void AccountsModule::onHandleVaildChanged(const bool isVaild)
266 {
267  if (isVaild) {
268  initFingerData();
269  } else {
270  for (const auto &user : m_userModel->userList()) {
271  disconnect(user, &User::nameChanged, m_fingerWorker, &FingerWorker::refreshUserEnrollList);
272  }
273  }
274 
275  //用户运行程序第一次点击账户初始化时
276  //调用过程：　initialize ->　refreshDevice -> GetDefaultDevice -> onGetFprDefaultDevFinished -> 获取指纹设备存在状态,发出vaildChanged信号 -> onHandleVaildChanged
277  //onHandleVaildChanged第一次被调用就行了，后面指纹设备状态发生变化时，有DevicesChanged信号来通知指纹页面。
278  disconnect(m_fingerModel, &FingerModel::vaildChanged, this, &AccountsModule::onHandleVaildChanged);
279 }
