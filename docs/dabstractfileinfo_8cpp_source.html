99 bool compareByString(const QString &str1, const QString &str2, Qt::SortOrder order)
100 {
101  thread_local static DCollator sortCollator;
102  //其他符号要排在最后，需要在中文前先做判断
103  if (DFMGlobal::startWithSymbol(str1)) {
104  if (!DFMGlobal::startWithSymbol(str2))
105  return order == Qt::DescendingOrder;
106  } else if (DFMGlobal::startWithSymbol(str2))
107  return order != Qt::DescendingOrder;
108 
109  if (DFMGlobal::startWithHanzi(str1)) {
110  if (!DFMGlobal::startWithHanzi(str2)) {
111  return order == Qt::DescendingOrder;
112  }
113  } else if (DFMGlobal::startWithHanzi(str2)) {
114  return order != Qt::DescendingOrder;
115  }
116 
117  return ((order == Qt::DescendingOrder) ^ (sortCollator.compare(str1, str2) < 0)) == 0x01;
118 }

789 bool DAbstractFileInfo::isAncestorsUrl(const DUrl &url, QList<DUrl> *ancestors) const
790 {
791  DUrl parentUrl = this->parentUrl();
792 
793  forever {
794  if (ancestors && parentUrl.isValid()) {
795  ancestors->append(parentUrl);
796  }
797 
798  if (parentUrl == url || DStorageInfo::isSameFile(parentUrl.path(), url.path())) {
799  return true;
800  }
801 
802  const DAbstractFileInfoPointer &fileInfo = DFileService::instance()->createFileInfo(Q_NULLPTR, parentUrl);
803 
804  if (!fileInfo) {
805  break;
806  }
807 
808  const DUrl &pu = fileInfo->parentUrl();
809 
810  if (pu == parentUrl) {
811  break;
812  }
813 
814  parentUrl = pu;
815  }
816 
817  return false;
818 }
