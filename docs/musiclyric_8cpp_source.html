76  //this->offset
77  this->line.clear();
78  this->postion.clear();
79  //先使用暴力的字符串匹配，还不会正则表达式
80  //时间复杂度O(n)
81  QString codeStr = getFileCodex(dir);
82  QFile file(dir);
83  if (!file.exists()) return;
84  if (!file.open(QIODevice::ReadOnly | QIODevice::Text))
85  return;
86  QTextStream read(&file);
87  if (!codeStr.isEmpty()) {
88  read.setCodec(QTextCodec::codecForName(codeStr.toStdString().c_str()));
89  }
90  qint64 mm;
91  double ss = 0.0;
92  QMap<qint64, QString> ans;
93  while (!read.atEnd()) {
94  QString lineStr = read.readLine();
95  if (lineStr.isEmpty() || lineStr[0] != '[')
96  continue;
97  QStringList curLineList = lineStr.split('[');
98  for (auto curLineStr : curLineList) {
99 
100  QStringList lineList = curLineStr.split(']');
101  if (lineList.isEmpty() || lineList[0].size() < 3)
102  continue;
103 
104  QString str;
105  if (lineList.size() == 2)
106  str = lineList[1];
107  QString t_timeStr = lineList[0].remove(0, 1);
108  QStringList t_timelist = t_timeStr.split(':');
109  if (t_timelist.size() != 2)
110  continue;
111  bool flag = false;
112  mm = t_timelist[0].toLongLong(&flag, 10);
113  if (flag) {
114  ss = t_timelist[1].toDouble(&flag);
115  if (flag) {
116  qint64 curtime = (qint64)(ss * 1000) + mm * 60 * 1000;
117  ans.insert(curtime, str);
118  }
119  }
120  }
121 
122  }
123 
124  QMap<qint64, QString>::iterator it;
125  for (it = ans.begin(); it != ans.end(); ++it) {
126  this->postion.push_back(it.key());
127  this->line.push_back(it.value());
128  }
129 }
