204 void UpdateSettings::setModel(UpdateModel *model)
205 {
206  if (m_model == model)
207  return;
208 
209  m_model = model;
210 
211  connect(model, &UpdateModel::autoCheckUpdatesChanged, m_autoCheckUpdate, &SwitchWidget::setChecked);
212  // connect(model, &UpdateModel::autoCheckUpdatesChanged, m_autoCheckSecureUpdate, &SwitchWidget::setVisible);
213  connect(model, &UpdateModel::autoCheckUpdatesChanged, m_autoCheckSystemUpdate, &SwitchWidget::setVisible);
214  connect(model, &UpdateModel::autoCheckSecureUpdatesChanged, m_autoCheckSecureUpdate, &SwitchWidget::setChecked);
215  connect(model, &UpdateModel::autoCheckSystemUpdatesChanged, m_autoCheckSystemUpdate, &SwitchWidget::setChecked);
216  connect(model, &UpdateModel::autoCheckSystemUpdatesChanged, m_autoCheckAppUpdate, &SwitchWidget::setVisible);
217  connect(model, &UpdateModel::autoCheckAppUpdatesChanged, m_autoCheckAppUpdate, &SwitchWidget::setChecked);
218  connect(model, &UpdateModel::updateNotifyChanged, m_updateNotify, &SwitchWidget::setChecked);
219  connect(model, &UpdateModel::updateNotifyChanged, m_autoDownloadUpdate, &SwitchWidget::setVisible);
220  connect(model, &UpdateModel::updateNotifyChanged, m_autoDownloadUpdateTips, &DTipLabel::setVisible);
221  connect(model, &UpdateModel::autoDownloadUpdatesChanged, m_autoDownloadUpdate, &SwitchWidget::setChecked);
222  connect(model, &UpdateModel::autoCleanCacheChanged, m_autoCleanCache, &SwitchWidget::setChecked);
223 
224  m_autoCheckUpdate->setChecked(model->autoCheckUpdates());
225  m_autoCheckSecureUpdate->setChecked(model->autoCheckSecureUpdates());
226  // m_autoCheckSecureUpdate->setVisible(model->autoCheckUpdates());
227  m_autoCheckSecureUpdate->setVisible(false); // 当前阶段暂时隐藏安全更新选项显示
228  m_autoCheckSystemUpdate->setChecked(model->autoCheckSystemUpdates());
229  m_autoCheckSystemUpdate->setVisible(model->autoCheckUpdates());
230  m_autoCheckAppUpdate->setChecked(model->autoCheckAppUpdates());
231  m_autoCheckAppUpdate->setVisible(model->autoCheckSystemUpdates());
232  m_updateNotify->setChecked(model->updateNotify());
233  m_autoDownloadUpdate->setChecked(model->autoDownloadUpdates());
234  m_autoDownloadUpdate->setVisible(model->updateNotify());
235  m_autoDownloadUpdateTips->setVisible(model->updateNotify());
236  m_autoCleanCache->setChecked(m_model->autoCleanCache());
237 
238 #ifndef DISABLE_SYS_UPDATE_SOURCE_CHECK
239  if (!IsServerSystem && !IsProfessionalSystem && !IsHomeSystem && !IsDeepinDesktop) {
240  connect(model, &UpdateModel::sourceCheckChanged, m_sourceCheck, &SwitchWidget::setChecked);
241  m_sourceCheck->setChecked(model->sourceCheck());
242  }
243 #endif
244 
245  if (IsCommunitySystem) {
246  auto setDefaultMirror = [this](const MirrorInfo & mirror) {
247  m_updateMirrors->setValue(mirror.m_name);
248  };
249 
250  if (!model->mirrorInfos().isEmpty()) {
251  setDefaultMirror(model->defaultMirror());
252  }
253 
254  connect(model, &UpdateModel::defaultMirrorChanged, this, setDefaultMirror);
255  connect(model, &UpdateModel::smartMirrorSwitchChanged, m_smartMirrorBtn, &SwitchWidget::setChecked);
256  m_smartMirrorBtn->setChecked(m_model->smartMirrorSwitch());
257 
258  auto setMirrorListVisible = [ = ](bool visible) {
259  m_updateMirrors->setVisible(!visible);
260  };
261 
262  connect(model, &UpdateModel::smartMirrorSwitchChanged, this, setMirrorListVisible);
263  setMirrorListVisible(model->smartMirrorSwitch());
264  }

267 void UpdateSettings::setUpdateMode()
268 {
269  quint64 updateMode = 0;
270  if (m_autoCheckUpdate->checked()) {
271  if (!m_autoCheckSystemUpdate->checked()) {
272  m_autoCheckAppUpdate->setChecked(false);
273  }
274  updateMode = updateMode | m_autoCheckSecureUpdate->checked();
275  updateMode = (updateMode << 1) | m_autoCheckAppUpdate->checked();
276  updateMode = (updateMode << 1) | m_autoCheckSystemUpdate->checked();
277  }
278  requestSetUpdateMode(updateMode);
279 }
