133 void SearchResultTabWidget::refreshListview()
134 {
135  if (m_switchtype == SearchMusicResultType) {
136  setLabelChecked(m_music);
137  m_StackedWidget->setCurrentWidget(m_musicListView);
138  m_musicListView->resetSonglistByStr(m_searchWord);
139  m_singerListView->resetSingerListDataBySongName(m_musicListView->getMusicListData());
140  m_albumListView->resetAlbumListDataBySongName(m_musicListView->getMusicListData());
141  emit sigSearchTypeChanged("musicResult");
142  } else if (m_switchtype == SearchSingerResultType) {
143  setLabelChecked(m_singer);
144  m_StackedWidget->setCurrentWidget(m_singerListView);
145  m_singerListView->resetSingerListDataByStr(m_searchWord);
146  m_musicListView->resetSonglistBySinger(m_singerListView->getSingerListData());
147  m_albumListView->resetAlbumListDataBySinger(m_singerListView->getSingerListData());
148  emit sigSearchTypeChanged("artistResult");
149  qDebug() << "artistResult";
150  } else if (m_switchtype == SearchAlbumResultType) {
151  setLabelChecked(m_album);
152  m_StackedWidget->setCurrentWidget(m_albumListView);
153  m_albumListView->resetAlbumListDataByStr(m_searchWord);
154  m_musicListView->resetSonglistByAlbum(m_albumListView->getAlbumListData());
155  m_singerListView->resetSingerListDataByAlbum(m_albumListView->getAlbumListData());
156  emit sigSearchTypeChanged("albumResult");
157  qDebug() << "albumResult";
158  }
159 }

305 void SearchResultTabWidget::setLabelChecked(Label *label)
306 {
307  if (m_music == label) {
308  m_singer->setEnabled(false);
309  m_album->setEnabled(false);
310  m_music->setEnabled(true);
311  m_LineWidget->show();
312  // 修改计算逻辑
313  m_LineWidget->move((m_music->x() + (m_music->width() - m_LineWidget->width()) / 2), m_singer->y() + m_singer->height() + 8);
314  m_StackedWidget->setCurrentWidget(m_musicListView);
315  } else if (m_singer == label) {
316  m_singer->setEnabled(true);
317  m_album->setEnabled(false);
318  m_music->setEnabled(false);
319  m_LineWidget->show();
320  m_LineWidget->move((m_singer->x() + (m_singer->width() - m_LineWidget->width()) / 2), m_singer->y() + m_singer->height() + 8);
321  m_StackedWidget->setCurrentWidget(m_singerListView);
322  } else if (m_album == label) {
323  m_singer->setEnabled(false);
324  m_album->setEnabled(true);
325  m_music->setEnabled(false);
326  m_LineWidget->show();
327  m_LineWidget->move((m_album->x() + (m_album->width() - m_LineWidget->width()) / 2), m_singer->y() + m_singer->height() + 8);
