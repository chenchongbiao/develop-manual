115 void SoundEffectsPage::startPlay(const QModelIndex &index)
116 {
117  if (m_playIdx.isValid()) {
118  auto item = static_cast<DStandardItem *>(m_listModel->itemFromIndex(m_playIdx));
119  auto aniAction = item->actionList(Qt::Edge::RightEdge)[0];
120  aniAction->setVisible(false);
121  m_effectList->update(m_playIdx);
122  }
123  m_playIdx = index;
124 
125  auto eff = m_model->soundEffectMap()[index.row()].second;
126  m_sound.reset(new QSound(m_model->soundEffectPathByType(eff)));
127  m_sound->stop();
128  m_sound->play();
129 
130  m_aniTimer->disconnect();
131  auto item = static_cast<DStandardItem *>(m_listModel->itemFromIndex(index));
132  auto aniAction = item->actionList(Qt::Edge::RightEdge)[0];
133  int intervalal = 300;
134  m_aniTimer->setInterval(intervalal);
135  aniAction->setVisible(true);
136  connect(m_aniTimer, &QTimer::timeout, this, [ = ] {
137  auto aniIdx = (m_aniDuration / intervalal) % 3 + 1;
138  auto icon = QIcon::fromTheme("dcc_volume" + QString::number(aniIdx));
139  aniAction->setIcon(icon);
140 
141  m_aniDuration += intervalal;
142  if (m_aniDuration > AnimationDuration)
143  {
144  aniAction->setVisible(false);
145  m_aniTimer->stop();
146  m_aniDuration = 0;
147  }
148  m_effectList->update(index);
149  });
150 
151  m_aniTimer->start();
152 }

154 void SoundEffectsPage::initList()
155 {
156  m_sw->setChecked(m_model->enableSoundEffect());
157 
158  m_listModel = new QStandardItemModel(this);
159  m_effectList->setModel(m_listModel);
160 
161  QSize size(16, 16);
162  DStandardItem *item = nullptr;
163  for (auto se : m_model->soundEffectMap()) {
164  item = new DStandardItem(se.first);
165  item->setFontSize(DFontSizeManager::T8);
166  auto action = new DViewItemAction(Qt::AlignVCenter, size, size, true);
167  auto checkstatus = m_model->queryEffectData(se.second) ?
168  DStyle::SP_IndicatorChecked : DStyle::SP_IndicatorUnchecked ;
169  auto icon = qobject_cast<DStyle *>(style())->standardIcon(checkstatus);
170  action->setIcon(icon);
171  auto aniAction = new DViewItemAction(Qt::AlignVCenter, size, size);
172  aniAction->setVisible(false);
173  item->setActionList(Qt::Edge::RightEdge, {aniAction, action});
174  m_listModel->appendRow(item);
175 
176  connect(action, &DViewItemAction::triggered, this, [ = ] {
177  auto isSelected = m_model->queryEffectData(se.second);
178  this->requestSetEffectAble(se.second, !isSelected);
179  requestRefreshList();
180  });
181  }
182 
183  connect(m_effectList, &DListView::clicked, this, &SoundEffectsPage::startPlay);
184  connect(m_effectList, &DListView::activated, m_effectList, &QListView::clicked);
185  connect(m_model, &SoundModel::soundEffectDataChanged, this,
186  [ = ](DDesktopServices::SystemSoundEffect effect, const bool enable) {
187  for (int idx = 0; idx < m_model->soundEffectMap().size(); ++idx) {
188  auto ite = m_model->soundEffectMap().at(idx);
189  if (ite.second != effect) {
190  continue;
191  }
192 
193 auto items = static_cast<DStandardItem *>(m_listModel->item(idx));
194  auto action = items->actionList(Qt::Edge::RightEdge)[1];
195  auto checkstatus = enable ?
196  DStyle::SP_IndicatorChecked : DStyle::SP_IndicatorUnchecked ;
197  auto icon = qobject_cast<DStyle *>(style())->standardIcon(checkstatus);
198  action->setIcon(icon);
199  m_effectList->update(items->index());
200  break;
201  }
202  });
203 }
