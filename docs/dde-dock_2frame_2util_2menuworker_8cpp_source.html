141 void MenuWorker::showDockSettingsMenu()
142 {
143  QTimer::singleShot(0, this, [=] {
144  onGSettingsChanged("enable");
145  });
146 
147  setAutoHide(false);
148 
149  bool hasComposite = DWindowManagerHelper::instance()->hasComposite();
150 
151  // create actions
152  QList<QAction *> actions;
153  for (auto *p : m_itemManager->pluginList()) {
154  if (!p->pluginIsAllowDisable())
155  continue;
156 
157  const bool enable = !p->pluginIsDisable();
158  const QString &name = p->pluginName();
159  const QString &display = p->pluginDisplayName();
160 
161  if (!m_trashPluginShow && name == "trash") {
162  continue;
163  }
164 
165  if (name == "multitasking" && !hasComposite) {
166  continue;
167  }
168 
169  if (name == "deepin-screen-recorder-plugin") {
170  continue;
171  }
172  QAction *act = new QAction(display, this);
173  act->setCheckable(true);
174  act->setChecked(enable);
175  act->setData(name);
176 
177  actions << act;
178  }
179 
180  // sort by name
181  std::sort(actions.begin(), actions.end(), [](QAction * a, QAction * b) -> bool {
182  return a->data().toString() > b->data().toString();
183  });
184 
185  // add actions
186  qDeleteAll(m_hideSubMenu->actions());
187  for (auto act : actions)
188  m_hideSubMenu->addAction(act);
189 
190  const DisplayMode displayMode = static_cast<DisplayMode>(m_dockInter->displayMode());
191  const Position position = static_cast<Position>(m_dockInter->position());
192  const HideMode hideMode = static_cast<HideMode>(m_dockInter->hideMode());
193 
194  m_fashionModeAct->setChecked(displayMode == Fashion);
195  m_efficientModeAct->setChecked(displayMode == Efficient);
196  m_topPosAct->setChecked(position == Top);
197  m_bottomPosAct->setChecked(position == Bottom);
198  m_leftPosAct->setChecked(position == Left);
199  m_rightPosAct->setChecked(position == Right);
200  m_keepShownAct->setChecked(hideMode == KeepShowing);
201  m_keepHiddenAct->setChecked(hideMode == KeepHidden);
202  m_smartHideAct->setChecked(hideMode == SmartHide);
203 
204  m_settingsMenu->exec(QCursor::pos());
205 
206  setAutoHide(true);
207 }

236 void MenuWorker::menuActionClicked(QAction *action)
237 {
238  Q_ASSERT(action);
239 
240  if (action == m_fashionModeAct)
241  return m_dockInter->setDisplayMode(DisplayMode::Fashion);
242  if (action == m_efficientModeAct)
243  return m_dockInter->setDisplayMode(Efficient);
244 
245  if (action == m_topPosAct)
246  return m_dockInter->setPosition(Top);
247  if (action == m_bottomPosAct)
248  return m_dockInter->setPosition(Bottom);
249  if (action == m_leftPosAct)
250  return m_dockInter->setPosition(Left);
251  if (action == m_rightPosAct)
252  return m_dockInter->setPosition(Right);
253 
254  if (action == m_keepShownAct)
255  return m_dockInter->setHideMode(KeepShowing);
256  if (action == m_keepHiddenAct)
257  return m_dockInter->setHideMode(KeepHidden);
258  if (action == m_smartHideAct)
259  return m_dockInter->setHideMode(SmartHide);
260 
261  // check plugin hide menu.
262  const QString &data = action->data().toString();
263  if (data.isEmpty())
264  return;
265  for (auto *p : m_itemManager->pluginList()) {
266  if (p->pluginName() == data)
267  return p->pluginStateSwitched();
268  }
269 }
