119 void SystemLanguageWidget::onEditClicked()
120 {
121  m_bEdit = !m_bEdit;
122  if (m_bEdit) {
123  m_editSystemLang->setText(tr("Done"));
124  int row_count = m_langItemModel->rowCount();
125  for (int i = 0; i < row_count; ++i) {
126  DStandardItem *item = dynamic_cast<DStandardItem *>(m_langItemModel->item(i, 0));
127  if (item && (item->checkState() == Qt::Unchecked)) {
128  DViewItemAction *iconAction = new DViewItemAction(Qt::AlignCenter | Qt::AlignRight, QSize(), QSize(), true);
129  iconAction->setIcon(DStyle::standardIcon(style(), DStyle::SP_DeleteButton));
130  item->setActionList(Qt::RightEdge, {iconAction});
131  connect(iconAction, &DViewItemAction::triggered, this, [this,item] {
132  m_sysLanglist.removeOne(item->text());
133  int idx = m_langItemModel->indexFromItem(item).row();
134  Q_EMIT delLocalLang(item->text());
135  m_langItemModel->removeRow(idx);
136  m_langListview->adjustSize();
137  m_langListview->update();
138  m_editSystemLang->setVisible(m_sysLanglist.size() > 1);
139  });
140  }
141  }
142  } else {
143  m_editSystemLang->setText(tr("Edit"));
144  int row_count = m_langItemModel->rowCount();
145  for (int i = 0; i < row_count; ++i) {
146  DStandardItem *item = dynamic_cast<DStandardItem *>(m_langItemModel->item(i, 0));
147  if (item && (item->checkState() == Qt::Unchecked)) {
148  item->setActionList(Qt::RightEdge, {});
149  }
150  }
151  }
152 }

154 void SystemLanguageWidget::onAddLanguage(const QString &localeLang)
155 {
156  if (m_sysLanglist.contains(localeLang))
157  return;
158 
159  DStandardItem *item = new DStandardItem(localeLang);
160  m_langItemModel->appendRow(item);
161  m_langListview->adjustSize();
162  m_langListview->update();
163  m_sysLanglist << localeLang;
164  m_editSystemLang->setVisible(m_sysLanglist.size() > 1);
165 }

167 void SystemLanguageWidget::setCurLangChecked(const QModelIndex &index)
168 {
169  if (m_bEdit) {
170  return;
171  }
172  int row_count = m_langItemModel->rowCount();
173  for (int i = 0; i < row_count; ++i) {
174  QStandardItem *item = m_langItemModel->item(i, 0);
175  if (item && (index.row() == i)) {
176  item->setCheckState(Qt::Checked);
177  QString langKey = m_model->langFromText(item->text());
178  Q_EMIT setCurLang(langKey);
179  } else if (item) { //如果不加此判断，item会出现空指针
180  item->setCheckState(Qt::Unchecked);
181  }
182  }
183 }

185 void SystemLanguageWidget::onDefault(const QString &curLang)
186 {
187  qDebug() << "curLang is " << curLang;
188  int row_count = m_langItemModel->rowCount();
189  for (int i = 0; i < row_count; ++i) {
190  QStandardItem *item = m_langItemModel->item(i, 0);
191  if (item && (item->text() == curLang)) {
192  item->setCheckState(Qt::Checked);
193  } else {
194  item->setCheckState(Qt::Unchecked);
195  }
196  }
