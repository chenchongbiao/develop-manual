65 void AppItemDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const
66 {
67  if (index.data(AppsListModel::AppItemIsDraggingRole).value<bool>() && !(option.features & QStyleOptionViewItem::HasDisplay))
68  return;
69 
70  painter->setRenderHints(QPainter::Antialiasing | QPainter::TextAntialiasing | QPainter::SmoothPixmapTransform);
71  painter->setPen(Qt::white);
72  painter->setBrush(QBrush(Qt::transparent));
73 
74 #ifdef QT_DEBUG
75  ItemInfo itemInfo = index.data(AppsListModel::AppRawItemInfoRole).value<ItemInfo>();
76 #else
77  const ItemInfo itemInfo = index.data(AppsListModel::AppRawItemInfoRole).value<ItemInfo>();
78 #endif
79  const int fontPixelSize = index.data(AppsListModel::AppFontSizeRole).value<int>();
80  const bool drawBlueDot = index.data(AppsListModel::AppNewInstallRole).toBool();
81  const bool is_current = CurrentIndex == index;
82  const QRect ibr = itemBoundingRect(option.rect);
83  const QSize iconSize = index.data(AppsListModel::AppIconSizeRole).toSize();
84 
85  // process font
86  QFont appNamefont(painter->font());
87  appNamefont.setPixelSize(fontPixelSize);
88  const QFontMetrics fm(appNamefont);
89  //分类模式且不是当前的分类就设置透明度
90  if( index.data(AppsListModel::AppGroupRole).toInt() >= 4 && index.data(AppsListModel::AppCategoryRole).toInt() != m_calcUtil->currentCategory()) {
91  painter->setOpacity(0.3);
92  } else {
93  painter->setOpacity(1);
94  }
95 
96  // Curve Fitting Result from MATLAB
97 // const int x = iconSize.width();
98 // const int margin = -0.000004236988913209739*x*x*x*x+0.0016406743692943455*x*x*x-0.22885856605074573*x*x+13.187308932617098*x-243.2646393941108;
99  const static double x1 = 0.26418192;
100  const static double x2 = -0.38890932;
101  const double result = x1 * ibr.width() + x2 * iconSize.width();
102  int margin = result > 0 ? result * 0.71 : 1;
103 
104  // adjust
105  QRect br;
106  QRect iconRect;
107  QRectF appNameRect;
108  QString appNameResolved;
109  bool adjust = false;
110  bool TextSecond = 0;
111  do {
112  // adjust
113  if (adjust)
114  --margin;
115  br = ibr.marginsRemoved(QMargins(margin, 1, margin, margin * 2));
116 
117  // calc icon rect
118  const int iconLeftMargins = (br.width() - iconSize.width()) / 2;
119  int iconTopMargin = ICONTOTOP;// ibr.height() * .2 - iconSize.height() * .3;
120  //iconTopMargin = 6; //std::max(iconTopMargin, 1.);
121  iconRect = QRect(br.topLeft() + QPoint(iconLeftMargins, iconTopMargin - 2), iconSize);
122 
123  //31是字体设置20的时候的高度
124  br.setHeight(ICONTOTOP + iconRect.height() + TEXTTOICON + 31 + fontPixelSize * TextSecond + TEXTTOLEFT);
125  if (br.height() > ibr.height()) {
126  br.setHeight(ibr.height() - 1);
127  }
128 
129  // calc text
130  appNameRect = itemTextRect(br, iconRect, drawBlueDot);
131  const QPair<QString, bool> appTextResolvedInfo = holdTextInRect(fm, itemInfo.m_name, appNameRect.toRect());
132  appNameResolved = appTextResolvedInfo.first;
133 
134  if(fm.width(appNameResolved) > appNameRect.width()){
135  TextSecond = 1;
136  }
137 
138  if (margin == 1 || appTextResolvedInfo.second)
139  break;
140 
141  // we need adjust again!
142  adjust = true;
143  } while (true);
144 // painter->fillRect(option.rect, Qt::gray);
145 // painter->fillRect(ibr, Qt::cyan);
146 // painter->fillRect(br, Qt::green);
147 // painter->fillRect(appNameRect, Qt::blue);
148 // painter->fillRect(iconRect, Qt::magenta);
149 
150 
151  // draw focus background
152  if (is_current && !(option.features & QStyleOptionViewItem::HasDisplay))
153  {
154  const int radius = 18;
155  const QColor brushColor(255, 255, 255, 51);
156 
157  painter->setPen(Qt::transparent);
158  painter->setBrush(brushColor);
159  int drawBlueDotWidth = 0;
160  if(drawBlueDot) {
161  drawBlueDotWidth = m_blueDotPixmap.width();
162  }
163 
164  if(iconSize.width() > (fm.width(appNameResolved) + drawBlueDotWidth)) {
165  br.setX(iconRect.x()-ICONTOLETF);
166  br.setWidth(iconSize.width()+ICONTOLETF*2);
167  }
168 
169  painter->drawRoundedRect(br, radius, radius);
170  }
171 
172  // draw app name
173  QTextOption appNameOption;
174  appNameOption.setAlignment(Qt::AlignHCenter | Qt::AlignTop);
175  appNameOption.setWrapMode(QTextOption::WordWrap);
176 
177  appNameRect.setY(br.y() + br.height() - TEXTTOLEFT + (fm.height() >= 28 ? 2 : 0)- fm.height() - fontPixelSize*TextSecond);
178 
179  painter->setFont(appNamefont);
180  painter->setBrush(QBrush(Qt::transparent));
181  painter->setPen(QColor(0, 0, 0, 80));
182  painter->drawText(appNameRect.adjusted(0.8, 1, 0.8, 1), appNameResolved, appNameOption);
183  painter->drawText(appNameRect.adjusted(-0.8, 1, -0.8, 1), appNameResolved, appNameOption);
184  painter->setPen(Qt::white);
185  painter->drawText(appNameRect, appNameResolved, appNameOption);
186 
187  // draw app icon
188  if("dde-calendar"==itemInfo.m_key)
189  {
190  //根据不同日期显示不同日历图表
191  int tw = iconRect.width();
192  int th = iconRect.height();
193  int tx = iconRect.x();
194  int ty = iconRect.y();
195  const double iconZoom = iconRect.width() /256.0;
196 
197  QStringList calIconList = m_calcUtil->calendarSelectIcon();
198  //绘制背景
199  QSvgRenderer renderer_bg(calIconList.at(0));
200  renderer_bg.render(painter, iconRect);
201  //绘制月份
202  QSvgRenderer renderer_month(calIconList.at(1));
203  renderer_month.render(painter, QRect(tx + (tw / 3.4), ty + (th / 5.4), 80 * iconZoom, 40 * iconZoom));
204  //绘制日
205  QSvgRenderer renderer_day(calIconList.at(2));
206  renderer_day.render(painter, QRect(tx + (tw / 3.5), ty + th / 3.1, 112 * iconZoom, 104 * iconZoom));
207  //绘制周
208  QSvgRenderer renderer_week(calIconList.at(3));
209  renderer_week.render(painter, QRect(tx + (tw / 2.3), ty + ((th / 3.9) * 2.8), 56 * iconZoom, 24 * iconZoom));
210  }else {
211  const QPixmap iconPix = index.data(AppsListModel::AppIconRole).value<QPixmap>();
212  painter->drawPixmap(iconRect, iconPix, iconPix.rect());
213  }
214 
215  // draw icon if app is auto startup
216  const QPoint autoStartIconPos = iconRect.bottomLeft()
217  // offset for auto-start mark itself
218  - QPoint(m_autoStartPixmap.height(), m_autoStartPixmap.width()) / m_autoStartPixmap.devicePixelRatioF() / 2
219  // extra offset
220  + QPoint(iconRect.width() / 10, -iconRect.height() / 10);
221 
222  if (index.data(AppsListModel::AppAutoStartRole).toBool())
223  painter->drawPixmap(autoStartIconPos, m_autoStartPixmap);
224 
225  // draw blue dot if needed
226  if (drawBlueDot)
227  {
228  const int marginRight = 2;
229  const QRectF textRect = fm.boundingRect(appNameRect.toRect(), Qt::AlignTop | Qt::AlignHCenter | Qt::TextWordWrap, appNameResolved);
230  const auto ratio = m_blueDotPixmap.devicePixelRatioF();
231  const QPointF blueDotPos = textRect.topLeft() + QPoint(-m_blueDotPixmap.width() / ratio - marginRight,
232  (fm.height() - m_blueDotPixmap.height() / ratio) / 2);
233  painter->drawPixmap(blueDotPos, m_blueDotPixmap);
234  }
235 }

249 const QRect AppItemDelegate::itemBoundingRect(const QRect &itemRect) const
250 {
251  const int w = itemRect.width();
252  const int h = itemRect.height();
253  const int sub = qAbs((w - h) / 2);
254 
255  if (w == h)
256  return itemRect;
257  else if (w > h)
258  return itemRect - QMargins(sub, 0, sub, 0);
259  else
260  return itemRect - QMargins(0, 0, 0, sub * 2);
261 }

279 const QPair<QString, bool> AppItemDelegate::holdTextInRect(const QFontMetrics &fm, const QString &text, const QRect &rect) const
280 {
281  const int textFlag = Qt::AlignTop | Qt::AlignHCenter | Qt::TextWordWrap;
282 
283  if (rect.contains(fm.boundingRect(rect, textFlag, text)))
284  return QPair<QString, bool>(text, true);
285 
286  QString str(text + "...");
287 
288  while (true)
289  {
290  if (str.size() < 4)
291  break;
292 
293  QRect boundingRect = fm.boundingRect(rect, textFlag, str);
294  if (rect.contains(boundingRect))
295  break;
296 
297  str.remove(str.size() - 4, 1);
298  }
299 
300  return QPair<QString, bool>(str, false);
301 }
