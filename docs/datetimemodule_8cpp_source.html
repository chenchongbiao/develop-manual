105 int DatetimeModule::load(const QString &path)
106 {
107  if (!m_widget) {
108  active();
109  }
110 
111  QListView *list = m_widget->getListViewPointer();
112  DatetimeType type = Default;
113 
114  if (!list) {
115  return 0;
116  }
117 
118  if (path == "Timezone List") {
119  type = ETimezoneList;
120  } else if (path == "Timezone List/Change System Timezone") {
121  type = ESystemTimezone;
122  } else if (path == "Time Settings") {
123  type = TimeSetting;
124  } else if (path == "Timezone List/Add Timezone") {
125  type = AddTimeZone;
126  } else if (path == "Format Settings") {
127  type = FormatSetting;
128  }
129 
130  QModelIndex index = list->model()->index(type, 0);
131  switch (type) {
132  case ETimezoneList:
133  case TimeSetting:
134  list->setCurrentIndex(index);
135  list->clicked(index);
136  break;
137  case ESystemTimezone:
138  //First enter timezoneList
139  index = list->model()->index(ETimezoneList, 0);
140  list->setCurrentIndex(index);
141  list->clicked(index);
142 
143  //Then enter systemTimezone
144  showSystemTimezone();
145  break;
146  case AddTimeZone:
147  //First enter timezoneList
148  index = list->model()->index(ETimezoneList, 0);
149  list->setCurrentIndex(index);
150  list->clicked(index);
151 
152  //Then enter addTimezone
153  showSystemTimezone();
154  Q_EMIT m_timezonelist->requestAddTimeZone();
155  break;
156  case FormatSetting:
157  //First enter FormatSetting
158  index = list->model()->index(FormatSetting, 0);
159  list->setCurrentIndex(index);
160  list->clicked(index);
161 
162  //Then enter FormatSetting
163  showFormatSetting();
164  break;
165  default:
166  break;
167  }
168 
169  return 0;
170 }

190  }
191

240 void DatetimeModule::showTimezoneList()
241 {

243  connect(sysTimezoneWidget, &SystemTimezone::requestSetSystemTimezone, this, &DatetimeModule::showSystemTimezone);
244  connect(m_model, &DatetimeModel::currentSystemTimeZoneChanged, sysTimezoneWidget, &SystemTimezone::setSystemTimezone);
245  sysTimezoneWidget->setSystemTimezone(m_model->currentSystemTimeZone());
246 
247  if (!m_timezonelist) {
248  m_timezonelist = new TimezoneList;
249 
250  //first into this page, update timezonelist data
251  updateSystemTimezone(m_model->systemTimeZoneId());
252 
253  connect(m_model, &DatetimeModel::userTimeZoneAdded,
254  m_timezonelist->getTimezoneContentListPtr(), &TimezoneContentList::addTimezone);
255  connect(m_model, &DatetimeModel::userTimeZoneRemoved,
256  m_timezonelist->getTimezoneContentListPtr(), &TimezoneContentList::removeTimezone);
257  connect(m_timezonelist, &TimezoneList::requestAddTimeZone, this, [this] {
258  ensureZoneChooserDialog();
259  m_dialog->setIsAddZone(true);
260  m_dialog->show();
261  });
262  connect(m_timezonelist->getTimezoneContentListPtr(), &TimezoneContentList::requestRemoveUserTimeZone,
263  this, &DatetimeModule::requestRemoveUserTimeZone);
264  connect(m_timezonelist, &TimezoneList::requestAddUserTimeZone,
265  this, &DatetimeModule::requestAddUserTimeZone);
266  }
267 
268  QWidget *widget = new QWidget;
269  widget->setVisible(false);
270  QVBoxLayout *layout = new QVBoxLayout;
271  layout->setMargin(0);
272  layout->addWidget(sysTimezoneWidget, Qt::AlignTop);
273  layout->addWidget(m_timezonelist);
274  widget->setLayout(layout);
275  m_frameProxy->pushWidget(this, widget);
276  widget->setVisible(true);
