112 void TouchscreenPage::onMonitorChanged()
113 {
114  m_buttonTuple->rightButton()->setEnabled(false);
115 
116  QWidget *oldWidget = m_contentArea->takeWidget();
117  if (oldWidget) {
118  oldWidget->deleteLater();
119  }
120  m_list.clear();
121  m_titleName.clear();
122  m_labels.clear();
123 
124  QWidget *contentWidget = new QWidget();
125  contentWidget->installEventFilter(this);
126 
127  m_contentArea->setWidget(contentWidget);
128 
129  QVBoxLayout *contentLayout = new QVBoxLayout();
130  contentWidget->setLayout(contentLayout);
131  contentLayout->setAlignment(Qt::AlignTop);
132  contentLayout->setSpacing(10);
133  contentLayout->setContentsMargins(0, 0, 0, 0);
134 
135  TouchscreenMap touchMap = m_model->touchMap();
136  TouchscreenInfoList touchscreenList = m_model->touchscreenList();
137  QList<Monitor *> monitorList = m_model->monitorList();
138  for (const auto &i : touchscreenList) {
139  QString touchscreenSerial = i.serialNumber;
140 
141  auto title = QString(tr("Touch Screen - %1 (%2)")).arg(i.name).arg(i.id);
142  auto *label = new QLabel(title);
143 
144  // 窗口显示前,窗口width()无法拿到正确值,导致eventFilter中初始化label文本时计算宽度错误,设置小的初始值,通过layout自动调整
145  label->setGeometry(0, 0, FontSpaceLenght, FontSpaceLenght);
146  label->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
147  label->setAlignment(Qt::AlignLeft);
148  label->setContentsMargins(10, 0, 0, 0);
149  contentLayout->addWidget(label);
150 
151  MCombobox *listCombo = new MCombobox();
152  listCombo->setProperty("touchscreenName", i.name);
153  listCombo->setProperty("touchscreenSerial", touchscreenSerial);
154  listCombo->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
155  for (Monitor *m : monitorList) {
156  listCombo->addItem(m->name());
157  }
158  if (touchMap.find(touchscreenSerial) != touchMap.end()) {
159  listCombo->setCurrentText(touchMap.value(touchscreenSerial));
160  }
161  contentLayout->addWidget(listCombo);
162 
163  m_list.push_back(listCombo);
164  m_labels.push_back(label);
165  m_titleName << title;
166 
167  connect(listCombo, static_cast<void (QComboBox::*)(int)>(&QComboBox::currentIndexChanged), [this] {
168  m_buttonTuple->rightButton()->setEnabled(true);
169  });
170  }
171 }

173 void TouchscreenPage::save()
174 {
175  m_buttonTuple->rightButton()->setEnabled(false);
176  bool changed = false;
177  for (auto i : m_list) {
178  QString touchscreenName = i->property("touchscreenName").toString();
179  QString touchscreenSerial = i->property("touchscreenSerial").toString();
180  QString output = i->currentText();
181  TouchscreenMap touchMap = m_model->touchMap();
182  if (touchMap.value(touchscreenSerial) == output) {
183  continue;
184  }
185 
186  changed = true;
187  Q_EMIT requestAssociateTouch(output, touchscreenSerial);
188  }
189 
190  if (changed) {
191  m_notifyInter->Notify("dde-control-center",
192  static_cast<uint>(QDateTime::currentMSecsSinceEpoch()),
193  "preferences-system",
194  tr("Touch Screen Settings"),
195  tr("The settings of touch screen changed"),
196  QStringList(),
197  QVariantMap(),
198  3000);
199  }
200 }

202 bool TouchscreenPage::eventFilter(QObject *obj, QEvent *event)
203 {
204  Q_UNUSED(obj);
205  if (event->type() == QEvent::Resize) {
206  for (int i = 0; i < m_labels.count(); i++)
207  {
208  QFontMetrics fontMetrics(m_labels[i]->font());
209  int fontSize = fontMetrics.boundingRect(m_titleName[i]).width();
210 
211  // QFontMetrics计算不精准，为保证视觉效果，设置右边空10（FontSpaceLenght）个像素作为余量
212  if ((fontSize + FontSpaceLenght) > m_labels[i]->width()) {
213  // 字体过长做省略显示处理时，以控件宽度减去空白处为标准截取
214  m_labels[i]->setText(fontMetrics.elidedText(m_titleName[i], Qt::ElideMiddle, m_labels[i]->width() - FontSpaceLenght * 2));
215  m_labels[i]->setToolTip(m_titleName[i]);
216  } else {
217  m_labels[i]->setText(m_titleName[i]);
218  m_labels[i]->setToolTip("");
219  }
220  }
221 
222  return true;
223  }
224  return dcc::ContentWidget::eventFilter(obj, event);
225 }
