68 bool CDDFManager::loadDDF(const QString &path, bool isOpenByDDF)
69 {
70  bool result = true;
71 
72  EDdfVersion ddfVersion = getDdfVersion(path);
73 
74  if (ddfVersion > EDdfCurVersion) {
75  qWarning() << "Cannot open higher version file!";
76 
77  auto pView = m_view;
78  //因为如果右键多个ddf文件进行打开的情况下,遇到加载文件失败时,如果是直接弹出对话框那么会阻塞后续文件的加载,通过队列的方式在加载完成后提示
79  QMetaObject::invokeMethod(this, [ = ]() {
80  //中文翻译:文件版本与当前应用不兼容，请安装最新版应用
81  drawApp->exeMessage(tr("The file is incompatible with the old app, please install the latest version"), Application::EWarningMsg, false);
82  this->closeViewPage(pView, true);
83  }, Qt::QueuedConnection);
84 
85  result = false;
86 
87  } else if (ddfVersion < 0) {
88  qWarning() << "Cannot open unknown version file!";
89  result = false;
90  } else {
91  if (ddfVersion >= EDdf5_9_0_3_LATER) {
92  loadDdfWithCombinGroup(path, isOpenByDDF);
93  } else {
94  loadDdfWithNoCombinGroup(path, isOpenByDDF);
95  }
96  }
97  return result;
98 }

146 bool CDDFManager::isDdfFileDirty(const QString &filePath)
147 {
148  QFile file(filePath);
149  if (file.exists()) {
150  if (file.open(QFile::ReadOnly)) {
151  QDataStream s(&file);
152  // 先通过版本号判断是否ddf拥有md5校验值
153  EDdfVersion ver = getVersion(s);
154  if (ver >= EDdf5_8_0_20) {
155  QByteArray allBins = file.readAll();
156  QByteArray md5 = allBins.right(16);
157 
158  qDebug() << "load head+bytes = " << (allBins.count() - md5.count()) << "md5 count = " << md5.count();
159  qDebug() << "direct read MD5 form ddffile file = " << filePath << " MD5 = " << md5.toHex().toUpper();
160 
161  QByteArray contex = allBins.left(allBins.size() - md5.size());
162 
163  QByteArray nMd5 = QCryptographicHash::hash(contex, QCryptographicHash::Md5);
164 
165  qDebug() << "recalculate MD5 form ddffile file = " << filePath << " MD5 = " << nMd5.toHex().toUpper();
166 
167  if (md5 != nMd5) {
168  return true;
169  }
170  } else if (ver == EDdfUnknowed) {
171  return true;
172  }
173  return false;
174  }
175  }
176  return true;
177 }

234 void serializationTree_helper(QDataStream &countStream, int &bzItemCount, int &groupCount,
235  const CGroupBzItemsTreeInfo &tree, std::function<void(int, int)> f = nullptr)
236 {
237  countStream << tree.childGroups.size();
238 
239  qDebug() << "save group count = " << tree.childGroups.size();
240 
241  for (auto p : tree.childGroups) {
242  serializationTree_helper(countStream, bzItemCount, groupCount, p, f);
243  }
244 
245  countStream << tree.bzItems.size();
246 
247  qDebug() << "save item count = " << tree.bzItems.size();
248 
249  for (auto i : tree.bzItems) {
250  countStream << i;
251  qDebug() << "save item info = " << i.head.dataType;
252 
253  ++bzItemCount;
254  if (f != nullptr) {
255  f(bzItemCount, groupCount);
256  }
257  }
258 
259  //bzItemCount += tree.bzItems.size();
260 
261  countStream << tree.data;
262 
263  qDebug() << "save group info = " << tree.data.head.dataType;
264 
265  ++groupCount;
266  if (f != nullptr) {
267  f(bzItemCount, groupCount);
268  }
269 }
270 CGroupBzItemsTreeInfo deserializationToTree_helper(QDataStream &inStream, int &outBzItemCount, int &outGroupCount,
271  std::function<void(int, int)> f = nullptr)
272 {
273  CGroupBzItemsTreeInfo result;
274  int groupCount = 0;
275  inStream >> groupCount;
276  qDebug() << "read group count = " << groupCount;
277  for (int i = 0; i < groupCount; ++i) {
278  CGroupBzItemsTreeInfo child = deserializationToTree_helper(inStream, outBzItemCount, outGroupCount, f);
279  result.childGroups.append(child);
280  }
281  int bzItemCount = 0;
282  inStream >> bzItemCount;
283  qDebug() << "read item count = " << bzItemCount;
284  for (int i = 0; i < bzItemCount; ++i) {
285  CGraphicsUnit unit;
286  unit.reson = ESaveToDDf;
287  inStream >> unit;
288  qDebug() << "read item info = " << unit.head.dataType;
289  result.bzItems.append(unit);
290  ++outBzItemCount;
291  if (f != nullptr) {
292  f(outBzItemCount, outGroupCount);
293  }
294  }
295  CGraphicsUnit unit;
296  unit.reson = ESaveToDDf;
297  inStream >> unit;
298  qDebug() << "read group info = " << unit.head.dataType;
299  result.data = unit;
300  ++outGroupCount;
301  if (f != nullptr) {
302  f(outBzItemCount, outGroupCount);
303  }
304  return result;
305 }

410 void CDDFManager::saveDdfWithCombinGroup(const QString &path, const QGraphicsScene *scene, bool finishedNeedClose)
411 {
412  if (scene == nullptr)
413  return;
414 
415  CDrawScene *pDrawScen = qobject_cast<CDrawScene *>(const_cast<QGraphicsScene *>(scene));
416 
417  if (pDrawScen == nullptr)
418  return;
419 
420  CGroupBzItemsTreeInfo treeInfo = pDrawScen->getGroupTreeInfo(nullptr, ESaveToDDf);
421 
422  if (treeInfo.childGroups.isEmpty() && treeInfo.bzItems.isEmpty()) {
423  return;
424  }
425 
426  //0.开始之前设置当前要保存的文件名,并去掉针对该文件的监视(自身程序修改该文件不需要提示文件被修改了)
427  m_path = path;
428  CManageViewSigleton::GetInstance()->removeWacthedFile(path);
429  m_finishedClose = finishedNeedClose;
430 
431  //接下来启动线程
432  QtConcurrent::run([ = ] {
433 
434  //1.将treeInfo进行序列化到内存这个过程很快(很多图片的时候内存是否足够?)
435  STreePlatInfo info = serializationTreeToBytes(treeInfo);
436 
437  //2.初始化文件的头信息(主要包含了ddf版本号,总的图元个数,场景的大小)并序列化到内存
438  m_graphics.version = qint32(EDdfCurVersion);
439  m_graphics.unitCount = info.groupCount + info.bzItemCount; //在加载时可快速获知图元总数
440  m_graphics.rect = scene->sceneRect(); //场景的大小
441  info.headBytes = serializationHeadToBytes(m_graphics);
442 
443  //3.将树结构的二进制流数据进行md5加密
444  info.md5 = QCryptographicHash::hash(info.headBytes + info.itemsBytes, QCryptographicHash::Md5);
445 
446  qDebug() << "save head size = " << info.headBytes.count() << "byte size = " << info.itemsBytes.count() << "md5 count = " << info.md5.count();
447 
448  //4.再进行磁盘检查(判断目标文件所处的磁盘分区空间是否足够),如果未通过,交互提示返回
449  if (!isVolumeSpaceAvailabel(path, info.itemsBytes.size() + info.md5.size()))
450  {
451  qWarning() << "volum space is not enough !!!";
452 
453  //4.1交互提示空间不足
454  QMetaObject::invokeMethod(this, [ = ]() {
455  drawApp->exeMessage(tr("Unable to save. There is not enough disk space."));
456  m_lastSaveStatus = true;
457  m_lastErrorString = "volum space is not enough!";
458  m_lastError = QFileDevice::WriteError;
459  m_graphics.vecGraphicsUnit.clear();
460  }, Qt::QueuedConnection);
461 
462  //磁盘检查不通过直接结束,不应该更新保存路径
463  //emit signalSaveDDFComplete();
464 
465  return;
466  }
467 
468  //5.展示保存的进度交互界面
469  QMetaObject::invokeMethod(this, [ = ]()
470  {
471  m_pSaveDialog->show();
472  m_pSaveDialog->setTitle(tr("Saving..."));
473  m_pSaveDialog->setProcess(0);
474  }, Qt::QueuedConnection);
475 
476  //6.将数据保存到文件(或者一开始就启动线程?)
477  QFile writeFile(path);
478  if (writeFile.open(QIODevice::WriteOnly | QIODevice::ReadOnly | QIODevice::Truncate))
479  {
480  QDataStream out(&writeFile);
481 
482  //6.1 首先保存沿用的老的头信息(主要包含了ddf版本号,总的图元个数,场景的大小)
483  int resultByteSize = out.writeRawData(info.headBytes.data(), info.headBytes.size());
484  qDebug() << "write heads bytes: " << info.headBytes.size() << "success byteSize: " << resultByteSize;
485 
486  emit signalUpdateProcessBar(10, true);
487 
488  //6.2 写入图元的数据(安排一下进度?)
489  int granularityByteSize = 200; //设置字节粒度
490  resultByteSize = 0;
491  while (resultByteSize < info.itemsBytes.size()) {
492  QByteArray bytes = info.itemsBytes.mid(resultByteSize, granularityByteSize);
493  resultByteSize += out.writeRawData(bytes.data(), bytes.size());
494  emit signalUpdateProcessBar(100 * resultByteSize / info.itemsBytes.size(), true);
495  }
496  //resultByteSize = out.writeRawData(info.itemsBytes.data(), info.itemsBytes.size());
497  qDebug() << "write items bytes: " << info.itemsBytes.size() << "success byteSize: " << resultByteSize;
498 
499  //6.3 写入md5数据
500  resultByteSize = out.writeRawData(info.md5.data(), info.md5.size());
501  qDebug() << "write md5 bytes: " << info.md5.size() << "success byteSize: " << resultByteSize << "md5 = " << info.md5.toHex().toUpper();
502 
503  //6.4 关闭文件(执行后才会真的写入到磁盘)
504  writeFile.close();
505 
506  emit signalUpdateProcessBar(100, true);
507 
508  //6.5 记录可能的错误原因,在结束后展示交互提示
509  m_lastSaveStatus = true;
510  m_lastErrorString = writeFile.errorString();
511  m_lastError = writeFile.error();
512 
513  //6.6 通知结束
514  emit signalSaveDDFComplete();
515 
516  qDebug() << "save ddf end----------------";
517  }
518  });
519 }

521 void CDDFManager::loadDdfWithNoCombinGroup(const QString &path, bool isOpenByDDF)
522 {
523  emit signalClearSceneBeforLoadDDF();
524 
525  m_CProgressDialog->showProgressDialog(CProgressDialog::LoadDDF, isOpenByDDF);
526  m_path = path;
527 
528  QtConcurrent::run([ = ] {
529 
530  //首先判断文件是否是脏的(是否是ddf合法的,判断方式为md5校验(如果ddf文件版本是老版本那么会直接返回false))
531  if (isDdfFileDirty(path))
532  {
533  //弹窗提示
534  emit signalLoadDDFComplete(path, false);
535  return;
536  }
537 
538  QFile readFile(path);
539  if (readFile.open(QIODevice::ReadOnly))
540  {
541  QDataStream in(&readFile);
542 
543  in >> m_graphics;
544  qDebug() << QString("load ddf(%1)").arg(path) << " ddf version = " << m_graphics.version << "graphics count = " << m_graphics.unitCount <<
545  "scene size = " << m_graphics.rect;
546  //qDebug() << "m_graphics.unitCount = " << m_graphics.unitCount;
547  emit signalStartLoadDDF(m_graphics.rect);
548 
549  int count = 0;
550  //int process = 0;
551 
552  for (int i = 0; i < m_graphics.unitCount; i++) {
553  CGraphicsUnit unit;
554  unit.reson = ESaveToDDf;
555  in >> unit;
556 
557  if (TextType == unit.head.dataType) {
558  emit signalAddTextItem(unit);
559  } else {
560  qDebug() << "unit.head.dataType != BlurType = " << (unit.head.dataType != BlurType);
561  //EDdf5_9_0_3_LATER之后的版本不再存在模糊图元,所以如果加载EDdf5_9_0_3_LATER之前的ddf文件那么不用加载模糊
562  if (unit.head.dataType != BlurType) {
563  CGraphicsItem *pItem = CGraphicsItem::creatItemInstance(unit.head.dataType, unit);
564  emit signalAddItem(pItem);
565  }
566  unit.release();
567  }
568 
569  //进度条处理
570  count ++;
571 // process = int(qreal(count) / qreal(m_graphics.unitCount) * 100.0);
572  emit signalUpdateProcessBar(int(qreal(count) / qreal(m_graphics.unitCount) * 100.0), false);
573  }
574  in >> m_graphics.version;
575  qDebug() << "loadDDF m_graphics.version = " << m_graphics.version << endl;
576  readFile.close();
577  emit signalLoadDDFComplete(path, true);
578  }
579  });
580 }

582 void CDDFManager::loadDdfWithCombinGroup(const QString &path, bool isOpenByDDF)
