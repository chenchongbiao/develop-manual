140 void BluetoothModule::showPinCode(const QDBusObjectPath &device, const QString &code)
141 {
142  qDebug() << "request confirmation: " << device.path() << code;
143 
144  PinCodeDialog *dialog = PinCodeDialog::instance(code);
145  m_dialogs[device] = dialog;
