146 template<typename T>
147 QVariant eventProcess(DFileService *service, const QSharedPointer<DFMEvent> &event, T function)
148 {
149  QSet<DAbstractFileController *> controller_set;
150  QSet<QString> scheme_set;
151 
152  for (const DUrl &url : event->handleUrlList()) {
153  QList<DAbstractFileController *> list = service->getHandlerTypeByUrl(url);
154  if (!scheme_set.contains(url.scheme())) {
155  list << service->getHandlerTypeByUrl(url, true);
156  } else {
157  scheme_set << url.scheme();
158  }
159 
160  for (DAbstractFileController *controller : list) {
161  if (controller_set.contains(controller)) {
162  continue;
163  }
164 
165  controller_set << controller;
166 
167  typedef typename std::remove_reference<typename QtPrivate::FunctionPointer<T>::Arguments::Car>::type::Type DFMEventType;
168 
169  const QVariant result = QVariant::fromValue((controller->*function)(event.staticCast<DFMEventType>()));
170 
171  if (event->isAccepted()) {
172  return result;
173  }
174  }
175  }
176 
177  return QVariant();
178 }

568 bool DFileService::setFileUrlHandler(const QString &scheme, const QString &host,
569  DAbstractFileController *controller)
570 {
571  if (DFileServicePrivate::handlerHash.contains(controller)) {
572  return true;
573  }
574 
575  const HandlerType &type = HandlerType(scheme, host);
576 
577  foreach (const DAbstractFileController *value, DFileServicePrivate::controllerHash.values(type)) {
578  if (typeid(*value) == typeid(*controller)) {
579  controller->deleteLater();
580  return false;
581  }
582  }
583 
584  DFileServicePrivate::handlerHash[controller] = type;
585  DFileServicePrivate::controllerHash.insertMulti(type, controller);
586 
587  return true;
588 }
