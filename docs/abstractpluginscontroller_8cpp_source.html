51 void AbstractPluginsController::saveValue(PluginsItemInterface *const itemInter, const QString &key, const QVariant &value)
52 {
53  // is it necessary?
54 // refreshPluginSettings();
55  int fixedPluginCount(0); // FixPlugin Counts
56 
57  // save to local cache
58  QJsonObject localObject = m_pluginSettingsObject.value(itemInter->pluginName()).toObject();
59  localObject.insert(key, QJsonValue::fromVariant(value)); //Note: QVariant::toJsonValue() not work in Qt 5.7
60 
61  // save to daemon
62  QJsonObject remoteObject, remoteObjectInter;
63  remoteObjectInter.insert(key, QJsonValue::fromVariant(value)); //Note: QVariant::toJsonValue() not work in Qt 5.7
64  remoteObject.insert(itemInter->pluginName(), remoteObjectInter);
65 
66  if (itemInter->type() == PluginsItemInterface::Fixed && key == "enable" && !value.toBool()) {
67  // 遍历FixPlugin插件个数
68  for (auto it(m_pluginsMap.begin()); it != m_pluginsMap.end();) {
69  if (it.key()->type() == PluginsItemInterface::Fixed) {
70  fixedPluginCount++;
71  }
72  ++it;
73  }
74  // 修改插件的order值，位置为队尾
75  QString name = localObject.keys().last();
76  localObject.insert(name, QJsonValue::fromVariant(fixedPluginCount)); //Note: QVariant::toJsonValue() not work in Qt 5.7
77 
78  // daemon中同样修改
79  remoteObjectInter.insert(name, QJsonValue::fromVariant(fixedPluginCount)); //Note: QVariant::toJsonValue() not work in Qt 5.7
80  remoteObject.insert(itemInter->pluginName(), remoteObjectInter);
81  }
82 
83  m_pluginSettingsObject.insert(itemInter->pluginName(), localObject);
84  m_dockDaemonInter->MergePluginSettings(QJsonDocument(remoteObject).toJson(QJsonDocument::JsonFormat::Compact));
85 }

181 void AbstractPluginsController::loadPlugin(const QString &pluginFile, bool lastone)
182 {
183  QPluginLoader *pluginLoader = new QPluginLoader(pluginFile);
184  const QJsonObject &meta = pluginLoader->metaData().value("MetaData").toObject();
185  const QString &pluginApi = meta.value("api").toString();
186  bool pluginIsValid = true;
187  if (pluginApi.isEmpty() || !CompatiblePluginApiList.contains(pluginApi)) {
188  qDebug() << objectName()
189  << "plugin api version not matched! expect versions:" << CompatiblePluginApiList
190  << ", got version:" << pluginApi
191  << ", the plugin file is:" << pluginFile;
192 
193  pluginIsValid = false;
194  }
195 
196  PluginsItemInterface *interface = qobject_cast<PluginsItemInterface *>(pluginLoader->instance());
197  if (!interface) {
198  qDebug() << objectName() << "load plugin failed!!!" << pluginLoader->errorString() << pluginFile;
199 
200  pluginLoader->unload();
201  pluginLoader->deleteLater();
202 
203  pluginIsValid = false;
204  }
205 
206  if (!pluginIsValid) {
207  QString notifyMessage(tr("The plugin %1 is not compatible with the system."));
208  Dtk::Core::DUtil::DNotifySender(notifyMessage.arg(QFileInfo(pluginFile).fileName())).appIcon("dialog-warning").call();
209  return;
210  }
211 
212  if (interface->pluginName() == "multitasking") {
213  if (qEnvironmentVariable("XDG_SESSION_TYPE").contains("wayland") or Dtk::Core::DSysInfo::deepinType() == Dtk::Core::DSysInfo::DeepinServer)
214  return;
215  }
216 
217  // 保存 PluginLoader 对象指针
218  QMap<QString, QObject *> interfaceData;
219  interfaceData["pluginloader"] = pluginLoader;
220  m_pluginsMap.insert(interface, interfaceData);
221 
222  QString dbusService = meta.value("depends-daemon-dbus-service").toString();
223  if (!dbusService.isEmpty() && !m_dbusDaemonInterface->isServiceRegistered(dbusService).value()) {
224  qDebug() << objectName() << dbusService << "daemon has not started, waiting for signal";
225  connect(m_dbusDaemonInterface, &QDBusConnectionInterface::serviceOwnerChanged, this,
226  [ = ](const QString & name, const QString & oldOwner, const QString & newOwner) {
227  Q_UNUSED(oldOwner);
228  if (name == dbusService && !newOwner.isEmpty()) {
229  qDebug() << objectName() << dbusService << "daemon started, init plugin and disconnect";
230  initPlugin(interface);
231  disconnect(m_dbusDaemonInterface);
232  }
233  }
234  );
235  return;
236  }
237 
238  // NOTE(justforlxz): 插件的所有初始化工作都在init函数中进行，
239  // loadPlugin函数是按队列执行的，initPlugin函数会有可能导致
240  // 函数执行被阻塞。
241  QTimer::singleShot(1, this, [ = ] {
242  initPlugin(interface);
243  if (lastone) {
244  emit pluginLoaderFinished();
245  }
246  });
247 }

256 void AbstractPluginsController::refreshPluginSettings()
257 {
258  const QString &pluginSettings = m_dockDaemonInter->GetPluginSettings().value();
259  qDebug() << pluginSettings;
260  if (pluginSettings.isEmpty()) {
261  qDebug() << "Error! get plugin settings from dbus failed!";
262  return;
263  }
264 
265  const QJsonObject &pluginSettingsObject = QJsonDocument::fromJson(pluginSettings.toLocal8Bit()).object();
266  if (pluginSettingsObject.isEmpty()) {
267  return;
268  }
269 
270  // nothing changed
271  if (pluginSettingsObject == m_pluginSettingsObject) {
272  return;
273  }
274 
275  for (auto pluginsIt = pluginSettingsObject.constBegin(); pluginsIt != pluginSettingsObject.constEnd(); ++pluginsIt) {
276  const QString &pluginName = pluginsIt.key();
277  const QJsonObject &settingsObject = pluginsIt.value().toObject();
278  QJsonObject newSettingsObject = m_pluginSettingsObject.value(pluginName).toObject();
279  for (auto settingsIt = settingsObject.constBegin(); settingsIt != settingsObject.constEnd(); ++settingsIt) {
280  newSettingsObject.insert(settingsIt.key(), settingsIt.value());
281  }
282  // TODO: remove not exists key-values
283  m_pluginSettingsObject.insert(pluginName, newSettingsObject);
284  }
285 
286  // not notify plugins to refresh settings if this update is not emit by dock daemon
287  if (sender() != m_dockDaemonInter) {
288  return;
289  }
290 
291  // notify all plugins to reload plugin settings
292  for (PluginsItemInterface *pluginInter : m_pluginsMap.keys()) {
293  pluginInter->pluginSettingsChanged();
294  }
295 
296  // reload all plugin items for sort order or container
297  QMap<PluginsItemInterface *, QMap<QString, QObject *>> pluginsMapTemp = m_pluginsMap;
298  for (auto it = pluginsMapTemp.constBegin(); it != pluginsMapTemp.constEnd(); ++it) {
299  const QList<QString> &itemKeyList = it.value().keys();
300  for (auto key : itemKeyList) {
301  itemRemoved(it.key(), key);
302  }
303  for (auto key : itemKeyList) {
304  itemAdded(it.key(), key);
305  }
306  }
307 }

309 bool AbstractPluginsController::eventFilter(QObject *o, QEvent *e)
310 {
311  if (o != qApp)
312  return false;
313  if (e->type() != QEvent::DynamicPropertyChange)
314  return false;
315 
316  QDynamicPropertyChangeEvent *const dpce = static_cast<QDynamicPropertyChangeEvent *>(e);
317  const QString propertyName = dpce->propertyName();
318 
319  if (propertyName == PROP_POSITION)
320  positionChanged();
321  else if (propertyName == PROP_DISPLAY_MODE)
322  displayModeChanged();
323 
324  return false;
325 }
