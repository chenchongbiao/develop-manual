92 void CSideWidthWidget::initConnection()
93 {
94  connect(m_menuComboBox, &DComboBox::currentTextChanged, [ = ](QString text) {
95  if (text.contains("px")) {
96  // 判断并且获取当前线宽度
97  bool flag = false;
98  int lineWidth = text.trimmed().toLower().replace("px", "").toInt(&flag);
99 
100  if (flag) {
101  if (CManageViewSigleton::GetInstance()->getCurView() != nullptr) {
102  CManageViewSigleton::GetInstance()->getCurView()->getDrawParam()->setLineWidth(lineWidth);
103  emit signalSideWidthChange();
104  emit sideWidthChanged(lineWidth);
105  }
106  }
107  }
108  });
109 
110  connect(m_menuComboBox, QOverload<const QString &>::of(&DComboBox::currentIndexChanged), [ = ](const QString & text) {
111  if (text.contains("px")) {
112  // 判断并且获取当前线宽度
113  bool flag = false;
114  int lineWidth = text.trimmed().toLower().replace("px", "").toInt(&flag);
115 
116  if (flag) {
117  emit signalSideWidthChoosed(lineWidth);
118  this->setMenuNoSelected(false);
119  }
120  }
121  });
122 
123  // 设置默认2px的线宽度
124  m_menuComboBox->setCurrentIndex(2);
125 }
