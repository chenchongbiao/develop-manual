153 QString DiskControlItem::sizeString(const QString &str)
154 {
155  int begin_pos = str.indexOf('.');
156 
157  if (begin_pos < 0)
158  return str;
159 
160  QString size = str;
161 
162  while (size.count() - 1 > begin_pos) {
163  if (!size.endsWith('0'))
164  return size;
165 
166  size = size.left(size.count() - 1);
167  }
168 
169  return size.left(size.count() - 1);
170 }

172 QString DiskControlItem::formatDiskSize(const quint64 num)
173 {
174  QStringList list {" B", " KB", " MB", " GB", " TB"};
175  qreal fileSize(num);
176 
177  QStringListIterator i(list);
178  QString unit = i.next();
179 
180  int index = 0;
181  while (i.hasNext()) {
182  if (fileSize < 1024) {
183  break;
184  }
185 
186  unit = i.next();
187  fileSize /= 1024;
188  index++;
189  }
190  return QString("%1%2").arg(sizeString(QString::number(fileSize, 'f', 1)), unit);
191 }

193 void DiskControlItem::mouseReleaseEvent(QMouseEvent *e)
194 {
195  QWidget::mouseReleaseEvent(e);
196 
197  DGioSettings gsettings("com.deepin.dde.dock.module.disk-mount", "/com/deepin/dde/dock/module/disk-mount/");
198  if (gsettings.value("filemanager-integration").toBool()) {
199  DUrl url = DUrl(attachedDevice->accessPointUrl());
200  if (url.scheme() == BURN_SCHEME) {
201  // 1. 当前熊默认文件管理器为 dde-file-manager 时，使用它打开光盘
202  // 2. 默认文件管理器为其他时，依然采用打开挂载点的方式
203  if (!QStandardPaths::findExecutable(QStringLiteral("dde-file-manager")).isEmpty()) {
204  QString &&path = url.path();
205  QString opticalPath = QString("burn://%1").arg(path);
206  qDebug() << "open optical path =>" << opticalPath;
207  QProcess::startDetached(QStringLiteral("dde-file-manager"), {opticalPath});
208  } else {
209  url = DUrl(attachedDevice->mountpointUrl());
210  DDesktopServices::showFolder(url);
211  }
212  } else {
213  DDesktopServices::showFolder(url);
214  }
215  }

218 void DiskControlItem::showEvent(QShowEvent *e)
219 {
220  m_diskName->setText(attachedDevice->displayName());
221 
222  if (attachedDevice->deviceUsageValid()) {
223  QString iconName = attachedDevice->iconName();
224  QPair<quint64, quint64> freeAndTotal = attachedDevice->deviceUsage();
225  quint64 bytesFree = freeAndTotal.first;
226  quint64 bytesTotal = freeAndTotal.second;
227 
228  if (iconName.simplified().toLower() == "media-optical") { // 光驱无法读取容量
229  //fix: 设置光盘容量属性
230  quint64 burnCapacityTotalSize = 0;
231  quint64 burnCapacityUsedSize = 0;
232  int burnStatus = 0;//光盘容量状态：0,光驱弹出状态 1,光驱弹入处于添加未挂载状态 2,光驱弹入处于添加后并挂载的状态
233  int burnExt = 0;
234  QFile burnCapacityFile(QString("%1/.config/deepin/dde-file-manager/dde-file-manager.json").arg(QStandardPaths::writableLocation(QStandardPaths::HomeLocation)));
235  if (!burnCapacityFile.open(QIODevice::ReadOnly)) {
236  qDebug() << "Couldn't open dde-file-manager.json burnCapacityFile!";
237  return;
238  }
239  QByteArray burnCapacityData = burnCapacityFile.readAll();
240  burnCapacityFile.close();
241 
242  QJsonParseError parseJsonErr;
243  QJsonDocument jsonDoc(QJsonDocument::fromJson(burnCapacityData, &parseJsonErr));
244  if (!(parseJsonErr.error == QJsonParseError::NoError)) {
245  qDebug() << "decode json file error！";
246  return;
247  }
248  QJsonObject tempBurnObjs = jsonDoc.object();
249  qDebug() << "tempBurnObjs==" << tempBurnObjs;
250  if (tempBurnObjs.contains(QStringLiteral("BurnCapacityAttribute"))) {
251  QJsonValue jsonBurnValueList = tempBurnObjs.value(QStringLiteral("BurnCapacityAttribute"));
252  QJsonObject tagItem = jsonBurnValueList.toObject();
253  if (tagItem.contains(m_tagName)) {
254  QJsonObject burnItem = tagItem[m_tagName].toObject();
255  burnCapacityTotalSize = static_cast<quint64>(burnItem["BurnCapacityTotalSize"].toDouble());
256  burnCapacityUsedSize = static_cast<quint64>(burnItem["BurnCapacityUsedSize"].toDouble());
257  burnStatus = burnItem["BurnCapacityStatus"].toInt();
258  burnExt = burnItem["BurnCapacityExt"].toInt();
259  }
260  }
261  Q_UNUSED(burnExt)//解决未用警告
262  bytesFree = burnCapacityTotalSize - burnCapacityUsedSize;
263  bytesTotal = burnCapacityTotalSize;
264  if (burnStatus == 2) {
265  m_diskCapacity->setText(QString("%1 / %2")
266  .arg(formatDiskSize(burnCapacityUsedSize))
267  .arg(formatDiskSize(burnCapacityTotalSize)));
268  } else {
269  m_diskCapacity->setText(tr("Unknown"));
270  }
271  } else {
272  m_diskCapacity->setText(QString("%1 / %2")
273  .arg(formatDiskSize(bytesTotal - bytesFree))
274  .arg(formatDiskSize(bytesTotal)));
275  }
276 
277  if (bytesTotal > 0) {
278  m_capacityValueBar->setValue(static_cast<int>(100 * (bytesTotal - bytesFree) / bytesTotal));
279  }
280  }
281 
282  QFrame::showEvent(e);
283 }
