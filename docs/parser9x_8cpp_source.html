45 Parser9x::Position::Position( U32 cp, const PLCF<Word97::PCD>* plcfpcd ) :
46  piece( 0 ), offset( cp )
47 {
48  PLCFIterator<Word97::PCD> it( *plcfpcd );
49  for ( ; it.current(); ++it, ++piece ) {
50  if ( it.currentLim() > cp && it.currentStart() <= cp )
51  break;
52  offset -= it.currentRun();
53  }
54 }

138 bool Parser9x::parse()
139 {
140  if ( !isOk() )
141  return false;
142 
143  if ( m_fib.fEncrypted ) {
144  // There is some code out there to break this "encryption", do we want
145  // to implement that?
146  // We could either ask for a password or cheat a bit :-)
147  wvlog << "Error: The document is encrypted." << std::endl;
148  return false;
149  }
150 
151  if ( m_fib.lcbClx == 0 )
152  fakePieceTable();
153  else {
154  // Get the piece table
155  if ( !readPieceTable() )
156  return false;
157  }
158 
159  // start parsing the body
160  if ( !parseBody() )
161  return false;
162  return true;
163 }

203  wvlog << "Parser9x::parseFootnote() #####################" << std::endl;
204 #endif
205  if ( data.limCP - data.startCP == 0 ) // shouldn't happen, but well...
206  return;

209  m_subDocumentHandler->footnoteStart();

214  parseHelper( Position( offset, m_plcfpcd ) );
215 
216  m_subDocumentHandler->footnoteEnd();

218 #ifdef WV2_DEBUG_FOOTNOTES
219  wvlog << "Parser9x::parseFootnote() done ################" << std::endl;

278 bool Parser9x::readPieceTable()
279 {
280  m_table->seek( m_fib.fcClx );
281  // first skip the leading grpprl blocks, we'll re-read them
282  // if we need them later (no caching here)
283  U8 blockType = m_table->readU8();
284  while ( blockType == wvWare::clxtGrpprl ) {
285  U16 size = m_table->readU16();
286 #if WV2_DUMP_PIECE_TABLE > 0
287  wvlog << "Found a clxtGrpprl (size=" << size << ")" << std::endl;
288 #endif
289  m_table->seek( size, SEEK_CUR );
290  blockType = m_table->readU8();
291  }
292  if ( blockType == wvWare::clxtPlcfpcd ) {
293  U32 size = m_table->readU32();
294 #if WV2_DUMP_PIECE_TABLE > 0
295  wvlog << "Found the clxtPlcfpcd (size=" << size << ")" << std::endl;
296 #endif
297  m_plcfpcd = new PLCF<Word97::PCD>( size, m_table, false );
298 
299 #if WV2_DUMP_PIECE_TABLE > 1
300  PLCFIterator<Word97::PCD> it( *m_plcfpcd );
301  for ( int i = 0; it.current(); ++it, ++i ) {
302  wvlog << "Piece Table Entry(" << i << "): " << std::endl;
303  wvlog << " start: " << it.currentStart() << std::endl;
304  wvlog << " lim: " << it.currentLim() << std::endl;
305  wvlog << " complex: " << it.current()->prm.fComplex << std::endl;
306  if ( it.current()->prm.fComplex )
307  wvlog << " igrpprl: " << it.current()->prm.toPRM2().igrpprl << std::endl;
308  else
309  wvlog << " isprm: " << it.current()->prm.isprm << std::endl;
310 
311  U32 fc = it.current()->fc;
312  U32 limit = it.currentRun() << 1;
313  wvlog << " value: " << fc << std::endl;
314  if ( fc & 0x40000000 ) {
315  fc = ( fc & 0xbfffffff ) >> 1;
316  limit >>= 1;
317  wvlog << " value (cleared 2nd MSB, div. by 2): " << fc << std::endl;
318  }
319  m_wordDocument->seek( fc );
320  wvlog << " position: " << m_wordDocument->tell() << ", limit: " << limit << std::endl;
321  for ( unsigned int j = 0; j < limit; ++j ) {
322  U8 foo = m_wordDocument->readU8();
323  if ( foo > 31 )
324  wvlog << static_cast<char>( foo );
325  else if ( foo == PARAGRAPH_MARK )
326  wvlog << std::endl;
327  else if ( foo > 0 )
328  wvlog << "{" << static_cast<int>( foo ) << "}";
329  else
330  wvlog << "_";
331  }
332  wvlog << std::endl << " position: " << m_wordDocument->tell() << ", limit: " << limit << std::endl;
333  }
334 #endif
335  }
336  else {
337  wvlog << "Oooops, couldn't find the piece table." << std::endl;
338  return false;
339  }
340  return true;
341 }

343 void Parser9x::fakePieceTable()
344 {
345  U32 fakePlcfPCD[ 4 ];
346  // The first CP is 0 (endianness doesn't matter :-)
347  fakePlcfPCD[ 0 ] = 0;
348  // The second CP corresponds to the length of the document
349  fakePlcfPCD[ 1 ] = toLittleEndian( m_fib.ccpText + m_fib.ccpFtn + m_fib.ccpHdd + m_fib.ccpMcr +
350  m_fib.ccpAtn + m_fib.ccpEdn + m_fib.ccpTxbx + m_fib.ccpHdrTxbx );
351 
352  // Now fake a matching PCD
353  U8* tmp( reinterpret_cast<U8*>( &fakePlcfPCD[0] ) );
354  tmp += 8;
355  *tmp++ = 0; // first the bitfields (unused)
356  *tmp++ = 0;
357  U32 fcMin = m_fib.fcMin << 1;
358  fcMin |= 0x40000000;
359  *tmp++ = static_cast<U8>( fcMin & 0x000000ff );
360  *tmp++ = static_cast<U8>( ( fcMin & 0x0000ff00 ) >> 8 ); // then store the
361  *tmp++ = static_cast<U8>( ( fcMin & 0x00ff0000 ) >> 16 ); // fc in little
362  *tmp++ = static_cast<U8>( ( fcMin & 0xff000000 ) >> 24 ); // endian style
363  *tmp++ = 0; // then an empty PRM
364  *tmp++ = 0;
365 
366  tmp = reinterpret_cast<U8*>( &fakePlcfPCD[0] );
367  m_plcfpcd = new PLCF<Word97::PCD>( 16, tmp );
368 }

370 bool Parser9x::parseBody()
371 {
372  saveState( m_fib.ccpText, Main );
373  m_subDocumentHandler->bodyStart();
374 
375  SharedPtr<const Word97::SEP> sep( m_properties->sepForCP( 0 ) );
376  if ( !sep )
377  sep = new Word97::SEP(); // don't pass 0 pointers in any case
378  m_textHandler->sectionStart( sep ); // First section, starting at CP 0
379  emitHeaderData( sep );
380  sep = 0; // get rid of the huge SEP
381 
382  // Process all the pieces belonging to the main document text
383  parseHelper( Position( 0, static_cast<U32>( 0 ) ) );
384 
385  // Implicit end of the section
386  m_textHandler->sectionEnd();
387  m_subDocumentHandler->bodyEnd();
388  restoreState();
389  return true;
390 }

392 void Parser9x::parseHelper( Position startPos )
393 {
394  PLCFIterator<Word97::PCD> it( m_plcfpcd->at( startPos.piece ) );
395 
396  while ( m_remainingChars > 0 && it.current() ) {
397  U32 fc = it.current()->fc; // Start FC of this piece
398  bool unicode;
399  realFC( fc, unicode );
400 
401  U32 limit = it.currentRun(); // Number of characters in this piece
402 
403  // Check whether the text starts somewhere within the piece, reset at the end of the loop body
404  if ( startPos.offset != 0 ) {
405  fc += unicode ? startPos.offset * 2 : startPos.offset;
406  limit -= startPos.offset;
407  }
408 
409  limit = limit > m_remainingChars ? m_remainingChars : limit;
410  m_wordDocument->seek( fc );
411 
412  if ( unicode ) {
413  XCHAR* string = new XCHAR[ limit ];
414  // First read the whole piece
415  for ( unsigned int j = 0; j < limit; ++j ) {
416  string[ j ] = m_wordDocument->readU16();
417  if ( ( string[ j ] & 0xff00 ) == 0xf000 ) {
418  // Microsoft uses a Private Unicode Area (PUA) to store the characters of the
419  // Symbol and the Wingdings font. We simply clear these bits to shift the
420  // characters to 0x00XX and hope the correct font is installed. If the font
421  // isn't there, the user will get some ASCII text instead of symbols :}
422  //wvlog << "private unicode area detected -- cropping" << std::endl;
423  string[ j ] &= 0x00ff;
424  }
425  }
426  processPiece<XCHAR>( string, fc, limit, startPos ); // also takes care to delete [] string
427  }
428  else {
429  U8* string = new U8[ limit ];
430  m_wordDocument->read( string, limit );
431  processPiece<U8>( string, fc, limit, startPos ); // also takes care to delete [] string
432  }
433  m_remainingChars -= limit;
434  ++it;
435  ++startPos.piece;
436  startPos.offset = 0; // just in case it was != 0 in the first iteration
437  }
438 }

440 template<typename String>
441 void Parser9x::processPiece( String* string, U32 fc, U32 limit, const Position& position )
442 {
443  // Take a closer look at the piece we just read. "start" and "index" are
444  // counted in character positions (take care!)
445  unsigned int start = 0;
446  unsigned int index = 0;
447  while ( index < limit ) {
448  switch( string[ index ] ) {
449  case SECTION_MARK:
450  {
451  if ( !m_currentParagraph->empty() || start != index ) {
452  // No "index - start + 1" here, as we don't want to copy the section mark!
453  UString ustring( processPieceStringHelper( string, start, index ) );
454  m_currentParagraph->push_back( Chunk( ustring, Position( position.piece, position.offset + start ),
455  fc + start * sizeof( String ), sizeof( String ) == sizeof( XCHAR ) ) );
456  processParagraph( fc + index * sizeof( String ) );
457  }
458  start = ++index;
459 
460  SharedPtr<const Word97::SEP> sep( m_properties->sepForCP( m_fib.ccpText - m_remainingChars + index ) );
461  if ( sep ) {
462  // It's not only a page break, it's a new section
463  m_textHandler->sectionEnd();
464  m_textHandler->sectionStart( sep );
465  emitHeaderData( sep );
466  }
467  else
468  m_textHandler->pageBreak();
469  break;
470  }
471  case CELL_MARK: // same ASCII code as a ROW_MARK
472  m_cellMarkFound = true;
473  // Fall-through intended. A row/cell end is also a paragraph end.
474  case PARAGRAPH_MARK:
475  {
476  // No "index - start + 1" here, as we don't want to copy the paragraph mark!
477  UString ustring( processPieceStringHelper( string, start, index ) );
478  m_currentParagraph->push_back( Chunk( ustring, Position( position.piece, position.offset + start ),
479  fc + start * sizeof( String ), sizeof( String ) == sizeof( XCHAR ) ) );
480  processParagraph( fc + index * sizeof( String ) );
481  m_cellMarkFound = false;
482  start = ++index;
483  break;
484  }
485  // "Special" characters
486  case TAB:
487  string[ index ] = m_inlineHandler->tab();
488  ++index;
489  break;
490  case HARD_LINE_BREAK:
491  string[ index ] = m_inlineHandler->hardLineBreak();
492  ++index;
493  break;
494  case COLUMN_BREAK:
495  string[ index ] = m_inlineHandler->columnBreak();
496  ++index;
497  break;
498  case NON_BREAKING_HYPHEN:
499  string[ index ] = m_inlineHandler->nonBreakingHyphen();
500  ++index;
501  break;
502  case NON_REQUIRED_HYPHEN:
503  string[ index ] = m_inlineHandler->nonRequiredHyphen();
504  ++index;
505  break;
506  case NON_BREAKING_SPACE:
507  string[ index ] = m_inlineHandler->nonBreakingSpace();
508  ++index;
509  break;
510  default:
511  ++index;
512  break;
513  }
514  }
515  if ( start < limit ) {
516  // Finally we have to add the remaining text to the current paragaph (if there is any)
517  UString ustring( processPieceStringHelper( string, start, limit ) );
518  m_currentParagraph->push_back( Chunk( ustring, Position( position.piece, position.offset + start ),
519  fc + start * sizeof( String ), sizeof( String ) == sizeof( XCHAR ) ) );
520  }
521  delete [] string;
522 }

534 void Parser9x::processParagraph( U32 fc )
535 {
536  // Get the PAP structure as it was at the last full-save
537  ParagraphProperties* props( m_properties->fullSavedPap( fc, m_data ) );
538  // ...and apply the latest changes, then the PAP is completely restored
539  m_properties->applyClxGrpprl( m_plcfpcd->at( m_currentParagraph->back().m_position.piece ).current(), m_fib.fcClx, props );
540 
541  // Skim the tables first, as soon as the functor is invoked we have to
542  // parse them and emit the text
543  if ( m_parsingMode == Default && props->pap().fInTable ) {
544  if ( !m_tableRowStart ) {
545  m_tableRowStart = new Position( m_currentParagraph->front().m_position );
546  m_tableRowLength = 0;
547 #ifdef WV2_DEBUG_TABLES
548  wvlog << "Start of a table row: piece=" << m_tableRowStart->piece << " offset="
549  << m_tableRowStart->offset << std::endl;
550 #endif
551  }
552  m_tableRowLength += std::accumulate( m_currentParagraph->begin(), m_currentParagraph->end(),
553  1, &Parser9x::accumulativeLength ); // init == 1 because of the parag. mark!
554  if ( props->pap().fTtp ) {
555  // Restore the table properties of this row
556  Word97::TAP* tap = m_properties->fullSavedTap( fc, m_data );
557  m_properties->applyClxGrpprl( m_plcfpcd->at( m_currentParagraph->back().m_position.piece ).current(),
558  m_fib.fcClx, tap, m_properties->styleByIndex( props->pap().istd ) );
559 
560  SharedPtr<const Word97::TAP> sharedTap( tap );
561  // We decrement the length by 1 that the trailing row mark doesn't emit
562  // one empty paragraph during parsing.
563  m_textHandler->tableRowFound( make_functor( *this, &Parser9x::parseTableRow,
564  TableRowData( m_tableRowStart->piece, m_tableRowStart->offset,
565  m_tableRowLength - 1, static_cast<int>( m_subDocument ),
566  sharedTap ) ),
567  sharedTap );
568  delete m_tableRowStart;
569  m_tableRowStart = 0;
570  }
571  delete props;
572  }
573  else {
574  // Now that we have the complete PAP, let's see if this paragraph belongs to a list
575  props->createListInfo( *m_lists );
576 
577  SharedPtr<const ParagraphProperties> sharedProps( props ); // keep it that way, else the ParagraphProperties get deleted!
578  m_textHandler->paragraphStart( sharedProps );
579 
580  // Get the appropriate style for this paragraph
581  const Style* style = m_properties->styleByIndex( props->pap().istd );
582  if ( !style ) {
583  wvlog << "Huh, really obscure error, couldn't find the Style for the current PAP -- skipping" << std::endl;
584  return;
585  }
586 
587  // Now walk the paragraph, chunk for chunk
588  std::list<Chunk>::const_iterator it = m_currentParagraph->begin();
589  std::list<Chunk>::const_iterator end = m_currentParagraph->end();
590  for ( ; it != end; ++it ) {
591  U32 index = 0;
592  const U32 limit = ( *it ).m_text.length();
593  const PLCFIterator<Word97::PCD> pcdIt( m_plcfpcd->at( ( *it ).m_position.piece ) );
594 
595  while ( index < limit ) {
596  Word97::CHP* chp = new Word97::CHP( style->chp() );
597  U32 length = m_properties->fullSavedChp( ( *it ).m_startFC + index * ( ( *it ).m_isUnicode ? 2 : 1 ), chp, style );
598  if ( ( *it ).m_isUnicode )
599  length >>= 1;
600  length = length > limit - index ? limit - index : length;
601  if (length == 0) //we got stuck here! File is corrupted probably
602  {
603  delete chp;
604  chp = NULL;
605  break;
606  }
607  m_properties->applyClxGrpprl( pcdIt.current(), m_fib.fcClx, chp, style );
608  SharedPtr<const Word97::CHP> sharedChp( chp ); // keep it that way, else the CHP gets deleted!
609  processChunk( *it, chp, length, index, pcdIt.currentStart() );
610  index += length;
611  }
612  }
613  m_textHandler->paragraphEnd();
614 
615  if ( m_cellMarkFound ) {
616  m_tableHandler->tableCellEnd();
617  if ( --m_remainingCells )
618  m_tableHandler->tableCellStart();
619  }
620  }
621  m_currentParagraph->clear();

624 void Parser9x::processChunk( const Chunk& chunk, SharedPtr<const Word97::CHP> chp,
625  U32 length, U32 index, U32 currentStart )
626 {
627  // Some characters have a special meaning (e.g. a footnote is anchored at some
628  // position inside the text) and they *don't* have the fSpec flag set. This means
629  // that we have to watch out for such characters even in plain text. Slooow :}
630  //
631  // For now we only have to handle footnote and endnote references that way. Due to that
632  // the code below is a bit simpler right now, but I fear we have to extend that later on.
633  // (We will have to keep track of the type of disruption, footnote() takes care of all now)
634  //
635  // A precondition for the footnote/endnote implementation below is, that footnote and
636  // endnote references only occur in the main body text. The reason is that we only check
637  // for the next footnote inside the PLCF and don't take subdocuments into account. If
638  // it turns out that this precondition is not satisfied we would have to change the
639  // O(1) nextFootnote() call to something like an O(n) containsFootnote( start, lim )
640  // Up to now Word 97, 2000, and 2002 seem to be bug compatible and fullfill that precondition.
641  //
642  while ( length > 0 ) {
643  U32 disruption = 0xffffffff; // "infinity"
644  if ( m_footnotes ) {
645  U32 nextFtn = m_footnotes->nextFootnote();
646  U32 nextEnd = m_footnotes->nextEndnote();
647  disruption = nextFtn < nextEnd ? nextFtn : nextEnd;
648 #ifdef WV2_DEBUG_FOOTNOTES
649  wvlog << "nextFtn=" << nextFtn << " nextEnd=" << nextEnd << " disruption="
650  << disruption << " length=" << length << std::endl;
651 #endif
652  }
653  U32 startCP = currentStart + chunk.m_position.offset + index;
654 
655  if ( disruption >= startCP && disruption < startCP + length ) {
656 #ifdef WV2_DEBUG_FOOTNOTES
657  wvlog << "startCP=" << startCP << " len=" << length << " disruption=" << disruption << std::endl;
658 #endif
659  U32 disLen = disruption - startCP;
660  if ( disLen != 0 )
661  processRun( chunk, chp, disLen, index, currentStart );
662  length -= disLen;
663  index += disLen;
664  processFootnote( chunk.m_text[ index ], disruption, chp );
665  --length;
666  ++index;
667  }
668  else {
669  // common case, no disruption at all (or the end of a disrupted chunk)
670  processRun( chunk, chp, length, index, currentStart );
671  break; // should be faster than messing with length...
672  }
673  }
674 }

855 void Parser9x::parseHeader( const HeaderData& data, unsigned char mask )
856 {
857 #ifdef WV2_DEBUG_HEADERS
858  wvlog << "parsing one header for section " << data.sectionNumber << ": mask=0x"
859  << std::hex << static_cast<int>( mask ) << std::dec << std::endl;
860 #endif
861 
862  // First we have to determine the CP start/lim for the header text. From what I
863  // found out Word 8 does it that way:
864  // - At the begin of the plcfhdd there are always 6 "0 fields" (stoppers)
865  // - The number of headers modulo 6 is always 0
866  // Word 6 does it completely different, of course :-}
867  std::pair<U32, U32> range( m_headers->findHeader( data.sectionNumber, mask ) );
868 
869  int length = range.second - range.first;
870 #ifdef WV2_DEBUG_HEADERS
871  wvlog << "found a range: start=" << range.first << " lim=" << range.second << std::endl
872  << "length: " << length << std::endl;
873 #endif
874  if ( length < 1 ) {
875 #ifdef WV2_DEBUG_HEADERS
876  wvlog << "Warning: Didn't find a valid CPs for this header -- faking it" << std::endl;
877 #endif
878  m_subDocumentHandler->headerStart( static_cast<HeaderData::Type>( mask ) );
879  SharedPtr<const ParagraphProperties> sharedProps( new ParagraphProperties );
880  m_textHandler->paragraphStart( sharedProps );
881  m_textHandler->paragraphEnd();
882  m_subDocumentHandler->headerEnd();
883  return;
884  }
885  else if ( length > 1 )
886  --length; // get rid of the trailing "end of header/footer" character
887 
888  saveState( length, Header );
889 
890  m_subDocumentHandler->headerStart( static_cast<HeaderData::Type>( mask ) );
891  parseHelper( Position( m_fib.ccpText + m_fib.ccpFtn + range.first, m_plcfpcd ) );
892  m_subDocumentHandler->headerEnd();
893 
894  restoreState();
895 }
