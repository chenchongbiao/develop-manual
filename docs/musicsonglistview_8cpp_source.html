153 void MusicSongListView::showContextMenu(const QPoint &pos)
154 {
155  auto index = indexAt(pos);
156  if (!index.isValid())
157  return;
158 
159  auto item = m_model->itemFromIndex(index);
160  if (!item) {
161  return;
162  }
163 
164  if (index.row() == 0 && index.data(Qt::UserRole + CDA_USER_ROLE_OFFSET).toString() == CDA_USER_ROLE) {
165  return;//cda item不做任何处理
166  }
167 
168  QPoint globalPos = this->mapToGlobal(pos);
169 
170 
171  DMenu menu;
172  connect(&menu, &DMenu::triggered, this, &MusicSongListView::slotMenuTriggered);
173 
174  QAction *playact = nullptr;
175  QAction *pauseact = nullptr;
176 
177  QString hash = index.data(Qt::UserRole).value<QString>();
178  emit CommonService::getInstance()->signalSwitchToView(ListPageSwitchType::CustomType, hash);
179 
180  if (hash == Player::getInstance()->getCurrentPlayListHash() && Player::getInstance()->status() == Player::Playing) {
181  pauseact = menu.addAction(tr("Pause"));
182  pauseact->setDisabled(0 == DataBaseService::getInstance()->customizeMusicInfos(hash).size());
183  } else {
184  playact = menu.addAction(tr("Play"));
185  playact->setDisabled(0 == DataBaseService::getInstance()->customizeMusicInfos(hash).size());
186  }
187 
188  if (hash != "album" || hash != "artist" || hash != "all" || hash != "fav") {
189  menu.addAction(tr("Rename"));
190  menu.addAction(tr("Delete"));
191  }
192 
193  menu.exec(globalPos);
194 }

299 void MusicSongListView::rmvSongList()
300 {
301  QModelIndex index = this->currentIndex();
302  // 当前选中行
303  int rowCurrent = index.row();
304  // 总行数，删除后焦点定位使用
305  int rowCount = m_model->rowCount();
306  if (rowCurrent == 0 && index.data(Qt::UserRole + CDA_USER_ROLE_OFFSET).toString() == CDA_USER_ROLE) {
307  return;//cda item不做任何处理
308  }
309  if (rowCurrent >= 0) {
310  QString message = QString(tr("Are you sure you want to delete this playlist?"));
311 
312  DDialog warnDlg(this);
313  warnDlg.setIcon(QIcon::fromTheme("deepin-music"));
314  warnDlg.setTextFormat(Qt::AutoText);
315  warnDlg.setTitle(message);
316  warnDlg.addSpacing(20);
317  warnDlg.addButton(tr("Cancel"), false, Dtk::Widget::DDialog::ButtonNormal);
318  warnDlg.addButton(tr("Delete"), true, Dtk::Widget::DDialog::ButtonWarning);
319  warnDlg.setObjectName(AC_MessageBox);
320  if (1 == warnDlg.exec()) {
321  QString hash = index.data(Qt::UserRole).value<QString>();
322  bool isMediaMetaInSonglist = false;
323  isMediaMetaInSonglist = DataBaseService::getInstance()->isMediaMetaInSonglist(hash, Player::getInstance()->getActiveMeta().hash);
324  // 数据库删除歌单
325  if (DataBaseService::getInstance()->deletePlaylist(hash)) {
326  // 判断当前是否播放歌曲，如果正在播放，则判断当前歌曲是否在此歌单
327  if (!Player::getInstance()->getActiveMeta().hash.isEmpty() && isMediaMetaInSonglist) {
328  Player::getInstance()->clearPlayList();
329  // 设置所有音乐播放状态
330  Player::getInstance()->setCurrentPlayListHash("", false);
331  }
332  if (rowCount == 1) {
333  // 没有播放且只有一个歌单,删除后焦点定位到所有音乐
334  this->setCurrentIndex(QModelIndex());
335  emit CommonService::getInstance()->signalSwitchToView(AllSongListType, "all");
336  this->clearSelection();
337  } else if (rowCurrent < (rowCount - 1)) {
338  // 删除后焦点定位到下一个歌单
339  QModelIndex nextIndex = m_model->index((rowCurrent + 1), 0);
340  QString nextHash = nextIndex.data(Qt::UserRole).value<QString>();
341  this->setCurrentIndex(nextIndex);
342  emit CommonService::getInstance()->signalSwitchToView(CustomType, nextHash);
343  } else {
344  // 没有后一个歌单,删除后焦点定位到上一个歌单
345  QModelIndex nextIndex = m_model->index((rowCurrent - 1), 0);
346  QString nextHash = nextIndex.data(Qt::UserRole).value<QString>();
347  this->setCurrentIndex(nextIndex);
348  emit CommonService::getInstance()->signalSwitchToView(CustomType, nextHash);
349  }
350  // 删除成功，移除歌单
351  m_model->removeRow(rowCurrent);
352  }
353  }
354  //删除消息，让scroll自动刷新
355  m_heightChangeToMax = false;
356  adjustHeight();
357  }
358 }

360 void MusicSongListView::slotUpdatePlayingIcon()
361 {
362  for (int i = 0; i < m_model->rowCount(); i++) {
363  QModelIndex index = m_model->index(i, 0);
364  DStandardItem *item = dynamic_cast<DStandardItem *>(m_model->item(i, 0));
365  if (item == nullptr) {
366  continue;
367  }
368  QString hash = index.data(Qt::UserRole).value<QString>();
369  //获取cda的hash值，用于处理cd歌单的波浪条
370  QString cdahash = index.data(Qt::UserRole + CDA_USER_ROLE_OFFSET).value<QString>();
371  if (!cdahash.isEmpty())
372  hash = cdahash;
373  if (hash == Player::getInstance()->getCurrentPlayListHash()) {
374  QPixmap playingPixmap = QPixmap(ItemIconSide, ItemIconSide);
375  playingPixmap.fill(Qt::transparent);
376  QPainter painter(&playingPixmap);
377  DTK_NAMESPACE::Gui::DPalette pa;// = this->palette();
378  if (selectedIndexes().size() > 0 && (selectedIndexes().at(0) == index)) {
379  painter.setPen(QColor(Qt::white));
380  } else {
381  painter.setPen(pa.color(QPalette::Active, DTK_NAMESPACE::Gui::DPalette::Highlight));
382  }
383  Player::getInstance()->playingIcon().paint(&painter, QRect(0, 0, ItemIconSide, ItemIconSide), Qt::AlignCenter, QIcon::Active, QIcon::On);
384 
385  QIcon playingIcon(playingPixmap);
386  DViewItemActionList actionList = item->actionList(Qt::RightEdge);
387  if (!actionList.isEmpty()) {
388  actionList.first()->setIcon(playingIcon);
389  } else {
390  actionList.clear();
391  auto viewItemAction = new DViewItemAction(Qt::AlignCenter, QSize(ItemIconSide, ItemIconSide));
392  viewItemAction->setIcon(playingIcon);
393  actionList.append(viewItemAction);
394  dynamic_cast<DStandardItem *>(item)->setActionList(Qt::RightEdge, actionList);
395  }
396  } else {
397  QIcon playingIcon;
398  DViewItemActionList actionList = item->actionList(Qt::RightEdge);
399  if (!actionList.isEmpty()) {
400  actionList.first()->setIcon(playingIcon);
401  } else {
402  actionList.clear();
403  auto viewItemAction = new DViewItemAction(Qt::AlignCenter, QSize(ItemIconSide, ItemIconSide));
404  viewItemAction->setIcon(playingIcon);
405  actionList.append(viewItemAction);
406  dynamic_cast<DStandardItem *>(item)->setActionList(Qt::RightEdge, actionList);
407  }
408  }
409  }
410  update();
411 }

494 void MusicSongListView::slotPopMessageWindow(int stat)
495 {
496  //初始cd的track索引
497  Player::getInstance()->initCddTrack();
498 
499  MediaMeta tmpMeta = Player::getInstance()->getActiveMeta();
500  //CD退出时若不关闭弹窗，再次接入cd后，此弹窗自动隐藏，列表数据刷新
501  QList<QWidget *> oldMsgList = this->findChildren<QWidget *>("_d_message_cda_deepin_music");
502  if (oldMsgList.size() > 0) {
503  oldMsgList.first()->setProperty("cda", 1);
504  oldMsgList.first()->close();
505  oldMsgList.first()->deleteLater();
506  //执行添加/移出CD歌单,cd弹出
507  changeCdaSongList(stat);
508  return;
509  }
510 
511  //只处理弹出cd的弹窗
512  if (stat == 1) {
513  changeCdaSongList(stat);
514  if (DataBaseService::getInstance()->allMusicInfosCount() == 0) {
515  //进入主页面
516  emit CommonService::getInstance()->signalCdaImportFinished();
517  }
518  return;
519  }
520 
521  //停止音乐
522  Player::getInstance()->stop();
523 
524  int retexec = -1;
525  int iprop = 0;
526  QString popStrMsg = "";
527 
528  popStrMsg = tr("The CD has been removed");
529  if (tmpMeta.mmType == MIMETYPE_CDA) {
530  popStrMsg = tr("Play failed, as the CD has been removed");
531  }
532 
533  Dtk::Widget::DDialog tipsDlg(this);
534  tipsDlg.setObjectName("_d_message_cda_deepin_music");
535  tipsDlg.setIcon(QIcon::fromTheme("deepin-music"));
536  tipsDlg.setTextFormat(Qt::RichText);
537  tipsDlg.addButton(tr("OK"), true, Dtk::Widget::DDialog::ButtonNormal);
538  tipsDlg.setMessage(popStrMsg);
539  retexec = tipsDlg.exec();
540  iprop = tipsDlg.property("cda").toInt();
544  qDebug() << __FUNCTION__ << "prop:" << iprop << "exec" << retexec;
545  if (retexec != 0 && iprop == 1) {
546  qDebug() << "________tipsDlg_______property" << retexec;
547  return;
548  }
549 
553  qDebug() << __FUNCTION__ << stat;
554  int allsize = DataBaseService::getInstance()->allMusicInfosCount();
555  if (allsize == 0) {
556  //清空列表
557  Player::getInstance()->clearPlayList();
558  //回到初始页面
559  emit DataBaseService::getInstance()->signalAllMusicCleared();
560  } else {
561  if (tmpMeta.mmType == MIMETYPE_CDA) {
562  Player::getInstance()->setActiveMeta(MediaMeta());
563  //设置当前页面，刷新播放队列和显示列表
564  Player::getInstance()->setCurrentPlayListHash("all", true);
565  //清空列表
566  Player::getInstance()->clearPlayList();
567  //播放歌单第一首歌曲
568  Player::getInstance()->forcePlayMeta();
569  }
570 
571  if (m_model->rowCount() > 0 && tmpMeta.mmType != MIMETYPE_CDA) {
572  int row = this->currentIndex().row();
573  QString strrole = m_model->index(row, 0).data(Qt::UserRole + CDA_USER_ROLE_OFFSET).toString();
574  if (strrole == CDA_USER_ROLE) {
575  //设置当前页面，刷新播放队列和显示列表
576  Player::getInstance()->setCurrentPlayListHash("all", false);
577  }
578  }
579  // 切换到所有音乐界面
580  emit CommonService::getInstance()->signalSwitchToView(AllSongListType, "all");
581  }
582 
583  //执行添加/移出CD歌单,cd弹出
584  changeCdaSongList(stat);
585  //清空选中
586  this->clearSelection();
587 }

624 void MusicSongListView::dropEvent(QDropEvent *event)

626  QModelIndex indexDrop = indexAt(event->pos());
627  if (!indexDrop.isValid())
628  return;
629  QString hash = indexDrop.data(Qt::UserRole).value<QString>();
630 
631  // auto t_playlistPtr = playlistPtr(index);
632  if (/*t_playlistPtr == nullptr || */(!event->mimeData()->hasFormat("text/uri-list") && !event->mimeData()->hasFormat("application/x-qabstractitemmodeldatalist"))) {
633  return;
634  }
635 
636  if (indexDrop.data(Qt::UserRole + CDA_USER_ROLE_OFFSET).toString() == CDA_USER_ROLE) {
637  return;//cda item不做任何处理
638  }
639 
640  if (event->mimeData()->hasFormat("text/uri-list")) {
641  auto urls = event->mimeData()->urls();
642  QStringList localpaths;
643  for (auto &url : urls) {
644  localpaths << url.toLocalFile();
645  }
646 
647  if (!localpaths.isEmpty()) {
648  DataBaseService::getInstance()->importMedias(hash, localpaths);
649  }
650  } else {
651  PlayListView *source = qobject_cast<PlayListView *>(event->source());
652  if (source != nullptr) {
653  QList<MediaMeta> metas;
654  for (auto index : source->selectionModel()->selectedIndexes()) {
655  MediaMeta imt = index.data(Qt::UserRole).value<MediaMeta>();
656  if (imt.mmType != MIMETYPE_CDA)
657  metas.append(imt);
658  }
659 
660  if (!metas.isEmpty()) {
661  int insertCount = DataBaseService::getInstance()->addMetaToPlaylist(hash, metas);
662  CommonService::getInstance()->signalShowPopupMessage(m_model->itemFromIndex(indexDrop)->text(), metas.size(), insertCount);
663  }
664  }
665  }
