123 void MainWindow::closeTabViews()
124 {
125  qDebug() << "close views begin = " << m_closeViews << m_closeUUids;
126 
127  // 此函数的作用是关闭 m_closeTabList 中的标签
128  // 需要每次在保存或者不保存后进行调用判断
129 
130  int count = m_closeViews.size();
131 
132  if (m_closeUUids.size() != count)
133  return;
134 
135  for (int i = 0; i < count; i++) {
136  QString current_name = m_closeViews[i];
137  QString current_uuid = m_closeUUids[i];
138 
139  m_centralWidget->setCurrentViewByUUID(current_uuid);
140  CGraphicsView *pView = CManageViewSigleton::GetInstance()->getCurView();
141  CManageViewSigleton::GetInstance()->setCurView(pView);
142  CGraphicsView *closeView = CManageViewSigleton::GetInstance()->getViewByUUID(current_uuid);
143  if (closeView == nullptr) {
144  qDebug() << "close error view:" << current_name;
145  continue;
146  } else {
147  // [0] 关闭标签前需要判断是否保存裁剪状态
148  m_centralWidget->slotJudgeCutStatusAndPopSaveDialog();
149  //bool editFlag = closeView->getDrawParam()->isModified();
150  bool editFlag = CManageViewSigleton::GetInstance()->getCurView()->drawScene()->getBzItems().count() > 0 ? true : false;
151  if (editFlag) {
152 
153  //qt qtabbar的bug,弹窗响应了leave但是未响应Hoverleave导致某个一个标签的按钮未清理掉高亮,
154  //所以这里手动清理一下以解决 BUG 43546 https://pms.uniontech.com/zentao/bug-view-43546.html
155  if (m_centralWidget->multipTabBarWidget() != nullptr) {
156  m_centralWidget->multipTabBarWidget()->clearHoverState();
157  }
158 
159  int ret = showSaveQuestionDialog();
160  if (ret <= 0) {
161  //结束关闭，同时结束其他标签页的关闭(因为取消了)
162  qDebug() << "cancel close view at view = " << closeView->getDrawParam()->viewName() << "uuid = " << closeView->getDrawParam()->uuid();
163  break;
164  }
165  } else {
166  m_centralWidget->closeCurrentScenseView();
167  }
168  }
169  }
170 
171  qDebug() << "close views end ------";
172 
173  //如果未保存直接抛弃那么这里的判断可以直接推出程序(如果有view标签页需要保存那么就要等待保存完成后推出程序)
174  CManageViewSigleton::GetInstance()->quitIfEmpty();
175 }

463 bool MainWindow::event(QEvent *event)
464 {
465  if (event->type() == QEvent::ChildAdded) {
466  QChildEvent *addEvent = dynamic_cast<QChildEvent *>(event);
467  if (addEvent != nullptr) {
468  QWidget *pWidget = dynamic_cast<QWidget *>(addEvent->child());
469 
470  if (pWidget != nullptr && !pWidget->isModal()) {
471  QLabel *pLabe = pWidget->findChild<QLabel *>("WebsiteLabel");
472  if (pLabe != nullptr) {
473  m_pWebsiteLabe = pLabe;
474  }
475  pWidget->installEventFilter(this);
476  m_allChilds.insert(pWidget);
477  }
478  }
479 
480  } else if (event->type() == QEvent::ChildRemoved) {
481  QChildEvent *addEvent = dynamic_cast<QChildEvent *>(event);
482  if (addEvent != nullptr) {
483  QWidget *pWidget = qobject_cast<QWidget *>(addEvent->child());
484  if (pWidget != nullptr) {
485  m_allChilds.remove(pWidget);
486  }
487  }
488  }
489  return DMainWindow::event(event);
490 }

492 bool MainWindow::eventFilter(QObject *o, QEvent *e)
493 {
494  if (o->isWidgetType()) {
495  QWidget *pWidget = qobject_cast<QWidget *>(o);
496 
497  if (titlebar() == o) {
498  //实现m_topToolbar的正确位置放置
499  if (e->type() == QEvent::Resize) {
500  if (m_topToolbar != nullptr) {
501  m_topToolbar->setGeometry(60, 0, titlebar()->width() - 60 - 200, titlebar()->height());
502  //m_topToolbar->setFixedWidth(100);
503  }
504  }
505  } else {
506  if (m_pWebsiteLabe == o) {
507  //找到这个东西了
508  static QCursor s_temp;
509  if (e->type() == QEvent::Enter) {
510  s_temp = (qApp->overrideCursor() == nullptr ? QCursor(Qt::ArrowCursor) : *qApp->overrideCursor());
511  drawApp->setApplicationCursor(Qt::PointingHandCursor);
512  } else if (e->type() == QEvent::Leave) {
513  drawApp->setApplicationCursor(s_temp);
514  }
515  } else if (m_allChilds.contains(pWidget)) {
516  if (e->type() == QEvent::ChildAdded) {
517  QChildEvent *addEvent = dynamic_cast<QChildEvent *>(e);
518 
519  QWidget *pWidgets = qobject_cast<QWidget *>(addEvent->child());
520  if (pWidgets != nullptr) {
521  QLabel *pLabe = pWidgets->findChild<QLabel *>("WebsiteLabel");
522  if (pLabe != nullptr) {
523  m_pWebsiteLabe = pLabe;
524  m_pWebsiteLabe->installEventFilter(this);
525  }
526  }
527  } else if (e->type() == QEvent::ChildRemoved) {
528  QChildEvent *addEvent = dynamic_cast<QChildEvent *>(e);
529  if (addEvent != nullptr && addEvent->child() == m_pWebsiteLabe) {
530  m_pWebsiteLabe->removeEventFilter(this);
531  m_pWebsiteLabe = nullptr;
532  }
533  }
534  }
535  }
536  }
537  return DMainWindow::eventFilter(o, e);
538 }
