60 void DatetimePlugin::init(PluginProxyInterface *proxyInter)
61 {
62  m_proxyInter = proxyInter;
63 
64  // transfer config
65  QSettings settings("deepin", "dde-dock-datetime");
66  if (QFile::exists(settings.fileName())) {
67  Dock::DisplayMode mode = displayMode();
68  const QString key = QString("pos_%1_%2").arg(pluginName()).arg(mode);
69  proxyInter->saveValue(this, key, settings.value(key, mode == Dock::DisplayMode::Fashion ? 6 : -1));
70  QFile::remove(settings.fileName());
71  }
72 
73  if (pluginIsDisable()) {
74  return;
75  }
76 
77  loadPlugin();
78 }

80 void DatetimePlugin::loadPlugin()
81 {
82  if (m_pluginLoaded)
83  return;
84 
85  m_pluginLoaded = true;
86  m_dateTipsLabel = new TipsWidget;
87  m_refershTimer = new QTimer(this);
88  m_dateTipsLabel->setObjectName("datetime");
89 
90  m_refershTimer->setInterval(1000);
91  m_refershTimer->start();
92 
93  m_centralWidget = new DatetimeWidget;
94 
95  connect(m_centralWidget, &DatetimeWidget::requestUpdateGeometry, [this] { m_proxyInter->itemUpdate(this, pluginName()); });
96  connect(m_refershTimer, &QTimer::timeout, this, &DatetimePlugin::updateCurrentTimeString);
97 
98  m_proxyInter->itemAdded(this, pluginName());
99 
100  pluginSettingsChanged();
101 }

217 void DatetimePlugin::updateCurrentTimeString()
218 {
219  const QDateTime currentDateTime = QDateTime::currentDateTime();
220 
221  if (m_centralWidget->is24HourFormat())
222  m_dateTipsLabel->setText(currentDateTime.date().toString(Qt::SystemLocaleLongDate) + currentDateTime.toString(" HH:mm:ss"));
223  else
224  m_dateTipsLabel->setText(currentDateTime.date().toString(Qt::SystemLocaleLongDate) + currentDateTime.toString(" hh:mm:ss A"));
225 
226  const QString currentString = currentDateTime.toString("yyyy/MM/dd hh:mm");
227 
228  if (currentString == m_currentTimeString)
229  return;
230 
231  m_currentTimeString = currentString;
232  m_centralWidget->update();
233 }

254 QDBusInterface* DatetimePlugin::timedateInterface()
255 {
256  if (!m_interface) {
257  if (QDBusConnection::sessionBus().interface()->isServiceRegistered("com.deepin.daemon.Timedate")) {
258  m_interface = new QDBusInterface("com.deepin.daemon.Timedate", "/com/deepin/daemon/Timedate", "com.deepin.daemon.Timedate");
259  } else {
260  const QString path = QString("/com/deepin/daemon/Accounts/User%1").arg(QString::number(getuid()));
261  QDBusInterface * systemInterface = new QDBusInterface("com.deepin.daemon.Accounts", path, "com.deepin.daemon.Accounts.User",
262  QDBusConnection::systemBus(), this);
263  return systemInterface;
264  }
265  }
266 
267  return m_interface;
268 }
