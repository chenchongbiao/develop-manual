111 void PalmDetectSetting::setModel(MouseModel * const model)
112 {
113  connect(model, &MouseModel::palmDetectChanged, this, &PalmDetectSetting::setDetectState);
114  connect(model, &MouseModel::palmMinWidthChanged, this, &PalmDetectSetting::setContactValue);
115  connect(model, &MouseModel::palmMinzChanged, this, &PalmDetectSetting::setPressureValue);
