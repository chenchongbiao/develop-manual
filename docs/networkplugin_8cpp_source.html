164 void NetworkPlugin::onDeviceListChanged(const QList<NetworkDevice *> devices)
165 {
166  QMap<QString, WirelessItem*> wirelessItems;
167  QMap<QString, WiredItem *> wiredItems;
168 
169  int wiredDeviceCnt = 0;
170  int wirelessDeviceCnt = 0;
171  for (auto device : devices) {
172  if (device && device->type() == NetworkDevice::Wired)
173  wiredDeviceCnt++;
174  else
175  wirelessDeviceCnt++;
176  }
177 
178  // 编号 (与控制中心有线设备保持一致命名)
179  int wiredNum = 0;
180  int wirelessNum = 0;
181  QString text;
182 
183  for (auto device : devices) {
184  const QString &path = device->path();
185  // new device
186  DeviceItem *item = nullptr;
187  switch (device->type()) {
188  case NetworkDevice::Wired:
189  wiredNum++;
190  if (wiredDeviceCnt == 1)
191  text = tr("Wired Network");
192  else
193  text = tr("Wired Network %1").arg(wiredNum);
194  item = new WiredItem(static_cast<WiredDevice *>(device), text);
195  wiredItems.insert(path, static_cast<WiredItem *>(item));
196 
197  connect(static_cast<WiredItem *>(item), &WiredItem::wiredStateChanged,
198  m_networkItem, &NetworkItem::updateSelf);
199  connect(static_cast<WiredItem *>(item), &WiredItem::enableChanged,
200  m_networkItem, &NetworkItem::updateSelf);
201  connect(static_cast<WiredItem *>(item), &WiredItem::activeConnectionChanged,
202  m_networkItem, &NetworkItem::updateSelf);
203  break;
204  case NetworkDevice::Wireless:
205  item = new WirelessItem(static_cast<WirelessDevice *>(device));
206  static_cast<WirelessItem *>(item)->setDeviceInfo(wirelessDeviceCnt == 1 ? -1 : ++wirelessNum);
207  wirelessItems.insert(path, static_cast<WirelessItem *>(item));
208 
209  connect(static_cast<WirelessItem *>(item), &WirelessItem::queryActiveConnInfo,
210  m_networkWorker, &NetworkWorker::queryActiveConnInfo);
211  connect(static_cast<WirelessItem *>(item), &WirelessItem::requestActiveAP,
212  m_networkWorker, &NetworkWorker::activateAccessPoint);
213  connect(static_cast<WirelessItem *>(item), &WirelessItem::requestDeactiveAP,
214  m_networkWorker, &NetworkWorker::disconnectDevice);
215  connect(static_cast<WirelessItem *>(item), &WirelessItem::feedSecret,
216  m_networkWorker, &NetworkWorker::feedSecret);
217  connect(static_cast<WirelessItem *>(item), &WirelessItem::cancelSecret,
218  m_networkWorker, &NetworkWorker::cancelSecret);
219  connect(static_cast<WirelessItem *>(item), &WirelessItem::requestWirelessScan,
220  m_networkWorker, &NetworkWorker::requestWirelessScan);
221  connect(static_cast<WirelessItem *>(item), &WirelessItem::createApConfig,
222  m_networkWorker, &NetworkWorker::createApConfig);
223  connect(static_cast<WirelessItem *>(item), &WirelessItem::queryConnectionSession,
224  m_networkWorker, &NetworkWorker::queryConnectionSession);
225 
226  connect(static_cast<WirelessItem *>(item), &WirelessItem::deviceStateChanged,
227  m_networkItem, &NetworkItem::updateSelf);
228  connect(static_cast<WirelessItem *>(item), &WirelessItem::requestWirelessScan,
229  m_networkItem, &NetworkItem::wirelessScan);
230 
231  m_networkWorker->queryAccessPoints(path);
232  m_networkWorker->requestWirelessScan();
233  break;
234  default:
235  Q_UNREACHABLE();
236  }
237 
238  connect(item, &DeviceItem::requestSetDeviceEnable, m_networkWorker, &NetworkWorker::setDeviceEnable);
239  connect(m_networkModel, &NetworkModel::connectivityChanged, item, &DeviceItem::refreshConnectivity);
240  connect(m_networkModel, &NetworkModel::connectivityChanged, m_networkItem, &NetworkItem::updateSelf);
241  }
242 
243  m_hasDevice = wiredItems.size() || wirelessItems.size();
244  m_networkItem->updateDeviceItems(wiredItems, wirelessItems);
245 }

247 void NetworkPlugin::loadPlugin()
248 {
249  m_networkModel = new NetworkModel;
250  m_networkWorker = new NetworkWorker(m_networkModel);
251 
252  connect(m_networkModel, &NetworkModel::deviceListChanged, this, &NetworkPlugin::onDeviceListChanged);
253 
254  m_networkModel->moveToThread(qApp->thread());
255  m_networkWorker->moveToThread(qApp->thread());
256 
257  onDeviceListChanged(m_networkModel->devices());
258 
259  m_proxyInter->itemAdded(this, NETWORK_KEY);
260 }
