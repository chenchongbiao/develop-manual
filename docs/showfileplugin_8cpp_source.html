193 int ShowFilePlugin::analyzeFile(const QString &filename, int &a, int &b)
194 {
195  //测试文件格式为，首行 test[01]开始的，数字0为0-4, 并将这2个位数字分别放在 a, b中
196  QFileInfo info( filename );
197  if(!info.isFile()) // 为目录
198  {
199  return 2;
200  }
201 
202  QFile file(filename);
203  if (!file.open(QIODevice::ReadOnly | QIODevice::Text))
204  {
205  return 2;
206  }
207 
208  QTextStream in(&file);
209  while (!in.atEnd())
210  {
211  QString line = in.readLine(); //只处理第一行
212  bool ok;
213  QString str = line.mid( 5, 2 );
214  ushort dec = str.toUShort(&ok);
215  if( ok )
216  {
217  a = dec/10;
218  b = dec%10;
219  if( a > 4 || a < 0 )
220  {
221  return 1;
222  }
223  if( b > 4 || b < 0 )
224  {
225  return 1;
226  }
227  str = QString("test[%1%2]").arg( a ).arg( b );
228  int posi = line.indexOf( str );
229  if( posi == 0 )
230  {
231  return 0;
232  }
233  else
234  {
235  return 1;
236  }
237  }
238 
239  break;
240  }
241  return 1;
242 }
