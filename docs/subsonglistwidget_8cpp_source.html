239 void SubSonglistWidget::flushDialog(QMap<QString, MediaMeta> musicinfos, ListPageSwitchType listPageType)
240 {
241  if (musicinfos.size() > 0) {
242  auto titleFont = m_titleLabel->font();
243  auto infoFont = m_infoLabel->font();
244  // 根据进入二级页面场景，设置hash值，二级页面删除时判断使用
245  switch (listPageType) {
246  case AlbumType: {
247  m_hash = "album";
248  break;
249  }
250  case SearchAlbumResultType: {
251  m_hash = "albumResult";
252  break;
253  }
254  case SingerType: {
255  m_hash = "artist";
256  break;
257  }
258  case SearchSingerResultType: {
259  m_hash = "artistResult";
260  break;
261  }
262  default: {
263  m_hash = "album";
264  break;
265  }
266  }
267  if (listPageType == AlbumType || listPageType == SearchAlbumResultType) {
268 // 解决字体不会根据系统字体大小改变问题
269 // titleFont.setPixelSize(24);
270 // infoFont.setPixelSize(18);
271  m_titleLabel->setText(musicinfos.first().album);
272  m_infoLabel->setText(musicinfos.first().singer);
273  } else if (listPageType == SingerType || listPageType == SearchSingerResultType) {
274 // titleFont.setPixelSize(36);
275  m_titleLabel->setText(musicinfos.first().singer);
276  m_infoLabel->hide();
277  }
278 
279  DFontSizeManager::instance()->bind(m_titleLabel, DFontSizeManager::T3, QFont::DemiBold);
280  m_infoLabel->setFont(infoFont);
281  DFontSizeManager::instance()->bind(m_infoLabel, DFontSizeManager::T5, QFont::Normal);
282  m_titleLabel->setForegroundRole(DPalette::TextTitle);
283 
284  // 设置titleImage
285  QPixmap img;
286  for (auto meta : musicinfos) {
287  QString imagesDirPath = Global::cacheDir() + "/images/" + meta.hash + ".jpg";
288  // 查找可用图片
289  if (QFileInfo(imagesDirPath).exists()) {
290  img = MetaDetector::getCoverDataPixmap(meta);
291  break;
292  }
293  }
294 
295  if (img.isNull()) {
296  img = QIcon::fromTheme("cover_max").pixmap(QSize(200, 200));
297  }
298 
299  setTitleImage(img);
300  if (listPageType == AlbumType) {
301  m_musicListInfoView->setMusicListView(musicinfos, "album");
302  } else if (listPageType == SingerType) {
303  m_musicListInfoView->setMusicListView(musicinfos, "artist");
304  } else {
305  m_musicListInfoView->setMusicListView(musicinfos, "all");
306  }
307  }
308 }

310 void SubSonglistWidget::slotPlayAllClicked()
311 {
312  QList<MediaMeta> musicList = m_musicListInfoView->getMusicListData();
313  if (musicList.size() > 0) {
314  emit CommonService::getInstance()->signalSetPlayModel(Player::RepeatAll);
315  Player::getInstance()->setCurrentPlayListHash(m_hash, false);
316  Player::getInstance()->setPlayList(musicList);
317  Player::getInstance()->playMeta(musicList.first());
318  emit Player::getInstance()->signalPlayListChanged();
319  }
