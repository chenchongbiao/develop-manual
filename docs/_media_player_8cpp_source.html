349 void VlcMediaPlayer::libvlc_callback(const libvlc_event_t *event,
350  void *data)
351 {
352  VlcMediaPlayer *core = static_cast<VlcMediaPlayer *>(data);
353  switch (event->type) {
354  case libvlc_MediaPlayerMediaChanged:
355  emit core->mediaChanged(event->u.media_player_media_changed.new_media);
356  break;
357  case libvlc_MediaPlayerNothingSpecial:
358  emit core->nothingSpecial();
359  break;
360  case libvlc_MediaPlayerOpening:
361  emit core->opening();
362  break;
363  case libvlc_MediaPlayerBuffering:
364  emit core->buffering(event->u.media_player_buffering.new_cache);
365  emit core->buffering(qRound(event->u.media_player_buffering.new_cache));
366  break;
367  case libvlc_MediaPlayerPlaying:
368  emit core->playing();
369  break;
370  case libvlc_MediaPlayerPaused:
371  emit core->paused();
372  break;
373  case libvlc_MediaPlayerStopped:
374  emit core->stopped();
375  break;
376  case libvlc_MediaPlayerForward:
377  emit core->forward();
378  break;
379  case libvlc_MediaPlayerBackward:
380  emit core->backward();
381  break;
382  case libvlc_MediaPlayerEndReached:
383  emit core->end(); //play end
384  break;
385  case libvlc_MediaPlayerEncounteredError:
386  emit core->error();
387  break;
388  case libvlc_MediaPlayerTimeChanged:
389  emit core->timeChanged(event->u.media_player_time_changed.new_time);
390  break;
391  case libvlc_MediaPlayerPositionChanged:
392  emit core->positionChanged(event->u.media_player_position_changed.new_position);
393  break;
394  case libvlc_MediaPlayerSeekableChanged:
395  emit core->seekableChanged(event->u.media_player_seekable_changed.new_seekable);
396  break;
397  case libvlc_MediaPlayerPausableChanged:
398  emit core->pausableChanged(event->u.media_player_pausable_changed.new_pausable);
399  break;
400  case libvlc_MediaPlayerTitleChanged:
401  emit core->titleChanged(event->u.media_player_title_changed.new_title);
402  break;
403  case libvlc_MediaPlayerSnapshotTaken:
404  emit core->snapshotTaken(event->u.media_player_snapshot_taken.psz_filename);
405  break;
406  case libvlc_MediaPlayerLengthChanged:
407  emit core->lengthChanged(event->u.media_player_length_changed.new_length);
408  break;
409  case libvlc_MediaPlayerVout:
410  emit core->vout(event->u.media_player_vout.new_count);
411  break;
412  default:
413  break;
414  }
415 
416  if (event->type >= libvlc_MediaPlayerNothingSpecial
417  && event->type <= libvlc_MediaPlayerEncounteredError) {
418  emit core->stateChanged();
419  }
420 }
