140 void DisplayWorker::onMonitorListChanged(const QList<QDBusObjectPath> &mons)
141 {
142  QList<QString> ops;
143  for (const auto *mon : m_monitors.keys())
144  ops << mon->path();
145 
146  qDebug() << mons.size();
147  QList<QString> pathList;
148  for (const auto &op : mons) {
149  const QString path = op.path();
150  pathList << path;
151  if (!ops.contains(path))

157  monitorRemoved(op);

260 void DisplayWorker::setUiScale(const double value)
261 {
262  double rv = value;
263  if (rv < 0)
264  rv = m_model->uiScale();
265 
266  for (auto &mm : m_model->monitorList()) {
267  mm->setScale(-1);
268  }
269  QDBusPendingCall call = m_appearanceInter->SetScaleFactor(rv);
270 
271  QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(call, this);
272  watcher->waitForFinished();
273  if (!watcher->isError()) {
274  m_model->setUIScale(rv);
275  }
276  watcher->deleteLater();

314 void DisplayWorker::monitorAdded(const QString &path)
315 {
316  MonitorInter *inter = new MonitorInter(DisplayInterface, path, QDBusConnection::sessionBus(), this);
317  Monitor *mon = new Monitor(this);
318 
319  connect(inter, &MonitorInter::XChanged, mon, &Monitor::setX);
320  connect(inter, &MonitorInter::YChanged, mon, &Monitor::setY);
321  connect(inter, &MonitorInter::WidthChanged, mon, &Monitor::setW);
322  connect(inter, &MonitorInter::HeightChanged, mon, &Monitor::setH);
323  connect(inter, &MonitorInter::MmWidthChanged, mon, &Monitor::setMmWidth);
324  connect(inter, &MonitorInter::MmHeightChanged, mon, &Monitor::setMmHeight);
325  connect(inter, &MonitorInter::RotationChanged, mon, &Monitor::setRotate);
326  connect(inter, &MonitorInter::NameChanged, mon, &Monitor::setName);
327  connect(inter, &MonitorInter::CurrentModeChanged, mon, &Monitor::setCurrentMode);
328  connect(inter, &MonitorInter::BestModeChanged, mon, &Monitor::setBestMode);
329  connect(inter, &MonitorInter::CurrentModeChanged, this, [=](Resolution value) {
330  if (value.id() == 0) {
331  return;
332  }
333  auto maxWScale = value.width() / 1024.0;
334  auto maxHScale = value.height() / 768.0;
335  auto maxScale = maxWScale < maxHScale ? maxWScale : maxHScale;
336  if ((m_model->uiScale() - maxScale) > 0.01 && maxScale >= 1.0) {
337  m_currentScale = 1.0;
338  for (int idx = 0; idx * 0.25 + 1.0 <= maxScale; ++idx) {
339  m_currentScale = idx * 0.25 + 1.0;
340  }
341  m_updateScale = true;
342  }
343  });
344 
345  connect(inter, &MonitorInter::ModesChanged, mon, &Monitor::setModeList);
346  connect(inter, &MonitorInter::RotationsChanged, mon, &Monitor::setRotateList);
347  connect(inter, &MonitorInter::EnabledChanged, mon, &Monitor::setMonitorEnable);
348  connect(&m_displayInter, static_cast<void (DisplayInter::*)(const QString &) const>(&DisplayInter::PrimaryChanged), mon, &Monitor::setPrimary);
349 
350  // NOTE: DO NOT using async dbus call. because we need to have a unique name to distinguish each monitor
351  Q_ASSERT(inter->isValid());
352  mon->setName(inter->name());
353  mon->setManufacturer(inter->manufacturer());
354  mon->setModel(inter->model());
355  QDBusReply<bool> reply = m_displayDBusInter->call("CanSetBrightness", inter->name());
356  mon->setCanBrightness(reply.value());
357  mon->setMonitorEnable(inter->enabled());
358  mon->setPath(path);
359  mon->setX(inter->x());
360  mon->setY(inter->y());
361  mon->setW(inter->width());
362  mon->setH(inter->height());
363  mon->setRotate(inter->rotation());
364  mon->setCurrentMode(inter->currentMode());
365  mon->setBestMode(inter->bestMode());
366  mon->setModeList(inter->modes());
367  if (m_model->isRefreshRateEnable() == false) {
368  for (auto resolutionModel : mon->modeList()) {
369  if (qFuzzyCompare(resolutionModel.rate(), 0.0) == false) {
370  m_model->setRefreshRateEnable(true);
371  }
372  }
373  }
374  mon->setRotateList(inter->rotations());
375  mon->setPrimary(m_displayInter.primary());
376  mon->setMmWidth(inter->mmWidth());
377  mon->setMmHeight(inter->mmHeight());
378 
379  if (!m_model->brightnessMap().isEmpty()) {
380  mon->setBrightness(m_model->brightnessMap()[mon->name()]);
381  }
382 
383  m_model->monitorAdded(mon);
384  m_monitors.insert(mon, inter);
385 
386  inter->setSync(false);
387 }

389 void DisplayWorker::monitorRemoved(const QString &path)
390 {
391  Monitor *monitor = nullptr;
392  for (auto it(m_monitors.cbegin()); it != m_monitors.cend(); ++it) {
393  if (it.key()->path() == path) {
394  monitor = it.key();

398  if (!monitor)
399  return;
400 
401  m_model->monitorRemoved(monitor);
402 
403  m_monitors[monitor]->deleteLater();
404  m_monitors.remove(monitor);
