107 void AccessPointWidget::updateAP(const AccessPoint &ap)
108 {
109  m_ap = ap;
110 
111  QString strSsid = ap.ssid();
112  m_ssidBtn->setText(strSsid);
113 
114  QFontMetrics fontMetrics(m_ssidBtn->font());
115  if(fontMetrics.width(strSsid) > m_ssidBtn->width())
116  {
117  strSsid = QFontMetrics(m_ssidBtn->font()).elidedText(strSsid, Qt::ElideRight, m_ssidBtn->width());
118  }
119  m_ssidBtn->setText(strSsid);
120 
121  setStrengthIcon(ap.strength());
122 
123  if (!ap.secured()) {
124  m_securityLabel->clear();
125  } else if(!m_securityLabel->pixmap()) {
126  m_securityLabel->setPixmap(m_securityPixmap);
127  }
128 
129  // reset state
130  setActiveState(NetworkDevice::Unknown);
131 }

158 void AccessPointWidget::setStrengthIcon(const int strength)
159 {
160  QPixmap iconPix;
161  const QSize s = QSize(16, 16);
162 
163  QString type;
164  if (strength > 65)
165  type = "80";
166  else if (strength > 55)
167  type = "60";
168  else if (strength > 30)
169  type = "40";
170  else if (strength > 5)
171  type = "20";
172  else
173  type = "0";
174 
175  QString iconString = QString("wireless-%1-symbolic").arg(type);
176  bool isLight = (DGuiApplicationHelper::instance()->themeType() == DGuiApplicationHelper::LightType);
177 
178  if (isLight) {
179  iconString.append("-dark");
180  }
181 
182  const auto ratio = devicePixelRatioF();
183  iconPix = ImageUtil::loadSvg(iconString, ":/wireless/resources/wireless/", s.width(), ratio);
184 
185  m_strengthLabel->setPixmap(iconPix);
186 
187  m_securityPixmap = QIcon::fromTheme(isLight ? ":/wireless/resources/wireless/security_dark.svg" : ":/wireless/resources/wireless/security.svg").pixmap(s * devicePixelRatioF());
188  m_securityPixmap.setDevicePixelRatio(devicePixelRatioF());
189  m_securityLabel->setPixmap(m_securityPixmap);
190 }
