98 MenuAction takeAvailableUserActionType()
99 {
100  if (availableUserActionQueue.isEmpty()) {
101  availableUserActionQueue.append(MenuAction(MenuAction::UserMenuAction + 1));
102 
103  return MenuAction::UserMenuAction;
104  }
105 
106  MenuAction type = availableUserActionQueue.dequeue();
107 
108  if (availableUserActionQueue.isEmpty()) {
109  availableUserActionQueue.append(MenuAction(type + 1));
110  }
111 
112  return type;
113 }
114 void recycleUserActionType(MenuAction type)
115 {
116  availableUserActionQueue.prepend(type);
117  QAction *action = actions.take(type);
118 
119  if (action) {
120  actionToMenuAction.remove(action);
121  }
122 }
