46 void ToolButton::paintEvent(QPaintEvent *e)
47 {
48  Q_UNUSED(e)
49 
50  QStylePainter p(this);
51  QStyleOptionButton option;
52  initStyleOption(&option);
53 
54  QPainter painter(this);
55  //绘制背景
56  painter.save();
57  //非禁用状态绘制背景
58  if (option.state & QStyle::State_Enabled) {
59  if (option.state & QStyle::State_MouseOver) {
60  painter.setPen(Qt::NoPen);
61  //获取系统主题颜色
62  QColor hovertColor(option.palette.highlight().color());
63  painter.setBrush(hovertColor);
64  painter.drawRect(this->rect());
65  }
66  }
67  painter.restore();
68 
69  if (CManageViewSigleton::GetInstance()->getThemeType() == 1) {
70  //控件禁用样式
71  if (!(option.state & QStyle::State_Enabled)) {
72  painter.setPen(QColor("#9C9C9C"));
73  }
74 
75  //鼠标悬停画笔颜色
76  else if (option.state & QStyle::State_MouseOver) {
77  painter.setPen(QColor(Qt::white));
78  }
79 
80  //鼠标按下画笔颜色
81  else if (option.state & QStyle::State_Sunken) {
82  painter.setPen(QColor("#99cdff"));
83  } else {
84  painter.setPen(QColor("#343434"));
85  }
86 
87  } else {
88 
89  //控件禁用样式
90  if (!(option.state & QStyle::State_Enabled)) {
91  painter.setPen(QColor("#8D8D8D"));
92  }
93 
94  //鼠标悬停画笔颜色
95  else if (option.state & QStyle::State_MouseOver) {
96  painter.setPen(QColor(Qt::white));
97  }
98 
99  //鼠标按下画笔颜色
100  else if (option.state & QStyle::State_Sunken) {
101  painter.setPen(QColor("#99cdff"));
102  } else {
103 
104  painter.setPen(QColor(Qt::white));
105  }
106  }
107 
108 
109  // 绘制图片
110  painter.save();
111  m_icon.paint(&painter, QRect(10, -6, 45, 45));
112  painter.restore();
113 
114  //绘制文字
115  painter.save();
116  QFont ft;
117  ft.setPixelSize(14);
118  painter.setFont(ft);
119  painter.drawText(50, 5, 80, 22, 0, m_text);
120  painter.restore();
121 }
