80 void DFMVfsManagerPrivate::GVolumeMonitorMountAddedCb(GVolumeMonitor *, GMount *mount, DFMVfsManager* managerPointer)
81 {
82  DFMGFile rootFile(g_mount_get_root(mount));
83  DFMGCChar rootUriCStr(g_file_get_uri(rootFile.data()));
84  QString rootUrlStr(rootUriCStr.data());
85  QUrl url(rootUrlStr);
86  if (url.scheme() == "file") return;
87 // if (g_mount_is_shadowed(mount)) return; // is_shadowed
88 
89  QUrl deviceUrl;
90  deviceUrl.setScheme("device");
91  deviceUrl.setPath(rootUrlStr);
92 
93  emit managerPointer->vfsAttached(deviceUrl);
94  emit managerPointer->vfsDeviceListInfoChanged();
95 }

176 const QList<QUrl> DFMVfsManager::getVfsList()
177 {
178  Q_D(DFMVfsManager);
179 
180  QSet<QUrl> result;
181 
182  DFMGMountList mountList(g_volume_monitor_get_mounts(d->m_GVolumeMonitor.data()));
183  GMount* mount = nullptr;
184  GList* iter;
185  for (iter = mountList.data(); iter != nullptr; iter = iter->next) {
186  mount = (GMount*)iter->data;
187  DFMGFile rootFile(g_mount_get_root(mount));
188  DFMGCChar rootUriCStr(g_file_get_uri(rootFile.data()));
189  QString rootUrlStr(rootUriCStr.data());
190  QUrl urlForCheck(rootUrlStr);
191  if (urlForCheck.scheme() == "file") continue;
192  if (g_mount_is_shadowed(mount)) continue;
193  QUrl url;
194  url.setScheme("device");
195  url.setPath(rootUrlStr);
196  result << url;
197  }
198 
199  return result.toList();
200 }

202 bool DFMVfsManager::attach(const QUrl &url)
203 {
204  if (!url.isValid() || url.scheme() == "file" || url.scheme().isEmpty()) {
205  return false;
206  }
207 
208  QScopedPointer<DFMVfsDevice> dev(DFMVfsDevice::createUnsafe(url, nullptr));
209 
210  if (eventHandler()) {
211  dev->setEventHandler(eventHandler());
212  }
213 
214  return dev->attach();
215 }
