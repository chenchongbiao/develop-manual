57 static tm rfc_date_time_to_tm(const DateTime& date_time)
58 {
59  tm tm_date_time;
60  tm_date_time.tm_year = date_time.year() - 1900;
61  tm_date_time.tm_mon = date_time.month().ordinal() - 1;
62  tm_date_time.tm_mday = date_time.day();
63  tm_date_time.tm_hour = date_time.hour();
64  tm_date_time.tm_min = date_time.minute();
65  tm_date_time.tm_sec = date_time.second();
66  return tm_date_time;
67 }

79  std::string getFilename(const MimeEntity& mime_entity)
80  {
81  if (mime_entity.hasField(ContentDisposition::label))
82  {
83  const ContentDisposition& cd = mime_entity.header().contentDisposition();
84  if (cd.param("filename").length())
85  return std::string(cd.param("filename").c_str());
86  }
87  else if (mime_entity.hasField(ContentType::label))
88  {
89  const ContentType& ct = mime_entity.header().contentType();
90  return std::string("unnamed_" + ct.type() + "." + ct.subtype()).c_str();
91  }
92  return std::string("Unknown attachment");
93  }

108  void decodeBin(const MimeEntity& me, std::string& out)
109  {
110  std::stringstream os(std::ios::out);
111  std::ostreambuf_iterator<char> oi(os);
112  const ContentTransferEncoding& cte = me.header().contentTransferEncoding();
113  mimetic::istring enc_algo = cte.mechanism();
114  if (enc_algo == ContentTransferEncoding::base64)
115  {
116  if (m_verbose_logging)
117  *m_log_stream << "Using base64 decoding\n";
118  Base64::Decoder b64;
119  decode(me.body().begin(), me.body().end(), b64, oi);
120  }
121  else if (enc_algo == ContentTransferEncoding::quoted_printable)
122  {
123  if (m_verbose_logging)
124  *m_log_stream << "Using quoted_printable decoding\n";
125  QP::Decoder qp;
126  decode(me.body().begin(), me.body().end(), qp, oi);
127  }
128  else if (enc_algo == ContentTransferEncoding::eightbit ||
129  enc_algo == ContentTransferEncoding::sevenbit ||
130  enc_algo == ContentTransferEncoding::binary)
131  {
132  if (m_verbose_logging)
133  *m_log_stream << "Using eightbit/sevenbit/binary decoding\n";
134  copy(me.body().begin(), me.body().end(), oi);
135  }
136  else
137  {
138  *m_log_stream << "Unknown encoding: " + enc_algo + "\n";
139  copy(me.body().begin(), me.body().end(), oi);
140  }
141  os.flush();
142  out = os.rdbuf()->str();
143  }

145  void extractPlainText(const MimeEntity& mime_entity, std::string& output, const FormattingStyle& formatting)
146  {
147  const Header& header = mime_entity.header();
148  if (header.contentDisposition().str().find(attachment.c_str()) == std::string::npos &&
149  header.contentType().str().substr(0, text.length()) == text)
150  {
151  std::string plain;
152  decodeBin(mime_entity, plain);
153 
154  bool skip_charset_decoding = false;
155  if (header.contentType().param("charset").length())
156  {
157  std::string charset(header.contentType().param("charset").c_str());
158  convertToUtf8(charset, plain);
159  skip_charset_decoding = true;
160  }
161  if (header.contentType().str().substr(0, text_html.length()) == text_html ||
162  header.contentType().str().substr(0, text_xhtml.length()) == text_xhtml)
163  {
164  HTMLParser html(plain.c_str(), plain.length());
165  html.setLogStream(*m_log_stream);
166  html.setVerboseLogging(m_verbose_logging);
167  if (skip_charset_decoding)
168  html.skipCharsetDecoding();
169  plain = html.plainText(formatting);
170  html.getLinks(m_links);
171  if (html.error())
172  *m_log_stream << "Warning: Error while parsing html\n";
173  //Update positions of the links.
174  if (m_links.size() > 0)
175  {
176  size_t link_offset = output.length();
177  for (size_t i = 0; i <m_links.size(); ++i)
178  m_links[i].setLinkTextPosition(m_links[i].getLinkTextPosition() + link_offset);
179  }
180  }
181  else
182  {
183  if (!skip_charset_decoding)
184  {
185  TXTParser txt(plain.c_str(), plain.length());
186  txt.setLogStream(*m_log_stream);
187  txt.setVerboseLogging(m_verbose_logging);
188  plain = txt.plainText();
189  if (txt.error())
190  *m_log_stream << "Warning: Error while parsing txt";
191  }
192  }
193  output += plain;
194  output += "\n\n";
195  return;
196  }
197  else if (header.contentType().str().substr(0, multipart.length()) != multipart)
198  {
199  std::string plain;
200  decodeBin(mime_entity, plain);
201 
202  m_attachments.push_back(doctotext::Attachment(getFilename(mime_entity)));
203  m_attachments[m_attachments.size() - 1].setBinaryContent(plain);
204  std::string content_type = header.contentType().str();
205  std::string content_disposition = header.contentDisposition().str();
206  std::string content_description = header.contentDescription().str();
207  std::string content_id = header.contentId().str();
208  if (!content_type.empty())
209  m_attachments[m_attachments.size() - 1].addField("Content-Type", content_type);
210  if (!content_disposition.empty())
211  m_attachments[m_attachments.size() - 1].addField("Content-Disposition", content_disposition);
212  if (!content_description.empty())
213  m_attachments[m_attachments.size() - 1].addField("Content-Description", content_description);
214  if (!content_id.empty())
215  m_attachments[m_attachments.size() - 1].addField("Content-ID", content_id);
216  }
217  const MimeEntityList& parts = mime_entity.body().parts();
218  MimeEntityList::const_iterator mbit = parts.begin(), meit = parts.end();
219  for(; mbit != meit; ++mbit)
220  extractPlainText(**mbit, output, formatting);
221  }

292 bool EMLParser::isEML()
293 {
294  if (!impl->m_data_stream->good())
295  {
296  *impl->m_log_stream << "Error opening file " + impl->m_file_name + "\n";
297  impl->m_error = true;
298  return false;
299  }
300  MimeEntity mime_entity(*impl->m_data_stream);
301  //"From" and "Date" are obligatory according to the RFC standard.
302  Header header = mime_entity.header();
303  if (!header.hasField("From") || !header.hasField("Date"))
304  return false;
305  return true;
306 }

332 Metadata EMLParser::metaData()
333 {
334  Metadata metadata;
335  if (!isEML())
336  {
337  impl->m_error = true;
338  return metadata;
339  }
340  impl->m_data_stream->seekg(0, std::ios_base::beg);
341  MimeEntity mime_entity(*impl->m_data_stream);
342  Header header = mime_entity.header();
343  metadata.setAuthor(header.from().str());
344  DateTime creation_date(header.field("Date").value());
345  metadata.setCreationDate(rfc_date_time_to_tm(creation_date));
346 
347  //in EML file format author is visible under key "From". And creation date is visible under key "Data".
348  //So, should I repeat the same values or skip them?
349  metadata.addField("From", header.from().str());
350  metadata.addField("Date", rfc_date_time_to_tm(creation_date));
351 
352  std::string to = header.to().str();
353  if (!to.empty())
354  metadata.addField("To", to);
355 
356  std::string subject = header.subject();
357  if (!subject.empty())
358  metadata.addField("Subject", subject);
359  std::string mime_version = header.mimeVersion().str();
360  if (!mime_version.empty())
361  metadata.addField("MIME-Version", mime_version);
362  std::string content_type = header.contentType().str();
363  if (!content_type.empty())
364  metadata.addField("Content-Type", content_type);
365  std::string content_disposition = header.contentDisposition().str();
366  if (!content_disposition.empty())
367  metadata.addField("Content-Disposition", content_disposition);
368  std::string content_description = header.contentDescription().str();
369  if (!content_description.empty())
370  metadata.addField("Content-Description", content_description);
371  std::string content_id = header.contentId().str();
372  if (!content_id.empty())
373  metadata.addField("Content-ID", content_id);
374  std::string message_id = header.messageid().str();
375  if (!message_id.empty())
376  metadata.addField("Message-ID", message_id);
377  std::string reply_to = header.replyto().str();
378  if (!reply_to.empty())
379  metadata.addField("Reply-To", reply_to);
380  std::string sender = header.sender().str();
381  if (!sender.empty())
