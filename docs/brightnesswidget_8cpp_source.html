93 void BrightnessWidget::setMode(DisplayModel *model)
94 {
95  m_displayModel = model;
96 
97  connect(m_autoLightMode, &SwitchWidget::checkedChanged, this, &BrightnessWidget::requestAmbientLightAdjustBrightness);
98  connect(m_displayModel, &DisplayModel::adjustCCTmodeChanged, this, &BrightnessWidget::setAdjustCCTmode);
99  connect(m_displayModel, &DisplayModel::redshiftVaildChanged, m_nightShift, &SwitchWidget::setVisible);
100  connect(m_displayModel, &DisplayModel::redshiftVaildChanged, m_tempratureColorTitle, &TitleLabel::setVisible);
101  connect(m_displayModel, &DisplayModel::redshiftVaildChanged, m_nightTips, &QLabel::setVisible);
102  connect(m_nightManual, &SwitchWidget::checkedChanged, this, [=](const bool enable) {
103  if (enable) {
104  Q_EMIT requestSetMethodAdjustCCT(2);
105  } else {
106  Q_EMIT requestSetMethodAdjustCCT(0);
107  }
108  });
109  connect(m_nightShift, &SwitchWidget::checkedChanged, this, [=](const bool enable) {
110  if (enable) {
111  Q_EMIT requestSetMethodAdjustCCT(1);
112  } else {
113  Q_EMIT requestSetMethodAdjustCCT(0);
114  }
115  });
116  connect(m_displayModel, &DisplayModel::autoLightAdjustVaildChanged, m_autoLightMode, &SwitchWidget::setVisible);
117  connect(m_displayModel, &DisplayModel::autoLightAdjustSettingChanged, m_autoLightMode, &SwitchWidget::setChecked);
118 
119  m_autoLightMode->setVisible(model->autoLightAdjustIsValid());
120  m_autoLightMode->setChecked(model->isAudtoLightAdjust());
121  int cctmode = model->adjustCCTMode(); //0不调节色温 1 自动调节 2手动调节
122  m_nightShift->setVisible(model->redshiftIsValid());
123  m_tempratureColorTitle->setVisible(model->redshiftIsValid());
124  m_nightManual->setVisible(model->redshiftIsValid());
125  m_nightTips->setVisible(model->redshiftIsValid());
126  setAdjustCCTmode(cctmode);
127  addSlider();
128 }

154 void BrightnessWidget::addSlider()
155 {
156  auto monList = m_displayModel->monitorList();
157  for (auto monitor : monList) {
158  if (!monitor->canBrightness()) {
159  monList.removeOne(monitor);
160  }
161  }
162 
163  TitleLabel *headTitle = new TitleLabel(tr("Brightness")); //亮度
164  m_centralLayout->insertWidget(0, headTitle);
165  for (int i = 0; i < monList.size(); ++i) {
166  //单独显示每个亮度调节名
167  TitledSliderItem *slideritem = new TitledSliderItem(monList[i]->name());
168 
169  slideritem->addBackground();
170  DCCSlider *slider = slideritem->slider();
171  int maxBacklight = static_cast<int>(m_displayModel->maxBacklightBrightness());
172  if (maxBacklight == 0) {
173  qDebug() << "MinimumBrightness: " << m_displayModel->minimumBrightnessScale();
174  int miniScale = int(m_displayModel->minimumBrightnessScale() * BrightnessMaxScale);
175  double brightness = monList[i]->brightness();
176  slideritem->setValueLiteral(brightnessToTickInterval(brightness));
177  slider->setRange(miniScale, int(BrightnessMaxScale));
178  slider->setType(DCCSlider::Vernier);
179  slider->setTickPosition(QSlider::TicksBelow);
180  slider->setLeftIcon(QIcon::fromTheme("dcc_brightnesslow"));
181  slider->setRightIcon(QIcon::fromTheme("dcc_brightnesshigh"));
182  slider->setIconSize(QSize(24, 24));
183  slider->setTickInterval(int((BrightnessMaxScale - miniScale) / 5.0));
184  slider->setValue(int(brightness * BrightnessMaxScale));
185  slider->setPageStep(1);
186 
187  auto onValueChanged = [=](int pos) {
188  this->requestSetMonitorBrightness(monList[i], pos / BrightnessMaxScale);
189  this->requestAmbientLightAdjustBrightness(false);
190  };
191 
192  connect(slider, &DCCSlider::valueChanged, this, onValueChanged);
193  connect(slider, &DCCSlider::sliderMoved, this, onValueChanged);
194 
195  connect(monList[i], &Monitor::brightnessChanged, this, [=](const double rb) {
196  slider->blockSignals(true);
197  if ((rb - m_displayModel->minimumBrightnessScale()) < 0.00001) {
198  slideritem->setValueLiteral(QString("%1%").arg(int(m_displayModel->minimumBrightnessScale() * BrightnessMaxScale)));
199  slider->setValue(int(m_displayModel->minimumBrightnessScale() * BrightnessMaxScale));
200  } else {
201  slideritem->setValueLiteral(QString("%1%").arg(int(rb * BrightnessMaxScale)));
202  slider->setValue(int(rb * BrightnessMaxScale));
203  }
204  slider->blockSignals(false);
205  });
206 
207  connect(m_displayModel, &DisplayModel::minimumBrightnessScaleChanged,
208  this, [=](const double ms) {
209  double rb = monList[i]->brightness();
210  int tmini = int(ms * PercentageNum);
211  slider->setMinimum(tmini);
212  slider->setTickInterval(int((BrightnessMaxScale - tmini) / 5.0));
213 
214  slider->blockSignals(true);
215  slideritem->setValueLiteral(brightnessToTickInterval(rb));
216  slider->setValue(int(rb * BrightnessMaxScale));
217  slider->blockSignals(false);
218  });
219  } else {
220  qDebug() << "MinimumBrightness: " << m_displayModel->minimumBrightnessScale();
221  m_miniScales = int(m_displayModel->minimumBrightnessScale() * maxBacklight);
222  if (m_miniScales == 0) {
223  m_miniScales = 1;
224  }
225  double brightness = monList[i]->brightness();
226  slider->setRange(m_miniScales, maxBacklight);
227  slider->setType(DCCSlider::Vernier);
228  slider->setTickPosition(QSlider::TicksBelow);
229  slider->setLeftIcon(QIcon::fromTheme("dcc_brightnesslow"));
230  slider->setRightIcon(QIcon::fromTheme("dcc_brightnesshigh"));
231  slider->setIconSize(QSize(24, 24));
232  slider->setTickInterval(1);
233  slider->setValue(int((brightness + DoubleZero) * maxBacklight));
234  slider->setPageStep(1);
235  QStringList speedList;
236  int j = m_miniScales;
237  for (; j <= maxBacklight; j++) {
238  speedList << "";
239  }
240  slideritem->setAnnotations(speedList);
241 
242  auto onValueChanged = [=](int pos) {
243  this->requestSetMonitorBrightness(monList[i], pos / double(maxBacklight));
244  this->requestAmbientLightAdjustBrightness(false);
245  };
246 
247  connect(slider, &DCCSlider::valueChanged, this, onValueChanged);
248  connect(slider, &DCCSlider::sliderMoved, this, onValueChanged);
249 
250  connect(monList[i], &Monitor::brightnessChanged, this, [=](const double rb) {
251  slider->blockSignals(true);
252  if ((rb - m_displayModel->minimumBrightnessScale()) < 0.00001) {
253  slider->setValue(int((m_displayModel->minimumBrightnessScale() + DoubleZero) * maxBacklight));
254  } else {
255  slider->setValue(int((rb + DoubleZero) * maxBacklight));
256  }
257  slider->blockSignals(false);
258  });
259 
260  connect(m_displayModel, &DisplayModel::minimumBrightnessScaleChanged,
261  this, [=](const double ms) {
262  double rb = monList[i]->brightness();
263  int tmini = int(ms * PercentageNum);
264  slider->setMinimum(tmini);
265  slider->setTickInterval(int((BrightnessMaxScale - tmini) / 5.0));
266 
267  slider->blockSignals(true);
268  slideritem->setValueLiteral(brightnessToTickInterval(rb));
269  slider->setValue(int((rb + DoubleZero) * BrightnessMaxScale));
270  slider->blockSignals(false);
271  });
272  }
273  m_centralLayout->insertWidget(1, slideritem);
274  m_monitorBrightnessMap[monList[i]] = slideritem;
275  }
276 
277  DCCSlider *cctSlider = m_cctItem->slider();
278  cctSlider->setRange(0, 100);
279  cctSlider->setType(DCCSlider::Vernier);
280  cctSlider->setTickPosition(QSlider::TicksBelow);
281  cctSlider->setTickInterval(10);
282  cctSlider->setPageStep(1);
283  cctSlider->setValue(colorTemperatureToValue(m_displayModel->colorTemperature()));
284  if (m_displayModel->redshiftIsValid()) {
285  m_cctItem->setVisible(m_displayModel->adjustCCTMode() == 2);
286  } else
287  m_cctItem->setVisible(false);
288  connect(m_displayModel, &DisplayModel::colorTemperatureChanged, this, [=](int value) {
289  cctSlider->blockSignals(true);
290  cctSlider->setValue(colorTemperatureToValue(value));
291  cctSlider->blockSignals(false);
292  });
293 
294  connect(cctSlider, &DCCSlider::valueChanged, this, [=](int pos) {
295  int kelvin = pos > 50 ? (6500 - (pos - 50) * 100) : (6500 + (50 - pos) * 300);
296  this->requestSetColorTemperature(kelvin);
297  });
298  connect(cctSlider, &DCCSlider::sliderMoved, this, [=](int pos) {
299  int kelvin = pos > 50 ? (6500 - (pos - 50) * 100) : (6500 + (50 - pos) * 300);
300  this->requestSetColorTemperature(kelvin);
301  });
302 }

304 int BrightnessWidget::colorTemperatureToValue(int kelvin)
305 {
306  //色温范围有效值10000-25000 值越大，色温越冷，不开启色温时值为6500,超过18000基本看不出变化，小于1500色温实际效果没法看，无实际价值
307  //此处取有效至1500-21500
308  if (kelvin >= 6500)
309  return 50 - (kelvin - 6500) / 300;
310  else if (kelvin < 6500 && kelvin >= 1000)
311  return 50 - (kelvin - 6500) / 100;
312  else
313  return 0;
314 }

316 QString BrightnessWidget::brightnessToTickInterval(const double tb) const
317 {
318  int tmini = int(m_displayModel->minimumBrightnessScale() * BrightnessMaxScale);
319  int tnum = int(tb * BrightnessMaxScale);
320  tnum = tnum > tmini ? tnum : tmini;
321  return QString::number(int(tnum)) + "%";
322 }
