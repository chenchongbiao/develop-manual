47 void AvatarItemDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const
48 {
49  painter->setRenderHints(painter->renderHints()
50  | QPainter::Antialiasing
51  | QPainter::SmoothPixmapTransform);
52 
53  if (!index.isValid())
54  return;
55 
56  QStyleOptionViewItem opt(option);
57  initStyleOption(&opt, index);
58  auto style = opt.widget->style();
59 
60  auto pm = static_cast<QStyle::PixelMetric>(DStyle::PM_FocusBorderWidth);
61  int borderWidth = style->pixelMetric(pm, &opt, nullptr);
62  pm = static_cast<QStyle::PixelMetric>(DStyle::PM_FocusBorderSpacing);
63  int borderSpacing = style->pixelMetric(pm, &opt, nullptr);
64  const QMargins margins(borderWidth + borderSpacing, borderWidth + borderSpacing,
65  borderWidth + borderSpacing, borderWidth + borderSpacing);
66  QPixmap pixmap = index.data(Qt::DecorationRole).value<QPixmap>();
67  QPainterPath path;
68  path.addEllipse(opt.rect.marginsRemoved(margins));
69  painter->setClipPath(path);
70 
71  if (!pixmap.isNull()) {
72  painter->drawPixmap(opt.rect.marginsRemoved(margins), pixmap);
73  painter->setClipping(false);
74  } else {
75  qreal tw = opt.rect.width() / 3.0;
76  qreal th = opt.rect.height() / 3.0;
77 
78  //绘制背景
79  DStyleHelper dh(style);
80  QRectF rect(tw + opt.rect.x(), th + opt.rect.y(), tw, th);
81  rect.moveCenter(QRect(opt.rect).center());
82  painter->setPen(Qt::NoPen);
83  painter->setBrush(dh.getColor(&opt, QPalette::Button));
84  painter->drawEllipse(opt.rect.marginsRemoved(margins));
85 
86  //画+号
87  qreal x1 = opt.rect.x() + tw ;
88  qreal y1 = opt.rect.y() + opt.rect.height() / 2.0 - 0.5;
89  qreal x2 = opt.rect.x() + opt.rect.width() / 2.0 - 0.5;
90  qreal y2 = opt.rect.y() + th;
91  painter->setBrush(dh.getColor(&opt, QPalette::Text));
92  painter->drawRect(QRectF(x1, y1, tw, 1.0));
93  painter->drawRect(QRectF(x2, y2, 1.0, th));
94  }
95 
96  if (index.data(Qt::CheckStateRole) == Qt::Checked) {
97  painter->setPen(QPen(opt.palette.highlight(), borderWidth));
98  painter->setBrush(Qt::NoBrush);
99  painter->drawEllipse(opt.rect.adjusted(1, 1, -1, -1));
100 
101  //在中间绘制选中小图标
102  int radius = 8;
103  int cx = opt.rect.marginsRemoved(margins).center().x();
104  int cy = opt.rect.marginsRemoved(margins).center().y();
105  QRect crect(QPoint(cx - radius, cy - radius), QPoint(cx + radius, cy + radius));
106  opt.rect = crect;
107  opt.state |= QStyle::State_On;
108  style->drawPrimitive(DStyle::PE_IndicatorItemViewItemCheck, &opt, painter, nullptr);
109  return;
110  }
111  // draw + in the end
112 }
