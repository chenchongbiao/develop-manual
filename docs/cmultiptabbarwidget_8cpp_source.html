85 void CMultipTabBarWidget::addTabBarItem(QString name, const QString &uuid, bool emitNewScene)
86 {
87  if (name.isEmpty()) {
88  emit this->tabAddRequested();
89  return;
90  }
91 
92  // 添加一个标签页之前如果是空的那么要再次发送页面变化的信号(以保证uuid能获取从而获得到正确的view场景，比如程序刚启动初始化的情形)
93  bool isEmptyBeforAdded = (this->count() == 0);
94 
95  int index = this->addTab(name);
96 
97  setTabData(index, uuid);
98 
99  //设置悬停提示消息
100  setTabBarTooltipName(uuid, name);
101 
102  if (emitNewScene)
103  emit signalNewAddItem(name, uuid);
104 
105  //保证uuid能获取到
106  if (isEmptyBeforAdded) {
107  emit currentChanged(index);
108  }
109 
110  this->setCurrentIndex(index);
111  this->setTabMinimumSize(index, TabBarMiniSize);
112 }

114 void CMultipTabBarWidget::initConnection()
115 {
116  // [0] 连接点击 + 按钮响应事件
117  connect(this, &DTabBar::tabAddRequested, this, [ = ]() {
118  // 组装标签名字
119  m_nameCounter++;
120  QString divName = m_defaultName + QString::number(m_nameCounter);
121 
122 // // 如果已经存在标签则显示存在的标签
123 // if (tabBarNameIsExist(divName)) {
124 // m_nameCounter--;
125 // return ;
126 // }
127 
128  QString uuid = CDrawParamSigleton::creatUUID();
129  addTabBarItem(divName, uuid);
130  });
131 
132  // [1] 连接点击关闭响应事件
133  connect(this, &DTabBar::tabCloseRequested, this, [ = ](int index) {
134  QString closeItemName = this->tabText(index);
135  QString closeItemUUID = this->tabData(index).toString();
136  emit signalTabItemCloseRequested(closeItemName, closeItemUUID);
137  });
138 
139  // [2] 连接标签选择改变响应事件
140  connect(this, &DTabBar::currentChanged, this, [ = ](int index) {
141  QString currentName = this->tabText(index);
142  QString closeItemUUID = this->tabData(index).toString();
143  emit signalItemChanged(currentName, closeItemUUID);
144  });
145 
146  // [3] 连接新增加标签页设置到当前新增加标签
147  connect(this, &DTabBar::tabIsInserted, this, [ = ](int index) {
148  this->setCurrentIndex(index);
149  }, Qt::QueuedConnection);
150 }

261 bool CMultipTabBarWidget::eventFilter(QObject *o, QEvent *event)
262 {
263  if (event->type() == QEvent::MouseButtonPress) {
264  QMouseEvent *mouseEvent = static_cast<QMouseEvent *>(event);
265 
266  if (mouseEvent->button() == Qt::RightButton) {
267  QPoint position = mouseEvent->pos();
268  m_rightClickTab = -1;
269 
270  // 判断当前是否找到该标签
271  for (int i = 0; i < count(); i++) {
272  if (this->tabRect(i).contains(position)) {
273  m_rightClickTab = i;
274  break;
275  }
276  }
277 
278  // popup right menu on tab.
279  if (m_rightClickTab >= 0) {
280  m_rightMenu = new DMenu;
281 
282  m_closeTabAction = new QAction(tr("Close tab"), this);
283  m_closeOtherTabAction = new QAction(tr("Close other tabs"), this);
284 
285  connect(m_closeTabAction, &QAction::triggered, this, [ = ] {
286  // 跳转到需要关闭的标签
287  this->setCurrentIndex(m_rightClickTab);
288 
289  emit signalTabItemCloseRequested(this->tabText(m_rightClickTab), this->tabData(m_rightClickTab).toString());
290  });
291 
292  connect(m_closeOtherTabAction, &QAction::triggered, this, [ = ] {
293  QStringList closeNames;
294  QStringList closeUUIDs;
295  for (int i = 0; i < this->count(); i++)
296  {
297  if (m_rightClickTab == i) {
298  continue;
299  }
300  closeNames << this->tabText(i);
301  closeUUIDs << this->tabData(i).toString();
302  }
303  emit signalTabItemsCloseRequested(closeNames, closeUUIDs);
304  });
305 
306  m_rightMenu->addAction(m_closeTabAction);
307  m_rightMenu->addAction(m_closeOtherTabAction);
308 
309  m_rightMenu->exec(mapToGlobal(position));
310 
311  //return true;
312  }
313  }
314  } /*else {
315  event->accept();
316  }*/
317  //return false;
318  return DTabBar::eventFilter(o, event);
319 }
