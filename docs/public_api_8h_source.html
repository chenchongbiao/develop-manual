107 class DMouseMoveEvent: public QTestMouseEvent
108 {
109 public:
110  //继承父亲的构造函数
111  using QTestMouseEvent::QTestMouseEvent;
112 
113  void simulate(QWidget *w) override
114  {
115  if (_action == QTest::MouseMove) {
116  if (_delay == -1 || _delay < 100)
117  _delay = 100;
118  if (_delay > 0) {
119  QTest::qWait(_delay);
120  QTest::lastMouseTimestamp += _delay;
121  }
122 
123  QMouseEvent me(QEvent::MouseMove, _pos, _button, _button, _modifiers);
124  me.setTimestamp(ulong(++QTest::lastMouseTimestamp));
125  QSpontaneKeyEvent::setSpontaneous(&me);
126  if (!dApp->notify(w, &me)) {
127  static const char *const mouseActionNames[] =
128  { "MousePress", "MouseRelease", "MouseClick", "MouseDClick", "MouseMove" };
129  QString warning = QString::fromLatin1("Mouse event \"%1\" not accepted by receiving widget");
130  QTest::qWarn(warning.arg(QString::fromLatin1(mouseActionNames[static_cast<int>(_action)])).toLatin1().data());
131  }
132  } else {
133  QTestMouseEvent::simulate(w);
134  }
135  }
136 
137  QTestEvent *clone() const override
138  {
139  return new DMouseMoveEvent(*this);
140  }
141 };

143 class DTestEventList: public QTestEventList
144 {
145 public:
146  inline void addMousePress(Qt::MouseButton button, Qt::KeyboardModifiers stateKey = Qt::KeyboardModifiers(),
147  QPoint pos = QPoint(), int delay = -1)
148  {
149  _pressedMouseButton = button;
150  QTestEventList::addMousePress(button, stateKey, pos, delay);
151  }
152  inline void addMouseRelease(Qt::MouseButton button, Qt::KeyboardModifiers stateKey = Qt::KeyboardModifiers(),
153  QPoint pos = QPoint(), int delay = -1)
154  {
155  _pressedMouseButton = Qt::NoButton;
156  QTestEventList::addMouseRelease(button, stateKey, pos, delay);
157  }
158  inline void addMouseMove(QPoint pos = QPoint(), int delay = -1)
159  {
160  append(new DMouseMoveEvent(QTest::MouseMove, _pressedMouseButton, Qt::KeyboardModifiers(), pos, delay));
161  }

163  {
164  //保证离屏时的焦点问题
165  {
166  qApp->setActiveWindow(drawApp->topMainWindowWidget());
167  w->setFocus();
168  qApp->processEvents();
169  }
170 
171  for (int i = 0; i < count(); ++i)
172  at(i)->simulate(w);
173  }
174 private:
175  Qt::MouseButton _pressedMouseButton = Qt::NoButton;
176 };

315 inline void resizeItem()
316 {
317  CGraphicsView *view = getCurView();
318  ASSERT_NE(view, nullptr);
319 
320  CGraphicsItem *pItem = dynamic_cast<CGraphicsItem *>(view->drawScene()->getBzItems().first());
321  ASSERT_NE(pItem, nullptr);
322 
323  view->drawScene()->clearSelectGroup();
324  view->drawScene()->selectItem(pItem);
325 
326  QVector<CSizeHandleRect *> handles = view->drawScene()->selectGroup()->handleNodes();
327 
328  int delay = 50;
329 
330  // 普通拉伸
331  for (int i = 0; i < handles.size(); ++i) {
332  CSizeHandleRect *pNode = handles[i];
333  QPoint posInView = view->mapFromScene(pNode->mapToScene(pNode->boundingRect().center()));
334  DTestEventList e;
335  e.addMouseMove(posInView, delay);
336  e.addMousePress(Qt::LeftButton, Qt::ShiftModifier, posInView, delay);
337  e.addMouseMove(posInView + QPoint(20, 20), delay);
338  e.addMouseRelease(Qt::LeftButton, Qt::ShiftModifier, posInView + QPoint(20, 20), delay);
339  e.simulate(view->viewport());
340 
341  // Undo Redo
342  e.clear();
343  e.addKeyPress(Qt::Key_Z, Qt::ControlModifier, delay);
344  e.addKeyRelease(Qt::Key_Z, Qt::ControlModifier, delay);
345  e.addDelay(100);
346  e.clear();
347  e.addKeyPress(Qt::Key_Y, Qt::ControlModifier, delay);
348  e.addKeyRelease(Qt::Key_Y, Qt::ControlModifier, delay);
349  e.simulate(view->viewport());
350  }
351 
352  // SHIFT ALT拉伸: QTestEvent mouseMove 中移动鼠标的实现是直接设置全局鼠标位置 5.15中解决了此问题
353  for (int i = 0; i < handles.size(); ++i) {
354  CSizeHandleRect *pNode = handles[i];
355  QPoint posInView = view->mapFromScene(pNode->mapToScene(pNode->boundingRect().center()));
356  QMouseEvent mouseEvent(QEvent::MouseButtonPress, posInView, Qt::LeftButton, Qt::LeftButton, Qt::ShiftModifier);
357  QApplication::sendEvent(view->viewport(), &mouseEvent);
358  QTest::qWait(delay);
359  QMouseEvent mouseEvent1(QEvent::MouseMove, posInView + QPoint(20, 20), Qt::LeftButton, Qt::LeftButton, Qt::ShiftModifier);
360  QApplication::sendEvent(view->viewport(), &mouseEvent1);
361  QTest::qWait(delay);
362  }
363  for (int i = 0; i < handles.size(); ++i) {
364  CSizeHandleRect *pNode = handles[i];
365  QPoint posInView = view->mapFromScene(pNode->mapToScene(pNode->boundingRect().center()));
366  QMouseEvent mouseEvent(QEvent::MouseButtonPress, posInView, Qt::LeftButton, Qt::LeftButton, Qt::AltModifier);
367  QApplication::sendEvent(view->viewport(), &mouseEvent);
368  QTest::qWait(delay);
369  QMouseEvent mouseEvent1(QEvent::MouseMove, posInView - QPoint(20, 20), Qt::LeftButton, Qt::LeftButton, Qt::AltModifier);
370  QApplication::sendEvent(view->viewport(), &mouseEvent1);
371  QTest::qWait(delay);
372  }
373  for (int i = 0; i < handles.size(); ++i) {
374  CSizeHandleRect *pNode = handles[i];
375  QPoint posInView = view->mapFromScene(pNode->mapToScene(pNode->boundingRect().center()));
376  QMouseEvent mouseEvent(QEvent::MouseButtonPress, posInView, Qt::LeftButton, Qt::LeftButton, Qt::ShiftModifier | Qt::AltModifier);
377  QApplication::sendEvent(view->viewport(), &mouseEvent);
378  QTest::qWait(delay);
379  QMouseEvent mouseEvent1(QEvent::MouseMove, posInView + QPoint(20, 20), Qt::LeftButton, Qt::LeftButton, Qt::ShiftModifier | Qt::AltModifier);
380  QApplication::sendEvent(view->viewport(), &mouseEvent1);
381  QTest::qWait(delay);
382  }
383 
384  // 全选拉伸
385  view->slotOnSelectAll();
386  handles = view->drawScene()->selectGroup()->handleNodes();
387  for (int i = 0; i < handles.size(); ++i) {
388  CSizeHandleRect *pNode = handles[i];
389  QPoint posInView = view->mapFromScene(pNode->mapToScene(pNode->boundingRect().center()));
390  DTestEventList e;
391  e.addMouseMove(posInView, 100);
392  e.addMousePress(Qt::LeftButton, Qt::ShiftModifier, posInView, 100);
393  e.addMouseMove(posInView - QPoint(50, 50), 100);
394  e.addMouseRelease(Qt::LeftButton, Qt::ShiftModifier, posInView - QPoint(50, 50), 100);
395  e.simulate(view->viewport());
396  }
397 }
