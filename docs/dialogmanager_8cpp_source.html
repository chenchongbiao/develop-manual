218 QPoint DialogManager::getPerportyPos(int dialogWidth, int dialogHeight, int count, int index)
219 {
220  Q_UNUSED(dialogHeight)
221  const QScreen *cursor_screen = Q_NULLPTR;
222  const QPoint &cursor_pos = QCursor::pos();
223 
224  for (const QScreen *screen : qApp->screens()) {
225  if (screen->geometry().contains(cursor_pos)) {
226  cursor_screen = screen;
227  break;
228  }
229  }
230 
231  if (!cursor_screen) {
232  cursor_screen = qApp->primaryScreen();
233  }
234 
235  int desktopWidth = cursor_screen->size().width();
236 // int desktopHeight = cursor_screen->size().height();//后面未用，注释掉
237  int SpaceWidth = 20;
238  int SpaceHeight = 70;
239  int row, x, y;
240 
241  int numberPerRow = desktopWidth / (dialogWidth + SpaceWidth);
242 
243 
244  if (count % numberPerRow == 0) {
245  row = count / numberPerRow;
246 
247  } else {
248  row = count / numberPerRow + 1;
249 
250  }
251  Q_UNUSED(row)
252  int dialogsWidth;
253  if (count / numberPerRow > 0) {
254  dialogsWidth = dialogWidth * numberPerRow + SpaceWidth * (numberPerRow - 1);
255  } else {
256  dialogsWidth = dialogWidth * (count % numberPerRow) + SpaceWidth * (count % numberPerRow - 1);
257  }
258 
259 // int dialogsHeight = dialogHeight + SpaceHeight * (row - 1);//未用注释掉
260 
261  x = (desktopWidth - dialogsWidth) / 2 + (dialogWidth + SpaceWidth) * (index % numberPerRow);
262 
263  y = 5 + (index / numberPerRow) * SpaceHeight;
264 
265  return QPoint(x, y) + cursor_screen->geometry().topLeft();

577 {

585  break;
586  case FileJob::OpticalBurn:
587  case FileJob::OpticalImageBurn:
588  failure_type = tr("Burn process failed");
589  break;
590  case FileJob::OpticalCheck:
591  failure_type = tr("Data verification failed");
592  break;
593  }
594  QString failure_str = QString(tr("%1: %2")).arg(failure_type).arg(err);
595  d.setTitle(failure_str);
596  QWidget *detailsw = new QWidget(&d);
