150 void HotZone::addButton(QString buttonName, QString actionStr, bool lastAddedButton)
151 {
152  m_pushButtonList->addButton(buttonName, actionStr, lastAddedButton);
153 
154  if (lastAddedButton) {
155  // m_ButtonsNum is used to record the number of QPushButton* , to calculate m_ButtonColumnBottomDistance.

178 QPixmap HotZone::getPixmap(const QString &file)
179 {
180  const qreal devicePixelRatio = devicePixelRatioF();
181 
182  QPixmap pixmap;
183  QImageReader reader;
184  reader.setFileName(file);
185  if (reader.canRead()) {
186  reader.setScaledSize(reader.size() * devicePixelRatio);
187  pixmap = QPixmap::fromImage(reader.read());
188  pixmap.setDevicePixelRatio(devicePixelRatio);
189  } else {
190  pixmap.load(file);
191  }
192 
193  return pixmap;

293 void HotZone::onMouseHasLeaved()
294 {
295  QPoint p = m_ButtonColumnItem->mapFromGlobal(QCursor::pos());
296  if (m_ButtonColumnItem->rect().contains(p)) {
297  if (m_pushButtonList->buttonClickStatus()) {
298  m_pushButtonList->setButtonClickStatus(false);
299  updateLeavePic();
300  } else {
301  m_hasAnimationRun = false;
302  updateEnterPic();
303  }
304  } else {
305  m_ButtonColumnItem->setHidden(true);
306  m_hasAnimationRun = true;
307  m_pushButtonList->setButtonClickStatus(false);
308 
309  updateLeavePic();
310  }
311 
312  m_mouseIsEnterFromButtonColumnAfterButtonClick = false;
313 }
