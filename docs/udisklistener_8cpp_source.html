232 bool UDiskListener::renameFile(const QSharedPointer<DFMRenameEvent> &event) const
233 {
234  const DUrl &oldUrl = event->fromUrl();
235  const DUrl &newUrl = event->toUrl();
236 
237  const QSharedPointer<DFMCreateFileInfoEvent> e(new DFMCreateFileInfoEvent(nullptr, oldUrl));
238  const DAbstractFileInfoPointer &oldDevicePointer = UDiskListener::createFileInfo(e);
239 
240  DAbstractFileInfo *info = oldDevicePointer.data();
241  UDiskDeviceInfo *udiskInfo = dynamic_cast<UDiskDeviceInfo *>(info);
242  QString devicePath = udiskInfo->getDBusPath();
243  QUrlQuery query(newUrl);
244  QString newName = query.queryItemValue("new_name");
245  DBlockDevice *partition = DDiskManager::createBlockDevice(devicePath, nullptr);
246 
247  if (!partition) {
248  return false;
249  }
250 
251  // set new label name
252  partition->setLabel(newName, {});
253 
254  // check if we got error
255  QDBusError err = DDiskManager::lastError();
256  qDebug() << err.type();
257  switch (err.type()) {
258  case QDBusError::NoReply:
259  // blumia: user doesn't do action and caused a dbus message reply timeout error (org.freedesktop.DBus.Error.NoReply).
260  // just return it and don't show any error message dialog.
261  // notice that don't be confused with another timeout-releated error type (org.freedesktop.DBus.Error.Timeout
262  // and org.freedesktop.DBus.Error.TimedOut), the `Timeout` one *possibly* means a socket ETIMEDOUT error
263  // and the `Timedout` one means certain timeout errors.
264  // see also: https://dbus.freedesktop.org/doc/api/html/group__DBusProtocol.html
265  // https://www.freedesktop.org/wiki/Software/DBusBindingErrors/ (¯\_(ツ)_/¯)
266  return false;
267  case QDBusError::NoError:
268  // blumia: when renamed, a DBus signal will be sent by udisks2, and it will trigger `fileSystemDeviceIdLabelChanged()`
269  // we don't need do rename here so just return true.
270  return true;
271  default:
272  dialogManager->showErrorDialog(tr("Failed to rename the label"), err.message());
273  return false;
274  }
275 
276  return false;
277 }

394 UDiskDeviceInfoPointer UDiskListener::getDeviceByMountPoint(const QString &mountPoint)
395 {
396  for (int i = 0; i < m_list.size(); i++) {
397  UDiskDeviceInfoPointer info = m_list.at(i);
398  if (info && !info->getMountPoint().isEmpty()) {
399  if (DUrl(info->getMountPoint()) == DUrl(mountPoint)) {
400  return info;
401  }
402 
403  // TODO: this is a polyfill since getting GFileInfo from network devices and mounted
404  // devices (both provided by gio) will return different uri (case insensitive
405  // when getting uri from network device, or say, samba/smb uri).
406  // this should be fixed or report to upstream.
407  if (mountPoint.startsWith("smb://")) {
408  QString source = mountPoint;
409  QString target = info->getMountPoint();
410 
411  if (source.endsWith("/")) {
412  source.chop(1);
413  }
414 
415  source = QUrl::fromPercentEncoding(source.toUtf8());
416 
417  if (target.endsWith("/")) {
418  target.chop(1);
419  }
420 
421  target = QUrl::fromPercentEncoding(target.toUtf8());
422 
423  if (QString::compare(source, target, Qt::CaseInsensitive) == 0) {
424  return info;
425  }
426  }
427  }
428  }
429  return UDiskDeviceInfoPointer();
430 }
