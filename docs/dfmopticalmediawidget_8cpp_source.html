96  // 1、获取暂存区内文件列表信息，去除与当前光盘中有交集的部分（当前 isomaster 库不提供覆盖写入的选项，后或可优化）
97  QDir dirMnt(d->strMntPath);

99  {
100  statusInfo.bReadyToBurn = false;
101  qWarning() << "Mount points doesn't exist: " << d->strMntPath;

103  }
104  // 如果放入空盘是没有挂载点的，此时给QDir传入空的path将导致QDir获取到的是程序运行目录的Dir，之后的去重会产生不正常的结果
105  QFileInfoList lstFilesOnDisc = d->strMntPath.isEmpty() ? QFileInfoList() : dirMnt.entryInfoList(filter);

107  QDir dirStage(urlOfStage.path());
108  if (!dirStage.exists())
109  {
110  statusInfo.bReadyToBurn = false;
111  return;
112  }
113  QFileInfoList lstFilesInStage = dirStage.entryInfoList(filter);
114  if (lstFilesInStage.count() == 0)
115  {
116  statusInfo.bReadyToBurn = false;
117  dialogManager->showMessageDialog(DialogManager::msgWarn, tr("No files to burn"));
118  return;
119  }
120 
121  bool bDeletedValidFile = false; // 在点击进入光驱中文件夹时，因解决bug#27870时，在暂存区中手动创建了本不存在的目录
122  // 如果光盘挂载根目录与暂存区根目录中有同名文件或文件夹，则移除暂存区中的相关文件或文件夹；
123  for (QFileInfo fOn : lstFilesOnDisc)
124  {
125  for (QFileInfo fStage : lstFilesInStage) {

128

130  dirStage.remove(fStage.fileName());
131  else {
132  if (!bDeletedValidFile)

134  QDir(fStage.absoluteFilePath()).removeRecursively();
135  }
136  }
137  }

140  if (lstFilesInStage.count() == 0)

143  dialogManager->showMessageDialog(DialogManager::msgWarn, errTitle);
144  statusInfo.bReadyToBurn = false;

187 void DFMOpticalMediaWidget::setBurnCapacity(int status, QString strVolTag)
188 {
189  QFile burnCapacityFile(QString("%1/dde-file-manager.json").arg(DFMStandardPaths::location(DFMStandardPaths::ApplicationConfigPath)));
190  if (!burnCapacityFile.open(QIODevice::ReadOnly)) {
191  qDebug() << "Couldn't open dde-file-manager.json burnCapacityFile!";
192  return;
193  }
194  QByteArray burnCapacityData = burnCapacityFile.readAll();
195  burnCapacityFile.close();
196 
197  double burnTotalSize = DFMOpticalMediaWidget::g_mapCdStatusInfo[strVolTag].nTotal;
198  double burnUsedSize = DFMOpticalMediaWidget::g_mapCdStatusInfo[strVolTag].nUsage;
199 
200  QJsonObject rootObj;
201  QJsonObject tagItem;
202  QJsonObject burnItem;
203  burnItem.insert("BurnCapacityTotalSize", burnTotalSize); //光盘容量总大小字节
204  burnItem.insert("BurnCapacityUsedSize", burnUsedSize); //光盘容量已使用大小字节
205  burnItem.insert("BurnCapacityStatus", status); //光盘容量状态：0,光驱弹出状态 1,光驱弹入处于添加未挂载状态 2,光驱弹入处于添加后并挂载的状态
206  burnItem.insert("BurnCapacityExt", 0); //光盘容量扩展预留属性
207 
208  QJsonParseError parseJsonErr;
209  QJsonDocument jsonDoc(QJsonDocument::fromJson(burnCapacityData, &parseJsonErr));
210  if (!(parseJsonErr.error == QJsonParseError::NoError)) {
211  qDebug() << "decode json file error, create new json data！";
212  //第一次如果没有这个属性需要创建
213  tagItem[strVolTag] = burnItem;
214  } else {
215  rootObj = jsonDoc.object();
216  if (rootObj.contains("BurnCapacityAttribute")) {
217  tagItem = rootObj["BurnCapacityAttribute"].toObject();
218  }
219  tagItem[strVolTag] = burnItem;
220  }
221  rootObj["BurnCapacityAttribute"] = tagItem;
222  jsonDoc.setObject(rootObj);
223  QFile burnCapacityFile2(QString("%1/dde-file-manager.json").arg(DFMStandardPaths::location(DFMStandardPaths::ApplicationConfigPath)));
224  if (!burnCapacityFile2.open(QIODevice::WriteOnly)) {
225  qDebug() << "Couldn't open dde-file-manager.json burnCapacityFile2!";
226  return;
227  }
228  burnCapacityFile2.write(jsonDoc.toJson());
229  burnCapacityFile2.close();
230 }
