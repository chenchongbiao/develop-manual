113 void OnboardPlugin::invokedMenuItem(const QString &itemKey, const QString &menuId, const bool checked)
114 {
115  Q_UNUSED(itemKey)
116  Q_UNUSED(checked)
117 
118  if (menuId != "onboard-settings") return;
119  if(!m_startupState) {
120  QProcess *process = new QProcess;
121  connect(process,&QProcess::started, this, [ = ] {
122  m_startupState = true;
123  });
124 
125  connect(process, QOverload<int, QProcess::ExitStatus>::of(&QProcess::finished),
126  [ = ](int exitCode, QProcess::ExitStatus exitStatus){
127  Q_UNUSED(exitCode)
128  Q_UNUSED(exitStatus)
129 
130  m_startupState = false;
131  process->close();
132  process->deleteLater();
133  });
134  process->start("onboard-settings");
135  }
136 
137  DBusDock DockInter("com.deepin.dde.daemon.Dock",
138  "/com/deepin/dde/daemon/Dock",
139  QDBusConnection::sessionBus(), this);
140 
141  for (auto entry : DockInter.entries()) {
142  DockEntryInter AppInter("com.deepin.dde.daemon.Dock",
143  entry.path(),
144  QDBusConnection::sessionBus(), this);
145  if(AppInter.name() == "Onboard-Settings" && !AppInter.isActive()) {
146  AppInter.Activate(0);
147  break;
148  }
149  }
150 }

193 void OnboardPlugin::refreshPluginItemsVisible()
194 {
