104 void AdapterWidget::setAdapter(const Adapter *adapter)
105 {
106  connect(adapter, &Adapter::nameChanged, m_titleEdit, &TitleEdit::setTitle, Qt::QueuedConnection);
107  connect(adapter, &Adapter::deviceAdded, this, &AdapterWidget::addDevice, Qt::QueuedConnection);
108  connect(adapter, &Adapter::deviceRemoved, this, &AdapterWidget::removeDevice, Qt::QueuedConnection);
109  connect(adapter, &Adapter::poweredChanged, m_switch, &SwitchWidget::setChecked, Qt::QueuedConnection);
110 
111  m_switch->blockSignals(true);
112  m_titleEdit->setTitle(adapter->name());
113  m_switch->setChecked(adapter->powered());
114  m_tip->setVisible(!m_switch->checked());
115  m_switch->blockSignals(false);
116 
117  blockSignals(true);
118  toggleSwitch(adapter->powered());
119  blockSignals(false);
120 
121  for (const Device *device : adapter->devices()) {
122  addDevice(device);
123  }
124 }

134 void AdapterWidget::addDevice(const Device *device)
135 {
136  DeviceSettingsItem *w = new DeviceSettingsItem(device);
137 
138  auto CategoryDevice = [this, w] (const bool paired) {
139  if (paired) {
140  m_myDevices << w;
141  m_myDevicesGroup->appendItem(w);
142  } else {
143  m_otherDevicesGroup->appendItem(w);
144  }
145 
146  m_myDevicesGroup->setVisible(!m_myDevices.isEmpty() && m_switch->checked());
147  };
148  CategoryDevice(device->paired());
149 
150  connect(w, &DeviceSettingsItem::requestConnectDevice, this, &AdapterWidget::requestConnectDevice);
151  connect(device, &Device::pairedChanged, CategoryDevice);
152  connect(w, &DeviceSettingsItem::requestShowDetail, [this] (const Device *device) {
153  Q_EMIT requestShowDetail(m_adapter, device);
154  });
155 
156  m_deviceLists << w;
157 }

159 void AdapterWidget::removeDevice(const QString &deviceId)
160 {
161  for (auto it = m_deviceLists.begin(); it != m_deviceLists.end(); ++it) {
162  if ((*it)->device()->id() == deviceId) {
163  DeviceSettingsItem *item = *it;
164  m_myDevices.removeOne(item);
165  m_myDevicesGroup->removeItem(item);
166  m_otherDevicesGroup->removeItem(item);
167  m_deviceLists.erase(it);
168  item->deleteLater();
169  break;
170  }
171  }
172 
173  if (m_myDevices.isEmpty()) {
174  m_myDevicesGroup->hide();
175  }
176 }
