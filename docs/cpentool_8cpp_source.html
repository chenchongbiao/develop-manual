37 void CPenTool::toolCreatItemUpdate(IDrawTool::CDrawToolEvent *event, ITERecordInfo *pInfo)
38 {
39  if (pInfo != nullptr) {
40  CGraphicsPenItem *pPenIem = dynamic_cast<CGraphicsPenItem *>(pInfo->businessItem);
41  if (nullptr != pPenIem) {
42  QPointF pointMouse = event->pos();
43  bool shiftKeyPress = event->keyboardModifiers() & Qt::ShiftModifier;
44  pPenIem->updatePenPath(pPenIem->mapFromScene(pointMouse), shiftKeyPress);
45  event->setAccepted(true);
46 
47  QPixmap &pix = event->view()->cachedPixmap();
48  QPainter painter(&pix);
49  QPen p = pPenIem->pen();
50  qreal penW = p.widthF() * event->view()->getScale();
51  p.setWidthF(penW);
52  //修复调节画笔粗细较粗,不透明度较小时绘制画笔图元,绘制过程中画笔中有很多点
53  QColor updateQColor = p.color();
54  updateQColor.setAlpha(255);
55  p.setColor(updateQColor);
56 
57  painter.setPen(p);
58  if (event->keyboardModifiers() != Qt::ShiftModifier) {
59  auto activeLineIt = _activePaintedLines.find(event->uuid());
60  if (activeLineIt != _activePaintedLines.end()) {
61  QLineF activeLine = activeLineIt.value();
62  if (!activeLine.isNull()) {
63  painter.drawLine(activeLine);
64  _activePaintedLines.erase(activeLineIt);
65  }
66  }
67  painter.drawLine(event->view()->mapFromScene(pInfo->_prePos), event->view()->mapFromScene(event->pos()));
68  } else {
69  auto activeLineIt = _activePaintedLines.find(event->uuid());
70  if (activeLineIt == _activePaintedLines.end()) {
71  QLineF activeLine;
72  activeLine.setP1(event->view()->mapFromScene(pInfo->_prePos));
73  activeLine.setP2(event->view()->mapFromScene(event->pos()));
74  _activePaintedLines.insert(event->uuid(), activeLine);
75  } else {
76  QLineF &activeLine = activeLineIt.value();
77  activeLine.setP2(event->view()->mapFromScene(event->pos()));
78  }
79  }
80  event->view()->update();
81  event->view()->viewport()->update();
82  }
83  }
84 }

86 void CPenTool::toolCreatItemFinish(IDrawTool::CDrawToolEvent *event, ITERecordInfo *pInfo)
87 {
88  if (pInfo != nullptr) {
89  CGraphicsPenItem *pPenIem = dynamic_cast<CGraphicsPenItem *>(pInfo->businessItem);
90  if (nullptr != pPenIem) {
91  if (!pInfo->hasMoved()) {
92  event->scene()->removeCItem(pPenIem);
93  delete pPenIem;
94  pInfo->businessItem = nullptr;
95  } else {
96  pPenIem->drawComplete();
97  if (pPenIem->scene() == nullptr) {
98  pPenIem->drawScene()->addCItem(pPenIem);
99  }
100  // [BUG 28087] 所绘制的画笔未默认呈现选中状态
101  //pPenIem->setSelected(true);
102  pPenIem->setDrawFlag(false);
103  }
104  }
105  //1.取消缓存，恢复到正常绘制
106  event->view()->setCacheEnable(false);
107  }
108 
109  IDrawTool::toolCreatItemFinish(event, pInfo);
110 }

112 void CPenTool::drawMore(QPainter *painter, const QRectF &rect, CDrawScene *scene)
113 {
114  Q_UNUSED(rect)
115  Q_UNUSED(scene)
116  for (auto it = _allITERecordInfo.begin(); it != _allITERecordInfo.end(); ++it) {
117  ITERecordInfo &pInfo = it.value();
118  const CDrawToolEvent &curEvnt = pInfo._curEvent;
119  CGraphicsPenItem *penItem = dynamic_cast<CGraphicsPenItem *>(pInfo.businessItem);
120  if (penItem != nullptr) {
121  QPen p = penItem->pen();
122  //qreal penWMin = qMax(p.widthF(), 1.0);
123  qreal penW = p.widthF() * scene->drawView()->getScale();
124  p.setWidthF(penW);
125  painter->setPen(p);
126  if (curEvnt.keyboardModifiers() & Qt::ShiftModifier) {
127 
128  //要绘制悬而未决的直线
129  for (auto b = _activePaintedLines.begin(); b != _activePaintedLines.end(); ++b) {
130  QLineF activeLine = b.value();
131  if (!activeLine.isNull())
132  painter->drawLine(activeLine);
133  }
134  }
135 
136  painter->save();
137  if (soildRing == penItem->getPenStartType() || soildArrow == penItem->getPenStartType()) {
138  painter->setBrush(penItem->pen().color());
139  painter->setPen(penItem->pen());
140  }
141 
142  if (penItem->getPenStartType() != noneLine)
143  painter->drawPath(scene->drawView()->mapFromScene(penItem->mapToScene(penItem->getPenStartpath())));
144  painter->restore();
145 
146 
147  if (soildRing == penItem->getPenEndType() || soildArrow == penItem->getPenEndType()) {
148  painter->setBrush(penItem->pen().color());
149  painter->setPen(penItem->pen());
150  }
151 
152  if (penItem->getPenEndType() != noneLine)
153  painter->drawPath(scene->drawView()->mapFromScene(penItem->mapToScene(penItem->getPenEndpath())));
154  }
155  }
156 }

158 CGraphicsItem *CPenTool::creatItem(CDrawToolEvent *event, ITERecordInfo *pInfo)
159 {
160  Q_UNUSED(pInfo)
161  if ((event->eventType() == CDrawToolEvent::EMouseEvent && event->mouseButtons() == Qt::LeftButton)
162  || event->eventType() == CDrawToolEvent::ETouchEvent) {
163 
164  // 连续画线之前清除之前的选中图元
165  event->scene()->clearSelectGroup();
166 
167  //启动缓冲绘制(会生成一张位图了替代原先的绘制)
168  event->view()->setCacheEnable(true);
169 
170  CGraphicsPenItem *pPenItem = new CGraphicsPenItem(event->pos());
171  pPenItem->setDrawFlag(true);
172 
173  CGraphicsView *pView = event->scene()->drawView();
174  QPen pen = pView->getDrawParam()->getPen();
175  pPenItem->setPen(pen);
176  pPenItem->setBrush(pView->getDrawParam()->getBrush());
177  pPenItem->setPenStartType(pView->getDrawParam()->getPenStartType());
178  pPenItem->setPenEndType(pView->getDrawParam()->getPenEndType());
179  event->scene()->addCItem(pPenItem);
180 
181  return pPenItem;
182  }
183  return nullptr;
184 }
