131 void CustomContent::setBottomTip(ShortcutInfo *conflict)
132 {
133  m_conflict = conflict;
134  if (conflict) {
135  QString accels = conflict->accels;
136  accels = accels.replace("<", "");
137  accels = accels.replace(">", "+");
138  accels = accels.replace("_L", "");
139  accels = accels.replace("_R", "");
140  accels = accels.replace("Control", "Ctrl");
141 
142  QString str = tr("This shortcut conflicts with %1, click on Add to make this shortcut effective immediately")
143  .arg(QString("<span style=\"color: rgba(255, 90, 90, 1);\">%1 %2</span>").arg(conflict->name).arg(QString("[%1]").arg(accels)));
144  m_bottomTip->setText(str);
145  m_bottomTip->show();
146  } else {
147  m_bottomTip->clear();
148  m_bottomTip->hide();
149  }
150 }

152 void CustomContent::onShortcut()
153 {
154  m_shortCutNameEdit->setAlert(m_shortCutNameEdit->text().isEmpty());
155  m_shortCutCmdEdit->setAlert(m_shortCutCmdEdit->lineEdit()->text().isEmpty());
156  m_shortcut->setAlert(m_shortcut->text().isEmpty());
157 
158  if (m_shortcut->text().isEmpty() || m_shortCutCmdEdit->lineEdit()->text().isEmpty() || m_shortCutNameEdit->text().isEmpty()) {
159  return;
160  }
161 
162  if (m_conflict)
163  Q_EMIT requestForceSubs(m_conflict);
164 
165  Q_EMIT requestAddKey(m_shortCutNameEdit->text(), m_shortCutCmdEdit->text(), m_shortcut->text());
166  Q_EMIT back();
167 }

169 void CustomContent::keyEvent(bool press, const QString &shortcut)
170 {
171  if (!press) {
172  if (shortcut.isEmpty() || shortcut == "BackSpace" || shortcut == "Delete") {
173  m_shortcut->setShortcut("");
174  setBottomTip(nullptr);
175  return;
176  }
177 
178  // check conflict
179  ShortcutInfo *conflict = m_model->getInfo(shortcut);
180  setBottomTip(conflict);
181  }
182 
183  m_shortcut->setShortcut(shortcut);
184 }
