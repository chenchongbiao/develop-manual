137 void BootWidget::setDefaultEntry(const QString &value)
138 {
139  m_defaultEntry = value;
140 
141  blockSignals(true);
142  int row_count = m_bootItemModel->rowCount();
143  for (int i = 0; i < row_count; ++i) {
144  QStandardItem *item = m_bootItemModel->item(i, 0);
145  if (item->text() == value) {
146  m_curSelectedIndex = item->index();
147  item->setCheckState(Qt::CheckState::Checked);
148  } else {
149  item->setCheckState(Qt::CheckState::Unchecked);
150  }
151  }
152  blockSignals(false);
153 }

188 void BootWidget::setEntryList(const QStringList &list)
189 {
190  m_bootItemModel = new QStandardItemModel(this);
191  m_bootList->setModel(m_bootItemModel);
192 
193  for (int i = 0; i < list.count(); i++) {
194  const QString entry = list.at(i);
195 
196  DStandardItem *item = new DStandardItem();
197  item->setText(entry);
198  item->setCheckable(false); // for Bug 2449
199  item->setData(VListViewItemMargin, Dtk::MarginsRole);
200 
201  m_bootItemModel->appendRow(item);
202 
203  if (m_defaultEntry == entry) {
204  m_curSelectedIndex = item->index();
205  item->setCheckState(Qt::CheckState::Checked);
206  } else {
207  item->setCheckState(Qt::CheckState::Unchecked);
208  }
209  }
210 
211  setBootList();
212 }

225 {
226  QString curText = curIndex.data().toString();

228  return;
229 
230  // 获取当前被选项
231  QString selectedText = m_curSelectedIndex.data().toString();
232  if (selectedText.isEmpty())

235  Q_EMIT defaultEntry(curText);
