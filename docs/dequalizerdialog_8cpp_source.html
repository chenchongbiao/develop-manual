436 void DequalizerDialog::SliderOneEntry(DSlider *slider, int value)
437 {
438  this->selectSlider(slider, QString::number(value));
439  if (slider == slider_pre) {
440  Q_EMIT this->setEqualizerpre(value);

442  int bandIndex = -1;
443  if (slider == slider_60) {
444  bandIndex = 0;
445  } else if (slider == slider_170) {
446  bandIndex = 1;

448  bandIndex = 2;
449  } else if (slider == slider_600) {
450  bandIndex = 3;

452  bandIndex = 4;
453  } else if (slider == slider_3K) {

462  bandIndex = 9;

465  return;
