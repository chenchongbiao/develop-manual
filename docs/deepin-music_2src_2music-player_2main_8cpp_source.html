186 bool checkOnly()
187 {
188  //single
189  QString userName = QDir::homePath().section("/", -1, -1);
190  std::string path = ("/home/" + userName + "/.cache/deepin/deepin-music/").toStdString();
191  QDir tdir(path.c_str());
192  if (!tdir.exists()) {
193  bool ret = tdir.mkpath(path.c_str());
194  qDebug() << ret ;
195  }
196 
197  path += "single";
198  int fd = open(path.c_str(), O_WRONLY | O_CREAT, 0644);
199  int flock = lockf(fd, F_TLOCK, 0);
200 
201  if (fd == -1) {
202  perror("open lockfile/n");
203  return false;
204  }
205  if (flock == -1) {
206  perror("lock file error/n");
207  return false;
208  }
209  return true;
210 }
