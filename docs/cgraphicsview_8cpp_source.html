177 void CGraphicsView::scale(qreal scale, EScaleCenter center, const QPoint &viewPos)
178 {
179  qreal multiple = scale / m_scale;
180 
181  QPoint centerViewPos = viewPos;
182  switch (center) {
183  case EViewCenter:
184  centerViewPos = viewport()->rect().center();
185  break;
186  case ESceneCenter:
187  centerViewPos = mapFromScene(sceneRect().center());
188  break;
189  case EMousePos:
190  centerViewPos = viewport()->mapFromGlobal(QCursor::pos());
191  break;
192  default:
193  centerViewPos = viewPos;
194  break;
195  }
196  scaleWithCenter(multiple, centerViewPos);

204 void CGraphicsView::scaleWithCenter(qreal factor, const QPoint viewPos)
205 {
206  qreal wantedTotalScaled = m_scale * factor;
207 
208  if (wantedTotalScaled < 0.1) {
209  wantedTotalScaled = 0.1;
210  } else if (wantedTotalScaled > 20) {
211  wantedTotalScaled = 20;
212  }
213  factor = wantedTotalScaled / m_scale;
214 
215  //最最最完美的方案！！！
216  QPoint centerViewPos = viewPos.isNull() ? viewport()->mapFromGlobal(QCursor::pos()) : viewPos;
217 
218  QPointF targetScenePos = mapToScene(centerViewPos);
219  ViewportAnchor oldAnchor = this->transformationAnchor();
220  setTransformationAnchor(QGraphicsView::NoAnchor);
221 
222  QTransform matrix = transform();
223  matrix.translate(targetScenePos.x(), targetScenePos.y())
224  .scale(factor, factor)
225  .translate(-targetScenePos.x(), -targetScenePos.y());
226 
227  setTransform(matrix);
228 
229  setTransformationAnchor(oldAnchor);
230 
231  //保存缩放值
232  m_scale *= factor;
233  getDrawParam()->setScale(m_scale);
234  emit signalSetScale(m_scale);
235 
236  if (drawScene() != nullptr)
237  drawScene()->selectGroup()->updateBoundingRect();
238 }

240 void CGraphicsView::wheelEvent(QWheelEvent *event)
241 {
242  Q_UNUSED(event)
243  if (activeProxItem() != nullptr) {
244  auto proxyItem = activeProxItem();
245  QPointF scenePos = this->mapToScene(this->viewport()->mapFromGlobal(QCursor::pos()));
246  if (proxyItem->sceneBoundingRect().contains(scenePos)) {
247  QPointF p = proxyItem->mapFromScene(scenePos);
248  QGraphicsSceneWheelEvent gsEvent(QEvent::GraphicsSceneWheel);
249  gsEvent.setPos(p);
250  gsEvent.setScenePos(scenePos);
251  gsEvent.setScreenPos(QCursor::pos());
252  gsEvent.setButtons(event->buttons());
253  gsEvent.setModifiers(event->modifiers());
254  gsEvent.setDelta(event->delta());
255  gsEvent.setOrientation(event->orientation());
256  drawScene()->sendEvent(proxyItem, &gsEvent);
257  return;
258  }
259  }
260  CGraphicsView *pCurView = this;
261  int delayValue = event->delta();
262  if (pCurView != nullptr) {
263  if (event->modifiers() == Qt::NoModifier) {
264  //滚动view的垂直scrollbar
265  int curValue = pCurView->verticalScrollBar()->value();
266  pCurView->verticalScrollBar()->setValue(curValue - delayValue / 12);
267  } else if (event->modifiers() == Qt::ShiftModifier) {
268  //滚动view的水平scrollbar
269  int curValue = pCurView->horizontalScrollBar()->value();
270  pCurView->horizontalScrollBar()->setValue(curValue - delayValue / 12);
271  } else if (event->modifiers()& Qt::ControlModifier) {
272  //如果按住CTRL那么就是放大缩小
273  if (event->delta() > 0) {
274  pCurView->zoomOut(CGraphicsView::EMousePos);
275  } else {
276  pCurView->zoomIn(CGraphicsView::EMousePos);
277  }
278  }
279  }

281 }

502  auto currScene = dynamic_cast<CDrawScene *>(scene());
503  QList<CGraphicsItem *> allitems = currScene->selectGroup()->items();
504 
505  // [1] 图元个数大于3个才可以进行对齐
506  if (allitems.size() < 3)

508  return ;
509  }
510 
511  // [2] 对图元X进行从小到大排序
512  qSort(allitems.begin(), allitems.end(), yValueSortDES);
513 
514  // [3] 统计所有图元占有的高度
515  qreal sum_items_height = 0;
516  for (int i = 0; i < allitems.size(); i++)
517  {
518  sum_items_height += allitems.at(i)->sceneBoundingRect().height();
519  }
520 
521  // [3] 计算每两个之间的间隔距离
522  auto curScene = dynamic_cast<CDrawScene *>(scene());
523  QRectF scence_BR = curScene->selectGroup()->sceneBoundingRect();
524 
525  // [4] 用于记录保存图元的位置，便于撤销和返回
526  QMap<CGraphicsItem *, QPointF> startPos;
527  QMap<CGraphicsItem *, QPointF> endPos;
528 
529  if (sum_items_height > scence_BR.height())
530  {
531  // [5] 按照相邻进行移动位置
532  for (int i = 1; i < allitems.size(); i++) {
533  startPos.insert(allitems.at(i), allitems.at(i)->sceneBoundingRect().topLeft());
534  QPointF endPoint(allitems.at(i)->sceneBoundingRect().left()
535  , allitems.at(i - 1)->sceneBoundingRect().bottom());
536  allitems.at(i)->move(allitems.at(i)->sceneBoundingRect().topLeft(), endPoint);
537  endPos.insert(allitems.at(i), allitems.at(i)->sceneBoundingRect().topLeft());
538  }
539  } else
540  {
541  qreal space_height = (scence_BR.height() - sum_items_height)
542  / (allitems.size() - 1);
543 
544  // [6] 按照y值进行移动位置
545  for (int i = 1; i < allitems.size() - 1; i++) {
546  startPos.insert(allitems.at(i), allitems.at(i)->sceneBoundingRect().topLeft());
547  QPointF endPoint(allitems.at(i)->sceneBoundingRect().left()
