61 int main(int argc, char *argv[])
62 {
63  DGuiApplicationHelper::setUseInactiveColorGroup(false);
64  DGuiApplicationHelper::setColorCompositingEnabled(true);
65  DApplication *app = DApplication::globalApplication(argc, argv);
66  app->setQuitOnLastWindowClosed(false);
67  app->setOrganizationName("deepin");
68  app->setApplicationName("dde-launcher");
69  app->setApplicationVersion("3.0");
70  app->loadTranslator();
71  app->setAttribute(Qt::AA_UseHighDpiPixmaps);
72 
73  DLogManager::registerConsoleAppender();
74  // 加载 Accessible 插件 测试使用 用sniff查看效果
75  QAccessible::installFactory(accessibleFactory);
76 
77 #ifndef QT_DEBUG
78  DLogManager::registerFileAppender();
79 #endif
80 
81  bool quit = !app->setSingleInstance(QString("dde-launcher_%1").arg(getuid()));
82 
83  QCommandLineOption showOption(QStringList() << "s" << "show", "show launcher(hide for default.)");
84  QCommandLineOption toggleOption(QStringList() << "t" << "toggle", "toggle launcher visible.");
85  QCommandLineOption dumpPresetOrder(QStringList() << "d" << "dump", "dump user-specificed preset order list and exit.");
86 
87  QCommandLineParser cmdParser;
88  cmdParser.setApplicationDescription("DDE Launcher");
89  cmdParser.addHelpOption();
90  cmdParser.addVersionOption();
91  cmdParser.addOption(showOption);
92  cmdParser.addOption(toggleOption);
93  cmdParser.addOption(dumpPresetOrder);
94  cmdParser.process(*app);
95 
96  if (cmdParser.isSet(dumpPresetOrder))
97  {
98  quit = true;
99  dump_user_apss_preset_order_list();
100  }
101 
102  if (quit)
103  {
104  DBusLauncherFrame launcherFrame;
105 
106  do {
107  if (!launcherFrame.isValid())
108  break;
109 
110  if (cmdParser.isSet(toggleOption))
111  launcherFrame.Toggle();
112  else if (cmdParser.isSet(showOption))
113  launcherFrame.Show();
114 
115  } while (false);
116 
117  return 0;
118  }
119 
120  // INFO: what's this?
121  setlocale(LC_ALL, "");
122 
123  LauncherSys launcher;
124  DBusLauncherService service(&launcher);
125  Q_UNUSED(service);
126  QDBusConnection connection = QDBusConnection::sessionBus();
127  if (!connection.registerService("com.deepin.dde.Launcher") ||
128  !connection.registerObject("/com/deepin/dde/Launcher", &launcher))
129  qWarning() << "register dbus service failed";
130 
131 #ifndef QT_DEBUG
132  if (cmdParser.isSet(showOption))
133 #endif
134  QTimer::singleShot(1, &launcher, &LauncherSys::showLauncher);
135 
136  return app->exec();
137 }
