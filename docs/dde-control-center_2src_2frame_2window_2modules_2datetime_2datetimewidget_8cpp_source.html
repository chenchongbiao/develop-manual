57 void DatetimeWidget::init()
58 {
59  QList<QPair<QString, QString>> menuIconText;
60  menuIconText = {
61  { "dcc_time_zone", tr("Timezone List")},
62  //~ contents_path /datetime/Time Settings
63  { "dcc_setting", tr("Time Settings")} ,
64  //~ contents_path /datetime/Format Settings
65  { "dcc_time_format", tr("Time Format")}
66  };
67 
68  DStandardItem *item = nullptr;
69  auto model = new QStandardItemModel(this);
70  for (auto it = menuIconText.cbegin(); it != menuIconText.cend(); ++it) {
71  item = new DStandardItem(QIcon::fromTheme(it->first), it->second);
72  item->setData(VListViewItemMargin, Dtk::MarginsRole);
73  model->appendRow(item);
74  }
75 
76  m_itemList.append({menuIconText.at(0).first, menuIconText.at(0).second,QMetaMethod::fromSignal(&DatetimeWidget::requestTimezoneList)});
77  m_itemList.append({menuIconText.at(1).first, menuIconText.at(1).second,QMetaMethod::fromSignal(&DatetimeWidget::requestTimeSetting)});
78  m_itemList.append({menuIconText.at(2).first, menuIconText.at(2).second,QMetaMethod::fromSignal(&DatetimeWidget::requestFormatSetting)});
79 
80  if(InsertPlugin::instance()->needPushPlugin("datetime"))
81  InsertPlugin::instance()->pushPlugin(model,m_itemList);
82 
83  m_listview->setAccessibleName("List_datetimesettingList");
84  m_listview->setFrameShape(QFrame::NoFrame);
85  m_listview->setModel(model);
86  m_listview->setEditTriggers(QAbstractItemView::NoEditTriggers);
87  m_listview->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Expanding);
88  m_listview->setCurrentIndex(model->index(0, 0));
89  m_listview->setItemSpacing(10);
90  m_listview->setViewportMargins(0, 0, 0, 0);
91  m_listview->setIconSize(ListViweIconSize);
92 
93  //default open 24 hour type : set hourTypeSwitch(true) , then set ClockItem TimeHourType
94  //~ contents_path /datetime/Timezone List
95  m_hourTypeSwitch = new SwitchWidget(tr("24-hour Time"));
96  m_hourTypeSwitch->setContentsMargins(0, 0, 0, 0);
97  m_hourTypeSwitch->addBackground();
98  m_clockItem->setTimeHourType(m_hourTypeSwitch->checked());
99  m_clockItem->setContentsMargins(0, 0, 0, 0);
100 
101  QVBoxLayout *layout = new QVBoxLayout;
102  layout->setContentsMargins(10, 0, 10, 0);
103  layout->setSpacing(10);
104  layout->addWidget(m_clockItem);
105  layout->addWidget(m_hourTypeSwitch);
106  layout->addWidget(m_listview);
107  setLayout(layout);
108 
109  connect(m_listview, &DListView::clicked, this, &DatetimeWidget::onItemClicked);
110  connect(m_listview, &DListView::activated, m_listview, &QListView::clicked);
111  // true : 24 hour type , false : 12 hour type ; All use the system time can recive DatetimeWidget::requestSetHourType signal
112  connect(m_hourTypeSwitch, &SwitchWidget::checkedChanged, this, &DatetimeWidget::requestSetHourType);
113 }
