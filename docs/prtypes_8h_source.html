63 /***********************************************************************
64 ** MACROS: PR_EXTERN
65 ** PR_IMPLEMENT
66 ** DESCRIPTION:
67 ** These are only for externally visible routines and globals. For
68 ** internal routines, just use "extern" for type checking and that
69 ** will not export internal cross-file or forward-declared symbols.
70 ** Define a macro for declaring procedures return types. We use this to
71 ** deal with windoze specific type hackery for DLL definitions. Use
72 ** PR_EXTERN when the prototype for the method is declared. Use
73 ** PR_IMPLEMENT for the implementation of the method.
74 **
75 ** Example:
76 ** in dowhim.h
77 ** PR_EXTERN( void ) DoWhatIMean( void );
78 ** in dowhim.c
79 ** PR_IMPLEMENT( void ) DoWhatIMean( void ) { return; }

81 **

330 /************************************************************************
331 ** TYPES: PRUint64
332 ** PRInt64
333 ** DESCRIPTION:
334 ** The int64 types are known to be 64 bits each. Care must be used when
335 ** declaring variables of type PRUint64 or PRInt64. Different hardware
336 ** architectures and even different compilers have varying support for
337 ** 64 bit values. The only guaranteed portability requires the use of
338 ** the LL_ macros (see prlong.h).
339 ************************************************************************/
340 #ifdef HAVE_LONG_LONG
341 /* Keep this in sync with prlong.h. */
342 /*
343  * On 64-bit Mac OS X, uint64 needs to be defined as unsigned long long to
344  * match uint64_t, otherwise our uint64 typedef conflicts with the uint64
345  * typedef in cssmconfig.h, which CoreServices.h includes indirectly.
346  */
347 #if PR_BYTES_PER_LONG == 8 && !defined(__APPLE__)
348 typedef long PRInt64;
349 typedef unsigned long PRUint64;
350 #elif defined(WIN32) && !defined(__GNUC__)
351 typedef __int64 PRInt64;
352 typedef unsigned __int64 PRUint64;
353 #else
354 typedef long long PRInt64;
355 typedef unsigned long long PRUint64;
356 #endif /* PR_BYTES_PER_LONG == 8 */
357 #else /* !HAVE_LONG_LONG */
358 typedef struct {
359 #ifdef IS_LITTLE_ENDIAN
360  PRUint32 lo, hi;
361 #else
362  PRUint32 hi, lo;
363 #endif
364 } PRInt64;
365 typedef PRInt64 PRUint64;
366 #endif /* !HAVE_LONG_LONG */
