53 void DFMAdvanceSearchBar::onOptionChanged()
54 {
55  QMap<int, QVariant> formData;
56 
57  bool searchRangeChanged = formData[SEARCH_RANGE] != asbCombos[SEARCH_RANGE]->currentData();
58 
59  formData[SEARCH_RANGE] = asbCombos[SEARCH_RANGE]->currentData();
60  formData[FILE_TYPE] = asbCombos[FILE_TYPE]->currentData();
61  formData[SIZE_RANGE] = asbCombos[SIZE_RANGE]->currentData();
62  formData[DATE_RANGE] = asbCombos[DATE_RANGE]->currentData();
63  formData[ACCESS_DATE_RANGE] = asbCombos[ACCESS_DATE_RANGE]->currentData();
64  formData[CREATE_DATE_RANGE] = asbCombos[CREATE_DATE_RANGE]->currentData();
65 
66  bool triggerSearch = false;
67 
68  if (needSearchAgain) {
69  needSearchAgain = false;
70  triggerSearch = true;
71  }
72 
73  if (searchRangeChanged) {
74  triggerSearch = true;
75  }
76 
77  formData[TRIGGER_SEARCH] = QVariant::fromValue(triggerSearch);
78 
79  emit optionChanged(formData, allowUpdateView);
80 }
