155 QString UpdateItem::elidedChangelog() const
156 {
157  QString text = m_info.m_changelog;
158 
159  const QFontMetrics fm(m_appChangelog->fontMetrics());
160  const QRect rect(0, 0, m_appChangelog->width(), fontMetrics().height());
161  const int textFlag = Qt::AlignTop | Qt::AlignLeft | Qt::TextWordWrap;
162 
163  if (rect.contains(fm.boundingRect(rect, textFlag, text)))
164  return text;
165 
166  QString str(text + "...");
167 
168  while (true) {
169  if (str.size() < 4)
170  break;
171 
172  QRect boundingRect = fm.boundingRect(rect, textFlag, str);
173  if (rect.contains(boundingRect))
174  break;
175 
176  str.remove(str.size() - 4, 1);
177  }
178 
179  return str;
180 }

187 const QString UpdateItem::getExtendLog() const
188 {
189  const QFontMetrics fm(m_appChangelog->fontMetrics());
190  QString retLog = m_info.m_changelog;
191 
192  int fontWidth = fm.width(retLog);
193 
194  retLog.prepend("<style type=\"text/css\"></style>");
195 
196  const int labelWidth = m_appChangelog->width();
197  if (fontWidth <= labelWidth) {
198  return retLog.replace('\n', "<br/>");
199  }
200 
201  // 当前行字符串
202  QString curLine = m_info.m_changelog;
203  // 返回字符中的一个记录序号
204  int pos = QString("<style type=\"text/css\"></style>").size();
205  // 文本宽度计数
206  int widthCount = 0;
207  while (true) {
208  if (curLine.isEmpty()) {
209  break;
210  }
211  int logSize = curLine.size();
212  for (int i = 0; i < logSize; ++i) {
213  const QChar word = curLine.at(i);
214 
215  // '\n'转换行
216  if (word == '\n') {
217  curLine = curLine.right(curLine.size() - i -1);
218  if (curLine.isEmpty()) {
219  return retLog;
220  }
221  retLog.replace(pos, 1, "<br/>");
222  // 5个字符替换一个字符加4
223  pos += 4;
224  widthCount = 0;
225  break;
226  }
227 
228  widthCount += fm.width(word);
229 
230  // 字符行宽大于标签宽度添加换行
231  if (widthCount >= labelWidth) {
232  curLine = curLine.right(curLine.size() - i -1);
233  if (curLine.isEmpty()) {
234  return retLog;
235  }
236 
237  retLog.insert(pos, "<br/>");
238  // 插入5个字符加5
239  pos += 5;
240  widthCount = 0;
241  break;
242  }
243 
244  // 不足一行时返回
245  if (i == logSize - 1) {
246  return retLog;
247  }
248  ++pos;
249  }
250  }
251 
252  return retLog;
253 }

255 void UpdateItem::updateChangelogDisplay()
256 {

259  m_details->setVisible(needElide);

261  if (m_isLogExpand) {

263  } else {
264  m_appChangelog->setText(elidedText);
