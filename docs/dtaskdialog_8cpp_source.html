59 DFileCopyMoveJob::Action ErrorHandle::handleError(DFileCopyMoveJob *job, DFileCopyMoveJob::Error error,
60  const DAbstractFileInfoPointer sourceInfo,
61  const DAbstractFileInfoPointer targetInfo)
62 {
63  if (m_actionOfError != DFileCopyMoveJob::NoAction) {
64  DFileCopyMoveJob::Action action = m_actionOfError;
65  m_actionOfError = DFileCopyMoveJob::NoAction;
66 
67  return action;
68  }
69  switch (error) {
70  case DFileCopyMoveJob::FileExistsError:
71  case DFileCopyMoveJob::DirectoryExistsError: {
72  if (sourceInfo->fileUrl() == targetInfo->fileUrl() || DStorageInfo::isSameFile(sourceInfo->fileUrl().path(), targetInfo->fileUrl().path())) {
73  return DFileCopyMoveJob::CoexistAction;
74  }
75 
76  emit onConflict(sourceInfo->fileUrl(), targetInfo->fileUrl());
77  emit job->currentJobChanged(sourceInfo ? sourceInfo->fileUrl() : DUrl(), sourceInfo ? targetInfo->fileUrl() : DUrl(), true);
78  if (job->state() != DFileCopyMoveJob::PausedState) {
79  job->togglePause();
80  }
81  }
82  break;
83  case DFileCopyMoveJob::UnknowUrlError: {
84  DDialog dialog("Error", QCoreApplication::translate("DTaskDialog", "This action is not supported"));
85  dialog.setIcon(QIcon::fromTheme("dialog-error"), QSize(64, 64));
86  dialog.exec();
87  }
88  // fall-through
89  case DFileCopyMoveJob::UnknowError:
90  return DFileCopyMoveJob::CancelAction;
91  default:
92  emit job->currentJobChanged(sourceInfo ? sourceInfo->fileUrl() : DUrl(), targetInfo ? targetInfo->fileUrl() : DUrl(), true);
93  if (job->state() != DFileCopyMoveJob::PausedState) {
94  job->togglePause();
95  }
96  emit onError(job->errorString());
97  break;
98  }
99 
100  return DFileCopyMoveJob::NoAction;
101 }

103 static QString formatTime(int second)
104 {
105  quint8 s = second % 60;
106  quint8 m = static_cast<quint8>(second / 60);
107  quint8 h = m / 60;
108  quint8 d = h / 24;
109 
110  m = m % 60;
111  h = h % 24;
112 
113  QString time_string;
114 
115  if (d > 0) {
116  time_string.append(QString::number(d)).append(" d");
117  }
118 
119  if (h > 0) {
120  if (!time_string.isEmpty()) {
121  time_string.append(' ');
122  }
123 
124  time_string.append(QString::number(h)).append(" h");
125  }
126 
127  if (m > 0) {
128  if (!time_string.isEmpty()) {
129  time_string.append(' ');
130  }
131 
132  time_string.append(QString::number(m)).append(" m");
133  }
134 
135  if (s > 0 || time_string.isEmpty()) {
136  if (!time_string.isEmpty()) {
137  time_string.append(' ');
138  }
139 
140  time_string.append(QString::number(s)).append(" s");
141  }
142 
143  return time_string;
144 }

218 void DTaskDialog::addTask(const QMap<QString, QString> &jobDetail)
219 {

236  }
237  }
238  connect(wid, &DFMTaskWidget::heightChanged, this, &DTaskDialog::adjustSize);
239  connect(wid, &DFMTaskWidget::butonClicked, this, [this, wid, jobDetail](DFMTaskWidget::BUTTON bt) {
240  int code = -1;
241  if (bt == DFMTaskWidget::STOP) {
242  this->close();
243  handleTaskClose(jobDetail);
244  } else if (bt == DFMTaskWidget::SKIP) {
245  code = 2;
246  } else if (bt == DFMTaskWidget::REPLACE) {
247  code = 1;
248  } else if (bt == DFMTaskWidget::COEXIST) {
249  code = 0;
250  }
251 
252  if (code != -1) {
253  QMap<QString, QVariant> response;
254  response.insert("code", code);
255  response.insert("applyToAll", wid->getButton(DFMTaskWidget::CHECKBOX_NOASK)->isChecked());
256  emit conflictRepsonseConfirmed(jobDetail, response);
257  }
258  });
259 
260  // P.S. conflictHided and conflictShowed never used..we don't emit this signal
261  addTaskWidget(wid);
262  }

265 void DTaskDialog::blockShutdown()
266 {
267  if (m_reply.value().isValid()) {
268  return ;
269  }
270 
271  QDBusInterface loginManager("org.freedesktop.login1",
272  "/org/freedesktop/login1",
273  "org.freedesktop.login1.Manager",
274  QDBusConnection::systemBus());
275 
276  QList<QVariant> arg;
277  arg << QString("shutdown:sleep:") // what
278  << qApp->applicationDisplayName() // who
279  << QObject::tr("Files are being processed") // why
280  << QString("block"); // mode
281 
282  int fd = -1;
283  m_reply = loginManager.callWithArgumentList(QDBus::Block, "Inhibit", arg);
284  if (m_reply.isValid()) {
285  fd = m_reply.value().fileDescriptor();
286  }
287 
288  if (fd > 0) {
289  QObject::connect(this, &DTaskDialog::closed, this, [this]() {
290  QDBusReply<QDBusUnixFileDescriptor> tmp = m_reply; //::close(fd);
291  m_reply = QDBusReply<QDBusUnixFileDescriptor>();
292  });
293  }

363 DFileCopyMoveJob::Handle *DTaskDialog::addTaskJob(DFileCopyMoveJob *job, const bool ischecksamejob)

366  DFMTaskWidget *wid = nullptr;
367  bool haswid = false;
368  if (ischecksamejob && m_jobIdItems.contains(QString::number(quintptr(job), 16))) {

370  wid = item ? static_cast<DFMTaskWidget *>(item->listWidget()->itemWidget(item)) : nullptr;
371 
372  }
373  if (nullptr == wid) {
374  wid = new DFMTaskWidget;
375  } else {
376  wid->disconnect();
377  haswid = true;
378  }
379 
380  wid->setTaskId(QString::number(quintptr(job), 16));
381 
382  // 判断任务是否属于保险箱任务,如果是，记录到容器
383  if (isHaveVaultTask(job->sourceUrlList(), job->targetUrl())) {
384  mapNotCompleteVaultTask.insert(job);
385 
386  // 结束当前保险箱未完成任务时，将保险箱任务记录移除

388  mapNotCompleteVaultTask.remove(job);

392 
393  ErrorHandle *handle = new ErrorHandle(wid);
394  job->setErrorHandle(handle, thread());
395  connect(handle, &ErrorHandle::onConflict, wid, &DFMTaskWidget::setConflictMsg);
396  connect(handle, &ErrorHandle::onError, wid, &DFMTaskWidget::setErrorMsg);

399  connect(this, &DTaskDialog::closed, job, &DFileCopyMoveJob::taskDailogClose);
400 
401  connect(wid, &DFMTaskWidget::butonClicked, job, [job, wid, handle, this](DFMTaskWidget::BUTTON bt) {

403  QCloseEvent event;

407  break;

409 
410 // this->closeEvent(&event);
411

434  break;
435  default:
436  break;
437  }
438 
439  iserroroc.remove(QString::number(quintptr(job), 16));
440 
441  if (action == DFileCopyMoveJob::NoAction) {
442  return;
443  }
444 
445  handle->setActionOfError(action);
446  if (wid->getButton(DFMTaskWidget::CHECKBOX_NOASK)->isChecked()) {
447  job->setActionOfErrorType(job->error(), action);
448  }
449  if (job->state() == DFileCopyMoveJob::PausedState) {
450  job->togglePause();
451  }
452  });
453 
454  connect(job, &DFileCopyMoveJob::progressChanged, wid, &DFMTaskWidget::onProgressChanged);
455  connect(job, &DFileCopyMoveJob::speedUpdated, wid, [wid](qint64 speed) {
456  //wid->onSpeedUpdated(speed);

458  QString rmTime;
459  qint64 totalsize = wid->property("totalDataSize").toLongLong();

462  rmTime = formatTime(int(totalsize * (1 - progress) / speed));
463  }
464  wid->setSpeedText(sp, rmTime);
465  });
466 
467  connect(job, &DFileCopyMoveJob::errorCanClear, wid, [job, this]() {
468  if (iserroroc.contains(QString::number(quintptr(job), 16))) {
469  iserroroc.remove(QString::number(quintptr(job), 16));
470  }
471  });

481  QMutexLocker lk(&currentjobchangemutex);

532  }
533 
534  if (job->state() != DFileCopyMoveJob::RunningState) {

538  iserroroc.insert(QString::number(quintptr(job), 16), true);
539  } else if (job->error() != DFileCopyMoveJob::NoError) {
540  data["status"] = "error";
541  bool supprotRetry = job->supportActions(job->error()).testFlag(DFileCopyMoveJob::RetryAction);
542  data["supprotRetry"] = supprotRetry ? "true" : "false";
543  data["errorMsg"] = job->errorString();
544  iserroroc.insert(QString::number(quintptr(job), 16), true);
545  }
546  }
547 
548  this->updateData(wid, data);
549  //正在执行当前槽函数时，job线程一结束，判断job线程是否结束
550  if (job->isFinished()) {
551  this->removeTaskJob(job);
552 
553  // 判断当前任务是否是保险箱任务，如果是，则消除记录
554  if (mapNotCompleteVaultTask.contains(job)) {
555  mapNotCompleteVaultTask.remove(job);
556  }
557  }
558 
560  m_flagMap[from] = true;
561  if (m_flag && getFlagMapValueIsTrue()) {
563  if (isHaveVaultTask(job->sourceUrlList(), job->targetUrl())) {
564  this->close();
565  emit closed();
566  }
567  }
568  });
569  connect(job, &DFileCopyMoveJob::errorChanged, wid, [wid](DFileCopyMoveJob::Error error) {
570  wid->getButton(DFMTaskWidget::PAUSE)->setEnabled(error == DFileCopyMoveJob::NoError);
571  }, Qt::QueuedConnection);
572 
573 
574  /*之前的显示进度百分比的元件CircleProgressAnimatePad，

576  if (!job->fileStatisticsIsFinished()) {
577  wid->setProgressValue(0); // start
578  }
579 
580  wid->getButton(DFMTaskWidget::PAUSE)->setEnabled(job->error() == DFileCopyMoveJob::NoError);
581 
582  if (isHaveVaultTask(job->sourceUrlList(), job->targetUrl())
583  && job->mode() != DFileCopyMoveJob::CopyMode) {
584  if (!job->targetUrl().isValid()) {

594  }
