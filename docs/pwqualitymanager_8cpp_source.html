50 {

54  {PW_ERR_LENGTH_LONG, tr("Password must be no more than %1 characters").arg(m_passwordMaxLen)},

58  {PW_ERR_PW_REPEAT, tr("New password should differ from the current one")}
59  };
60 
61  if (PasswordFlagsStrMap.value(type).isEmpty()) {
