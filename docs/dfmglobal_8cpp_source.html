109 void onClipboardDataChanged()
110 {
111  clipboardFileUrls.clear();
112  clipbordFileinode.clear();
113  const QByteArray &data = qApp->clipboard()->mimeData()->data("x-special/gnome-copied-files");
114 
115  if (data.startsWith("cut")) {
116  clipboardAction = DFMGlobal::CutAction;
117  } else if (data.startsWith("copy")) {
118  clipboardAction = DFMGlobal::CopyAction;
119  } else {
120  clipboardAction = DFMGlobal::UnknowAction;
121  }
122  clipboardFileUrls = qApp->clipboard()->mimeData()->urls();
123  for (QUrl url : qApp->clipboard()->mimeData()->urls()) {
124  //链接文件的inode不加入clipbordFileinode，只用url判断clip，避免多个同源链接文件的逻辑误判
125  struct stat statInfo;
126  int fileStat = stat(url.path().toStdString().c_str(), &statInfo);
127  if (0 == fileStat && !S_ISLNK(statInfo.st_mode)) {
128  clipbordFileinode << statInfo.st_ino;
129  }
130  }
131 }

182 void DFMGlobal::setUrlsToClipboard(const QList<QUrl> &list, DFMGlobal::ClipboardAction action, QMimeData *mimeData)
183 {
184  if (action == UnknowAction)
185  return;
186 
187  if (!mimeData)
188  mimeData = new QMimeData;
189 
190  QByteArray ba = (action == DFMGlobal::CutAction) ? "cut" : "copy";
191  QString text;
192  QByteArray iconBa;
193  QDataStream stream(&iconBa, QIODevice::WriteOnly);
194 
195  int maxIconsNum = 3;
196  for (const QUrl &url : list) {
197  ba.append("\n");
198  ba.append(url.toString());
199 
200  const QString &path = url.toLocalFile();
201  const DAbstractFileInfoPointer &fileInfo = DFileService::instance()->createFileInfo(nullptr, DUrl(url));
202 
203  if (!fileInfo)
204  continue;
205  if (maxIconsNum-- > 0) {
206  QStringList iconList;
207  if (fileInfo->isSymLink()) {
208  iconList << "emblem-symbolic-link";
209  }
210  if (!fileInfo->isWritable()) {
211  iconList << "emblem-readonly";
212  }
213  if (!fileInfo->isReadable()) {
214  iconList << "emblem-unreadable";
215  }
216 
217  // 多文件时只显示文件图标, 一个文件时显示缩略图(如果有的话)
218  DFileInfo *fi = dynamic_cast<DFileInfo *>(fileInfo.data());
219  QIcon icon = fi ? DFileIconProvider::globalProvider()->icon(*fi) :
220  DFileIconProvider::globalProvider()->icon(fileInfo->toQFileInfo());
221  DAbstractFileInfo::FileType fileType = mimeTypeDisplayManager->displayNameToEnum(fileInfo->mimeTypeName());
222  if (list.size() == 1 && fileType == DAbstractFileInfo::FileType::Images) {
223  QIcon thumb(DThumbnailProvider::instance()->thumbnailFilePath(fileInfo->toQFileInfo(), DThumbnailProvider::Large));
224  if (thumb.isNull()) {
225  //qWarning() << "thumbnail file faild " << fileInfo->absoluteFilePath();
226  } else {
227  icon = thumb;
228  }
229  }
230  stream << iconList << icon;
231  }
232 
233  if (!path.isEmpty()) {
234  text += path + '\n';
235  }
236  }
237 
238  mimeData->setText(text.endsWith('\n') ? text.left(text.length() - 1) : text);
239  mimeData->setData("x-special/gnome-copied-files", ba);
240  mimeData->setData("x-dfm-copied/file-icons", iconBa);
241  mimeData->setUrls(list);
242 
243  qApp->clipboard()->setMimeData(mimeData);
244 }

392 void DFMGlobal::initTagManagerConnect()
393 {
394  connect(TagManager::instance(), static_cast<void(TagManager::*)(const QMap<QString, QString>&)>(&TagManager::changeTagColor),
395  [](const QMap<QString, QString> &tag_and_new_color) {
396  for (auto i = tag_and_new_color.constBegin(); i != tag_and_new_color.constEnd(); ++i) {
397  const QString &tag_name = i.key();
398  const QStringList &files = TagManager::instance()->getFilesThroughTag(tag_name);
399 
400  for (const QString &file : files) {
401  DUrl url = DUrl::fromLocalFile(file);
402  DAbstractFileWatcher::ghostSignal(url.parentUrl(), &DAbstractFileWatcher::fileAttributeChanged, url);
403  }
404 
405  // for tag watcher
406  const DUrl &file_url = DUrl::fromUserTaggedFile(tag_name, QString());
407 
408  DAbstractFileWatcher::ghostSignal(DUrl(TAG_ROOT), &DAbstractFileWatcher::fileAttributeChanged, file_url);
409  }
410  });
411  connect(TagManager::instance(), &TagManager::filesWereTagged, [](const QMap<QString, QList<QString>> &files_were_tagged) {
412  for (auto i = files_were_tagged.constBegin(); i != files_were_tagged.constEnd(); ++i) {
413  // is trash files
414  if (i.key().startsWith(DFMStandardPaths::location(DFMStandardPaths::TrashFilesPath)))
415  return;
416 
417  DUrl url = DUrl::fromLocalFile(i.key());
418  DAbstractFileWatcher::ghostSignal(url.parentUrl(), &DAbstractFileWatcher::fileAttributeChanged, url);
419 
420  // for tag watcher
421  for (const QString &tag : i.value()) {
422  const DUrl &parent_url = DUrl::fromUserTaggedFile(tag, QString());
423  const DUrl &file_url = DUrl::fromUserTaggedFile(tag, i.key());
424 
425  DAbstractFileWatcher::ghostSignal(parent_url, &DAbstractFileWatcher::subfileCreated, file_url);
426  }
427  }
428  });
429  connect(TagManager::instance(), &TagManager::untagFiles, [](const QMap<QString, QList<QString>> &tag_be_removed_files) {
430  for (auto i = tag_be_removed_files.constBegin(); i != tag_be_removed_files.constEnd(); ++i) {
431  DUrl url = DUrl::fromLocalFile(i.key());
432  DAbstractFileWatcher::ghostSignal(url.parentUrl(), &DAbstractFileWatcher::fileAttributeChanged, url);
433 
434  // for tag watcher
435  for (const QString &tag : i.value()) {
436  const DUrl &parent_url = DUrl::fromUserTaggedFile(tag, QString());
437  const DUrl &file_url = DUrl::fromUserTaggedFile(tag, i.key());
438 
439  DAbstractFileWatcher::ghostSignal(parent_url, &DAbstractFileWatcher::fileDeleted, file_url);
440  }
441  }
442  });
443 
444  // for tag watcher
445  connect(TagManager::instance(), &TagManager::addNewTag, [](const QList<QString> &new_tags) {
446  for (const QString &tag : new_tags) {
447  DAbstractFileWatcher::ghostSignal(DUrl(TAG_ROOT), &DAbstractFileWatcher::subfileCreated, DUrl::fromUserTaggedFile(tag, QString()));
448  }
449  });
450  connect(TagManager::instance(), &TagManager::deleteTag, [](const QList<QString> &new_tags) {
451  for (const QString &tag : new_tags) {
452  DAbstractFileWatcher::ghostSignal(DUrl(TAG_ROOT), &DAbstractFileWatcher::fileDeleted, DUrl::fromUserTaggedFile(tag, QString()));
453  }
454  });
455  connect(TagManager::instance(), static_cast<void(TagManager::*)(const QMap<QString, QString>&)>(&TagManager::changeTagName),
456  [](const QMap<QString, QString> &old_and_new_name) {
457  for (auto i = old_and_new_name.constBegin(); i != old_and_new_name.constEnd(); ++i) {
458  const DUrl &old_url = DUrl::fromUserTaggedFile(i.key(), QString());
459  const DUrl &new_url = DUrl::fromUserTaggedFile(i.value(), QString());
460 
461  DAbstractFileWatcher::ghostSignal(DUrl(TAG_ROOT), &DAbstractFileWatcher::fileMoved, old_url, new_url);
462  }
463  });
464 }

599 void DFMGlobal::elideText(QTextLayout *layout, const QSizeF &size, QTextOption::WrapMode wordWrap,
600  Qt::TextElideMode mode, qreal lineHeight, int flags, QStringList *lines,
601  QPainter *painter, QPointF offset, const QColor &shadowColor, const QPointF &shadowOffset,
602  const QBrush &background, qreal backgroundRadius, QList<QRectF> *boundingRegion)
603 {
604  qreal height = 0;
605  bool drawBackground = background.style() != Qt::NoBrush;
606  bool drawShadow = shadowColor.isValid();
607 
608  QString text = layout->engine()->hasFormats() ? layout->engine()->block.text() : layout->text();
609  QTextOption &text_option = *const_cast<QTextOption *>(&layout->textOption());
610 
611  text_option.setWrapMode(wordWrap);
612 
613  if (flags & Qt::AlignRight)
614  text_option.setAlignment(Qt::AlignRight);
615  else if (flags & Qt::AlignHCenter)
616  text_option.setAlignment(Qt::AlignHCenter);
617 
618  if (painter) {
619  text_option.setTextDirection(painter->layoutDirection());
620  layout->setFont(painter->font());
621  } else {
622  // dont paint
623  layout->engine()->ignoreBidi = true;
624  }
625 
626  auto naturalTextRect = [&](const QRectF rect) {
627  QRectF new_rect = rect;
628 
629  new_rect.setHeight(lineHeight);
630 
631  return new_rect;
632  };
633 
634  auto drawShadowFun = [&](const QTextLine & line) {
635  const QPen pen = painter->pen();
636 
637  painter->setPen(shadowColor);
638  line.draw(painter, shadowOffset);
639 
640  // restore
641  painter->setPen(pen);
642  };
643 
644  layout->beginLayout();
645 
646  QTextLine line = layout->createLine();
647  QRectF lastLineRect;
648 
649  while (line.isValid()) {
650  height += lineHeight;
651 
652  if (height + lineHeight > size.height()) {
653  const QString &end_str = layout->engine()->elidedText(mode, qRound(size.width()), flags, line.textStart());
654 
655  layout->endLayout();
656  layout->setText(end_str);
657 
658  if (layout->engine()->block.docHandle()) {
659  const_cast<QTextDocument *>(layout->engine()->block.document())->setPlainText(end_str);
660  }
661 
662  text_option.setWrapMode(QTextOption::NoWrap);
663  layout->beginLayout();
664  line = layout->createLine();
665  line.setLineWidth(size.width() - 1);
666  text = end_str;
667  } else {
668  line.setLineWidth(size.width());
669  }
670 
671  line.setPosition(offset);
672 
673  const QRectF rect = naturalTextRect(line.naturalTextRect());
674 
675  if (painter) {
676  if (drawBackground) {
677  const QMarginsF margins(backgroundRadius, 0, backgroundRadius, 0);
678  QRectF backBounding = rect;
679  QPainterPath path;
680 
681  if (lastLineRect.isValid()) {
682  if (qAbs(rect.width() - lastLineRect.width()) < backgroundRadius * 2) {
683  backBounding.setWidth(lastLineRect.width());
684  backBounding.moveCenter(rect.center());
685  path.moveTo(lastLineRect.x() - backgroundRadius, lastLineRect.bottom() - backgroundRadius);
686  path.lineTo(lastLineRect.x(), lastLineRect.bottom() - 1);
687  path.lineTo(lastLineRect.right(), lastLineRect.bottom() - 1);
688  path.lineTo(lastLineRect.right() + backgroundRadius, lastLineRect.bottom() - backgroundRadius);
689  path.lineTo(lastLineRect.right() + backgroundRadius, backBounding.bottom() - backgroundRadius);

977  prober_encoding = pre_encoding;

980  }
981

1194 {

1196  int non_base_latin_count = 0;
1197  qreal unidentification_count = 0;

1202

1205 
1206  if (ch.unicode() > 0x7f)
1207  ++non_base_latin_count;
1208 
1209  switch (ch.script()) {
1210  case QChar::Script_Hiragana:
1211  case QChar::Script_Katakana:
1212  hep_count += country == QLocale::Japan ? 1.2 : 0.5;
1213  unidentification_count += country == QLocale::Japan ? 0 : 0.3;
1214  break;
1215  case QChar::Script_Han:
1216  hep_count += country == QLocale::China ? 1.2 : 0.5;
1217  unidentification_count += country == QLocale::China ? 0 : 0.3;

1219  case QChar::Script_Hangul:
1220  hep_count += (country == QLocale::NorthKorea) || (country == QLocale::SouthKorea) ? 1.2 : 0.5;
1221  unidentification_count += (country == QLocale::NorthKorea) || (country == QLocale::SouthKorea) ? 0 : 0.3;
1222  break;
1223  case QChar::Script_Cyrillic:
1224  hep_count += (country == QLocale::Russia) ? 1.2 : 0.5;
1225  unidentification_count += (country == QLocale::Russia) ? 0 : 0.3;

1227  case QChar::Script_Greek:
1228  hep_count += (country == QLocale::Greece) ? 1.2 : 0.5;
1229  unidentification_count += (country == QLocale::Greece) ? 0 : 0.3;
1230  break;
1231  default:
1232  // full-width character, emoji, 常用标点, 拉丁文补充1，天城文及其补充，CJK符号和标点符号（如：【】）
1233  if ((ch.unicode() >= 0xff00 && ch <= 0xffef)
1234  || (ch.unicode() >= 0x2600 && ch.unicode() <= 0x27ff)
1235  || (ch.unicode() >= 0x2000 && ch.unicode() <= 0x206f)
1236  || (ch.unicode() >= 0x80 && ch.unicode() <= 0xff)
1237  || (ch.unicode() >= 0xa8e0 && ch.unicode() <= 0xa8ff)
1238  || (ch.unicode() >= 0x0900 && ch.unicode() <= 0x097f)
1239  || (ch.unicode() >= 0x3000 && ch.unicode() <= 0x303f)) {
1240  ++hep_count;
1241  } else if (ch.isSurrogate() && ch.isHighSurrogate()) {
1242  ++i;
1243 
1244  if (i < unicode_data.size()) {
1245  const QChar &next_ch = unicode_data.at(i);
1246 
1247  if (!next_ch.isLowSurrogate()) {
1248  --i;

1253

1272  return 1.0f;

1276
