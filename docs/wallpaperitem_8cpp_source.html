42 class WrapperWidget : public QWidget
43 {
44 public:
45  explicit WrapperWidget(QWidget *parent = nullptr) : QWidget(parent) {}
46 
47 protected:
48  void paintEvent(QPaintEvent *event) override
49  {
50  if (m_pixmap.isNull()) return;
51 
52  QRect pixmap_geometry = m_pixmapBoxGeometry;
53 
54  pixmap_geometry.setSize(m_pixmap.size());
55  pixmap_geometry.moveCenter(m_pixmapBoxGeometry.center());
56 
57  QPainter pa(this);
58 
59  pa.setOpacity(m_opacity);
60  pa.drawPixmap(pixmap_geometry.topLeft(), m_pixmap, pixmap_geometry.united(event->rect()));
61  pa.setPen(QColor(255, 255, 255, 25));
62  pa.drawRect(m_pixmapBoxGeometry.adjusted(0, 0, -1, -1));
63  }
64 
65 private:
66  QPixmap m_pixmap;
67  qreal m_opacity = 1;
68  QRect m_pixmapBoxGeometry;
69 
70  friend class WallpaperItem;
71 };

117 void WallpaperItem::initPixmap()
118 {
119  if (m_useThumbnailManager) {
120  refindPixmap();
121  } else {
122  QIcon icon(m_path);
123  m_wrapper->m_pixmap = icon.pixmap(window()->windowHandle(), QSize(ItemWidth, ItemHeight));
124  // 由于QIcon::pixmap中计算图片缩放比例时的特殊计算方式，当请求的pixmap大小的比例和图片实际大小比例不一致时
125  // 计算出的pixmap缩放系数和希望的值不一致，所以此处纠正一下
126  m_wrapper->m_pixmap.setDevicePixelRatio(qMax(m_wrapper->m_pixmap.width() / qreal(ItemWidth),
127  m_wrapper->m_pixmap.height() / qreal(ItemHeight)));
128  m_wrapper->update();
129  }
130 }

159 void WallpaperItem::slideUp()
160 {
161  if (m_wrapper->y() < 0 && m_downAnim->state() == QAbstractAnimation::Stopped)
162  return;
163 
164  m_upAnim->setStartValue(QPoint(0, 0));
165  m_upAnim->setEndValue(QPoint(0, -ItemHeight / 2 * m_buttonLayout->count()));
166  m_upAnim->start();
167  //当按钮浮起，设置按钮可获得焦点
168  for (int i = 0; i < m_buttonLayout->count(); i++) {
169  m_buttonLayout->itemAt(i)->widget()->setFocusPolicy(Qt::StrongFocus);
170  }
171  //设置第一个按钮为焦点
172  m_buttonLayout->itemAt(0)->widget()->setFocus();
173 }

175 void WallpaperItem::slideDown()

184  for (int i = 0; i < m_buttonLayout->count(); i++) {
185  m_buttonLayout->itemAt(i)->widget()->setFocusPolicy(Qt::NoFocus);

264 bool WallpaperItem::eventFilter(QObject *object, QEvent *event)
265 {
266  Q_UNUSED(object)
267  if (event->type() == QEvent::KeyPress) {
268  QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);

312 void WallpaperItem::setOpacity(qreal opacity)
313 {
314  //qreal类型是double类型，这里是设置透明度，根据Qt文档描述取值范围是0.0-1.0
315  //这里精度应当足够了。
316  const double tempEps = 1e-6;
317  if (fabs(m_wrapper->m_opacity - opacity) < tempEps)
318  return;
319 
320  m_wrapper->m_opacity = opacity;
321  m_wrapper->update();
322 }
