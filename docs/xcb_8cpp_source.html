84 {

87  xcb_generic_error_t *e;
88  xcb_get_property_cookie_t cookie = xcb_ewmh_get_wm_window_type(&m_ewmh_connection, winId);
89  if (1 == xcb_ewmh_get_wm_window_type_reply(&m_ewmh_connection,

92  &e)) {
93

95  xcb_atom_t t = name.atoms[i];

100  }

104 QRect get_window_rect(xcb_window_t winId)

122 QList<DockInfo> XcbMisc::find_dock_window()

130  if (1 == xcb_ewmh_get_client_list_reply(&m_ewmh_connection, cookie, &clients, &e)) {
131 
132  for (uint32_t wi = 0; wi < clients.windows_len; ++wi) {
133  xcb_window_t winId = clients.windows[wi];
134  if (is_dock_window(winId)) {

137  info.screenId = i;

141  }
