79 void CategoryTitleWidget::updatePosition(const QPoint pos, int w, int posType)
80 {
81  int x = width() / 2 - m_title->width() / 2;
82 
83  int boxWidth = m_calcUtil->getAppBoxSize().width();
84  int canMoveWidth = boxWidth / 2 - m_title->width() / 2 - DLauncher::APPS_CATEGORY_TITLE_SPACING;
85 
86  int leftX = w / 2 - boxWidth / 2 - DLauncher::APPHBOX_SPACING;
87  int rightX = w / 2 + boxWidth / 2 + DLauncher::APPHBOX_SPACING;
88 
89  //因为单元循环引用冲突，没有直接引用scrollwidgetagent.h 单元中的枚举PosType
90  //其中0 = Pos_None,1 = Pos_LL, 2 = Pos_L,3 = Pos_M,4 = Pos_R,5 = Pos_RR
91  if (posType == 1) {
92  x = width() - m_title->width() - DLauncher::APPS_CATEGORY_TITLE_SPACING;
93  } else if (posType == 2) {
94  x = width() - m_title->width() - DLauncher::APPS_CATEGORY_TITLE_SPACING;
95 
96  int movedDiff = pos.x() + boxWidth - leftX;
97 
98  movedDiff = movedDiff < 0 ? 0 : movedDiff;
99  movedDiff = movedDiff > canMoveWidth ? canMoveWidth : movedDiff;
100 
101  x = x - movedDiff;
102  } else if (posType == 4) {
103  x = DLauncher::APPS_CATEGORY_TITLE_SPACING;
104 
105  int movedDiff = pos.x() - rightX;
106  movedDiff = movedDiff > 0 ? 0 : movedDiff;
107  movedDiff = movedDiff < -canMoveWidth ? -canMoveWidth : movedDiff;
108 
109  x = x - movedDiff;
110  } else if (posType == 5) {
111  x = DLauncher::APPS_CATEGORY_TITLE_SPACING;
112  }
113 
114  QPoint p(x, 0);
115  m_title->move(p);
116  update();
117 }
