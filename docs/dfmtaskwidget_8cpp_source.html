385 void DFMTaskWidget::setProgressValue(int value)
386 {
387  Q_D(DFMTaskWidget);
388 
389  if (value > 100) {
390  value = 100;
391  }
392 
393  //大量快速的设置进度条会导致进度数值不刷新
394  //这里通过对相等值判断和定时器降低刷新频率
395  //并且手动调用update强制刷新界面
396  if (d->m_isSettingValue && value != 100) { // fix : 删除文件最后要显示100%
397  return;
398  }
399 
400  if (value > 0 && value == d->m_progress->value()) {
401  return;
402  }
403 
404  d->m_timer->start(100);
405  d->m_isSettingValue = true;
406  if (value >= 0 && d->m_progress->value() == 0) {
407  d->m_progress->start();
408  d->m_progress->setValue(value);
409  return;
410  }
411 
412  if (value < 0) {
413  d->m_progress->stop();
414  d->m_isSettingValue = false;
415  d->m_timer->stop();
416  } else {
417  d->m_progress->setValue(value);
418  d->m_progress->update();
419  }
420 }

585 void DFMTaskWidget::paintEvent(QPaintEvent *event)
586 {
587  QStyleOption opt;
588  opt.initFrom(this);
589  QPainter painter(this);
590  if (opt.state & QStyle::State_MouseOver) {
591  int radius = 8;
592  QRectF bgRect;
593  bgRect.setSize(size());
594  QPainterPath path;
595  path.addRoundedRect(bgRect, radius, radius);
596  QColor bgColor = palette().background().color();
597  if (DGuiApplicationHelper::instance()->themeType() == DGuiApplicationHelper::LightType) {
598  bgColor = QColor(0, 0, 0, 13); //rgba(0,0,0,13); border-radius: 8px
599  } else {
600  bgColor = QColor(255, 255, 255, 13); //rgba(255,255,255,13); border-radius: 8px
601  }
602 
603  // drawbackground color
604  painter.save();
605  painter.setRenderHint(QPainter::Antialiasing, true);
606  painter.fillPath(path, bgColor);
607  painter.setRenderHint(QPainter::Antialiasing, false);
608  painter.restore();
609  } else if (property("drawSeparator").toBool()) {
610  QPoint p1, p2;
611  p1 = opt.rect.topLeft();
612  p2 = opt.rect.topRight();
613  QPen oldPen = painter.pen();
614  painter.setPen(QPen(opt.palette.brush(foregroundRole()), 1));
615  painter.drawLine(p1, p2);
616  painter.setPen(oldPen);
617  }
618 
619  QWidget::paintEvent(event);
620 }
