233  m_wakeComputerNeedPassword->setChecked(model->sleepLock());

236 #ifndef DCC_DISABLE_POWERSAVE
237  m_autoIntoSaveEnergyMode->setChecked(model->autoPowerSaveMode());
238 #endif
239 
240  m_wakeComputerNeedPassword->setVisible(model->canSleep() && model->getSuspend());
241 
242  //-------------sp2 add-------------------------
243  m_swLowPowerAutoIntoSaveEnergyMode->setChecked(model->powerSavingModeAutoWhenQuantifyLow());
244  connect(model, &PowerModel::powerSavingModeAutoWhenQuantifyLowChanged, m_swLowPowerAutoIntoSaveEnergyMode, &SwitchWidget::setChecked);
245 
246  QDBusInterface Interface("com.deepin.daemon.Display",
247  "/com/deepin/daemon/Display",
248  "com.deepin.daemon.Display",
249  QDBusConnection::sessionBus());
250  int maxBacklight = Interface.property("MaxBacklightBrightness").toInt();
251  m_sldLowerBrightness->setVisible(maxBacklight >= 100 || maxBacklight == 0);
252  m_sldLowerBrightness->slider()->setValue(model->powerSavingModeLowerBrightnessThreshold() / 10);

322 void GeneralWidget::onHighPerformanceSupportChanged(const bool isSupport)
323 {
324  int row_count = m_powerPlanModel->rowCount();
325  if (!isSupport) {
326  for (int i = 0; i < row_count; ++i) {
327  QStandardItem *items = m_powerPlanModel->item(i, 0);
328  if (items->data(PowerPlanRole).toString() == PERFORMANCE) {
329  m_powerPlanModel->removeRow(i);
330  break;
331  }
332  }
333  } else {
334  bool findHighPerform = false;
335  for (int i = 0; i < row_count; ++i) {
336  QStandardItem *items = m_powerPlanModel->item(i, 0);
337  if (items->data(PowerPlanRole).toString() == PERFORMANCE) {
338  findHighPerform = true;
339  break;
340  }
341  }
342  if (!findHighPerform) {
343  DStandardItem *powerPlanItem = new DStandardItem(m_powerPlanMap.value(PERFORMANCE));
344  powerPlanItem->setData(PERFORMANCE, PowerPlanRole);
345  m_powerPlanModel->insertRow(1, powerPlanItem);
346  }
347  }
348 }
