31 WacomWorker::WacomWorker(WacomModel *model, QObject *parent)
32  : QObject(parent)
33  , m_dbusWacom(new Wacom(Service, ServicePath, QDBusConnection::sessionBus(), this))
34  , m_model(model)
35 {
36  connect(m_dbusWacom, &Wacom::StylusPressureSensitiveChanged, this, &WacomWorker::setPressureSensitive);
37  connect(m_dbusWacom, &Wacom::ExistChanged, m_model, &WacomModel::setExist);
38  connect(m_dbusWacom, &Wacom::CursorModeChanged, this, &WacomWorker::onCursorModeChanged);
39 
40  //get real data
41  m_model->setExist(m_dbusWacom->exist());
42  m_dbusWacom->setSync(false);
43 }

45 void WacomWorker::active()
46 {
47  m_dbusWacom->blockSignals(false);
48  uint pressure = m_dbusWacom->stylusPressureSensitive();
49  if (pressure < 1 || pressure > 7) {
50  pressure = m_dbusWacom->stylusPressureSensitive();
51  }
52  m_model->setPressureValue(pressure);
53  m_model->setExist(m_dbusWacom->exist());
54  m_model->setCursorMode(m_dbusWacom->cursorMode());
55 }
56
