109 TEST_F(TestMusicMessageView, can_is_chinese)
110 {
111  MediaMeta meta;
112  QUrl url(m_url);
113  TagLib::FileRef f(url.toLocalFile().toLocal8Bit());
114  TagLib::Tag *tag = f.tag();
115  bool encode = true;
116  encode &= tag->title().isNull() ? true : tag->title().isLatin1();
117  encode &= tag->artist().isNull() ? true : tag->artist().isLatin1();
118  encode &= tag->album().isNull() ? true : tag->album().isLatin1();
119 
120  QByteArray detectByte;
121  QByteArray detectCodec;
122  if (encode)
123  {
124  if (detectCodec.isEmpty())
125  {
126  detectByte += tag->title().toCString();
127  detectByte += tag->artist().toCString();
128  detectByte += tag->album().toCString();
129  auto allDetectCodecs = m_musicMessageView->detectEncodings(detectByte);
130  auto localeCode = m_musicMessageView->localeCodes.value(QLocale::system().name());
131 
132  for (auto curDetext : allDetectCodecs)
133  {
134  if (curDetext == "Big5" || curDetext == localeCode)
135  {
136  detectCodec = curDetext;
137  break;
138  }
139  }
140  if (detectCodec.isEmpty())
141  detectCodec = allDetectCodecs.value(0);
142 
143  QString curStr = QString::fromLocal8Bit(tag->title().toCString());
144  if (curStr.isEmpty())
145  curStr = QString::fromLocal8Bit(tag->artist().toCString());
146  if (curStr.isEmpty())
147  curStr = QString::fromLocal8Bit(tag->album().toCString());
148  for (auto ch : curStr)
149  {
150  EXPECT_TRUE(m_musicMessageView->isChinese(ch));
151  if (m_musicMessageView->isChinese(ch))
152  {
153  detectCodec = "GB18030";
154  break;
155  }
156  }
157  }
158  }
159 }
