78 void MultiPagesView::updateGradient(QPixmap &pixmap, QPoint topLeftImg, QPoint topRightImg)
79 {
80  m_pLeftGradient->setDirection(GradientLabel::LeftToRight);
81  m_pRightGradient->setDirection(GradientLabel::RightToLeft);
82 
83  const qreal ratio = devicePixelRatioF();
84  pixmap.setDevicePixelRatio(1);
85 
86  int nWidth = DLauncher::TOP_BOTTOM_GRADIENT_HEIGHT * m_calcUtil->getScreenScaleX();
87  QSize gradientSize(nWidth, height());
88 
89  QPoint topLeft = mapTo(this, QPoint(0, 0));
90  QRect topRect(topLeftImg * ratio, gradientSize * ratio);
91  QPixmap topCache = pixmap.copy(topRect);
92  topCache.setDevicePixelRatio(ratio);
93 
94  m_pLeftGradient->setPixmap(topCache);
95  m_pLeftGradient->resize(gradientSize);
96  m_pLeftGradient->move(topLeft);
97  m_pLeftGradient->raise();
98 
99  QPoint topRight(topLeft.x() + width() - gradientSize.width(), topLeft.y());
100  QPoint imgTopRight(topRightImg.x() - gradientSize.width(), topRightImg.y());
101 
102  QRect RightRect(imgTopRight * ratio, gradientSize * ratio);
103  QPixmap bottomCache = pixmap.copy(RightRect);
104 
105  m_pRightGradient->setPixmap(bottomCache);
106  m_pRightGradient->resize(gradientSize);
107  m_pRightGradient->move(topRight);
108  m_pRightGradient->raise();
109  setGradientVisible(true);
110 }

112 void MultiPagesView::updatePageCount(AppsListModel::AppCategory category)
113 {
114  int pageCount = m_appsManager->getPageCount(category == AppsListModel::All ? m_category : category);
115  if (pageCount < 1)
116  pageCount = 1;
117 
118  if (pageCount == m_pageCount)
119  return;
120 
121  if (pageCount > m_pageCount) {
122  while (pageCount > m_pageCount) {
123  AppsListModel *pModel = new AppsListModel(m_category);
124  pModel->setPageIndex(m_pageCount);
125  m_pageAppsModelList.push_back(pModel);
126 
127  AppGridView *pageView = new AppGridView;
128  pageView->setModel(pModel);
129  pageView->setItemDelegate(m_delegate);
130  pageView->setContainerBox(m_appListArea);
131  pageView->installEventFilter(this);
132  pageView->setDelegate(this);
133  m_appGridViewList.push_back(pageView);
134 
135  m_viewBox->layout()->insertWidget(m_pageCount, pageView);
136 
137  m_pageCount ++;
138 
139  connect(pageView, &AppGridView::requestScrollLeft, this, &MultiPagesView::dragToLeft);
140  connect(pageView, &AppGridView::requestScrollRight, this, &MultiPagesView::dragToRight);
141  connect(pageView, &AppGridView::requestScrollStop, [this] {
142  m_bDragStart = false;
143  setGradientVisible(false);
144  });
145  connect(pageView, &AppGridView::dragEnd, this, &MultiPagesView::dragStop);
146  connect(m_pageSwitchAnimation, &QPropertyAnimation::finished,pageView,&AppGridView::setDragAnimationEnable);
147  connect(m_pageSwitchAnimation, &QPropertyAnimation::finished,this, [ = ]{
148  setGradientVisible(false);
149  });
150  emit connectViewEvent(pageView);
151  }
152  } else {
153  while (pageCount < m_pageCount) {
154  AppGridView *pageView = qobject_cast<AppGridView *>(m_viewBox->layout()->itemAt(m_pageCount - 1)->widget());
155  m_viewBox->layout()->removeWidget(pageView);
156  pageView->deleteLater();
157 
158  m_pageAppsModelList.removeLast();
159  m_appGridViewList.removeLast();
160 
161  m_pageCount --;
162  }
163  }
164 
165  m_pageControl->setPageCount(m_pageCount > 1 ? pageCount : 0);
166 }

168 void MultiPagesView::dragToLeft(const QModelIndex &index)
169 {
170  Q_UNUSED(index);
171  if (m_pageIndex <= 0)
172  return;
173 
174  if (m_pageSwitchAnimation->state() == QPropertyAnimation::Running || m_bDragStart)
175  return;
176 
177  m_appGridViewList[m_pageIndex]->dragOut(-1);
178 
179  showCurrentPage(m_pageIndex - 1);
180 
181  int lastApp = m_pageAppsModelList[m_pageIndex]->rowCount(QModelIndex());
182  QModelIndex firstModel = m_appGridViewList[m_pageIndex]->indexAt(lastApp - 1);
183  m_appGridViewList[m_pageIndex]->dragIn(firstModel);
184 
185  m_bDragStart = true;
186 }

287 QModelIndex MultiPagesView::selectApp(const int key)
288 {
289  int page = m_pageIndex;
290  int itemSelect = -1;
291  if (Qt::Key_Left == key || Qt::Key_Up == key) {
292  if (page - 1 >= 0) {
293  -- page;
294  itemSelect = m_calcUtil->appPageItemCount(m_category) - 1;
295  } else {
296  page = m_pageCount - 1;
297  itemSelect = m_pageAppsModelList[page]->rowCount(QModelIndex()) - 1;
298  }
299  } else {
300  if (page + 1 < m_pageCount) {
301  ++ page;
302  itemSelect = 0;
303  } else {
304  page = 0;
305  itemSelect = 0;
306  }
307  }
308  if (page != m_pageIndex)
309  showCurrentPage(page);
310 
311  return m_appGridViewList[m_pageIndex]->indexAt(itemSelect);
312 }

376 void MultiPagesView::mouseRelease(QMouseEvent *e)
377 {
378  int nDiff = m_nMousePos - e->x();
379  if (nDiff > DLauncher::TOUCH_DIFF_THRESH) { // 加大范围来避免手指点击触摸屏抖动问题
380  showCurrentPage(m_pageIndex + 1);
381  } else if (nDiff < -DLauncher::TOUCH_DIFF_THRESH) { // 加大范围来避免手指点击触摸屏抖动问题
382  showCurrentPage(m_pageIndex - 1);
383  } else {
384  int nScroll = m_appListArea->horizontalScrollBar()->value();
385  //多个分页是点击直接隐藏
386  if (nScroll == m_scrollStart && m_pageCount != 1)
387  emit m_appGridViewList[m_pageIndex]->clicked(QModelIndex());
388  else if (nScroll - m_scrollStart > DLauncher::MOUSE_MOVE_TO_NEXT)
389  showCurrentPage(m_pageIndex + 1);
390  else if (nScroll - m_scrollStart < -DLauncher::MOUSE_MOVE_TO_NEXT)
391  showCurrentPage(m_pageIndex - 1);
392  else
393  showCurrentPage(m_pageIndex);
394  }
395  m_bMousePress = false;
396 
397  setGradientVisible(false);
398 
399  if(m_pageCount == 1)
400  QWidget::mouseReleaseEvent(e);
401 }

409 // 更新边框渐变，在屏幕变化时需要更新，类别拖动时需要隐藏
410 void MultiPagesView::updateGradient()
411 {
412  // 找到背景widget
413  QWidget *backgroundWidget = parentWidget();
414  while (backgroundWidget) {
415  if (qobject_cast<FullScreenFrame *>(backgroundWidget))
416  break;
417 
418  if (qobject_cast<BlurBoxWidget *>(backgroundWidget))
419  return;
420 
421  backgroundWidget = backgroundWidget->parentWidget();
422  }
423 
424  // 获取当前屏幕的高度
425  QScreen *screen = QGuiApplication::primaryScreen();
426  int screen_height = screen->availableGeometry().height();
427  // 计算过渡动画开始的位置 区分左右位置与上下位置
428  int paddingL = m_calcUtil->getScreenSize().width() * SIDES_SPACE_SCALE;
429  int paddingR = m_calcUtil->getScreenSize().width() - paddingL - 1;
430  int topPos = mapToGlobal(rect().topLeft()).y() > screen_height ? mapToGlobal(rect().topLeft()).y() - screen_height : mapToGlobal(rect().topLeft()).y();
431 
432  QPoint left(paddingL,topPos);
433  QPoint right(paddingR,topPos);
434 
435  QPixmap background = backgroundWidget->grab();
436  updateGradient(background, left, right);
437 }
