183 QList<CrumbData> DFMCrumbInterface::seprateUrl(const DUrl &url)
184 {
185  QList<CrumbData> list;
186 
187  DAbstractFileInfoPointer info = DFileService::instance()->createFileInfo(nullptr, url);
188  if (!info) {
189  // blumia: it's possible if the url is not a file url at all, so we are not able
190  // to create a fileinfo for a non-file url. In this case we just return
191  // the empty list.
192  return list;
193  }
194 
195  DUrlList urlList;
196  urlList.append(url);
197  urlList.append(info->parentUrlList());
198 
199  DAbstractFileInfoPointer infoPointer;
200  // Push urls into crumb list
201  DUrlList::const_reverse_iterator iter = urlList.crbegin();
202  while (iter != urlList.crend()) {
203  const DUrl &oneUrl = *iter;
204 
205  QString displayText = oneUrl.fileName();
206  // Check for possible display text.
207  infoPointer = DFileService::instance()->createFileInfo(nullptr, oneUrl);
208  if (infoPointer) {
209  displayText = infoPointer->fileDisplayName();
210  }
211  CrumbData data(oneUrl, displayText);
212  list.append(data);
213 
214  ++iter;
215  }
216 
217  return list;
218 }

233 void DFMCrumbInterface::requestCompletionList(const DUrl &url)
234 {
235  Q_D(DFMCrumbInterface);
236 
237  if (d->folderCompleterJobPointer) {
238  d->folderCompleterJobPointer->disconnect();
239  d->folderCompleterJobPointer->stopAndDeleteLater();
240  }
241 
242  d->folderCompleterJobPointer = DFileService::instance()->getChildrenJob(this, url, QStringList(), QDir::Dirs | QDir::Hidden | QDir::NoDotAndDotDot, QDirIterator::NoIteratorFlags, true, false);
243  if (!d->folderCompleterJobPointer) {
244  return;
245  }
246 
247  connect(d->folderCompleterJobPointer, &JobController::addChildrenList, this, [this](const QList<DAbstractFileInfoPointer> &infoList) {
248  QStringList list;
249  for (const DAbstractFileInfoPointer &infoPointer : infoList) {
250  if (infoPointer.data())
251  list.append(infoPointer->fileName());
252  }
253  emit completionFound(list);
254  }, Qt::DirectConnection);
255 
256  connect(d->folderCompleterJobPointer, &JobController::finished, this, [this]() {
257  emit completionListTransmissionCompleted();
258  }, Qt::QueuedConnection);
259 
260  d->folderCompleterJobPointer->setTimeCeiling(1000);
261  d->folderCompleterJobPointer->setCountCeiling(500);
262  d->folderCompleterJobPointer->start();
263 }
