204 void DFMVfsDevicePrivate::GMountOperationAskPasswordCb(GMountOperation *op, const char *message, const char *default_user,
205  const char *default_domain, GAskPasswordFlags flags, gpointer vfsDevicePtr)
206 {
207  DFMVfsDevice* device = static_cast<DFMVfsDevice*>(vfsDevicePtr);
208  bool anonymous = g_mount_operation_get_anonymous(op);
209  GPasswordSave passwordSave = g_mount_operation_get_password_save(op);
210 
211  const char* default_password = g_mount_operation_get_password(op);
212 
213  QJsonObject obj;
214  obj.insert("message", message);
215  obj.insert("anonymous", anonymous);
216  obj.insert("username", default_user);
217  obj.insert("domain", default_domain);
218  obj.insert("password", default_password);
219  obj.insert("GAskPasswordFlags", flags);
220  obj.insert("passwordSave", passwordSave);
221 
222  qCDebug(vfsDevice()) << "GMountOperationAskPasswordCb() Default fields data" << obj;
223 
224  QJsonObject loginObj;
225  if (device->eventHandler()) {
226  DFMVfsAbstractEventHandler* handler = static_cast<DFMVfsAbstractEventHandler*>(device->eventHandler());
227  loginObj = handler->handleAskPassword(obj);
228  } else {
229  qCDebug(vfsDevice()) << "GMountOperationAskPasswordCb(): No event handler registered to DFMVfsManager, use the default action.";
230  }
231 
232  if (!loginObj.isEmpty()) {
233  anonymous = loginObj.value("anonymous").toBool();
234  QString username = loginObj.value("username").toString();
235  QString domain = loginObj.value("domain").toString();
236  QString password = loginObj.value("password").toString();
237  GPasswordSave passwordsaveFlag = static_cast<GPasswordSave>(loginObj.value("passwordSave").toInt());
238 
239  if ((flags & G_ASK_PASSWORD_ANONYMOUS_SUPPORTED) && anonymous) {
240  g_mount_operation_set_anonymous(op, TRUE);
241  } else {
242  if (flags & G_ASK_PASSWORD_NEED_USERNAME) {
243  g_mount_operation_set_username(op, username.toLocal8Bit().constData());
244  }
245 
246  if (flags & G_ASK_PASSWORD_NEED_DOMAIN) {
247  g_mount_operation_set_domain(op, domain.toLocal8Bit().constData());
248  }
249 
250  if (flags & G_ASK_PASSWORD_NEED_PASSWORD) {
251  g_mount_operation_set_password(op, password.toLocal8Bit().constData());
252  }
253 
254  if (flags & G_ASK_PASSWORD_SAVING_SUPPORTED) {
255  g_mount_operation_set_password_save(op, passwordsaveFlag);
256  }
257  }
258 
259  /* Only try anonymous access once. */
260  if (anonymous) {
261  g_mount_operation_reply(op, G_MOUNT_OPERATION_ABORTED);
262  } else {
263  g_mount_operation_reply(op, G_MOUNT_OPERATION_HANDLED);
264  }
265 
266  } else {
267  qCDebug(vfsDevice()) << "cancel connect";
268  g_mount_operation_reply(op, G_MOUNT_OPERATION_ABORTED);
269  }
270 }

272 // blumia: This callback is mainly for sftp fingerprint identity dialog, but should works on any ask-question signal.
273 // ref: https://www.freedesktop.org/software/gstreamer-sdk/data/docs/latest/gio/GMountOperation.html#GMountOperation-ask-question
274 void DFMVfsDevicePrivate::GMountOperationAskQuestionCb(GMountOperation *op, const char *message, const GStrv choices, gpointer vfsDevicePtr)
275 {
276  char **ptr = choices;
277  int choice = 0;
278  QStringList choiceList;
279  DFMVfsDevice* device = static_cast<DFMVfsDevice*>(vfsDevicePtr);
280 
281  QString oneMessage(message);
282  qCDebug(vfsDevice()) << "GMountOperationAskQuestionCb() message: " << message;
283 
284  while (*ptr) {
285  QString oneOption = QString::asprintf("%s", *ptr++);
286  qCDebug(vfsDevice()) << "GMountOperationAskQuestionCb() - option(s): " << oneOption;
287  choiceList << oneOption;
288  }
289 
290  if (device->eventHandler()) {
291  DFMVfsAbstractEventHandler* handler = static_cast<DFMVfsAbstractEventHandler*>(device->eventHandler());
292  choice = handler->handleAskQuestion(oneMessage, choiceList);
293  } else {
294  qCDebug(vfsDevice()) << "GMountOperationAskQuestionCb(): No event handler registered to DFMVfsManager, use the default action.";
295  }
296 
297  qCDebug(vfsDevice()) << "GMountOperationAskQuestionCb() user choice(start at 0): " << choice;
298 
299  // check if choose is invalid
300  if (choice < 0 && choice >= choiceList.count()) {
301  g_mount_operation_reply(op, G_MOUNT_OPERATION_ABORTED);
302  return;
303  }
304 
305  g_mount_operation_set_choice(op, choice);
306  g_mount_operation_reply(op, G_MOUNT_OPERATION_HANDLED);
307 
308  return;
309 }

311 void DFMVfsDevicePrivate::GFileMountDoneCb(GObject *object, GAsyncResult *res, gpointer vfsDevicePtr)
312 {
313  gboolean succeeded;
314  GError *error = NULL;
315  DFMVfsDevice* device = static_cast<DFMVfsDevice*>(vfsDevicePtr);
316 
317  succeeded = g_file_mount_enclosing_volume_finish(G_FILE(object), res, &error);
318 
319  if (!succeeded) {
320  Q_ASSERT(error->domain == G_IO_ERROR);
321 
322  int errorCode = error->code;
323  QString errorMsg(error->message);
324 
325  if (device->eventHandler()) {
326  DFMVfsAbstractEventHandler* handler = static_cast<DFMVfsAbstractEventHandler*>(device->eventHandler());
327  handler->handleMountError(errorCode, errorMsg);
328  } else {
329  qCDebug(vfsDevice()) << "GFileMountDoneCb(): No event handler registered to DFMVfsManager, use the default action.";
330 
331  switch (error->code) {
332  case G_IO_ERROR_FAILED_HANDLED: // Operation failed and a helper program has already interacted with the user. Do not display any error dialog.
333  break;
334  default:
335  qCDebug(vfsDevice()) << "GFileMountDoneCb() mount failed. reason: " << errorMsg;
336  break;
337  }
338  }
339 
340  g_error_free(error);
341  }
342 
343  if (device->d_ptr->m_eventLoop) {
344  device->d_ptr->m_eventLoop->exit(succeeded ? 0 : -1);
345  }
346 }

415 {
416  if (!url.isValid() || url.scheme() == "file" || url.scheme().isEmpty()) {
417  return nullptr;

419 
420  QString deviceId = url.scheme() == "device" ? url.path() : url.toString();
421 
422  QByteArray ba = deviceId.toLatin1();

425  GError *error = nullptr;

491 bool DFMVfsDevice::attach()
492 {
493  Q_D(DFMVfsDevice);
494 
495  QPointer<QEventLoop> oldEventLoop = d->m_eventLoop;
496  QEventLoop eventLoop;
497 
498  d->m_eventLoop = &eventLoop;
499 
500  DFMGFile file(g_file_new_for_uri(d->m_setupUrl.toUtf8().constData()));
501  if (!file) {
502  return false;
503  }
504 
505  GMountOperation *op = DFMVfsDevicePrivate::GMountOperationNewMountOp(this);
506  g_file_mount_enclosing_volume(file.data(), static_cast<GMountMountFlags>(0),
507  op, nullptr, (GAsyncReadyCallback)&DFMVfsDevicePrivate::GFileMountDoneCb, this);
508 
509  int ret = d->m_eventLoop->exec();
510  if (oldEventLoop) {
511  oldEventLoop->exit(ret);
512  }
513 
514  return ret == 0;
515 }
