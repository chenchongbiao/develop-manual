215 CGraphicsView *CCentralwidget::createNewScense(QString scenceName, const QString &uuid, bool isModified)
216 {
217  CGraphicsView *newview = new CGraphicsView(this);
218  CManageViewSigleton::GetInstance()->addView(newview);
219  auto curScene = new CDrawScene(newview, uuid, isModified);
220  newview->setFrameShape(QFrame::NoFrame);
221  newview->getDrawParam()->setViewName(scenceName);
222 
223  drawApp->setWidgetAccesibleName(newview, "view" + uuid);
224 
225  emit signalAddNewScence(curScene);
226 
227  //设置scene大小为屏幕分辨率
228  //获取屏幕分辨率
229  QDesktopWidget *desktopWidget = QApplication::desktop();
230  QRect screenRect = desktopWidget->screenGeometry();
231  //需要乘以系统缩放系数才是最终的大小
232  screenRect = QRect(screenRect.left(), screenRect.top(),
233  qRound(screenRect.width() * desktopWidget->devicePixelRatioF()),
234  qRound(screenRect.height() * desktopWidget->devicePixelRatioF()));
235  newview->getDrawParam()->setCutDefaultSize(QSize(screenRect.width(), screenRect.height()));
236  curScene->setSceneRect(QRectF(0, 0, screenRect.width(), screenRect.height()));
237 
238  if (CManageViewSigleton::GetInstance()->getThemeType() == 1) {
239  curScene->setBackgroundBrush(QColor(248, 248, 251));
240  } else {
241  curScene->setBackgroundBrush(QColor(35, 35, 35));
242  }
243 
244  newview->setAlignment(Qt::AlignCenter);
245  newview->setRenderHint(QPainter::Antialiasing);//设置反走样
246 
247  //自动设置滚动条
248  newview->setHorizontalScrollBarPolicy(Qt::ScrollBarAsNeeded);
249  newview->setVerticalScrollBarPolicy(Qt::ScrollBarAsNeeded);
250 
251  m_stackedLayout->addWidget(newview);
252 
253  QSize size = newview->getDrawParam()->getCutDefaultSize();
254  QRectF rc = QRectF(0, 0, size.width(), size.height());
255  static_cast<CDrawScene *>(newview->scene())->setSceneRect(rc);
256 
257  connect(curScene, SIGNAL(signalQuitCutAndChangeToSelect()), m_leftToolbar, SLOT(slotAfterQuitCut()));
258  connect(newview, SIGNAL(signalSetScale(const qreal)), this, SIGNAL(signalSetScale(const qreal)));
259  connect(curScene, &CDrawScene::signalAttributeChanged, this, &CCentralwidget::signalAttributeChangedFromScene);
260  connect(curScene, &CDrawScene::signalChangeToSelect, m_leftToolbar, &CLeftToolBar::slotShortCutSelect);
261 
262  connect(curScene, &CDrawScene::signalUpdateCutSize, this, &CCentralwidget::signalUpdateCutSize);
263  connect(curScene, &CDrawScene::signalUpdateTextFont, this, &CCentralwidget::signalUpdateTextFont);
264 
265  connect(newview, SIGNAL(signalLoadDragOrPasteFile(QString)), this, SLOT(slotLoadDragOrPasteFile(QString)));
266 
267  connect(newview, SIGNAL(signalPastePixmap(QPixmap, const QByteArray &)), this, SLOT(slotPastePixmap(QPixmap, const QByteArray &)));
268 
269  connect(newview, SIGNAL(signalTransmitContinueDoOtherThing()), this, SIGNAL(signalContinueDoOtherThing()));
270  connect(newview, SIGNAL(singalTransmitEndLoadDDF()), this, SLOT(slotTransmitEndLoadDDF()));
271 
272  //如果是裁剪模式点击左边工具栏按钮则执行裁剪
273  connect(m_leftToolbar, SIGNAL(singalDoCutFromLeftToolBar()), newview, SLOT(slotDoCutScene()));
274 
275  //如果是裁剪模式点击工具栏的菜单则执行裁剪
276  connect(this, SIGNAL(signalTransmitQuitCutModeFromTopBarMenu()), newview, SLOT(slotDoCutScene()));
277 
278  // 当场景内容被改变需要进行的操作
279  connect(curScene, SIGNAL(signalIsModify(bool)), this, SLOT(currentScenseViewIsModify(bool)));
280 
281  // 连接view保存文件状态
282  connect(newview, &CGraphicsView::signalSaveFileStatus, this, &CCentralwidget::slotOnFileSaveFinished);
283  //connect(newview, SIGNAL(signalSaveFileStatus(bool, QString, QFileDevice::FileError)), this, SLOT(slotSaveFileStatus(bool, QString, QFileDevice::FileError)));
284 
285  return newview;
286 }

305 void CCentralwidget::closeSceneView(CGraphicsView *pView, bool ifTabOnlyOneCloseAqq, bool deleteView)
306 {
307  CGraphicsView *closeView = pView;
308  if (nullptr != closeView) {
309  // [0] 判断是否处于裁剪状态
310  if (!slotJudgeCutStatusAndPopSaveDialog()) {
311  return;
312  }
313 
314  // [1] must set mouse tool to select
315  m_leftToolbar->slotShortCutSelect();
316 
317  QString viewname = closeView->getDrawParam()->viewName();
318  qDebug() << "closeCurrentScenseView:" << viewname;
319 
320  // 如果只剩一个画板并且没有进行修改且不是导入文件则不再创建新的画板
321  if (1 == m_topMutipTabBarWidget->count() && ifTabOnlyOneCloseAqq) {
322 
323  qDebug() << "closeSceneView:" << viewname << " not modify";
324  emit signalLastTabBarRequestClose();
325  return;
326  }
327 
328  closeView->setParent(nullptr);
329  m_stackedLayout->removeWidget(closeView);
330  CManageViewSigleton::GetInstance()->removeView(closeView);
331  m_topMutipTabBarWidget->closeTabBarItemByUUID(closeView->getDrawParam()->uuid());
332  }
333 
334  if (m_topMutipTabBarWidget->count() > 0) {
335  if (m_topMutipTabBarWidget->count() == 1) {
336  m_topMutipTabBarWidget->hide();
337  } else {
338  m_topMutipTabBarWidget->show();
339  }
340  }
341  if (deleteView) {
342  delete closeView;
343  closeView = nullptr;
344  }
345 
346  // Free optimized memory
347  malloc_trim(0);
348 
349  // BUG:39095
350  this->setFocus();
351 }

363 bool CCentralwidget::slotJudgeCutStatusAndPopSaveDialog()
364 {
365  if (nullptr == CManageViewSigleton::GetInstance()->getCurView()->scene())
366  return false;
367 
368  bool isNowCutStatus = getCutedStatus();
369  if (isNowCutStatus) {
370  CCutDialog dialog(this);
371  dialog.exec();
372  auto curScene = static_cast<CDrawScene *>(CManageViewSigleton::GetInstance()->getCurView()->scene());
373  if (CCutDialog::Save == dialog.getCutStatus()) {
374  curScene->doCutScene();
375  } else if (CCutDialog::Cancel == dialog.getCutStatus()) {
376  return false;
377  } else if (CCutDialog::Discard == dialog.getCutStatus()) {
378  curScene->quitCutMode();
379  }
380  CManageViewSigleton::GetInstance()->getCurView()->getDrawParam()->setCurrentDrawToolMode(selection);
381  }
382  return true;
383 }

385 void CCentralwidget::currentScenseViewIsModify(bool isModify)
386 {
387  //需要判断的当前的view是否是信号来源的同一个view
388  QGraphicsScene *pScene = qobject_cast<QGraphicsScene *>(sender());
389 
390  if (pScene != nullptr && !pScene->views().isEmpty()) {
391  CGraphicsView *pView = dynamic_cast<CGraphicsView *>(pScene->views().first()); /*CManageViewSigleton::GetInstance()->getCurView()*/
392 
393  if (pView != nullptr) {
394  QString viewName = pView->getDrawParam()->viewName();
395  qDebug() << "viewName：" << viewName << " modify:" << isModify;
396 
397  bool drawParamCurModified = pView->getDrawParam()->isModified();
398  if (isModify != drawParamCurModified) {
399 
400  //已经修改的状态和drawParam中的状态不同那么就要刷新drawParam的状态
401  QString uuid = pView->getDrawParam()->uuid();
402  pView->getDrawParam()->setModify(isModify);
403  QString newVName = pView->getDrawParam()->getShowViewNameByModifyState();
404 
405  //刷新标签的名字
406  updateTabName(uuid, newVName);
407 
408  //判断当前所有viewscene的修改状态是否需要通知系统阻塞关机
409  CManageViewSigleton::GetInstance()->updateBlockSystem();
410  }
411  }
412  }

416 void CCentralwidget::slotOnFileSaveFinished(const QString &savedFile,
417  bool success,
418  QString errorString,
419  QFileDevice::FileError error,
420  bool needClose)
421 {
422  Q_UNUSED(errorString)
423  CGraphicsView *pView = CManageViewSigleton::GetInstance()->getViewByFilePath(savedFile);
424  if (pView != nullptr) {
425  //如果是要保存后关闭
426  if (needClose) {
427 
428  //关闭这个view及其相关的数据
429  closeViewScense(pView);
430 
431  //如果所有标签页都被删除完了那么退出程序
432  CManageViewSigleton::GetInstance()->quitIfEmpty();
433 
434 
435  } else {
436  if (success) {
437  //保存成功后标签要更新成保存成的ddf文件的名字
438  QFileInfo info(savedFile);
439 
440  QString fileName = info.completeBaseName();
441 
442  pView->getDrawParam()->setViewName(fileName);
443 
444  updateTabName(pView->getDrawParam()->uuid(), pView->getDrawParam()->viewName());
445 
446  } else {
447  qDebug() << "save error:" << errorString << error;
448  }
449  }
450  }
