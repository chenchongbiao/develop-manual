195 void AdapterWidget::initConnect()
196 {
197  connect(m_titleEdit, &TitleEdit::requestSetBluetoothName, this, [ = ](const QString & alias) {
198  Q_EMIT requestSetAlias(m_adapter, alias);
199  });
200 
201  connect(m_myDeviceListView, &DListView::clicked, this, [this](const QModelIndex & idx) {
202  m_otherDeviceListView->clearSelection();
203  const QStandardItemModel *deviceModel = dynamic_cast<const QStandardItemModel *>(idx.model());
204  if (!deviceModel) {
205  return;
206  }
207  BtStandardItem *item = dynamic_cast<BtStandardItem *>(deviceModel->item(idx.row()));
208  if (!item) {
209  return;
210  }
211  for (auto it : m_myDevices) {
212  if (it && it->device() && it->getStandardItem() == item) {
213  if (it->device()->state() != Device::StateConnected) {
214  it->requestConnectDevice(it->device(), m_adapter);
215  }
216  Q_EMIT requestShowDetail(m_adapter, it->device());
217  break;
218  }
219  }
220  });
221 
222  connect(m_myDeviceListView, &DListView::activated, m_myDeviceListView, &DListView::clicked);
223 
224  connect(m_otherDeviceListView, &DListView::clicked, this, [this](const QModelIndex & idx) {
225  m_myDeviceListView->clearSelection();
226  const QStandardItemModel *deviceModel = dynamic_cast<const QStandardItemModel *>(idx.model());
227  if (!deviceModel) {
228  return;
229  }
230  BtStandardItem *item = dynamic_cast<BtStandardItem *>(deviceModel->item(idx.row()));
231  if (!item) {
232  return;
233  }
234  for (auto it : m_deviceLists) {
235  if (it && it->device() && it->getStandardItem() == item) {
236  it->requestConnectDevice(it->device(), m_adapter);
237  break;
238  }
239  }
240  });
241 
242  connect(m_otherDeviceListView, &DListView::activated, m_otherDeviceListView, &DListView::clicked);
243 
244  connect(m_refreshBtn, &DIconButton::clicked, this, [ = ] {
245  Q_EMIT requestRefresh(m_adapter);
246  });
247 
248  connect(m_powerSwitch, &SwitchWidget::checkedChanged, this, [ = ](const bool check) {
249  //关闭蓝牙的时候,直接隐藏列表
250  if (!check) {
251  onPowerStatus(false, false);
252  if (m_adapter) {
253  Q_EMIT m_adapter->closeDetailPage();
254  }
255  }
256  m_powerSwitch->switchButton()->setEnabled(false);
257  Q_EMIT requestSetToggleAdapter(m_adapter, check);
258  });
259 
260  connect(m_discoverySwitch, &SwitchWidget::checkedChanged, this, &AdapterWidget::toggleDiscoverableSwitch);
261 
262  connect(&m_bluetoothInter, &BluetoothInter::DisplaySwitchChanged, m_showAnonymousCheckBox, &DCheckBox::setChecked);
263  connect(m_showAnonymousCheckBox, &DCheckBox::stateChanged, this, [ = ](int state) {
264  if (state == Qt::CheckState::Unchecked) {
265  if (m_bluetoothInter.displaySwitch()) {
266  m_bluetoothInter.setDisplaySwitch(false);
267  }
268  // 将蓝牙名称为空的设备过滤掉
269  for (int i = 0; i < m_deviceLists.size(); i++) {
270  DeviceSettingsItem *pDeviceItem = m_deviceLists[i];
271 
272  if (pDeviceItem && pDeviceItem->device() && pDeviceItem->device()->paired())
273  continue;
274 
275  if (pDeviceItem) {
276  BtStandardItem *dListItem = pDeviceItem->getStandardItem();
277  QModelIndex index = m_otherDeviceModel->indexFromItem(dListItem);
278  if (index.isValid() && pDeviceItem->device() && pDeviceItem->device()->name().isEmpty()) {
279  m_otherDeviceModel->takeRow(index.row());
280  }
281  }
282  }
283  } else {
284  if (!m_bluetoothInter.displaySwitch()) {
285  m_bluetoothInter.setDisplaySwitch(true);
286  }
287  // 显示所有蓝牙设备
288  for (int i = 0; i < m_deviceLists.size(); i++) {
289  DeviceSettingsItem *pDeviceItem = m_deviceLists[i];
290  if (pDeviceItem && pDeviceItem->device() && pDeviceItem->device()->paired())
291  continue;
292 
293  if (pDeviceItem) {
294  BtStandardItem *dListItem = pDeviceItem->getStandardItem();
295  QModelIndex index = m_otherDeviceModel->indexFromItem(dListItem);
296  if (!index.isValid() && pDeviceItem->device() && pDeviceItem->device()->name().isEmpty()) {
297  m_otherDeviceModel->insertRow(0, dListItem);
298  }
299  }
300  }
301  }
302  });
303 }

311 void AdapterWidget::setAdapter(const Adapter *adapter)
312 {
313  connect(adapter, &Adapter::nameChanged, m_titleEdit, &TitleEdit::setTitle, Qt::QueuedConnection);
314  connect(adapter, &Adapter::deviceAdded, this, &AdapterWidget::addDevice, Qt::QueuedConnection);
315  connect(adapter, &Adapter::deviceRemoved, this, &AdapterWidget::removeDevice, Qt::QueuedConnection);
316  connect(adapter, &Adapter::poweredChanged, this, &AdapterWidget::onPowerStatus, Qt::QueuedConnection);
317  connect(m_model, &BluetoothModel::adpaterPowerChanged, m_powerSwitch, [ = ] {
318  m_powerSwitch->switchButton()->setEnabled(true);
319  m_spinnerBtn->hide();
320  });
321  connect(adapter, &Adapter::loadStatus, m_powerSwitch, [ = ] {
322  m_powerSwitch->switchButton()->setEnabled(false);
323  m_spinnerBtn->show();
324  });
325 
326  m_titleEdit->setTitle(adapter->name());
327  for (QString id : adapter->devicesId()) {
328  if (adapter->devices().contains(id)) {
329  const Device *device = adapter->devices()[id];
330  addDevice(device);
331  }
332  }
333  connect(adapter, &Adapter::discoverableChanged, m_discoverySwitch, [ = ] {
334  m_discoverySwitch->setChecked(adapter->discoverabled());
335  });
336  m_lastPowerCheck = adapter->powered();
337  m_discoverySwitch->setChecked(m_adapter->discoverabled());
338  onPowerStatus(adapter->powered(), adapter->discovering());
339 }

381 void AdapterWidget::categoryDevice(DeviceSettingsItem *deviceItem, const bool paired)
382 {
383  if (deviceItem) {
384  if (paired) {
385  BtStandardItem *dListItem = deviceItem->getStandardItem(m_myDeviceListView);
386  m_myDevices << deviceItem;
387  m_myDeviceModel->insertRow(0, dListItem);
388  } else {
389  BtStandardItem *dListItem = deviceItem->getStandardItem(m_otherDeviceListView);
390  if (m_showAnonymousCheckBox->checkState() == Qt::CheckState::Unchecked) {
391  if (deviceItem->device() && false == deviceItem->device()->name().isEmpty()) { // 只关注有名称的蓝牙设备,没有名称的忽略
392  m_otherDeviceModel->insertRow(0, dListItem);
393  }
394  } else {
395  m_otherDeviceModel->insertRow(0, dListItem);
396  }
397  }
398  }
399  bool isVisible = !m_myDevices.isEmpty() && m_powerSwitch->checked();
400  m_myDevicesGroup->setVisible(isVisible);
401  m_myDeviceListView->setVisible(isVisible);
402 }

404 void AdapterWidget::addDevice(const Device *device)
405 {
406  if (!device)
407  return;
408  QPointer<DeviceSettingsItem> deviceItem = new DeviceSettingsItem(device, style());
409  categoryDevice(deviceItem, device->paired());
410 
411  connect(deviceItem, &DeviceSettingsItem::requestConnectDevice, this, &AdapterWidget::requestConnectDevice);
412  connect(device, &Device::pairedChanged, this, [this, deviceItem](const bool paired) {
413  if (deviceItem && deviceItem->device()) {
414  if (paired) {
415  qDebug() << "paired :" << deviceItem->device()->name();
416  BtStandardItem *item = deviceItem->getStandardItem();
417  QModelIndex otherDeviceIndex = m_otherDeviceModel->indexFromItem(item);
418  m_otherDeviceModel->removeRow(otherDeviceIndex.row());
419  deviceItem->resetDeviceItem();
420  BtStandardItem *dListItem = deviceItem->createStandardItem(m_myDeviceListView);
421  m_myDevices << deviceItem;
422  m_myDeviceModel->insertRow(0, dListItem);
423  } else {
424  qDebug() << "unpaired :" << deviceItem->device()->name();
425  for (auto it : m_myDevices) {
426  if (it && it->device() && it == deviceItem) {
427  m_myDevices.removeOne(it);
428  break;
429  }
430  }
431  BtStandardItem *item = deviceItem->getStandardItem();
432  QModelIndex myDeviceIndex = m_myDeviceModel->indexFromItem(item);
433  if (myDeviceIndex.isValid()) {
434  m_myDeviceModel->removeRow(myDeviceIndex.row());
435  BtStandardItem *dListItem = deviceItem->createStandardItem(m_otherDeviceListView);
436  m_otherDeviceModel->insertRow(0, dListItem);
437  }
438  }
439  }
440  bool isVisible = !m_myDevices.isEmpty() && m_powerSwitch->checked();
441  m_myDevicesGroup->setVisible(isVisible);
442  m_myDeviceListView->setVisible(isVisible);
443  });
444  connect(deviceItem, &DeviceSettingsItem::requestShowDetail, this, [this](const Device * device) {
445  Q_EMIT requestShowDetail(m_adapter, device);
446  });
447 
448  m_deviceLists << deviceItem;
449 }

451 void AdapterWidget::removeDevice(const QString &deviceId)
452 {
453  bool isFind = false;
454  for (auto it : m_myDevices) {
455  if (it && it->device() && it->device()->id() == deviceId) {
456  BtStandardItem *item = it->getStandardItem();
457  QModelIndex myDeviceIndex = m_myDeviceModel->indexFromItem(item);
458  m_myDeviceModel->removeRow(myDeviceIndex.row());
459  m_myDevices.removeOne(it);
460  m_deviceLists.removeOne(it);
461  delete it;
462  it = nullptr;
463  Q_EMIT notifyRemoveDevice();
464  isFind = true;
465  break;
466  }
467  }
468  if (!isFind) {
469  for (auto it : m_deviceLists) {
470  if (it && it->device() && it->device()->id() == deviceId) {
471  BtStandardItem *item = it->getStandardItem();
472  QModelIndex otherDeviceIndex = m_otherDeviceModel->indexFromItem(item);
473  m_otherDeviceModel->removeRow(otherDeviceIndex.row());
474  m_deviceLists.removeOne(it);
475  delete it;
476  it = nullptr;
477  Q_EMIT notifyRemoveDevice();
478  break;
479  }
480  }
481  }
482  if (m_myDevices.isEmpty()) {
483  m_myDevicesGroup->hide();
484  m_myDeviceListView->hide();
485  }
