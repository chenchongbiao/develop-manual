130 void DFMCrumbBarPrivate::updateController(const DUrl &url)
131 {
132  Q_Q(DFMCrumbBar);
133 
134  if (!crumbController || !crumbController->supportedUrl(url)) {
135  if (crumbController) {

415 void DFMCrumbBar::updateCrumbs(const DUrl &url)
416 {
417  Q_D(DFMCrumbBar);
418 
419  d->clearCrumbs();
420 
421  if (!d->crumbController) {
422  qWarning("No controller found when trying to call DFMCrumbBar::updateCrumbs() !!!");
423  qDebug() << "updateCrumbs (no controller) : " << url;
424  return;
425  }
426 
427  // 回收站预览打开文件夹时传过来的是真实路径，所以将其转换为虚拟路径
428  DUrl fileUrl = url;
429  if (url.toLocalFile().startsWith(DFMStandardPaths::location(DFMStandardPaths::TrashFilesPath))) {
430  fileUrl = DUrl::fromTrashFile(url.toLocalFile().remove(DFMStandardPaths::location(DFMStandardPaths::TrashFilesPath)));
431  }
432 
433  QList<CrumbData> crumbDataList = d->crumbController->seprateUrl(fileUrl);
434  for (const CrumbData &c : crumbDataList) {
435  if (d->crumbListviewModel) {
436  QString iconName = getIconName(c);
437  QStandardItem *listitem = nullptr;
438  if (!iconName.isEmpty()) {
439  listitem = new QStandardItem(QIcon::fromTheme(iconName), QString());
440  } else {
441  listitem = new QStandardItem(c.displayText);
442  }
443 
444  listitem->setCheckable(false);
445  listitem->setData(c.url, DFMCrumbListviewModel::FileUrlRole);
446  d->crumbListviewModel->appendRow(listitem);
447  }
448  }
449 
450  if (d->crumbListView.selectionModel() && d->crumbListviewModel)
451  d->crumbListView.selectionModel()->select(d->crumbListviewModel->lastIndex(), QItemSelectionModel::Select);
452 
453  d->checkArrowVisiable();
454  d->crumbListView.horizontalScrollBar()->triggerAction(QScrollBar::SliderToMaximum);
455 }

544 bool DFMCrumbBar::eventFilter(QObject *watched, QEvent *event)
545 {
546  Q_D(DFMCrumbBar);
547  QMouseEvent *me = nullptr;
548  if (watched && watched->parent() == &d->crumbListView && (me = dynamic_cast<QMouseEvent *>(event))) {
549  QEvent::Type type = event->type();
550  bool isMousePressed = type == QEvent::MouseButtonPress || type == QEvent::MouseButtonDblClick;
551  static QPoint pos;
552  if (isMousePressed) {
553  pos = QCursor::pos();
554  }
555 
556  bool isIgnore = isMousePressed || type == QEvent::MouseMove;
557  if (isIgnore) {
558  event->ignore();
559  return true;
560  }
561 
562  bool isDragging = (pos - QCursor::pos()).manhattanLength() > QApplication::startDragDistance();
563  if (type == QEvent::MouseButtonRelease && me->button() == Qt::LeftButton && !isDragging) {
564  QModelIndex index = d->crumbListView.indexAt(me->pos());
565  if (index.isValid() && index != d->crumbListView.currentIndex()) {
566  d->crumbListView.clicked(index);
567  return true;
568  }
569  }
570  }
571 
572  return QFrame::eventFilter(watched, event);
573 }
