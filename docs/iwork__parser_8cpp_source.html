51 static std::string ParseDate(std::string& format, long value)
52 {
53  //by default value "0" is "1st january 2001 midnight". At least in iWorks
54  std::ostringstream os;
55  #warning TODO: mktime is not working for some reason... I will use my own function.
56  int months_length[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
57  int hour = 0;
58  int minute = 0;
59  int second = 0;
60  int day = 0;
61  int month = 0;
62  int year = 2001;
63  int year_in_seconds = 365 * 24 * 3600;
64  while (value < 0)
65  {
66  year -= 1;
67  if (((year % 100) != 0 && (year % 4) == 0) || ((year % 100) == 0 && ((year / 100) % 4) == 0))
68  {
69  year_in_seconds = 366 * 24 * 3600;
70  months_length[1] = 29;
71  }
72  else
73  {
74  year_in_seconds = 365 * 24 * 3600;
75  months_length[1] = 28;
76  }
77  value += year_in_seconds;
78  }
79  while (value >= year_in_seconds)
80  {
81  year += 1;
82  value -= year_in_seconds;
83  if (((year % 100) != 0 && (year % 4) == 0) || ((year % 100) == 0 && ((year / 100) % 4) == 0))
84  {
85  year_in_seconds = 366 * 24 * 3600;
86  months_length[1] = 29;
87  }
88  else
89  {
90  year_in_seconds = 365 * 24 * 3600;
91  months_length[1] = 28;
92  }
93  }
94  int month_in_seconds;
95  month_in_seconds = months_length[month] * 24 * 3600;
96  while (value >= month_in_seconds)
97  {
98  month += 1;
99  value -= month_in_seconds;
100  month_in_seconds = months_length[month] * 24 * 3600;
101  }
102  while (value >= 24 * 3600)
103  {
104  day += 1;
105  value -= 24 * 3600;
106  }
107  while (value >= 3600)
108  {
109  hour += 1;
110  value -= 3600;
111  }
112  while (value >= 60)
113  {
114  minute += 1;
115  value -= 60;
116  }
117  second = value;
118  std::string am_pm = "AM";
119  if (hour >= 12) //correct, we are using 12-hour clock in iWorks. Or am I missing something?
120  {
121  am_pm = "PM";
122  hour -= 12;
123  }
124  if (hour == 0)
125  hour = 12;
126  std::string element;
127  for (int i = 0; i < format.length(); ++i)
128  {
129  bool finished_element = false;
130  char c = format[i];
131 
132  if (c == 'd' || c == 'M' || c == 'y' || c == 'Y' || c == 'h' || c == 'm' || c == 's' || c == 'a')
133  element += c;
134  else
135  finished_element = true;
136 
137  if (i == format.length() - 1 || finished_element)
138  {
139  if (element == "d")
140  os << (day + 1);
141  else if (element == "dd")
142  {
143  if (day + 1 < 10)
144  os << 0;
145  os << (day + 1);
146  }
147  else if (element == "M")
148  os << (month + 1);
149  else if (element == "MM")
150  {
151  if (month + 1 < 10)
152  os << 0;
153  os << (month + 1);
154  }
155  else if (element == "yy" || element == "YY")
156  {
157  if (year % 100 < 10)
158  os << 0;
159  os << (year % 100);
160  }
161  else if (element == "yyyy" || element == "YYYY")
162  os << year;
163  else if (element == "h")
164  os << hour;
165  else if (element == "hh")
166  {
167  if (hour < 10)
168  os << 0;
169  os << hour;
170  }
171  else if (element == "m")
172  os << minute;
173  else if (element == "mm")
174  {
175  if (minute < 10)
176  os << 0;
177  os << minute;
178  }
179  else if (element == "s")
180  os << second;
181  else if (element == "ss")
182  {
183  if (second < 10)
184  os << 0;
185  os << second;
186  }
187  else if (element == "a")
188  os << am_pm;
189  element.clear();
190  }
191  if (finished_element)
192  os << format[i];
193  }
194  return os.str();

197 static std::string ParseDuration(std::string& format, long value)
198 {
199  std::ostringstream os;
200  std::string element;
201  for (int i = 0; i < format.length(); ++i)
202  {
203  bool finished_element = false;
204  char c = format[i];
205 
206  if (c == 'h' || c == 'm' || c == 's')
207  element += c;
208  else
209  finished_element = true;
210 
211  if (i == format.length() - 1 || finished_element)
212  {
213  if (element == "hhh" || element == "hhhh") //hour?
214  {
215  int h = value / 3600;
216  os << h;
217  os << "h";
218  value -= h * 3600;
219  }
220  else if (element == "mmm" || element == "mmmm") //minute?
221  {
222  int m = value / 60;
223  os << m;
224  os << "m";
225  value -= m * 60;
226  }
227  else if (element == "sss" || element == "ssss") //second?
228  {
229  int s = value;
230  os << s;
231  os << "s";
232  value -= s;
233  }
234  element.clear();
235  }
236  if (finished_element)
237  os << format[i];
238  }
239  return os.str();

367  {
368  if (!m_tag_opened)
369  return true;
370  bool done = false;
371  bool reading_attribute_name = true;
372  bool reading_attribute_value = false;
373  std::string attribute_name = "";
374  std::string* attribute_value;
375  attribute_name.reserve(50);
376  while (!done)
377  {
378  for (; m_pointer < m_chunk_len; ++m_pointer)
379  {

381  {
382  done = true;
383  m_tag_opened = false;
384  ++m_pointer;
385  return true;
386  }
387  else if (reading_attribute_value == false && m_chunk[m_pointer] == '/')
388  {
389  xml_element.m_kind = empty_element;
390  }
391  else if (reading_attribute_name && m_chunk[m_pointer] != ' ')
392  {
393  if (m_chunk[m_pointer] != '=')
394  attribute_name += m_chunk[m_pointer];
395  else
396  {
397  reading_attribute_name = false;
398  attribute_value = &xml_element.m_attributes[attribute_name];
399  attribute_value->reserve(50);
400  attribute_name.clear();
401  }
402  }
403  else
404  {
405  if (!reading_attribute_value)
406  {
407  if (m_chunk[m_pointer] == '"')
408  reading_attribute_value = true;
409  }
410  else
411  {
412  if (m_chunk[m_pointer] != '"')
413  *attribute_value += m_chunk[m_pointer];
414  else
415  {
416  reading_attribute_value = false;
417  reading_attribute_name = true;
418  }
419  }
420  }
421  }
422  if (!done)
423  {
424  if (!m_data_source->ReadChunk(m_chunk, 1024, true, m_chunk_len))
425  return false;

442  {
443  xml_element.Clear();
444  char c;
445  while (true)
446  {

451  {

453  {
454  xml_element.m_kind = characters;
455  return true;
456  }
457  m_tag_opened = true;
458  }
459  else if (c == '>')
460  {
461  m_tag_opened = false;
462  ++m_pointer;
463  return true;
464  }
465  else if (m_tag_opened == false) //some character beetwen > and <
466  {
467  xml_element.m_characters += c;
468  }
469  else if (c > 32 && c != '/') //some character beetwen < and > (but not '/' or space)
470  {
471  xml_element.m_name += c;
472  }
473  else if (c == '/')
474  {
475  if (xml_element.m_name.length() == 0) //</something>
476  xml_element.m_kind = closing_element;
477  else //<something/>
478  xml_element.m_kind = empty_element;
479  }
480  else
481  {
482  //this is an element with attributes: <something [attributes].
483  //For now I'm not going to read them (attributes)
484  ++m_pointer;
485  return true;
486  }
487  }
488  if (!m_data_source->ReadChunk(m_chunk, 1024, false, m_chunk_len))
489  return false;
490  m_pointer = 0;
491  if (m_chunk_len == 0) //end of data
492  {
493  xml_element.m_kind = no_element;

495  }
496  }

608  void AddCell()
609  {
610  CellStyle* cell_style = m_current_cell.m_cell_style;
611  NumberFormat* number_format = cell_style == NULL ? NULL : cell_style->m_number_format;
612  std::ostringstream os;
613 
614  if (m_current_cell.m_is_duration)
615  {
616  std::string duration_format = "hhh mmm sss";
617  if (cell_style != NULL && cell_style->m_duration_format.length() > 0)
618  duration_format = cell_style->m_duration_format;
619  long value = strtol(m_current_cell.m_value.c_str(), NULL, 0);
620  m_current_cell.m_value = ParseDuration(duration_format, value);
621  }
622  else if (m_current_cell.m_is_date)
623  {
624  std::string date_format = "dd.MM.yyyy hh:mm a";
625  if (cell_style != NULL && cell_style->m_date_format.length() > 0)
626  date_format = cell_style->m_date_format;
627  long value = strtol(m_current_cell.m_value.c_str(), NULL, 0);
628  m_current_cell.m_value = ParseDate(date_format, value);
629  }
630  else if (m_current_cell.m_is_number)
631  {
632  double value = strtod(m_current_cell.m_value.c_str(), NULL);
633  if (number_format == NULL)
634  os << std::fixed << (long)value;
635  else
636  {
637  //format string looks like: ¤#,##0.00;-¤#,##0.00
638  //after ';' we have style for negative number. If '-' is missing, we should get absolute value
639  if (value < 0.0 && number_format->m_format_string.find(";-") == std::string::npos)
640  value = -value;
641  switch (number_format->m_type_numbers)
642  {
643  default: //others
644  case 0: //number
645  {
646  if (number_format->m_fraction_precision > 0)
647  {
648  os.precision(number_format->m_fraction_precision);
649  os << std::fixed << value;
650  }
651  else
652  os << std::fixed << (long)round(value);
653  break;
654  }
655  case 1: //money
656  {
657  if (number_format->m_currency == "EUR")
658  os << "€";
659  else if (number_format->m_currency == "USD" || number_format->m_currency == "MXN")
660  os << "$";
661  if (number_format->m_fraction_precision > 0)
662  {
663  os.precision(number_format->m_fraction_precision);
664  os << std::fixed << value;
665  }
666  else
667  os << std::fixed << (long)round(value);
668  break;
669  }
670  case 2: //percentage
671  {
672  value *= 100;
673  if (number_format->m_fraction_precision > 0)
674  {
675  os.precision(number_format->m_fraction_precision);
676  os << std::fixed << value;
677  }
678  else
679  os << std::fixed << (long)round(value);
680  os << "%";
681  break;
682  }
683  }
684  m_current_cell.m_value = os.str();
685  }
686  }
687  if (!m_in_pm)
688  *m_text_pointer += m_current_cell.m_value;
689  else
690  m_pms[m_selected_pm] += m_current_cell.m_value;
691  m_current_cell.Clear();
692  }
