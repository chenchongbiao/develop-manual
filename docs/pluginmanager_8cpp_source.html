60 void PluginManager::loadPlugin()
61 {
62  Q_D(PluginManager);
63  QStringList pluginChildDirs;
64  d->expandInfoInterfaces.clear();
65  d->viewInterfaces.clear();
66  d->previewInterfaces.clear();
67 
68  QStringList pluginDirs = DFMGlobal::PluginLibraryPaths;
69 
70  foreach (QString dir, pluginDirs) {
71  QDir pluginDir(dir);
72  qDebug() << dir;
73  pluginChildDirs << "view" << "preview";
74  foreach (QString childDir, pluginChildDirs) {
75  QDir childPluginDir(pluginDir.absoluteFilePath(childDir));
76  qDebug() << "load plugin in: " << childPluginDir.absolutePath();
77  foreach (QString fileName, childPluginDir.entryList(QDir::Files)) {
78  QPluginLoader pluginLoader(childPluginDir.absoluteFilePath(fileName));
79  QObject *plugin = pluginLoader.instance();
80  if (plugin) {
81  qDebug() << plugin;
82 
83  PropertyDialogExpandInfoInterface *expandInfoInterface = qobject_cast<PropertyDialogExpandInfoInterface *>(plugin);
84  if (expandInfoInterface) {
85  d->expandInfoInterfaces.append(expandInfoInterface);
86  }
87 
88  ViewInterface *viewInterface = qobject_cast<ViewInterface *>(plugin);
89  if (viewInterface) {
90  d->viewInterfaces.append(viewInterface);
91  d->viewInterfacesMap.insert(viewInterface->scheme(), viewInterface);
92  }
93 
94  PreviewInterface *previewInterface = qobject_cast<PreviewInterface *> (plugin);
95  if (previewInterface) {
96  static int counter = 0;
97  counter ++;
98  d->previewInterfaces << previewInterface;
99  }
100  }
101  }
102  }
103  }
104  qDebug() << "expand info size:" << d->expandInfoInterfaces.size();
105  qDebug() << "view size:" << d->viewInterfaces.size();
106 }
