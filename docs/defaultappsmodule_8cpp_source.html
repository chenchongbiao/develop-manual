94 int DefaultAppsModule::load(const QString &path)
95 {
96  QMap<QString, dcc::defapp::DefAppWorker::DefaultAppsCategory> maps = {
97  { QStringLiteral("Webpage"), dcc::defapp::DefAppWorker::Browser},
98  { QStringLiteral("Mail"), dcc::defapp::DefAppWorker::Mail},
99  { QStringLiteral("Text"), dcc::defapp::DefAppWorker::Text},
100  { QStringLiteral("Music"), dcc::defapp::DefAppWorker::Music},
101  { QStringLiteral("Video"), dcc::defapp::DefAppWorker::Video},
102  { QStringLiteral("Picture"), dcc::defapp::DefAppWorker::Picture},
103  { QStringLiteral("Terminal"), dcc::defapp::DefAppWorker::Terminal},
104  };
105  QString loadPath = path.split("/").at(0);
106  dcc::defapp::DefAppWorker::DefaultAppsCategory currentCategory = maps.value(loadPath, dcc::defapp::DefAppWorker::Browser);
107 
108  showDetailWidget(currentCategory);
109  Q_EMIT requestSetDefappCategory(currentCategory);
110  return 0;
111 }
