405 void MainWindow::resetDragWindow()
406 {
407  switch (m_multiScreenWorker->position()) {
408  case Dock::Top:
409  m_dragWidget->setGeometry(0, height() - DRAG_AREA_SIZE, width(), DRAG_AREA_SIZE);
410  break;
411  case Dock::Bottom:
412  m_dragWidget->setGeometry(0, 0, width(), DRAG_AREA_SIZE);
413  break;
414  case Dock::Left:
415  m_dragWidget->setGeometry(width() - DRAG_AREA_SIZE, 0, DRAG_AREA_SIZE, height());
416  break;
417  case Dock::Right:
418  m_dragWidget->setGeometry(0, 0, DRAG_AREA_SIZE, height());
419  break;
420  }
421 
422  QRect rect = m_multiScreenWorker->dockRect(m_multiScreenWorker->deskScreen()
423  , m_multiScreenWorker->position()
424  , HideMode::KeepShowing
425  , m_multiScreenWorker->displayMode());
426 
427  // 这个时候屏幕有可能是隐藏的，不能直接使用this->width()这种去设置任务栏的高度，而应该保证原值
428  int dockSize = 0;
429  if (m_multiScreenWorker->position() == Position::Left
430  || m_multiScreenWorker->position() == Position::Right) {
431  dockSize = this->width() == 0 ? rect.width() : this->width();
432  } else {
433  dockSize = this->height() == 0 ? rect.height() : this->height();
434  }
435 
442  dockSize = qBound(MAINWINDOW_MIN_SIZE, dockSize, MAINWINDOW_MAX_SIZE);
443 
444  // 通知窗管和后端更新数据
445  m_multiScreenWorker->updateDaemonDockSize(dockSize); // 1.先更新任务栏高度
446  m_multiScreenWorker->requestUpdateFrontendGeometry(); // 2.再更新任务栏位置,保证先1再2
447  m_multiScreenWorker->requestNotifyWindowManager();
448  m_multiScreenWorker->requestUpdateRegionMonitor(); // 界面发生变化，应更新监控区域
449 
450  if ((Top == m_multiScreenWorker->position()) || (Bottom == m_multiScreenWorker->position())) {
451  m_dragWidget->setCursor(Qt::SizeVerCursor);
452  } else {
453  m_dragWidget->setCursor(Qt::SizeHorCursor);
454  }
455 }

457 void MainWindow::onMainWindowSizeChanged(QPoint offset)
458 {
459  const QRect &rect = m_multiScreenWorker->dockRect(m_multiScreenWorker->deskScreen()
460  , m_multiScreenWorker->position()
461  , HideMode::KeepShowing,
462  m_multiScreenWorker->displayMode());
463  QRect newRect;
464  switch (m_multiScreenWorker->position()) {
465  case Top: {
466  newRect.setX(rect.x());
467  newRect.setY(rect.y());
468  newRect.setWidth(rect.width());
469  newRect.setHeight(qBound(MAINWINDOW_MIN_SIZE, rect.height() + offset.y(), MAINWINDOW_MAX_SIZE));
470  }
471  break;
472  case Bottom: {
473  newRect.setX(rect.x());
474  newRect.setY(rect.y() + rect.height() - qBound(MAINWINDOW_MIN_SIZE, rect.height() - offset.y(), MAINWINDOW_MAX_SIZE));
475  newRect.setWidth(rect.width());
476  newRect.setHeight(qBound(MAINWINDOW_MIN_SIZE, rect.height() - offset.y(), MAINWINDOW_MAX_SIZE));
477  }
478  break;
479  case Left: {
480  newRect.setX(rect.x());
481  newRect.setY(rect.y());
482  newRect.setWidth(qBound(MAINWINDOW_MIN_SIZE, rect.width() + offset.x(), MAINWINDOW_MAX_SIZE));
483  newRect.setHeight(rect.height());
484  }
485  break;
486  case Right: {
487  newRect.setX(rect.x() + rect.width() - qBound(MAINWINDOW_MIN_SIZE, rect.width() - offset.x(), MAINWINDOW_MAX_SIZE));
488  newRect.setY(rect.y());
489  newRect.setWidth(qBound(MAINWINDOW_MIN_SIZE, rect.width() - offset.x(), MAINWINDOW_MAX_SIZE));
490  newRect.setHeight(rect.height());
491  }
492  break;
493  }
494 
495  // 更新界面大小
496  m_mainPanel->setFixedSize(newRect.size());
497  setFixedSize(newRect.size());
498  move(newRect.topLeft());
499 }

517 void MainWindow::touchRequestResizeDock()
518 {
519  const QPoint touchPos(QCursor::pos());
520  QRect dockRect = m_multiScreenWorker->dockRect(m_multiScreenWorker->deskScreen()
521  , m_multiScreenWorker->position()
522  , HideMode::KeepShowing
523  , m_multiScreenWorker->displayMode());
524 
525  // 隐藏状态返回
526  if (width() == 0 || height() == 0) {
527  return;
528  }
529 
530  QGSettings settings("com.deepin.dde.dock.touch", QByteArray(), this);
531  int resizeHeight = settings.get("resizeHeight").toInt();
532 
533  QRect touchRect;
534  // 任务栏屏幕 内侧边线 内外resizeHeight距离矩形区域内长按可拖动任务栏高度
535  switch (m_multiScreenWorker->position()) {
536  case Position::Top:
537  touchRect = QRect(dockRect.x(), dockRect.y() + dockRect.height() - resizeHeight, dockRect.width(), resizeHeight * 2);
538  break;
539  case Position::Bottom:
540  touchRect = QRect(dockRect.x(), dockRect.y() - resizeHeight, dockRect.width(), resizeHeight * 2);
541  break;
542  case Position::Left:
543  touchRect = QRect(dockRect.x() + dockRect.width() - resizeHeight, dockRect.y(), resizeHeight * 2, dockRect.height());
544  break;
545  case Position::Right:
546  touchRect = QRect(dockRect.x() - resizeHeight, dockRect.y(), resizeHeight * 2, dockRect.height());
547  break;
548  }
549 
550  if (!touchRect.contains(touchPos)) {
551  return;
552  }
553  qApp->postEvent(m_dragWidget, new QMouseEvent(QEvent::MouseButtonPress, m_dragWidget->mapFromGlobal(touchPos)
554  , QPoint(), touchPos, Qt::LeftButton, Qt::NoButton
555  , Qt::NoModifier, Qt::MouseEventSynthesizedByApplication));
556 }
