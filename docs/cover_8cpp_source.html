109 void Cover::setCoverPixmap(const QPixmap &pixmap)
110 {
111  int radius = m_radius;
112 
113  auto ratio = this->devicePixelRatioF();
114  auto coverRect = QRectF(rect()).marginsRemoved(outterMargins);
115  auto sz = coverRect.size().toSize() * ratio;
116 
117  // 添加非空判断，减少警告日志输出与不必要的性能损耗
118  QPixmap backgroundPixmap;
119  if (!pixmap.isNull()) {
120  backgroundPixmap = pixmap.scaled(sz, Qt::KeepAspectRatioByExpanding);
121  }
122  // TODO: fix ratio by Dtk::Widget::DHiDPIHelper
123  backgroundPixmap.setDevicePixelRatio(1);
124  QPixmap maskPixmap(sz);
125  maskPixmap.fill(Qt::transparent);
126  QPainterPath path;
127  path.addRoundedRect(QRectF(0, 0, sz.width(), sz.height()), double(radius), double(radius));
128  QPainter bkPainter(&maskPixmap);
129  bkPainter.setRenderHints(QPainter::Antialiasing | QPainter::HighQualityAntialiasing);
130  bkPainter.fillPath(path, QBrush(Qt::red));
131 
132  QPainter::CompositionMode mode = QPainter::CompositionMode_SourceIn;
133  QImage resultImage = QImage(sz, QImage::Format_ARGB32_Premultiplied);
134  QPainter painter(&resultImage);
135  painter.setCompositionMode(QPainter::CompositionMode_Source);
136  painter.fillRect(resultImage.rect(), Qt::transparent);
137  painter.setCompositionMode(QPainter::CompositionMode_SourceOver);
138  painter.drawImage(0, 0, maskPixmap.toImage());
139  painter.setCompositionMode(mode);
140  painter.drawImage(0, 0, backgroundPixmap.toImage());
141  painter.setCompositionMode(QPainter::CompositionMode_DestinationOver);
142  painter.end();
143 
144  m_Background = QPixmap::fromImage(resultImage);
145 }
