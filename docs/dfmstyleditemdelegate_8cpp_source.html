219 QPixmap DFMStyledItemDelegate::getIconPixmap(const QIcon &icon, const QSize &size, qreal pixelRatio = 1.0, QIcon::Mode mode, QIcon::State state)
220 {
221  // ###(zccrs): 开启Qt::AA_UseHighDpiPixmaps后，QIcon::pixmap会自动执行 pixmapSize *= qApp->devicePixelRatio()
222  // 而且，在有些QIconEngine的实现中，会去调用另一个QIcon::pixmap，导致 pixmapSize 在这种嵌套调用中越来越大
223  // 最终会获取到一个是期望大小几倍的图片，由于图片太大，会很快将 QPixmapCache 塞满，导致后面再调用QIcon::pixmap
224  // 读取新的图片时无法缓存，非常影响图片绘制性能。此处在获取图片前禁用 Qt::AA_UseHighDpiPixmaps，自行处理图片大小问题
225  bool useHighDpiPixmaps = qApp->testAttribute(Qt::AA_UseHighDpiPixmaps);
226  qApp->setAttribute(Qt::AA_UseHighDpiPixmaps, false);
227 
228  if (icon.isNull())
229  return QPixmap();
230 
231  QSize icon_size = icon.actualSize(size, mode, state);
232  //取出icon的真实大小
233  QList<QSize> iconSizeList = icon.availableSizes();
234  QSize iconRealSize;
235  if (iconSizeList.count() > 0)
236  iconRealSize = iconSizeList.first();
237  else
238  iconRealSize = icon_size;
239  if (iconRealSize.width() <= 0 || iconRealSize.height() <= 0) {
240 // return icon.pixmap(iconRealSize);
241  return icon.pixmap(icon_size);
242  }
243 
244  //确保特殊比例icon的高或宽不为0
245  bool isSpecialSize = false;
246  QSize tempSize(size.width(), size.height());
247  while (icon_size.width() < 1) {
248  tempSize.setHeight(tempSize.height() * 2);
249  icon_size = icon.actualSize(tempSize, mode, state);
250  isSpecialSize = true;
251  }
252  while (icon_size.height() < 1) {
253  tempSize.setWidth(tempSize.width() * 2);
254  icon_size = icon.actualSize(tempSize, mode, state);
255  isSpecialSize = true;
256  }
257 
258  if ((icon_size.width() > size.width() || icon_size.height() > size.height()) && !isSpecialSize)
259  icon_size.scale(size, Qt::KeepAspectRatio);
260 
261  QSize pixmapSize = icon_size * pixelRatio;
262  QPixmap px = icon.pixmap(pixmapSize, mode, state);
263 
264  // restore the value
265  qApp->setAttribute(Qt::AA_UseHighDpiPixmaps, useHighDpiPixmaps);
266 
267 // if (icon_size.width() == 0) {
268 // icon_size.setWidth(1);
269 // }
270 // if (icon_size.height() == 0) {
271 // icon_size.setHeight(1);
272 // }
273 
274 // icon_size = icon_size * pixelRatio;
275 // px = px.scaled(icon_size.width(), icon_size.height(), Qt::IgnoreAspectRatio, Qt::SmoothTransformation);
276 
277  //约束特殊比例icon的尺寸
278  if (isSpecialSize) {
279  if (px.width() > size.width() * pixelRatio) {
280  px = px.scaled(size.width() * pixelRatio, px.height(), Qt::IgnoreAspectRatio);
281  } else if (px.height() > size.height() * pixelRatio) {
282  px = px.scaled(px.width(), size.height() * pixelRatio, Qt::IgnoreAspectRatio);
283  }
284  }
285 
286  if (px.width() > icon_size.width() * pixelRatio) {
287  px.setDevicePixelRatio(px.width() / (qreal)icon_size.width());
288  } else if (px.height() > icon_size.height() * pixelRatio) {
289  px.setDevicePixelRatio(px.height() / (qreal)icon_size.height());
290  } else {
291  px.setDevicePixelRatio(pixelRatio);
292  }
293 
294  return px;
295 }

298 {

315 void DFMStyledItemDelegate::paintIcon(QPainter *painter, const QIcon &icon, const QRectF &rect, Qt::Alignment alignment, QIcon::Mode mode, QIcon::State state)
316 {
317  if (DEEPIN_QT_THEME::followColorScheme
318  && (*DEEPIN_QT_THEME::followColorScheme)()
319  && painter->device()->devType() == QInternal::Widget) {
320  const QWidget *widget = dynamic_cast<QWidget *>(painter->device());
321  const QPalette &pal = widget->palette();
322  DEEPIN_QT_THEME::colorScheme.setLocalData(mode == QIcon::Selected ? pal.highlightedText().color().name() : pal.windowText().color().name());
323  }
324 
325  // Copy of QStyle::alignedRect
326  alignment = visualAlignment(painter->layoutDirection(), alignment);
327  const qreal pixel_ratio = painter->device()->devicePixelRatioF();
328  const QPixmap &px = getIconPixmap(icon, rect.size().toSize(), pixel_ratio, mode, state);
329  qreal x = rect.x();
330  qreal y = rect.y();
331  qreal w = px.width() / px.devicePixelRatio();
332  qreal h = px.height() / px.devicePixelRatio();
333  if ((alignment & Qt::AlignVCenter) == Qt::AlignVCenter)
334  y += (rect.size().height() - h) / 2.0;
335  else if ((alignment & Qt::AlignBottom) == Qt::AlignBottom)
336  y += rect.size().height() - h;
337  if ((alignment & Qt::AlignRight) == Qt::AlignRight)
338  x += rect.size().width() - w;
339  else if ((alignment & Qt::AlignHCenter) == Qt::AlignHCenter)
340  x += (rect.size().width() - w) / 2.0;
341 
342  painter->drawPixmap(qRound(x), qRound(y), px);
343 }
