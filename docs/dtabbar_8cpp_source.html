168  QImage img(300, m_height, QImage::Format_ARGB32);

172  QColor color(Qt::yellow);
173  pen.setStyle(Qt::SolidLine);
174  pen.setWidth(1);

178  font.setPixelSize(12);

181  QString str = fm.elidedText(m_tabText, Qt::ElideRight, 300 - 10);

189  painter.drawText((300 - fm.width(str)) / 2, (m_height - fm.height()) / 2,

198  painter.drawPath(path);

225 QString Tab::getDisplayNameByUrl(const DUrl &url) const
226 {
227  QString urlDisplayName;
228  if (url.isComputerFile()) {
229  if (systemPathManager->isSystemPath(url.toString()))
230  urlDisplayName = systemPathManager->getSystemPathDisplayNameByPath(url.toString());
231  } else if (url == DUrl::fromTrashFile("/")) {
232  urlDisplayName = systemPathManager->getSystemPathDisplayName("Trash");
233  } else if (PluginManager::instance()->getViewInterfacesMap().keys().contains(url.scheme())) {
234  urlDisplayName = PluginManager::instance()->getViewInterfaceByScheme(url.scheme())->bookMarkText();
235  } else if (url == DUrl(RECENT_ROOT)) {
236  urlDisplayName = systemPathManager->getSystemPathDisplayName("Recent");
237  } else {
238  const DAbstractFileInfoPointer &fileInfo = fileService->createFileInfo(this, url);
239 
240  if (fileInfo)
241  urlDisplayName = fileInfo->fileDisplayName();
242  }
243  return urlDisplayName;
244 }

258 void Tab::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
259 {
260  Q_UNUSED(widget)
261  Q_UNUSED(option)
262 
263  if (m_dragOutSide)
264  return;
265  QPen pen = painter->pen();
266  pen.setWidth(1);
267 
268  //draw text
269  QFont font;
270  font.setPixelSize(12);
271 
272  if (isChecked()) {
273  font.setWeight(QFont::Normal + 10);
274  } else {
275  font.setWeight(QFont::Normal);
276  }
277 
278  painter->setFont(font);
279  QFontMetrics fm(font);
280  QString str = fm.elidedText(m_tabText, Qt::ElideRight, m_width - 10);
281 
282  DPalette pal = DApplicationHelper::instance()->palette(widget);
283  QColor color;
284 
285  //draw backgound
286  if (isChecked()) {
287  color = pal.color(QPalette::Active, QPalette::Base);
288  color = DGuiApplicationHelper::adjustColor(color, 0, 0, 0, 0, 0, 0, +51);
289  painter->fillRect(boundingRect(), color);
290 
291  color = pal.color(QPalette::Active, QPalette::Highlight);
292  QPen tPen = painter->pen();
293  tPen.setColor(color);
294  painter->setPen(tPen);
295  painter->drawText((m_width - fm.width(str)) / 2, (m_height - fm.height()) / 2,
296  fm.width(str), fm.height(), 0, str);
297  } else if (m_hovered || (m_hovered && !isChecked())) {
298  color = pal.color(QPalette::Active, QPalette::Light);
299  color = DGuiApplicationHelper::adjustColor(color, 0, 0, 0, +2, +2, +2, +51);
300  painter->fillRect(boundingRect(), color);
301  painter->drawText((m_width - fm.width(str)) / 2, (m_height - fm.height()) / 2,
302  fm.width(str), fm.height(), 0, str);
303  } else {
304  color = pal.color(QPalette::Active, QPalette::Base);
305  color = DGuiApplicationHelper::adjustColor(color, 0, 0, 0, 0, 0, 0, +51);
306  painter->fillRect(boundingRect(), color);
307  painter->drawText((m_width - fm.width(str)) / 2, (m_height - fm.height()) / 2,
308  fm.width(str), fm.height(), 0, str);
309  }
310 
311  // draw line
312  pen.setColor(pal.color(QPalette::Inactive, QPalette::Shadow));
313  painter->setPen(pen);
314  int y = static_cast<int>(boundingRect().height());
315  int x = static_cast<int>(boundingRect().width());
316  qDrawShadeLine(painter, QPoint(x, 0), QPoint(x, y), pal);
317  QPalette::ColorGroup cp = isChecked() || m_hovered ? QPalette::Active : QPalette::Inactive;
318  pen.setColor(pal.color(cp, QPalette::Foreground));
319  painter->setPen(pen);
320 }

338 void Tab::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
339 {
340  if (m_isDragging && !m_borderLeft) {
341  m_borderLeft = true;
342  update();
343  }
344 
345  if (event->pos().y() < -m_height || event->pos().y() > m_height * 2) {
346 
347  if (!m_dragOutSide) {
348  m_dragOutSide = true;
349  update();
350  emit aboutToNewWindow(this);
351  emit draggingFinished();
352  m_dragObject = new QDrag(this);
353  QMimeData *mimeData = new QMimeData;
354  int radius = 20;
355 
356  const QPixmap &pixmap = toPixmap(true);
357  QImage image = Dtk::Widget::dropShadow(pixmap, radius, QColor(0, 0, 0, static_cast<int>(0.2 * 255)));
358  QPainter pa(&image);
359 
360  pa.drawPixmap(radius, radius, pixmap);
361 
362  m_dragObject->setPixmap(QPixmap::fromImage(image));
363  m_dragObject->setMimeData(mimeData);
364  m_dragObject->setHotSpot(QPoint(150 + radius, 12 + radius));
365  m_dragObject->exec();
366  m_dragObject->deleteLater();
367  m_pressed = false;
368 
369  DFileManagerWindow::flagForNewWindowFromTab.store(true, std::memory_order_seq_cst);
370  emit requestNewWindow(currentUrl());
371  }
372  }
373 
374  if (m_dragOutSide) {
375  QGraphicsObject::mouseMoveEvent(event);
376  return;
377  }
378 
379  if (pos().x() == 0 && pos().x() == scene()->width() - m_width) {
380  QGraphicsObject::mouseMoveEvent(event);
381  return;
382  }
383  setPos(x() + event->pos().x() - event->lastPos().x(), 0);
384  draggingStarted();
385  m_isDragging = true;
386  if (pos().x() < 0)
387  setX(0);
388  else if (pos().x() > scene()->width() - m_width)
389  setX(scene()->width() - m_width);
390 
391  if (pos().x() > m_originPos.x() + m_width / 2) {
392  emit moveNext(this);
393  m_originPos.setX(m_originPos.x() + m_width);
394  } else if (pos().x() < m_originPos.x() - m_width / 2) {
395  emit movePrevius(this);
396  m_originPos.setX(m_originPos.x() - m_width);
397  }
398 
399  QGraphicsObject::mouseMoveEvent(event);
400 }

609 void TabBar::removeTab(const int index, const bool &remainState)
610 {
611  Tab *tab = m_tabs.at(index);
612 
613  m_tabs.removeAt(index);
614  tab->deleteLater();
615 
616  if (m_TabCloseButton->closingIndex() <= count() - 1 &&
617  m_TabCloseButton->closingIndex() >= 0) {
618  m_lastDeleteState = remainState;
619  } else {
620  m_lastAddTabState = false;
621  // handle tab close button display position
622  if (remainState) {
623  QMouseEvent *event = new QMouseEvent(QMouseEvent::MouseMove,
624  mapFromGlobal(QCursor::pos()),
625  Qt::NoButton,
626  Qt::NoButton,
627  Qt::NoModifier);
628  mouseMoveEvent(event);
629  }
630  }
631 
632  if (index < count())
633  setCurrentIndex(index);
634  else
635  setCurrentIndex(count() - 1);
636  emit tabAddableChanged(count() < TAB_MAX_COUNT);
637 
638  if (count() < 2) {
639  m_lastDeleteState = false;
640  hide();
641  emit tabBarHidden();
642  }
643 }
