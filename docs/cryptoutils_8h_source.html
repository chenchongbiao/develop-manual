49  pointer allocate (size_type n, const void* hint = nullptr) {
50  UNUSED(hint)
51  if (n > std::numeric_limits<size_type>::max() / sizeof(T))
52  throw std::bad_alloc();
53  return static_cast<pointer> (::operator new (n * sizeof (value_type)));
54  }
55 
56  void deallocate(pointer p, size_type n) {
57  OPENSSL_cleanse(p, n*sizeof(T));
58  ::operator delete(p);
59  }

61  size_type max_size() const {
62  return std::numeric_limits<size_type>::max() / sizeof (T);
63  }
