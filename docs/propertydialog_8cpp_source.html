108 bool DFMRoundBackground::eventFilter(QObject *watched, QEvent *event)
109 {
110  if (watched == parent() && event->type() == QEvent::Paint) {
111  QWidget *w = dynamic_cast<QWidget *>(watched);
112  if (!w) {
113  return false;
114  }

118  QRectF bgRect;

120  const QPalette pal = QGuiApplication::palette();
121  QColor bgColor = pal.color(QPalette::Base);
122 
123  QPainterPath path;
124  path.addRoundedRect(bgRect, radius, radius);
125  // drawbackground color

127  painter.fillPath(path, bgColor);
128  painter.setRenderHint(QPainter::Antialiasing, false);
129  return true;
130  }
131 
132  return QObject::eventFilter(watched, event);
