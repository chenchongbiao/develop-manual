75 void SingerListView::sortList(QList<SingerInfo> &singerInfos, const DataBaseService::ListSortType &sortType)
76 {
77  switch (sortType) {
78  case DataBaseService::SortByAddTimeASC: {
79  qSort(singerInfos.begin(), singerInfos.end(), moreThanTimestampASC);
80  break;
81  }
82  case DataBaseService::SortBySingerASC: {
83  qSort(singerInfos.begin(), singerInfos.end(), moreThanSingerASC);
84  break;
85  }
86  case DataBaseService::SortByAddTimeDES: {
87  qSort(singerInfos.begin(), singerInfos.end(), moreThanTimestampDES);
88  break;
89  }
90  case DataBaseService::SortBySingerDES: {
91  qSort(singerInfos.begin(), singerInfos.end(), moreThanSingerDES);
92  break;
93  }
94  default:
95  break;
96  }
97 }

599 void SingerListView::slotRemoveSelectedSongs(const QString &deleteHash, const QStringList &musicHashs, bool removeFromLocal)
600 {
601  Q_UNUSED(removeFromLocal)
602  if (deleteHash != "all") {
603  return;
604  }
605  if (Player::getInstance()->getCurrentPlayListHash() != m_hash) {
606  return;
607  }
608  if (musicHashs.size() == 0) {
609  return;
610  }
611  // 标志准备删除的歌曲,找到当前正在播放的专辑index
612  int playIndex = -1;
613  for (int i = 0; i < singerModel->rowCount(); i++) {
614  QModelIndex idx = singerModel->index(i, 0, QModelIndex());
615  SingerInfo singerTmp = idx.data(Qt::UserRole).value<SingerInfo>();
616  if (singerTmp.musicinfos.contains(Player::getInstance()->getActiveMeta().hash)) {
617  playIndex = i;
618  }
619  for (MediaMeta meta : singerTmp.musicinfos.values()) {
620  if (musicHashs.contains(meta.hash)) {
621  singerTmp.musicinfos[meta.hash].toDelete = true;
622  }
623  }
624  QVariant singerval;
625  singerval.setValue(singerTmp);
626  singerModel->setData(idx, singerval, Qt::UserRole);
627  }
628  // 记录切换歌单之前播放状态
629  Player::PlaybackStatus preStatue = Player::getInstance()->status();
630  // 找到要播放的专辑
631  for (int i = playIndex; i < singerModel->rowCount(); i++) {
632  QModelIndex idx = singerModel->index(i, 0, QModelIndex());
633  SingerInfo singerTmp = idx.data(Qt::UserRole).value<SingerInfo>();
634  // 查看专辑中除去要删除的还有没有剩余歌曲
635  bool isExsit = false;
636  for (MediaMeta meta : singerTmp.musicinfos.values()) {
637  if (!meta.toDelete) {
638  isExsit = true;
639  break;
640  }
641  }
642  if (isExsit) {
643  // 当前专辑存在歌曲，播放该专辑
644  QList<MediaMeta> list;
645  // 创建播放队列
646  for (MediaMeta meta : singerTmp.musicinfos.values()) {
