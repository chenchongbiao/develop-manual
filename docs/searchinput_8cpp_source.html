67 void SearchInput::paintEvent(QPaintEvent *e)
68 {
69  QLineEdit::paintEvent(e);
70 
71  if(!hasFocus() && text().isEmpty())
72  {
73  QRect rect = this->rect();
74  QPainter painter(this);
75  painter.setRenderHint(QPainter::Antialiasing);
76  painter.setOpacity(0.5);
77  if(m_iconVisible)
78  {
79  QFontMetrics fm(qApp->font());
80  int w = fm.width(m_search);
81  int iw = m_icon.width();
82  int x = (rect.width() - w - iw -8)/2;
83  QRect iconRect(x, 0, iw, rect.height());
84  QRect tmp(QPoint(0,0),m_icon.size() / devicePixelRatioF());
85  tmp.moveCenter(iconRect.center());
86  QRect searchRect(iconRect.right() + 2, 0, w, rect.height());
87  painter.drawPixmap(tmp, m_icon);
88  painter.drawText(searchRect, Qt::AlignCenter, m_search);
89  }
90  else
91  {
92  painter.drawText(rect, Qt::AlignCenter, m_search);
93  }
94  }
95 }
