35 DesktopFile::DesktopFile(const QString &fileName) {
36 
37  // Store file name
38  m_fileName = fileName;
39 
40  // File validity
41  if (m_fileName.isEmpty() || !QFile::exists(fileName)) {
42  return;
43  }
44 
45  QSettings settings(fileName, QSettings::IniFormat);
46  settings.beginGroup("Desktop Entry");
47  // Loads .desktop file (read from 'Desktop Entry' group)
48  Properties desktop(fileName, "Desktop Entry");
49 
50  if(desktop.contains("X-Deepin-AppID")){
51  m_deepinId = desktop.value("X-Deepin-AppID", settings.value("X-Deepin-AppID")).toString();
52  }
53 
54  if(desktop.contains("X-Deepin-Vendor")){
55  m_deepinVendor = desktop.value("X-Deepin-Vendor", settings.value("X-Deepin-Vendor")).toString();
56  }
57 
58  if(desktop.contains("NoDisplay")){
59  m_noDisplay = desktop.value("NoDisplay", settings.value("NoDisplay").toBool()).toBool();
60  }
61  if(desktop.contains("Hidden")){
62  m_hidden = desktop.value("Hidden", settings.value("Hidden").toBool()).toBool();
63  }
64 
65  //由于获取的系统语言简写与.desktop的语言简写存在不对应关系，经决定先采用获取的系统值匹配
66  //若没匹配到则采用系统值"_"左侧的字符串进行匹配，均为匹配到，才走原未匹配流程
67  auto getValueFromSys = [&desktop,&settings](const QString &type, const QString &sysName)->QString {
68  const QString key = QString("%0[%1]").arg(type).arg(sysName);
69  return desktop.value(key, settings.value(key)).toString();
70  };
71 
72  auto getNameByType = [&desktop,&settings,&getValueFromSys](const QString &type)->QString{
73  QString tempSysName = QLocale::system().name();
74  QString targetName = getValueFromSys(type, tempSysName);
75  if (targetName.isEmpty()) {
76  auto strSize = tempSysName.trimmed().split("_");
77  if (!strSize.isEmpty()){
78  tempSysName = strSize.first();
79  targetName = getValueFromSys(type,tempSysName);
80  }
81 
82  if (targetName.isEmpty())
83  targetName = desktop.value(type, settings.value(type)).toString();
84  }
85 
86  return targetName;
87  };
88  m_localName = getNameByType("Name");
89  m_genericName = getNameByType("GenericName");
90 
91  m_exec = desktop.value("Exec", settings.value("Exec")).toString();
92  m_icon = desktop.value("Icon", settings.value("Icon")).toString();
93  m_type = desktop.value("Type", settings.value("Type", "Application")).toString();
94  m_categories = desktop.value("Categories", settings.value("Categories").toString()).toString().remove(" ").split(";");
95 
96  QString mime_type = desktop.value("MimeType", settings.value("MimeType").toString()).toString().remove(" ");
97 
98  if (!mime_type.isEmpty())
99  m_mimeType = mime_type.split(";");
100  // Fix categories
101  if (m_categories.first().compare("") == 0) {
102  m_categories.removeFirst();
103  }
104 }
