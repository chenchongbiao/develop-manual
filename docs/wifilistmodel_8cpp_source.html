73 QVariant WifiListModel::data(const QModelIndex &index, int role) const
74 {
75  if (!index.isValid())
76  return QVariant();
77 
78  const ItemInfo info = indexInfo(index.row());
79 
80  switch (role) {
81  case Qt::DisplayRole:
82  {
83  if (rowCount(QModelIndex()) == 1)
84  return tr("Click icon to enable WLAN");
85 
86  if (!info.info && info.device)
87  return deviceName(info.device);
88  else if (!info.device)
89  return tr("Connect to hidden network");
90  else
91  return info.info->value("Ssid");
92  }
93  case Qt::SizeHintRole:
94  if (rowCount(QModelIndex()) == 1)
95  return QSize(0, 36);
96  if (!info.info)
97  return QSize(0, 24);
98  else
99  return QSize(0, 36);
100  case ItemInfoRole:
101  return info.info ? *info.info : QVariant();
102  case ItemHoveredRole:
103  return index == m_currentIndex;
104  case ItemIsHeaderRole:
105  return !info.info && info.device;
106  case ItemIsActiveRole:
107  return info.info && static_cast<const WirelessDevice *>(info.device)->activeConnName() == info.info->value("Ssid").toString();
108  case ItemIsActivatingRole:
109  return m_refreshTimer->isActive() && info.info && m_activatingSsid == info.info->value("Ssid").toString();
110  case ItemDevicePathRole:
111  return info.device->path();
112  case ItemApPathRole:
113  return info.info ? info.info->value("Path") : QVariant();
114  case ItemUuidRole:
115  return info.info ? m_networkModel->connectionUuidByApInfo(static_cast<const WirelessDevice *>(info.device), info.info->value("Ssid").toString()) : QVariant();
116  case ItemIsHiddenTipsRole:
117  return !info.info && !info.device;
118  case ItemNextRole:
119  return m_currentIndex.row() + 1 == index.row();
120  case ItemIsPowerOffRole:
121  return rowCount(QModelIndex()) == 1;
122  case ItemCountRole:
123  return m_apInfoList.keys().count();
124  default:;
125  }
126 
127  return QVariant();
128 }

169 const ItemInfo WifiListModel::indexInfo(const int index) const
170 {
171  ItemInfo info;
172  int r = index;
173  for (auto *dev : m_networkModel->devices())
174  {
175  if (dev->type() != NetworkDevice::Wireless || !dev->enabled())
176  continue;
177 
178  WirelessDevice *d = static_cast<WirelessDevice *>(dev);
179  if (!m_apInfoList.contains(d))
180  continue;
181 
182  info.device = dev;
183 
184  if (!r)
185  return info;
186  else
187  --r;
188 
189  const int s = m_apInfoList[d].size();
190  if (r < s)
191  {
192  info.info = &m_apInfoList[d][r];
193  return info;
194  } else {
195  r -= s;
196  }
197  }
198 
199  info.device = nullptr;
200 
201  return info;
202 }

222 void WifiListModel::onDeviceListChanged(const QList<NetworkDevice *> &devices)
223 {
224  for (auto *dev : devices)
225  {
226  if (dev->type() != NetworkDevice::Wireless || !dev->enabled())
227  continue;
228 
229  WirelessDevice *d = static_cast<WirelessDevice *>(dev);
230 
231  if (m_apInfoList.contains(d))
232  continue;
233 
234  connect(d, &WirelessDevice::enableChanged, this, &WifiListModel::onDeviceEnableChanged, Qt::UniqueConnection);
235  connect(d, &WirelessDevice::activeConnectionChanged, this, &WifiListModel::refershActivatingIndex, Qt::UniqueConnection);
236  connect(d, &WirelessDevice::apAdded, this, &WifiListModel::onDeviceApAdded, Qt::UniqueConnection);
237  connect(d, &WirelessDevice::apInfoChanged, this, &WifiListModel::onDeviceApInfoChanged, Qt::UniqueConnection);
238  connect(d, &WirelessDevice::activeConnectionChanged, this, &WifiListModel::onDeviceActiveApChanged, Qt::UniqueConnection);
239  connect(d, static_cast<void (WirelessDevice::*)(const NetworkDevice::DeviceStatus) const>(&WirelessDevice::statusChanged), this, &WifiListModel::onDeviceStateChanged, Qt::UniqueConnection);
240  connect(d, &WirelessDevice::apRemoved, d, [=](const QJsonObject &apInfo) { onDeviceApRemoved(d, apInfo); }, Qt::UniqueConnection);
241 
242  emit requestDeviceApList(d->path());
243  }
244 
245  // removed devices
246  QList<WirelessDevice *> removedDeviceList;
247  for (auto it(m_apInfoList.cbegin()); it != m_apInfoList.cend(); ++it)
248  if (!devices.contains(it.key()))
249  removedDeviceList.append(it.key());
250  for (auto *d : removedDeviceList)
251  m_apInfoList.remove(d);
252 
253  sortApList();
254 
255  emit layoutChanged();
256 }

258 void WifiListModel::onDeviceApAdded(const QJsonObject &info)
259 {
260  WirelessDevice *dev = static_cast<WirelessDevice *>(sender());
261  Q_ASSERT(dev);
262 
263  if (!dev->enabled())
264  return;
265 
266  const auto &path = info.value("Path").toString();
267  const auto &ssid = info.value("Ssid").toString();
268  const auto &strength = info.value("Strength").toInt();
269 
270  // test if ap already exist.
271  int row = indexOf(dev);
272  auto &list = m_apInfoList[dev];
273  for (int i(0); i != list.size(); ++i) {
274  const auto &mAp = list.at(i);
275  // same Path means same ap, return
276  if (mAp.value("Path").toString() == path)
277  return;
278  // different Path but use the same Ssid, then compare Theirs Strength
279  else if (mAp.value("Ssid").toString() == ssid) {
280  if (mAp.value("Strength").toInt() < strength) {
281  row += i + 1;
282  list.replace(i, info);
283  auto changedIndex = index(row);
284  Q_EMIT dataChanged(changedIndex, changedIndex);
285  }
286  return;
287  }
288  }
289 
290  // reach here means it is a new ap need to add
291  beginInsertRows(QModelIndex(), row, row);
292  if (info.value("Ssid").toString() == dev->activeConnName()) {
293  m_activeConnNameMap.insert(dev, dev->activeConnName());
294  m_apInfoList[dev].insert(0, info);
295  } else {
296  m_apInfoList[dev].append(info);
297  }
298  sortApList();
299  endInsertRows();
300 }

394 void WifiListModel::sortApList()
395 {
396  QString *activeSsid = new QString();
397 
398  auto cmpFunc = [=](const QJsonObject &a, const QJsonObject &b) {
399  // make sure active ap is the first one of ap list
400  bool aIsActive = (a.value("Ssid").toString() == *activeSsid);
401  if (aIsActive || b.value("Ssid").toString() == *activeSsid) {
402  return aIsActive;
403  }
404  return a.value("Strength").toInt() > b.value("Strength").toInt();
405  };
406 
407  for (auto dev : m_apInfoList.keys()) {
408  *activeSsid = m_activeConnNameMap.value(dev);
409  QList<QJsonObject> list = m_apInfoList.value(dev);
410  std::sort(list.begin(), list.end(), cmpFunc);
411  m_apInfoList.insert(dev, list);
412  }
413 
414  delete activeSsid;
415 }
