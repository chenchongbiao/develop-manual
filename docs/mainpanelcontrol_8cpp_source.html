279 void MainPanelControl::removePluginAreaItem(QWidget *wdg)
280 {

389 void MainPanelControl::moveItem(DockItem *sourceItem, DockItem *targetItem)
390 {
391  // get target index
392  int idx = -1;
393  if (targetItem->itemType() == DockItem::App)
394  idx = m_appAreaSonLayout->indexOf(targetItem);
395  else if (targetItem->itemType() == DockItem::Plugins){
396  //因为日期时间插件大小和其他插件大小有异，为了设置边距，在各插件中增加了一层布局
397  //因此有拖动图标时，需要从多的一层布局中判断是否相同插件而获取插件位置顺序
398  for (int i = 0; i < m_pluginLayout->count(); ++i) {
399  QLayout *layout = m_pluginLayout->itemAt(i)->layout();
400  if (layout && layout->itemAt(0)->widget() == targetItem) {
401  idx = i;
402  break;
403  }
404  }
405  } else if (targetItem->itemType() == DockItem::FixedPlugin)
406  idx = m_fixedAreaLayout->indexOf(targetItem);
407  else
408  return;
409 
410  // remove old item
411  removeItem(sourceItem);
412 
413  // insert new position
414  if (sourceItem->isDragging()) {
415  m_dragIndex = idx;
416  }
417  insertItem(idx, sourceItem);
418 }

461 void MainPanelControl::handleDragMove(QDragMoveEvent *e, bool isFilter)
462 {
463  if (!e->source()) {
464  // 应用程序拖到dock上
465  e->accept();
466 
467  DockItem *insertPositionItem = dropTargetItem(nullptr, e->pos());
468 
469  if (m_placeholderItem.isNull()) {
470 
471  m_placeholderItem = new PlaceholderItem;
472 
473  if (m_position == Dock::Top || m_position == Dock::Bottom) {
474  if (m_appAreaSonWidget->mapFromParent(e->pos()).x() > m_appAreaSonWidget->rect().right()) {
475  // 插入到最右侧
476  insertPositionItem = nullptr;
477  }
478  } else {
479  if (m_appAreaSonWidget->mapFromParent(e->pos()).y() > m_appAreaSonWidget->rect().bottom()) {
480  // 插入到最下测
481  insertPositionItem = nullptr;
482  }
483  }
484 
485  insertItem(m_appAreaSonLayout->indexOf(insertPositionItem), m_placeholderItem);
486 
487  } else if (insertPositionItem && m_placeholderItem != insertPositionItem) {
488  moveItem(m_placeholderItem, insertPositionItem);
489  }
490 
491  return;
492  }
493 
494  DockItem *sourceItem = qobject_cast<DockItem *>(e->source());
495 
496  if (!sourceItem) {
497  e->ignore();
498  return;
499  }
500 
501  DockItem *targetItem = nullptr;
502 
503  if (isFilter) {
504  // appItem调整顺序或者移除驻留
505  targetItem = dropTargetItem(sourceItem, mapFromGlobal(m_appDragWidget->mapToGlobal(e->pos())));
506 
507  if (targetItem) {
508  m_appDragWidget->setOriginPos((m_appAreaSonWidget->mapToGlobal(targetItem->pos())));
509  } else {
510  targetItem = sourceItem;
511  }
512 
513  } else {
514  // other dockItem调整顺序
515  targetItem = dropTargetItem(sourceItem, e->pos());
516  }
517 
518  if (!targetItem) {
519  e->ignore();
520  return;
521  }
522 
523  e->accept();
524 
525  if (targetItem == sourceItem) {
526  m_dragIndex = -1;
527  return;
528  }
529 
530  moveItem(sourceItem, targetItem);
531  emit itemMoved(sourceItem, targetItem);
532 }

534 void MainPanelControl::dragMoveEvent(QDragMoveEvent *e)
535 {
536  DockItem *sourceItem = qobject_cast<DockItem *>(e->source());
537  if (sourceItem) {
538  handleDragMove(e, false);
539  return;
540  }
541 
542  // 拖app到dock上
543  const char *RequestDockKey = "RequestDock";
544  const char *RequestDockKeyFallback = "text/plain";
545  const char *DesktopMimeType = "application/x-desktop";
546  auto DragmineData = e->mimeData();
547 
548  m_draggingMimeKey = DragmineData->formats().contains(RequestDockKey) ? RequestDockKey : RequestDockKeyFallback;
549 
550  // dragging item is NOT a desktop file
551  if (QMimeDatabase().mimeTypeForFile(DragmineData->data(m_draggingMimeKey)).name() != DesktopMimeType) {
552  m_draggingMimeKey.clear();
553  e->setAccepted(false);
554  qDebug() << "dragging item is NOT a desktop file";
555  return;
556  }
557 
558  //如果当前从桌面拖拽的的app是trash，则不能放入app任务栏中
559  QString str = "file://";
560  //启动器
561  QString str_t = "";
562 
563  str.append(QStandardPaths::locate(QStandardPaths::DesktopLocation, "dde-trash.desktop"));
564  str_t.append(QStandardPaths::locate(QStandardPaths::ApplicationsLocation, "dde-trash.desktop"));
565 
566  if ((str == DragmineData->data(m_draggingMimeKey)) || (str_t == DragmineData->data(m_draggingMimeKey))) {
567  e->setAccepted(false);
568  return;
569  }
570 
571  if (m_delegate && m_delegate->appIsOnDock(DragmineData->data(m_draggingMimeKey))) {
572  e->setAccepted(false);
573  return;
574  }
575 
576  handleDragMove(e, false);
577 }

579 bool MainPanelControl::eventFilter(QObject *watched, QEvent *event)
580 {
581  if (watched == m_appAreaSonWidget) {
582  switch (event->type()) {
583  case QEvent::LayoutRequest:
584  m_appAreaSonWidget->adjustSize();
585  resizeDockIcon();
586  break;
587  case QEvent::Resize:
588  resizeDockIcon();
589  break;
590  default:
591  moveAppSonWidget();
592  break;
593  }
594  }
595 
596  if (watched == m_desktopWidget) {
597  if (event->type() == QEvent::Enter) {
598  if (checkNeedShowDesktop()) {
599  m_needRecoveryWin = true;
600  QProcess::startDetached("/usr/lib/deepin-daemon/desktop-toggle");
601  }
602  m_isHover = true;
603  update();
604  } else if (event->type() == QEvent::Leave) {
605  // 鼠标移入时隐藏了窗口，移出时恢复
606  if (m_needRecoveryWin) {
607  QProcess::startDetached("/usr/lib/deepin-daemon/desktop-toggle");
608  }
609  m_isHover = false;
610  update();
611  }
612  }
613 
614  if (watched == m_appAreaWidget) {
615  if (event->type() == QEvent::Resize)
616  updateAppAreaSonWidgetSize();
617 
618  if (event->type() == QEvent::Move)
619  moveAppSonWidget();
620  }
621 
622  if (m_appDragWidget && watched == static_cast<QGraphicsView *>(m_appDragWidget)->viewport()) {
623  QDropEvent *e = static_cast<QDropEvent *>(event);
624  bool isContains = rect().contains(mapFromGlobal(m_appDragWidget->mapToGlobal(e->pos())));
625  if (isContains) {
626  if (event->type() == QEvent::DragMove) {
627  handleDragMove(static_cast<QDragMoveEvent *>(event), true);
628  } else if (event->type() == QEvent::Drop) {
629  m_appDragWidget->hide();
630  return true;
631  }
632  }
633 
634  return false;
635  }
636 
637  if (event->type() != QEvent::MouseMove)
638  return false;
639 
640  QMouseEvent *mouseEvent = dynamic_cast<QMouseEvent *>(event);
641  if (!mouseEvent || mouseEvent->buttons() != Qt::LeftButton)
642  return false;
643 
644  DockItem *item = qobject_cast<DockItem *>(watched);
645  if (!item)
646  return false;
647
