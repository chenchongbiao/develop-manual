87 bool FileUtils::removeRecurse(const QString &path, const QString &name)
88 {
89  // File location
90  QString url = path + QDir::separator() + name;
91 
92  // Check whether file or directory exists
93  QFileInfo file(url);
94  if (!file.exists()) {
95  return false;
96  }
97 
98  // List of files that will be deleted
99  QStringList files;
100 
101  // If given file is a directory, collect all children of given directory
102  if (file.isDir()) {
103  QDirIterator it(url, QDir::AllEntries | QDir::System | QDir::NoDotAndDotDot | QDir::Hidden, QDirIterator::Subdirectories);
104  while (it.hasNext()) {
105  files.prepend(it.next());
106  }
107  }
108 
109  // Append given file to the list of files and delete all
110  files.append(url);
111  foreach (QString file, files) {
112  QFile(file).remove();
113  }
114  return true;
115 }

414 QString FileUtils::formatSize(qint64 num, bool withUnitVisible, int precision, int forceUnit, QStringList unitList)
415 {
416  if (num < 0) {
417  qWarning() << "Negative number passed to formatSize():" << num;
418  num = 0;
419  }
420 
421  bool isForceUnit = (forceUnit >= 0);
422  QStringList list;
423  qreal fileSize(num);
424 
425  if (unitList.size() == 0) {
426  list << " B" << " KB" << " MB" << " GB" << " TB"; // should we use KiB since we use 1024 here?
427  } else {
428  list = unitList;
429  }
430 
431  QStringListIterator i(list);
432  QString unit = i.hasNext() ? i.next() : QStringLiteral(" B");
433 
434  int index = 0;
435  while (i.hasNext()) {
436  if (fileSize < 1024 && !isForceUnit) {
437  break;
438  }
439 
440  if (isForceUnit && index == forceUnit) {
441  break;
442  }
443 
444  unit = i.next();
445  fileSize /= 1024;
446  index++;
447  }
448  QString unitString = withUnitVisible ? unit : QString();
449  return QString("%1%2").arg(sizeString(QString::number(fileSize, 'f', precision)), unitString);
450 }

501 QString FileUtils::defaultOpticalSize(const QString &tagName, quint64 &usedSize, quint64 &totalSize)
502 {
503  QString size{"0M"};
504 
505  // todo: should be beautify, copy from `diskcontrolitem.cpp`
506  // 解析配置文件，获取光驱数据暂存的内容（包含挂载状态，大小）
507  QFile burnCapacityFile(QString("%1/dde-file-manager.json").arg(DFMStandardPaths::location(DFMStandardPaths::ApplicationConfigPath)));
508  if (burnCapacityFile.open(QIODevice::ReadOnly)) {
509  QByteArray burnCapacityData = burnCapacityFile.readAll();
510  burnCapacityFile.close();
511 
512  QJsonParseError parseJsonErr;
513  QJsonDocument jsonDoc(QJsonDocument::fromJson(burnCapacityData, &parseJsonErr));
514  if (!(parseJsonErr.error == QJsonParseError::NoError)) {
515  qDebug() << "decode json file error！";
516  return size;
517  }
518 
519  int burnStatus{DFMOpticalMediaWidget::BCSA_BurnCapacityStatusEjct};
520  quint64 curTotalSize{0}; // 光盘总大小
521  quint64 curUsedSize{0}; // 光盘已使用的大小
522  QJsonObject tempBurnObjs = jsonDoc.object();
523  if (tempBurnObjs.contains(QStringLiteral("BurnCapacityAttribute"))) {
524  QJsonValue jsonBurnValueList = tempBurnObjs.value(QStringLiteral("BurnCapacityAttribute"));
525  QJsonObject tagItem = jsonBurnValueList.toObject();
526  // 从配置文件读取光盘信息
527  if (tagItem.contains(tagName)) {
528  QJsonObject burnItem = tagItem[tagName].toObject();
529  curTotalSize = static_cast<quint64>(burnItem["BurnCapacityTotalSize"].toDouble());
530  curUsedSize = static_cast<quint64>(burnItem["BurnCapacityUsedSize"].toDouble());
531  burnStatus = burnItem["BurnCapacityStatus"].toInt();
532  }
533  }
534 
535  // 光盘已挂载状态
536  if (burnStatus == DFMOpticalMediaWidget::BCSA_BurnCapacityStatusAddMount && curUsedSize > 0 && curTotalSize > 8) {
537  const qint64 kb = 1024;
538  const qint64 mb = 1024 * kb;
539  const QStringList unitDisplayText = {"B", "K", "M", "G", "T"};
540  // 重置原始的size数据
541  usedSize = curUsedSize;
542  totalSize = curTotalSize;
543  size = QString("%1/%2").arg(FileUtils::formatSize(static_cast<qint64>(usedSize), true, 1, usedSize < mb ? 2 : -1, unitDisplayText),
544  FileUtils::formatSize(static_cast<qint64>(totalSize), true, 1, totalSize < mb ? 2 : -1, unitDisplayText));
545  }
546  }
547 
548  return size;
549 }
