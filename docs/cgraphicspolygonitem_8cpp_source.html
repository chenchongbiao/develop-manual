101 void CGraphicsPolygonItem::calcPoints_helper(QVector<QPointF> &outVector, int n, const QRectF &rect, qreal offset)
102 {
103  if (n <= 0)return;
104  outVector.clear();
105 
106  QList<QLineF> lines;
107 
108  qreal angle = 90. * M_PI / 180.;
109  QPointF pointCenter = rect.center();
110  qreal w = rect.width();
111  qreal h = rect.height();
112  if (n) {
113  qreal preAngle = 360. / n * M_PI / 180.;
114  for (int i = 0; i != n; i++) {
115 
116  qreal curAngleDgree = angle + preAngle * i;
117  qreal x = pointCenter.x() + w / 2 * cos(curAngleDgree);
118  qreal y = pointCenter.y() - h / 2 * sin(curAngleDgree);
119 
120  outVector.push_back(QPointF(x, y));
121 
122  if (i != 0) {
123  QLineF line(outVector.at(i - 1), outVector.at(i));
124  lines.append(line);
125  }
126  }
127  }
128  lines.push_front(QLineF(outVector.last(), outVector.first()));
129 
130  if (!qFuzzyIsNull(offset)) {
131  for (int i = 0; i < lines.size(); ++i) {
132  QLineF curLine = lines[i];
133  QLineF nextLine = (i == lines.size() - 1 ? lines[0] : lines[i + 1]);
134 
135  qreal finalDegree = 180 - curLine.angleTo(nextLine); //两条线相交的交角*/
136 
137  qreal offLen = offset / qSin(qDegreesToRadians(finalDegree / 2.));
138 
139  QLineF tempLine(nextLine);
140  qreal newAngle = tempLine.angle() + finalDegree / 2.0;
141  tempLine.setAngle(newAngle);
142  tempLine.setLength(qAbs(offLen));
143  outVector[i] = tempLine.p2();
144  }
145  }
146 }
