47 void CPictureTool::addLocalImages(QStringList filePathList
48  , CDrawScene *scene
49  , bool asFirstImageSize
50  , bool addUndoRedo
51  , bool hadCreatedOneViewScene
52  , bool appFirstExec)
53 {
54  QStringList filenames = filePathList;
55  int exitPicNum = 0;
56 
57  //获取已导入图片数量
58  QList<QGraphicsItem *> items = scene->items();
59 
60  if (items.count() != 0) {
61  for (int i = 0; i < items.size(); i++) {
62  if (items[i]->type() == PictureType) {
63  exitPicNum = exitPicNum + 1;
64  };
65  }
66  }
67 
68  //大于30张报错，主要是适应各种系统环境，不给内存太大压力
69  if (exitPicNum + filenames.size() > 30) {
70  Dtk::Widget::DDialog warnDlg(/*centralWindow*/drawApp->topMainWindowWidget());
71  warnDlg.setIcon(QIcon::fromTheme("dialog-warning"));
72  warnDlg.setTitle(tr("You can import up to 30 pictures, please try again!"));
73  warnDlg.addButtons(QStringList() << tr("OK"));
74  warnDlg.setDefaultButton(0);
75  warnDlg.exec();
76  return;
77  }
78 
79  getProgressLayout()->setRange(0, filenames.size());
80 
81  // [bug:26525] 设置默认值
82  getProgressLayout()->setProgressValue(1);
83  getProgressLayout()->show();
84  getProgressLayout()->raise();
85 
86  //启动图片导入线程
87  QtConcurrent::run([ = ] {
88  QList<QString> failedFiles;
89  QList<QString> successFiles;
90  int processValue = 1;
91  bool firestImageAdjustScence = asFirstImageSize;
92  for (int i = 0; i < filenames.size(); i++)
93  {
94  QString file = filenames[i];
95 
96  QPixmap pixmap = readPixMapQuickly(file);
97 
98  if (pixmap.isNull()) {
99  failedFiles.append(file);
100  continue;
101  }
102 
103  successFiles.append(file);
104 
105  QFile f(file);
106  QByteArray srcBytes;
107  if (f.open(QFile::ReadOnly)) {
108  srcBytes = f.readAll();
109  }
110 
111  //QThread::msleep(100);
112  QMetaObject::invokeMethod(this, "addImage", Qt::QueuedConnection,
113  Q_ARG(QPixmap, pixmap),
114  Q_ARG(CDrawScene *, scene),
115  Q_ARG(const QByteArray &, srcBytes),
116  Q_ARG(bool, firestImageAdjustScence),
117  Q_ARG(bool, addUndoRedo),
118  Q_ARG(bool, i == filenames.size() - 1));
119 
120 
121  // 多张图片只以第一张图片的大小进行自适应
122  if (firestImageAdjustScence) {
123  firestImageAdjustScence = false;
124  }
125 
126  processValue++;
127 
128  //invoke process
129  QMetaObject::invokeMethod(this, [ = ]() {
130  this->getProgressLayout()->setProgressValue(processValue);
131  }, Qt::QueuedConnection);
132  }
133 
134  //invoke finished
135  QMetaObject::invokeMethod(this, [ = ]()
136  {
137  this->onLoadImageFinished(successFiles, failedFiles, asFirstImageSize, hadCreatedOneViewScene, appFirstExec);
138  this->getProgressLayout()->setProgressValue(filenames.size());
139  }, Qt::QueuedConnection);
140  });
141 }

143 QPixmap CPictureTool::readPixMapQuickly(const QString &imagePath)
144 {
145  auto fOptimalConditions = [ = ](const QImageReader & reader) {
146  QSize orgSz = reader.size();
147  return (orgSz.width() > 1920 && orgSz.height() > 1080);
148  };
149 
150  QPixmap pixmap;
151  QImageReader reader;
152  reader.setFileName(imagePath);
153  bool shouldOptimal = fOptimalConditions(reader);
154  qreal radio = shouldOptimal ? 0.5 : 1.0;
155  QSize orgSize = reader.size();
156  QSize newSize = orgSize * radio;
157  reader.setScaledSize(newSize);
158 
159 
160  if (shouldOptimal)
161  reader.setQuality(30);
162 
163  //当不能读取数据时,设置以内容判断格式进行加载,这样可以避免后缀名修改后文件无法打开的情况
164  if (!reader.canRead()) {
165  reader.setAutoDetectImageFormat(true);
166  reader.setDecideFormatFromContent(true);
167  reader.setFileName(imagePath); //必须重新设置一下文件,才能触发内部加载方式的切换
168  }
169 
170  if (reader.canRead()) {
171  QImage img = reader.read();
172  pixmap = QPixmap::fromImage(img);
173 
174  //维持原大小
175  bool haveOptimal = shouldOptimal;
176  if (haveOptimal) {
177  pixmap = pixmap.scaled(orgSize);
178  }
179  }
180 
181  //qDebug() << " QImageReader load used ms ================ " << ti.elapsed();;
182 
183  return pixmap;
184 }
