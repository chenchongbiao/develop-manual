216 void CExportImageDialog::slotOnSavePathChange(int index)
217 {
218  switch (index) {
219  case Pictures:
220  m_savePath = QStandardPaths::writableLocation(QStandardPaths::PicturesLocation);
221  break;
222  case Documents:
223  m_savePath = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation);
224  break;
225  case Downloads:
226  m_savePath = QStandardPaths::writableLocation(QStandardPaths::DownloadLocation);
227  break;
228  case Desktop:
229  m_savePath = QStandardPaths::writableLocation(QStandardPaths::DesktopLocation);
230  break;
231  case Videos:
232  m_savePath = QStandardPaths::writableLocation(QStandardPaths::MoviesLocation);
233  break;
234  case Music:
235  m_savePath = QStandardPaths::writableLocation(QStandardPaths::MusicLocation);
236  break;
237  case UsrSelect:
238  showDirChoseDialog();
239  break;
240  case Other:
241  m_savePath = m_savePathCombox->itemText(index);
242  break;
243  default:
244  m_savePath = "";
245  break;
246  }
247 }

249 void CExportImageDialog::slotOnFormatChange(int index)
250 {
251  switch (index) {
252  case PDF:
253  case BMP:
254  case TIF:
255  m_qualitySlider->setValue(100);
256  m_qualitySlider->setEnabled(false);
257  break;
258  case JPG:
259  case PNG:
260  m_qualitySlider->setEnabled(true);
261  break;
262  default:
263  break;
264  }
265 
266  m_saveFormat = m_formatCombox->itemText(index);
267 
268  QString name = m_fileNameEdit->text().trimmed();
269 
270  if (isHaveSuffix(name)) {
271  name = name.mid(0, name.lastIndexOf(".") + 1);
272  name += m_saveFormat;
273  } else {
274  name = name + "." + m_saveFormat;
275  }
276 
277  m_fileNameEdit->setText(name);
278 }

280 void CExportImageDialog::slotOnDialogButtonClick(int index, const QString &text)
281 {
282  Q_UNUSED(text)
283 
284  if (index == 1) {
285  // [BUG: 30843] 希望在导出时，名字以点开头的图片，也能给到提示：以点开始会被隐藏
286  QString fileName = m_fileNameEdit->text().trimmed();
287  if (fileName.startsWith(".")) {
288  this->hide();
289  Dialog dialog(this);
290  dialog.setModal(true);
291  dialog.setIcon(QPixmap(":/icons/deepin/builtin/Bullet_window_warning.svg"));
292  dialog.setMessage(tr("This file will be hidden if the file name starts with a dot (.). Do you want to hide it?"));
293  dialog.addButton(tr("Cancel"), false, DDialog::ButtonNormal);
294  dialog.addButton(tr("Confirm"), true, DDialog::ButtonRecommend);
295  dialog.showInCenter(this);
296  int status = dialog.exec();
297  if (status != 1) {
298  this->show();
299  return;
300  }
301  }
302 
303  // 判断路径是否超过255字符
304  QString completePath = getCompleteSavePath();
305  QFileInfo info(completePath);
306  // NAME_MAX 文件名字最大长度
307  if (info.fileName().toLocal8Bit().length() > NAME_MAX) {
308  Dtk::Widget::DDialog dialog(this);
309  dialog.setTextFormat(Qt::RichText);
310  dialog.addButton(tr("OK"));
311  dialog.setIcon(QIcon(":/icons/deepin/builtin/Bullet_window_warning.svg"));
312  dialog.setMessage(tr("The file name is too long"));
313  dialog.exec();
314  return;
315  }
316 
317  if (completePath == "") {
318  return;
319  }
320  if (QFileInfo(completePath).exists()) {
321  hide();
322  showQuestionDialog(completePath);
323  } else {
324  emit signalDoSave(completePath);
325  hide();
326  }
327  } else if (index == 0) {
328  hide();
329  }
330 }

386 QString CExportImageDialog::getCompleteSavePath()
