57 QMenu *DFMSideBarTagItemHandler::contextMenu(const DFMSideBar *sidebar, const DFMSideBarItem *item)
58 {
59  DFileMenu *menu = new DFileMenu();
60  DFileManagerWindow *wnd = qobject_cast<DFileManagerWindow *>(sidebar->topLevelWidget());
61  bool shouldDisable = !WindowManager::tabAddableByWinId(wnd->windowId());
62  DTagActionWidget* tagWidget{ new DTagActionWidget };
63  QWidgetAction* tagAction{ new QWidgetAction{ nullptr } };
64 
65  menu->addAction(QObject::tr("Open in new window"), [item]() {
66  WindowManager::instance()->showNewWindow(item->url(), true);
67  });
68 
69  menu->addAction(QObject::tr("Open in new tab"), [wnd, item]() {
70  wnd->openNewTab(item->url());
71  })->setDisabled(shouldDisable);
72 
73  menu->addSeparator();
74 
75  menu->addAction(QObject::tr("Rename"), [sidebar, item]() {
76  int index = sidebar->findItem(item);
77  if (index >= 0) {
78  sidebar->openItemEditor(index);
79  }
80  });
81 
82  menu->addAction(QObject::tr("Remove"), [item]() {
83  DFileService::instance()->deleteFiles(nullptr, DUrlList{item->url()}, false);
84  });
85 
86  menu->addSeparator();
87 
88  tagAction->setDefaultWidget(tagWidget);
89  tagAction->setText("Change color of present tag");
90  tagWidget->setExclusive(true);
91  tagWidget->setToolTipVisible(false);
92 
93  menu->addAction(tagAction);
94  connect(tagAction, &QWidgetAction::triggered, this, [item, menu, tagWidget]() {
95  // should setIcon first, processEvent will reset checkedColorList
96  if (tagWidget->checkedColorList().size()>0) {
97  QString iconName = TagManager::instance()->getTagIconName(tagWidget->checkedColorList().first());
98  QIcon icon = QIcon::fromTheme(iconName);
99  DFMSideBarItem *it = const_cast<DFMSideBarItem *>(item);
100  it->setIcon(icon);
101  }
102 
103  DFMEventDispatcher::instance()->processEvent<DFMMenuActionEvent>(nullptr, menu, DUrl(TAG_ROOT),
104  DUrlList{item->url()}, DFMGlobal::ChangeTagColor);
105  });
106 
107  return menu;
108 }
