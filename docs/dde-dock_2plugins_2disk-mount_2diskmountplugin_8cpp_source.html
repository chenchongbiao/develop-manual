81 const QString DiskMountPlugin::itemContextMenu(const QString &itemKey)
82 {
83  Q_UNUSED(itemKey);
84 
85  QList<QVariant> items;
86  items.reserve(2);
87 
88  QMap<QString, QVariant> open;
89  open["itemId"] = OPEN;
90  open["itemText"] = tr("Open");
91  open["isActive"] = true;
92  items.push_back(open);
93 
94  QMap<QString, QVariant> unmountAll;
95  unmountAll["itemId"] = UNMOUNT_ALL;
96  unmountAll["itemText"] = tr("Unmount all");
97  unmountAll["isActive"] = true;
98  items.push_back(unmountAll);
99 
100  QMap<QString, QVariant> menu;
101  menu["items"] = items;
102  menu["checkableMenu"] = false;
103  menu["singleCheck"] = false;
104
