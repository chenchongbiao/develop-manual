80 // blumia: DDE not yet got fully support about session management, so when logout or shutdown,
81 // the config file won't save. On mips64el, sw, arm, there will be a "warm-up" process
82 // running in the background (dde-file-manager -d) and the file manager instance will
83 // not got exit when all visible DFM window got closed, so that means the file manager
84 // config file will never got saved.
85 // blumia: Handling SIGTERM is a workaround way to fix that issue, but we still need to add
86 // session management support to DDE.

97 // Within an SSH session, I can use gvfs-mount provided that
98 // dbus-daemon is launched first and the environment variable DBUS_SESSION_BUS_ADDRESS is set.
99 
100 // gvfs-mount and other GVFS utilities must all talk to the same D-Bus session. Hence,
101 // if you use multiple SSH sessions or otherwise use mounts across login sessions, you must:
102 // - start D-Bus the first time it is needed, at the latest;
103 // - take care not to let D-Bus end with the session, as long as there are mounted GVFS filesystems;
104 // - reuse the existing D-Bus session at login time if there is one.
105 // see https://unix.stackexchange.com/questions/44317/reuse-d-bus-sessions-across-login-sessions for that.
106 void handleEnvOfOpenAsAdmin()
107 {
108  QProcess p;
109  p.start("bash", QStringList() << "-c"
110  << "echo $(dbus-launch)");
111  p.waitForFinished();
112  QString envName("DBUS_SESSION_BUS_ADDRESS");
113  QString output(p.readAllStandardOutput());
114  QStringList group(output.split(" "));
115  for (const QString &vals : group) {
116  const QStringList &envGroup = vals.split(",");
117  for (const QString &env : envGroup) {
118  int mid = env.indexOf("=");
119  if (env.startsWith(envName) && mid >= envName.size()) {
120  QString v(env.mid(mid + 1));
121  int ret = setenv(envName.toLocal8Bit().data(), v.toLocal8Bit().data(), 1);
122  qDebug() << "set " << env << "=" << v << "ret=" << ret;
123  return;
124  }
125  }
126  }
127 }
