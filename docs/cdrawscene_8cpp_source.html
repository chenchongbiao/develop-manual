258 bool CDrawScene::event(QEvent *event)
259 {
260  QEvent::Type evType = event->type();
261  if (evType == QEvent::TouchBegin || evType == QEvent::TouchUpdate || evType == QEvent::TouchEnd) {
262 
263  //qDebug() << "CDrawScene:: touch event evType = " << evType;
264  QTouchEvent *touchEvent = dynamic_cast<QTouchEvent *>(event);
265 
266  QList<QTouchEvent::TouchPoint> touchPoints = touchEvent->touchPoints();
267 
268  EDrawToolMode currentMode = getDrawParam()->getCurrentDrawToolMode();
269 
270  IDrawTool *pTool = CDrawToolManagerSigleton::GetInstance()->getDrawTool(currentMode);
271 
272  if (nullptr == pTool || (touchPoints.count() > 1 && currentMode == selection)) {
273  //是一个手势，那么中断当前的updating操作
274  if (pTool != nullptr)
275  pTool->interrupt();
276  return QGraphicsScene::event(event);
277  }
278 
279  //解决触屏后鼠标隐藏但还可能鼠标的位置还是高亮的问题
280  if (evType == QEvent::TouchBegin) {
281  DToolButton *pBtn = drawApp->leftToolBar()->toolButton(currentMode);
282  if (pBtn != nullptr) {
283  pBtn->setAttribute(Qt::WA_UnderMouse, false);
284  pBtn->update();
285  }
286  }
287 
288  bool accept = true;
289  foreach (const QTouchEvent::TouchPoint tp, touchPoints) {
290  IDrawTool::CDrawToolEvent e = IDrawTool::CDrawToolEvent::fromTouchPoint(tp, this, event);
291  switch (tp.state()) {
292  case Qt::TouchPointPressed:
293  //表示触碰按下
294  pTool->toolDoStart(&e);
295  break;
296  case Qt::TouchPointMoved:
297  //触碰移动
298  pTool->toolDoUpdate(&e);
299  break;
300  case Qt::TouchPointReleased:
301  //触碰离开
302  pTool->toolDoFinish(&e);
303  break;
304  default:
305  break;
306  }
307  if (!e.isAccepted()) {
308  accept = false;
309  }
310  }
311  event->accept();
312 
313  if (evType == QEvent::TouchEnd)
314  pTool->clearITE();
315 
316  if (accept)
317  return true;
318 
319  } else if (event->type() == QEvent::Gesture) {
320  EDrawToolMode currentMode = getDrawParam()->getCurrentDrawToolMode();
321 
322  if (currentMode == selection) {
323  return drawView()->gestureEvent(static_cast<QGestureEvent *>(event));
324  }
325  }
326  return QGraphicsScene::event(event);
327 }

509  info.childGroups.append(getGroupTreeInfo(gp));

598 void CDrawScene::moveItemsZDown(const QList<CGraphicsItem *> &items, int step)
599 {
600  //1.首先判断图元们是否是一个组合的,要移动的图元需要是处于一个组合中的(或者都不处于组合)
601  CGraphicsItemGroup *pSameGroup = getSameGroup(items, false, false, true);
602  if (pSameGroup == nullptr)
603  return;
604 
605  //2.获取会受到影响的图元(按照z降序排列,即从z值大到小排序)
606  qreal minZ = 0;
607  QList<CGraphicsItem *> invokedItems;
608  if (pSameGroup->groupType() == CGraphicsItemGroup::ENormalGroup) {
609  invokedItems = pSameGroup->items();
610  sortZ(invokedItems, EDesSort);
611  minZ = invokedItems.last()->zValue();
612  } else {
613  invokedItems = getRootItems(EDesSort);
614  minZ = 0;
615  }
616  //3.进行排序
617  bool toLimitFirst = true;
618  //按照z值从低到高进行遍历,遇到想移动的图元见按照步数在invokedItems链表向后移动
619  for (int i = invokedItems.size() - 1; i >= 0; --i) {
620  CGraphicsItem *p = invokedItems.at(i);
621  if (items.contains(p)) {
622  //向下移动step个图层
623  int index = (i + step + 1);
624  if (step == -1) {
