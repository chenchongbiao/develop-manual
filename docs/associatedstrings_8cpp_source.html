27 AssociatedStrings::AssociatedStrings( U32 fcSttbfAssoc, U32 lcbSttbfAssoc, U16 lid, AbstractOLEStreamReader* tableStream ) :
28  m_sttbf( 0 )
29 {
30  tableStream->push();
31  tableStream->seek( fcSttbfAssoc );
32  m_sttbf = new STTBF( lid, tableStream );
33  if ( tableStream->tell() - fcSttbfAssoc != lcbSttbfAssoc )
34  wvlog << "Warning: Associated strings have a different size than expected!" << std::endl;
35  tableStream->pop();
36 }
