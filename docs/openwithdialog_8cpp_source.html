301 void OpenWithDialog::initData()
302 {
303  //在选择默认程序时，有多个url，要传多个url
304  if (m_url.isValid() && m_urllist.size() == 0) {
305  const DAbstractFileInfoPointer &file_info = DFileService::instance()->createFileInfo(this, m_url);
306 
307  if (!file_info)
308  return;
309 
310  m_mimeType = file_info->mimeType();
311 
312  if (file_info->isDesktopFile())
313  m_setToDefaultCheckBox->hide();
314  } else if (!m_url.isValid() && m_urllist.size() > 0) {
315  QList<DUrl> openlist;
316  bool bhide = true;
317  for (auto url : m_urllist) {
318  const DAbstractFileInfoPointer &file_info = DFileService::instance()->createFileInfo(this, url);
319 
320  if (!file_info) {
321  continue;
322  }
323  m_mimeType = file_info->mimeType();
324  if (!file_info->isDesktopFile()) {
325  bhide = false;
326  }
327  openlist.push_back(url);
328  }
329 
330  if (0 == openlist.size())
331  return;
332 
333  if (bhide)
334  m_setToDefaultCheckBox->hide();
335  }
336 
337 
338  const QString &default_app = mimeAppsManager->getDefaultAppByMimeType(m_mimeType);
339  const QStringList &recommendApps = mimeAppsManager->getRecommendedAppsByQio(m_mimeType);
340 
341  for (int i = 0; i < recommendApps.count(); ++i) {
342  const DesktopFile &desktop_info = mimeAppsManager->DesktopObjs.value(recommendApps.at(i));
343 
344  OpenWithDialogListItem *item = createItem(QIcon::fromTheme(desktop_info.getIcon()), desktop_info.getDisplayName(), recommendApps.at(i));
345  m_recommandLayout->addWidget(item);
346 
347  if (!default_app.isEmpty() && recommendApps.at(i).endsWith(default_app))
348  checkItem(item);
349  }
350 
351  QList<DesktopFile> other_app_list;
352 
353  foreach (const QString &f, mimeAppsManager->DesktopObjs.keys()) {
354  //filter recommend apps , no show apps and no mime support apps
355  const DesktopFile &app = mimeAppsManager->DesktopObjs.value(f);
356  if (recommendApps.contains(f))
357  continue;
358 
359  if (mimeAppsManager->DesktopObjs.value(f).getNoShow())
360  continue;
361 
362  if (mimeAppsManager->DesktopObjs.value(f).getMimeType().isEmpty())
363  continue;
364 
365  bool isSameDesktop = false;
366  foreach (const DesktopFile &otherApp, other_app_list) {
367  if (otherApp.getExec() == app.getExec() && otherApp.getLocalName() == app.getLocalName())
368  isSameDesktop = true;
369  }
370 
371  Properties desktop_info(f, "Desktop Entry");
372 
373  const QString &custom_open_desktop = desktop_info.value("X-DDE-File-Manager-Custom-Open").toString();
374 
375  // Filter self own desktop files for opening other types of files
376  if (!custom_open_desktop.isEmpty() && custom_open_desktop != m_mimeType.name())
377  continue;
378 
379  if (isSameDesktop)
380  continue;
381 
382  other_app_list << mimeAppsManager->DesktopObjs.value(f);
383  QString iconName = other_app_list.last().getIcon();
384  OpenWithDialogListItem *item = createItem(QIcon::fromTheme(iconName), other_app_list.last().getDisplayName(), f);
385  m_otherLayout->addWidget(item);
386 
387  if (!default_app.isEmpty() && f.endsWith(default_app))
388  checkItem(item);
389  }
390 }

401 void OpenWithDialog::useOtherApplication()
402 {
403  const QString &file_path = QFileDialog::getOpenFileName(this);
404 
405  if (file_path.isEmpty())
406  return;
407 
408  QFileInfo info(file_path);
409  QString target_desktop_file_name("%1/%2-custom-open-%3.desktop");
410 
411  target_desktop_file_name = target_desktop_file_name.arg(QStandardPaths::writableLocation(QStandardPaths::ApplicationsLocation)).arg(qApp->applicationName()).arg(m_mimeType.name().replace("/", "-"));
412 
413  if (file_path.endsWith(".desktop")) {
414  for (const OpenWithDialog *w : m_recommandLayout->parentWidget()->findChildren<OpenWithDialog *>()) {
415  if (w->property("app").toString() == file_path)
416  return;
417  }
418 
419  Properties desktop(file_path, "Desktop Entry");
420 
421  if (desktop.value("MimeType").toString().isEmpty())
422  return;
423 
424  if (!QFile::link(file_path, target_desktop_file_name))
425  return;
426  } else if (info.isExecutable()) {
427  Properties desktop;
428 
429  desktop.set("Type", "Application");
430  desktop.set("Name", info.fileName());
431  desktop.set("Icon", "application-x-desktop");
432  desktop.set("Exec", file_path);
433  desktop.set("MimeType", "*/*");
434  desktop.set("X-DDE-File-Manager-Custom-Open", m_mimeType.name());
435 
436  if (QFile::exists(target_desktop_file_name))
437  QFile(target_desktop_file_name).remove();
438 
439  if (!desktop.save(target_desktop_file_name, "Desktop Entry"))
440  return;
441  }
442 
443  // remove old custom item
444  for (int i = 0; i < m_otherLayout->count(); ++i) {
445  QWidget *w = m_otherLayout->itemAt(i)->widget();
446 
447  if (!w)
448  continue;
449 
450  if (w->property("app").toString() == target_desktop_file_name) {
451  m_otherLayout->removeWidget(w);
452  w->deleteLater();
453  }
454  }
455 
456  OpenWithDialogListItem *item = createItem(QIcon::fromTheme("application-x-desktop"), info.fileName(), target_desktop_file_name);
457 
458  int other_layout_sizeHint_height = m_otherLayout->sizeHint().height();
459  m_otherLayout->addWidget(item);
460  item->show();
461  m_otherLayout->parentWidget()->setFixedHeight(m_otherLayout->parentWidget()->height() + m_otherLayout->sizeHint().height() - other_layout_sizeHint_height);
462  checkItem(item);
463 }

476 void OpenWithDialog::openFileByApp()

512 bool OpenWithDialog::eventFilter(QObject *obj, QEvent *event)
513 {
514  // blumia: for m_scrollArea, to avoid touch screen scrolling cause window move
515  if (event->type() == QEvent::MouseMove) {
516  QMouseEvent *mouseEvent = static_cast<QMouseEvent *>(event);
517  if (mouseEvent->source() == Qt::MouseEventSynthesizedByQt) {
518  return true;
519  }
520  }
521 
522  if (event->type() == QEvent::MouseButtonPress) {
523 
524  if (static_cast<QMouseEvent *>(event)->button() == Qt::LeftButton) {
525  if (OpenWithDialogListItem *item = qobject_cast<OpenWithDialogListItem *>(obj))
526  checkItem(item);
527 
528  return true;
529  }
530  }
531 
532  return false;
533 }
