51 const QString compressedIpv6Addr(const QString &ipv6Adr)
52 {
53  if (ipv6Adr.contains("::"))
54  return ipv6Adr;
55 
56  // NOTE(sbw): Calculate longest zero sequence, ensure
57  int start = 0, len = 0;
58  int maxStart = 0, maxLen = 0;
59  const auto &sequence = ipv6Adr.split(':');
60  for (int i = 0; i != sequence.size(); ++i) {
61  if (sequence[i] == "0000") {
62  len += 5;
63  } else {
64  if (len > maxLen) {
65  maxStart = start;
66  maxLen = len;
67  }
68  start = (i + 1) * 5;
69  len = 0;
70  }
71  }
72 
73  if (maxLen) {
74  return QString(ipv6Adr).replace(maxStart, maxLen, ':');
75  } else if (len > maxLen) {
76  if (start) {
77  return QString(ipv6Adr).replace(start, len, ':');
78  } else {
79  return QString("::");
80  }
81  } else {
82  return ipv6Adr;
83  }
84 }

105 void NetworkDetailPage::updateNetworkInfo()
106 {
107  QDBusInterface *networkInter = new QDBusInterface("com.deepin.daemon.Network",
108  "/com/deepin/daemon/Network",
109  "com.deepin.daemon.Network",
110  QDBusConnection::sessionBus(), this);
111  QDBusPendingCall async = networkInter->asyncCall("GetActiveConnectionInfo");
112  QDBusPendingCallWatcher *w = new QDBusPendingCallWatcher(async, this);
113  connect(w, &QDBusPendingCallWatcher::finished, this, [this, w](){
114  QDBusPendingReply<QString> reply = *w;
115  if (reply.isError()) {
116  qDebug() << "GetActiveConnectionInfo error";
117  return;
118  }
119  QList<QJsonObject> activeinfos;
120  QJsonArray activeConns = QJsonDocument::fromJson(reply.value().toUtf8()).array();
121  for (const auto info : activeConns)
122  {
123  const auto &connInfo = info.toObject();
124  activeinfos << connInfo;
125  }
126  onActiveInfoChanged(activeinfos);
127  w->deleteLater();
128  });
129 }

131 void NetworkDetailPage::onActiveInfoChanged(const QList<QJsonObject> &infos)
132 {
133  auto appendInfo = [](SettingsGroup *g, const QString &t, const QString &v) {
134  TitleValueItem *i = new TitleValueItem;
135  i->setTitle(t);
136  i->setValue(v);
137  g->appendItem(i);
138  if (t == "IPv6") {
139  i->setWordWrap(false);
140  }
141  };
142 
143  // clear old infos
144  while (QLayoutItem *item = m_groupsLayout->takeAt(0)) {
145  if (item->widget()) {
146  item->widget()->deleteLater();
147  }
148  delete item;
149  }
150 
151  int infoCount = infos.count();
152  for (const auto &info : infos) {
153  SettingsGroup *grp = new SettingsGroup;
154  const QString type = info.value("ConnectionType").toString();
155  const bool isHotspot = type == "wireless-hotspot";
156  const bool isWireless = type == "wireless";
157  const QJsonObject &hotspotInfo = info.value("Hotspot").toObject();
158 
159  // 设置活跃网络标题
160  SettingsHead *head = new SettingsHead;
161  head->setEditEnable(false);
162  head->setContentsMargins(20,0,0,0);
163 
164  if (isHotspot) {
165  head->setTitle(tr("Hotspot"));
166  grp->appendItem(head, SettingsGroup::NoneBackground);
167 
168  const QString ssid = hotspotInfo.value("Ssid").toString();
169  appendInfo(grp, tr("SSID"), ssid);
170  } else {
171  const QString name = info.value("ConnectionName").toString();
172  head->setTitle(name);
173  grp->appendItem(head, SettingsGroup::NoneBackground);
174  }
175 
176  if (isWireless) {
177  // protocol
178  const QString &protocol = info.value("Protocol").toString();
179  if (!protocol.isEmpty())
180  appendInfo(grp, tr("Protocol"), protocol);
181 
182  // security type
183  const QString &securityType = info.value("Security").toString();
184  appendInfo(grp, tr("Security Type"), securityType);
185 
186  // band
187  const QString &band = hotspotInfo.value("Band").toString();
188  QString bandInfo = band == "a" ? "5G" : (band == "bg" ? "2.4G" : "automatic");
189  appendInfo(grp, tr("Band"), bandInfo);
190 
191  // channel
192  const QString &channel = QString::number(hotspotInfo.value("Channel").toInt());
193  if (!channel.isEmpty())
194  appendInfo(grp, tr("Channel"), channel);
195  }
196 
197  // encrypt method
198  if (isHotspot) {
199  const QString securityType = info.value("Security").toString();
200  appendInfo(grp, tr("Security Type"), securityType);
201  }
202  // device interface
203  const auto device = info.value("DeviceInterface").toString();
204  if (!device.isEmpty())
205  appendInfo(grp, tr("Interface"), device);
206  // mac info
207  const QString mac = info.value("HwAddress").toString();
208  if (!mac.isEmpty())
209  appendInfo(grp, tr("MAC"), mac);
210  // band
211  if (isHotspot) {
212  const QString band = hotspotInfo.value("Band").toString();
213  appendInfo(grp, tr("Band"), band);
214  } else {
215  // ipv4 info
216  const auto ipv4 = info.value("Ip4").toObject();
217  if (!ipv4.isEmpty()) {
218  // ipv4 address
219  const auto ip4Addr = ipv4.value("Address").toString();
220  if (!ip4Addr.isEmpty())
221  appendInfo(grp, tr("IPv4"), ip4Addr);
222  // ipv4 gateway
223  const auto gateway = ipv4.value("Gateways").toArray();
224  if (!gateway.isEmpty())
225  appendInfo(grp, tr("Gateway"), gateway.first().toString());
226  // ipv4 primary dns
227  const auto ip4PrimaryDns = ipv4.value("Dnses").toArray();
228  if (!ip4PrimaryDns.isEmpty())
229  appendInfo(grp, tr("Primary DNS"), ip4PrimaryDns.first().toString());
230  // ipv4 netmask
231  const auto ip4Netmask = ipv4.value("Mask").toString();
232  if (!ip4Netmask.isEmpty())
233  appendInfo(grp, tr("Netmask"), ip4Netmask);
234  }
235  // ipv6 info
236  const auto ipv6 = info.value("Ip6").toObject();
237  if (!ipv6.isEmpty()) {
238  appendInfo(grp, tr("IPv6"), compressedIpv6Addr(ipv6Infomation(info, NetworkDetailPage::Ip)));
239  appendInfo(grp, tr("Gateway"), compressedIpv6Addr(ipv6Infomation(info, NetworkDetailPage::Gateway)));
240 
241  // ipv6 primary dns
242  const auto ip6PrimaryDns = ipv6.value("Dnses").toArray();
243  if (!ip6PrimaryDns.isEmpty())
244  appendInfo(grp, tr("Primary DNS"), compressedIpv6Addr(ip6PrimaryDns.first().toString()));
245  // ipv6 netmask
246  const auto ip6Prefix = ipv6.value("Prefix").toString();
247  if (!ip6Prefix.isEmpty())
248  appendInfo(grp, tr("Prefix"), ip6Prefix);
249  }
250  // speed info
251  const QString speed = info.value("Speed").toString();
252  if (!speed.isEmpty())
253  appendInfo(grp, tr("Speed"), speed);
254  }
255  m_groupsLayout->addWidget(grp);
256  if (--infoCount > 0) {
257  m_groupsLayout->addSpacing(30);
258  }
259  }
260 
261  m_groupsLayout->addStretch();
262 }

264 QString NetworkDetailPage::ipv6Infomation(QJsonObject connectinfo, NetworkDetailPage::InfoType type)
265 {
266  NetworkManager::Connection::Ptr connection = findConnectionByUuid(connectinfo.value("ConnectionUuid").toString());
267  NetworkManager::ConnectionSettings::Ptr connectionSettings = connection->settings();
268  NetworkManager::Ipv6Setting::Ptr ipv6Setting = connectionSettings->setting(Setting::Ipv6).staticCast<NetworkManager::Ipv6Setting>();
269  QList<NetworkManager::IpAddress> addressInfos = ipv6Setting->addresses();
270  if (ipv6Setting->method() == Ipv6Setting::Manual) {
271 
272  if (addressInfos.count() == 0) {
273  qDebug() << "ipv6Setting error! ";
274  return "";
275  }
276 
277  switch (type) {
278  case Ip:
279  return addressInfos[0].ip().toString();
280  case Gateway:
281  return addressInfos[0].gateway().toString();
282  }
283  }
284 
285  if (ipv6Setting->method() == Ipv6Setting::Automatic) {
286  const auto ipv6 = connectinfo.value("Ip6").toObject();
287  switch (type) {
288  case Ip:
289  return ipv6.value("Address").toString();
290  case Gateway:
291  return ipv6.value("Address").toString();
292  }
293  }
294  return "";
295 }
296 }
