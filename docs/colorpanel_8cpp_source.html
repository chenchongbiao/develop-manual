66 void ColorButton::paintEvent(QPaintEvent *)
67 {
68  QPainter painter(this);
69  painter.setRenderHints(QPainter::Antialiasing);
70  painter.setBrush(QBrush(m_color));
71  QColor colorTrasparent(Qt::white);
72  colorTrasparent.setAlpha(0);
73 
74  QPen pen;
75  pen.setWidth(1);
76  if (m_color == colorTrasparent) {
77  pen.setColor(QColor("#cccccc"));
78  } else {
79  pen.setColor(QColor(0, 0, 0, 55));
80  }
81  painter.setPen(pen);
82  painter.drawRoundedRect(QRect(3, 3, this->width() - 6,
83  this->height() - 6),
84  RADIUS, RADIUS);
85  if (m_color == colorTrasparent) {
86  pen.setColor(QColor("#ff804d"));
87  painter.setPen(pen);
88  painter.drawLine(6, this->height() - 6, this->width() - 6, 6);
89  }
90 
91  if (isChecked()) {
92  painter.setBrush(QBrush());
93  QPen borderPen;
94  borderPen.setWidth(2);
95  borderPen.setColor("#008eff");
96  painter.setPen(borderPen);
97  painter.drawRoundedRect(QRect(1, 1, this->width() - 2,
98  this->height() - 2),
99  RADIUS, RADIUS);
100  }

221 void ColorPanel::initConnection()
222 {
223  //1.颜色按钮组
224  connect(m_colorsButtonGroup, QOverload<QAbstractButton *, bool>::of(&QButtonGroup::buttonToggled),
225  this, [ = ](QAbstractButton * pBtn, bool checked) {
226  ColorButton *pColorBtn = qobject_cast<ColorButton *>(pBtn);
227  if (checked) {
228  this->setColor(pColorBtn->color());
229  }
230  });
231 
232  //2.pick板颜色设置完成
233  connect(m_pickColWidget, &PickColorWidget::colorChanged, this, [ = ](const QColor & color) {
234  this->setColor(color);
235  });
236 
237  //3.pick板颜色预览
238  connect(m_pickColWidget, &PickColorWidget::previewedColorChanged, [ = ](const QColor & color) {
239  this->setColor(color, true, EChangedUpdate);
240  });
241 
242  //4.lineedit颜色设置
243  connect(m_colLineEdit, &DLineEdit::textChanged, this, [ = ](const QString & colorStr) {
244  if (colorStr.size() == 6) {
245  QColor c("#" + colorStr);
246  if (c.isValid()) {
247  this->setColor(c, true, EChanged);
248  }
249  }
250  });
251  connect(m_colLineEdit, &DLineEdit::editingFinished, this, [ = ]() {
252  qDebug() << "DLineEdit::editingFinished ------- ";
253  QString colorStr = m_colLineEdit->text();
254  if (colorStr.size() == 6) {
255  QColor c("#" + colorStr);
256  if (c.isValid()) {
257  this->setColor(c, true, EChanged);
258  }
259  }
260  });
261 
262  //5.设置透明度
263  connect(m_alphaControlWidget, &CAlphaControlWidget::alphaChanged, this, [ = ](int apl, EChangedPhase phase) {
264  QColor c = color();
265  c.setAlpha(apl);
266  qDebug() << "alphaChanged apl = " << c.alpha();
267  this->setColor(c, true, phase);
268  });
269 
270  //展开按钮
271  connect(m_colorfulBtn, &CIconButton::buttonClick, this, [ = ] {
272  s_expand = !s_expand;
273  this->updateExpendArea();
274  });
275 }

301 void ColorPanel::updateColor(const QColor &previewColor)
302 {
303  QColor c = previewColor.isValid() ? previewColor : curColor;
304 
305  //1.检查当前颜色是否是颜色组的颜色值
306  int id = m_colList.indexOf(curColor);
307  if (id != -1) {
308  m_colorsButtonGroup->blockSignals(true);
309  m_colorsButtonGroup->button(id)->setChecked(true);
310  m_colorsButtonGroup->blockSignals(false);
311  } else {
312  //m_colorsButtonGroup->
313  }
314 
315  //2.pick界面刷新设置为该颜色
316  m_pickColWidget->setColor(c, false);
317 
318  //3.设置当前颜色的透明度(被动设置改值)
319  m_alphaControlWidget->setAlpha(c.alpha(), false);
320 
321  //4.更新颜色名字
322  QString colorName = "";
323  if (c.name().contains("#")) {
324  colorName = c.name().split("#").last();
325  }
326  //qDebug() << "colorName ====== " << colorName << "alpha = " << c.alpha() << "curColor ap = " << curColor.alpha();
327  m_colLineEdit->blockSignals(true);
328  m_colLineEdit->setText(colorName);
329  m_colLineEdit->blockSignals(false);
