31 void CAlphaControlWidget::setAlpha(int alpha, bool internalChanged)
32 {
33  if (!internalChanged) {

35  m_alphaSlider->blockSignals(true);
36  this->blockSignals(true);

99 void CAlphaControlWidget::initConnection()
100 {
101  connect(m_alphaSlider, &DSlider::sliderPressed, this, [ = ]() {
102  emit alphaChanged(m_alphaSlider->value(), EChangedBegin);
103  });
104 
105  connect(m_alphaSlider, &DSlider::valueChanged, this, [ = ](int value) {
106  value = m_alphaSlider->value();
107  m_alphaLabel->setText(QString("%1%").arg(value * 100 / 255));
108  emit alphaChanged(value, EChangedUpdate);
109  });
110 
111  connect(m_alphaSlider, &DSlider::sliderReleased, this, [ = ]() {
112  emit alphaChanged(m_alphaSlider->value(), EChangedFinished);
113  });
114 }
