85 int KeyboardModule::load(const QString &path)
86 {
87  QStringList pathList = path.split("/");
88  QString loadPath = pathList.at(0);
89  int hasPage = -1;
90  if (loadPath == QStringLiteral("General")) {
91  m_keyboardWidget->initSetting(0);
92  hasPage = 0;
93  } else if (loadPath == QStringLiteral("Keyboard Layout")) {
94  m_keyboardWidget->initSetting(1);
95  hasPage = 0;
96  } else if (loadPath == QStringLiteral("System Language")) {
97  m_keyboardWidget->initSetting(2);
98  hasPage = 0;
99  } else if (loadPath == QStringLiteral("Shortcuts")) {
100  if ((pathList.length() > 1) && (pathList.at(1) == QStringLiteral("Custom Shortcut"))) {
101  showShortCutSetting();
102  m_shortcutSettingWidget->showCustomShotcut();
103  hasPage = 0;
104  } else {
105  m_keyboardWidget->initSetting(3);
106  hasPage = 0;
107  }
108  }
109 
110  return hasPage;
111 }

149 void KeyboardModule::onPushKeyboard(const QStringList &kblist)
150 {
151  m_work->onPinyin();
152  m_kbLayoutWidget = new KeyboardLayoutWidget();
153  m_kbLayoutWidget->setVisible(false);
154  auto dataControll = [ = ](QList<MetaData> datas) {
155  for (auto it(datas.begin()); it != datas.end();) {
156  const MetaData &data = *it;
157  if (kblist.contains(data.key()))
158  it = datas.erase(it);
159  else
160  ++it;
161  }
162 
163  m_kbLayoutWidget->setMetaData(datas);
164  };
165 
166  connect(m_work, &KeyboardWorker::onDatasChanged, m_kbLayoutWidget, dataControll);
167  connect(m_work, &KeyboardWorker::onLettersChanged, m_kbLayoutWidget, &KeyboardLayoutWidget::setLetters);
168 
169  dataControll(m_work->getDatas());
170  m_kbLayoutWidget->setLetters(m_work->getLetters());
171 
172  connect(m_kbLayoutWidget, &KeyboardLayoutWidget::layoutSelected, m_work, &KeyboardWorker::addUserLayout);
173  connect(m_kbLayoutWidget, &KeyboardLayoutWidget::back, this, &KeyboardModule::showKBLayoutSetting);
174 
175  m_frameProxy->pushWidget(this, m_kbLayoutWidget);
176  m_kbLayoutWidget->setVisible(true);
177 }
