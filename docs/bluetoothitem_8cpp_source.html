44 BluetoothItem::BluetoothItem(QWidget *parent)
45  : QWidget(parent)
46  , m_tipsLabel(new TipsWidget(this))
47  , m_applet(new BluetoothApplet(this))
48  , m_devState(Device::State::StateUnavailable)
49  , m_adapterPowered(m_applet->poweredInitState())
50 {
51  setAccessibleName("BluetoothPluginItem");
52  m_applet->setVisible(false);
53  m_tipsLabel->setVisible(false);
54  refreshIcon();
55 
56  connect(m_applet, &BluetoothApplet::powerChanged, [ & ] (bool powered) {
57  m_adapterPowered = powered;
58  refreshIcon();
59  });
60  connect(m_applet, &BluetoothApplet::deviceStateChanged, [ & ] (const Device* device) {
61  m_devState = device->state();

66  connect(m_applet, &BluetoothApplet::noAdapter, this, &BluetoothItem::noAdapter);

83 const QString BluetoothItem::contextMenu() const
84 {
85  QList<QVariant> items;
86  if (m_applet->hasAadapter()) {
87  items.reserve(2);
88 
89  QMap<QString, QVariant> shift;
90  shift["itemId"] = SHIFT;
91  if (m_adapterPowered)
92  shift["itemText"] = tr("Turn off");
93  else
94  shift["itemText"] = tr("Turn on");
95  shift["isActive"] = true;
96  items.push_back(shift);
97 
98  QMap<QString, QVariant> settings;
99  settings["itemId"] = SETTINGS;
100  settings["itemText"] = tr("Bluetooth settings");
101  settings["isActive"] = true;
102  items.push_back(settings);
103 
104  QMap<QString, QVariant> menu;
105  menu["items"] = items;
106  menu["checkableMenu"] = false;
107  menu["singleCheck"] = false;
108  return QJsonDocument::fromVariant(menu).toJson();
109  }
110  return QByteArray();
111 }

130 void BluetoothItem::refreshIcon()
131 {
132  QString stateString;
133  QString iconString;
134 
135  if (m_adapterPowered) {
136  if (m_applet->connectedDevicesName().isEmpty()) {
137  stateString = "disable";
138  } else {
139  stateString = "active";
140  }
141  } else {
142  stateString = "disable";
143  }
144 
145  iconString = QString("bluetooth-%1-symbolic").arg(stateString);
146 
147  const qreal ratio = devicePixelRatioF();
148  int iconSize = PLUGIN_ICON_MAX_SIZE;
149  if (height() <= PLUGIN_BACKGROUND_MIN_SIZE && DGuiApplicationHelper::instance()->themeType() == DGuiApplicationHelper::LightType)
150  iconString.append(PLUGIN_MIN_ICON_NAME);
151 
152  m_iconPixmap = ImageUtil::loadSvg(iconString, ":/", iconSize, ratio);
153 
154  update();
155 }

157 void BluetoothItem::refreshTips()
158 {
159  QString tipsText;
160 
161  if (m_adapterPowered) {
162  if (!m_applet->connectedDevicesName().isEmpty() && m_devState != Device::StateAvailable) {
163  QStringList textList;
164  for (QString devName : m_applet->connectedDevicesName()) {
165  textList << tr("%1 connected").arg(devName);
166  }
167  m_tipsLabel->setTextList(textList);
168  return;
169  }
170  if (m_devState == Device::StateAvailable) {
171  tipsText = tr("Connecting...");
172  } else {
173  tipsText = tr("Bluetooth");
174  }
175  } else {
176  tipsText = tr("Turned off");
177  }
178 
179  m_tipsLabel->setText(tipsText);
180 }
