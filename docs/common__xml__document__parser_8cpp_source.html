162  static void onODFOOXMLList(CommonXMLDocumentParser& parser, XmlStream& xml_stream, XmlParseMode mode,
163  const FormattingStyle& options, const DocToTextUnzip* zipfile, std::string& text,
164  bool& children_processed, std::string& level_suffix, bool first_on_level,
165  std::vector<Link>& links)
166  {
167  std::vector<std::string> list_vector;
168  if (parser.verbose())
169  parser.getLogStream() << "ODFOOXML_LIST command.\n";
170  ++parser.getListDepth();
171  std::string header;
172  CommonXMLDocumentParser::ODFOOXMLListStyle list_style = CommonXMLDocumentParser::bullet;
173  std::string style_name = xml_stream.attribute("style-name");
174  if (parser.getListDepth() <= 10 && !style_name.empty() && parser.getListStyles().find(style_name) != parser.getListStyles().end())
175  list_style = parser.getListStyles()[style_name].at(parser.getListDepth() - 1);
176  xml_stream.levelDown();
177  while (xml_stream)
178  {
179  std::string list_element_type = xml_stream.name();
180  xml_stream.levelDown();
181  if (list_element_type == "list-header" && xml_stream)
182  {
183  header = parser.parseXmlData(xml_stream, mode, options, zipfile, links);
184  }
185  else if (xml_stream)
186  {
187  list_vector.push_back(parser.parseXmlData(xml_stream, mode, options, zipfile, links));
188  }
189  xml_stream.levelUp();
190  xml_stream.next();
191  }
192  xml_stream.levelUp();
193 
194  if (header.length() > 0)
195  {
196  text += header;
197  if (list_vector.size() > 0)
198  text += "\n";
199  }
200  --parser.getListDepth();
201  if (list_style == CommonXMLDocumentParser::number)
202  text += formatNumberedList(list_vector);
203  else
204  text += formatList(list_vector, options);
205  children_processed = true;
206  }

363 std::string CommonXMLDocumentParser::parseXmlData(XmlStream& xml_stream, XmlParseMode mode, const FormattingStyle& options, const DocToTextUnzip* zipfile, std::vector<Link>& links)
364 {
365  std::string text;

367  bool first_on_level = true;
368

379  }
380  xml_stream.next();
381  first_on_level = false;
382  }
383  if (!level_suffix.empty())
384  text += level_suffix;
385  return text;

388 bool CommonXMLDocumentParser::extractText(const std::string& xml_contents, XmlParseMode mode, const FormattingStyle& options, const DocToTextUnzip* zipfile, std::string& text, std::vector<Link>& links)
389 {
390  if (mode == STRIP_XML)
391  {
392  text = "";
393  bool in_tag = false;
394  for (int i = 0; i < xml_contents.length(); i++)
395  {
396  char ch = xml_contents[i];
397  if (ch == '<')
398  in_tag = true;
399  else if (ch == '>')
400  in_tag = false;
401  else if (!in_tag)
402  text += ch;
403  }
404  return true;
405  }
406  std::string xml;
407  if (mode == FIX_XML)
408  {
409  DocToTextXmlFixer xml_fixer;
410  xml = xml_fixer.fix(xml_contents);
411  }
412  else
413  xml = xml_contents;
414  XmlStream xml_stream(xml, impl->m_manage_xml_parser, getXmlOptions());
415  if (!xml_stream)
416  {
417  *impl->m_log_stream << "Error parsing XML contents.\n";
418  return false;
419  }
420  text = parseXmlData(xml_stream, mode, options, zipfile, links);
421  return true;
422 }

424 bool CommonXMLDocumentParser::parseODFMetadata(const std::string &xml_content, Metadata &metadata)
425 {
426  XmlStream xml_stream(xml_content, impl->m_manage_xml_parser, XML_PARSE_NOBLANKS);
427  if (!xml_stream)
428  {
429  impl->m_error = true;
430  return false;
431  }
432  xml_stream.levelDown();
433  while (xml_stream)
434  {
435  if (xml_stream.name() == "meta")
436  {
437  xml_stream.levelDown();
438  while (xml_stream)
439  {
440  if (xml_stream.name() == "initial-creator")
441  metadata.setAuthor(xml_stream.stringValue());
442  if (xml_stream.name() == "creation-date")
443  {
444  tm creation_date;
445  string_to_date(xml_stream.stringValue(), creation_date);
446  metadata.setCreationDate(creation_date);
447  }
448  if (xml_stream.name() == "creator")
449  metadata.setLastModifiedBy(xml_stream.stringValue());
450  if (xml_stream.name() == "date")
451  {
452  tm last_modification_date;
453  string_to_date(xml_stream.stringValue(), last_modification_date);
454  metadata.setLastModificationDate(last_modification_date);
455  }
456  if (xml_stream.name() == "document-statistic")
457  {
458  std::string attr = xml_stream.attribute("meta::page-count"); // LibreOffice 3.5
459  if (attr.empty())
460  attr = xml_stream.attribute("page-count"); // older OpenOffice.org
461  if (!attr.empty())
462  metadata.setPageCount(str_to_int(attr));
463  attr = xml_stream.attribute("meta:word-count"); // LibreOffice 3.5
464  if (attr.empty())
465  attr = xml_stream.attribute("word-count"); // older OpenOffice.org
466  if (!attr.empty())
467  metadata.setWordCount(str_to_int(attr));
468  }
469  xml_stream.next();
470  }
471  xml_stream.levelUp();
472  }
473  xml_stream.next();
474  }
475  return true;
476 }
