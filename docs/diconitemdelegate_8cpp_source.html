60 QString trimmedEnd(QString str)
61 {
62  while (!str.isEmpty()) {
63  switch (str.at(str.count() - 1).toLatin1()) {
64  case '\t':
65  case '\n':
66  case '\r':
67  case '\v':
68  case '\f':
69  case ' ':
70  str.chop(1);
71  break;
72  default:
73  return str;
74  }
75  }
76 
77  return str;
78 }

80 QRectF boundingRect(const QList<QRectF> rects)
81 {
82  QRectF bounding;
83 
84  if (rects.isEmpty())
85  return bounding;
86 
87  bounding = rects.first();
88 
89  for (const QRectF &r : rects) {
90  if (r.top() < bounding.top()) {
91  bounding.setTop(r.top());
92  }
93 
94  if (r.left() < bounding.left()) {
95  bounding.setLeft(r.left());
96  }
97 
98  if (r.right() > bounding.right()) {
99  bounding.setRight(r.right());
100  }
101 
102  if (r.bottom() > bounding.bottom()) {
103  bounding.setBottom(r.bottom());
104  }
105  }
106 
107  return bounding;
108 }

110 QPainterPath boundingPath(QList<QRectF> rects, qreal radius, qreal padding)
111 {
112  QPainterPath path;
113  const QMarginsF margins(radius + padding, 0, radius + padding, 0);
114 
115  if (rects.count() == 1) {
116  path.addRoundedRect(rects.first().marginsAdded(margins).adjusted(0, -padding, 0, padding), radius, radius);
117 
118  return path;
119  }
120 
121  auto joinRightCorner = [&](const QRectF & rect, const QRectF & prevRect, const QRectF & nextRect) {
122  if (Q_LIKELY(prevRect.isValid())) {
123  qreal new_radius = qMin(radius, qAbs(prevRect.right() - rect.right()) / 2);
124 
125  if (rect.right() > prevRect.right()) {
126  path.arcTo(rect.right() - new_radius * 2, rect.y() - padding, new_radius * 2, new_radius * 2, 90, -90);
127  } else if (rect.right() < prevRect.right()) {
128  path.arcTo(rect.right(), rect.y() + padding, new_radius * 2, new_radius * 2, 90, 90);
129  }
130  } else {
131  path.arcTo(rect.right() - radius * 2, rect.y() - padding, radius * 2, radius * 2, 90, -90);
132  }
133 
134  if (Q_LIKELY(nextRect.isValid())) {
135  qreal new_radius = qMin(radius, qAbs(nextRect.right() - rect.right()) / 2);
136 
137  if (rect.right() > nextRect.right()) {
138  path.arcTo(rect.right() - new_radius * 2, rect.bottom() - new_radius * 2 + padding, new_radius * 2, new_radius * 2, 0, -90);
139  } else if (rect.right() < nextRect.right()) {
140  path.arcTo(rect.right(), rect.bottom() - new_radius * 2 - padding, new_radius * 2, new_radius * 2, 180, 90);
141  }
142  } else {
143  path.arcTo(rect.right() - radius * 2, rect.bottom() - radius * 2 + padding, radius * 2, radius * 2, 0, -90);
144  }
145  };
146 
147  auto joinLeftCorner = [&](const QRectF & rect, const QRectF & prevRect, const QRectF & nextRect) {
148  if (Q_LIKELY(nextRect.isValid())) {
149  qreal new_radius = qMin(radius, qAbs(nextRect.x() - rect.x()) / 2);
150 
151  if (rect.x() > nextRect.x()) {
152  path.arcTo(rect.x() - new_radius * 2, rect.bottom() - new_radius * 2 - padding, new_radius * 2, new_radius * 2, 270, 90);
153  } else if (rect.x() < nextRect.x()) {
154  path.arcTo(rect.x(), rect.bottom() - new_radius * 2 + padding, new_radius * 2, new_radius * 2, 270, -90);
155  }
156  } else {
157  path.arcTo(rect.x(), rect.bottom() - radius * 2 + padding, radius * 2, radius * 2, 270, -90);
158  }
159 
160  if (Q_LIKELY(prevRect.isValid())) {
161  qreal new_radius = qMin(radius, qAbs(prevRect.x() - rect.x()) / 2);
162 
163  if (rect.x() > prevRect.x()) {
164  path.arcTo(rect.x() - new_radius * 2, rect.y() + padding, new_radius * 2, new_radius * 2, 0, 90);
165  } else if (rect.x() < prevRect.x()) {
166  path.arcTo(rect.x(), rect.y() - padding, new_radius * 2, new_radius * 2, 180, -90);
167  }
168  } else {
169  path.arcTo(rect.x(), rect.y() - padding, radius * 2, radius * 2, 180, -90);
170  }
171  };
172 
173  auto preproccess = [&](QRectF & rect, const QRectF & prev) {
174  if (qAbs(rect.x() - prev.x()) < radius) {
175  rect.setLeft(prev.x());
176  }
177 
178  if (qAbs(rect.right() - prev.right()) < radius) {
179  rect.setRight(prev.right());
180  }
181  };
182 
183  for (int i = 1; i < rects.count(); ++i) {
184  preproccess(rects[i], rects.at(i - 1));
185  }
186 
187  const QRectF &first = rects.first().marginsAdded(margins);
188 
189  path.arcMoveTo(first.right() - radius * 2, first.y() - padding, radius * 2, radius * 2, 90);
190  joinRightCorner(first, QRectF(), rects.at(1).marginsAdded(margins));
191 
192  for (int i = 1; i < rects.count() - 1; ++i) {
193  joinRightCorner(rects.at(i) + margins, rects.at(i - 1).marginsAdded(margins), rects.at(i + 1).marginsAdded(margins));
194  }
195 
196  QRectF last = rects.last();
197  const QRectF &prevRect = rects.at(rects.count() - 2);
198 
199  joinRightCorner(last.marginsAdded(margins), prevRect.marginsAdded(margins), QRectF());
200  joinLeftCorner(last.marginsAdded(margins), prevRect.marginsAdded(margins), QRectF());
201 
202  for (int i = rects.count() - 2; i > 0; --i) {
203  joinLeftCorner(rects.at(i) + margins, rects.at(i - 1).marginsAdded(margins), rects.at(i + 1).marginsAdded(margins));
204  }
205 
206  joinLeftCorner(first, QRectF(), rects.at(1).marginsAdded(margins));
207 
208  path.closeSubpath();
209 
210  return path;
211 }

474 QSize DIconItemDelegatePrivate::textSize(const QString &text, const QFontMetrics &metrics, int lineHeight) const
475 {

478  if (str.endsWith('\n'))

484  for (const QString &line : str.split('\n')) {
485  max_width = qMax(metrics.width(line), max_width);

489  else
490  height += metrics.height();
491  }
492 
493  return QSize(max_width, height);

579 void DIconItemDelegate::paint(QPainter *painter,
580  const QStyleOptionViewItem &option,
581  const QModelIndex &index) const
582 {
583  QMutexLocker lk(&(const_cast<DIconItemDelegate *>(this)->m_mutex));
584  Q_D(const DIconItemDelegate);
585 
586  bool isCanvas = parent()->property("isCanvasViewHelper").toBool();
588  bool isDragMode = (static_cast<QPaintDevice *>(parent()->parent()->viewport()) != painter->device());
589  bool isEnabled = option.state & QStyle::State_Enabled;
590  bool hasFocus = option.state & QStyle::State_HasFocus;
591 
592  if (index == d->expandedIndex && !parent()->isSelected(index))
593  const_cast<DIconItemDelegate *>(this)->hideNotEditingIndexWidget();
594 
595  painter->setOpacity(parent()->isTransparent(index) ? 0.3 : 1.0);
596 
597  QStyleOptionViewItem opt = option;
598  initStyleOption(&opt, index);
599  painter->setFont(opt.font);
600 
601  static QFont old_font = opt.font;
602 
603  if (old_font != opt.font) {
604  if (d->expandedItem)
605  d->expandedItem->setFont(opt.font);
606 
607  QWidget *editing_widget = editingIndexWidget();
608 
609  if (editing_widget)
610  editing_widget->setFont(opt.font);
611 
612  const_cast<DIconItemDelegate *>(this)->updateItemSizeHint();
613 // parent()->parent()->updateEditorGeometries();
614  }
615 
616  old_font = opt.font;
617 
618  bool isSelected = !isDragMode && (opt.state & QStyle::State_Selected) && opt.showDecorationSelected;
619  bool isDropTarget = parent()->isDropTarget(index);
620 
621  DPalette pl(DApplicationHelper::instance()->palette(option.widget));
622  QColor c = pl.color(DPalette::ColorGroup::Active, DPalette::ColorType::ItemBackground);
623  QColor base_color = c;
624  if (option.widget) {
625  DPalette pa = DApplicationHelper::instance()->palette(option.widget);
626  base_color = option.widget->palette().base().color();
627  DGuiApplicationHelper::ColorType ct = DGuiApplicationHelper::toColorType(base_color);
628  if (ct == DGuiApplicationHelper::DarkType) {
629  base_color = DGuiApplicationHelper::adjustColor(base_color, 0, 0, +5, 0, 0, 0, 0);
630  }
631  }
632 
633  if ((isDropTarget && !isSelected) || option.state & QStyle::StateFlag::State_Selected) {
634  if (isCanvas) {
635  c = pl.color(DPalette::ColorGroup::Active, QPalette::ColorRole::Highlight);
636  } else {
637  c.setAlpha(c.alpha() + 30);
638  }
639  } else if (option.state & QStyle::StateFlag::State_MouseOver) {
640  c = c.lighter();
641  } else if (!isCanvas) {
642  c = base_color;
643  }
644 
645  QRectF rect = opt.rect;
646  int backgroundMargin = isCanvas ? 0 : COLUMU_PADDING;
647  if (!isCanvas)
648  rect.adjust(backgroundMargin, backgroundMargin, -backgroundMargin, -backgroundMargin); // 为了让对勾右上角， 缩小框框
