52 void MenuWorker::showMenuByAppItem(QPoint pos, const QModelIndex &index) {
53  setCurrentModelIndex(index);
54 
55  m_appKey = m_currentModelIndex.data(AppsListModel::AppKeyRole).toString();
56  m_appDesktop = m_currentModelIndex.data(AppsListModel::AppDesktopRole).toString();
57  m_isItemOnDesktop = m_currentModelIndex.data(AppsListModel::AppIsOnDesktopRole).toBool();
58  m_isItemOnDock = m_currentModelIndex.data(AppsListModel::AppIsOnDockRole).toBool();
59  m_isItemStartup = m_currentModelIndex.data(AppsListModel::AppAutoStartRole).toBool();
60  m_isRemovable = m_currentModelIndex.data(AppsListModel::AppIsRemovableRole).toBool();
61  m_isItemProxy = m_currentModelIndex.data(AppsListModel::AppIsProxyRole).toBool();
62  m_isItemEnableScaling = m_currentModelIndex.data(AppsListModel::AppEnableScalingRole).toBool();
63 
64  qDebug() << "appKey" << m_appKey;
65 
66  QMenu *menu = new QMenu;
67  menu->setAccessibleName("popmenu");
68 
69  QSignalMapper *signalMapper = new QSignalMapper(menu);
70 
71  QAction *open;
72  QAction *desktop;
73  QAction *dock;
74  QAction *startup;
75  QAction *proxy;
76  QAction *scale;
77  QAction *uninstall;
78 
79  open = new QAction(tr("Open"), menu);
80 
81  desktop = new QAction(m_isItemOnDesktop ?
82  tr("Remove from desktop") :
83  tr("Send to desktop"),
84  menu);
85 
86  dock = new QAction(m_isItemOnDock ?
87  tr("Remove from dock") :
88  tr("Send to dock"),
89  menu);
90 
91  startup = new QAction(m_isItemStartup ?
92  tr("Remove from startup") :
93  tr("Add to startup"),
94  menu);
95 
96 
97  uninstall = new QAction(tr("Uninstall"), menu);
98 
99  menu->addAction(open);
100  menu->addSeparator();
101  menu->addAction(desktop);
102  menu->addAction(dock);
103  menu->addSeparator();
104  menu->addAction(startup);
105 
106  if (QFile::exists(ChainsProxy_path)) {
107  proxy = new QAction(tr("Use a proxy"), menu);
108  proxy->setCheckable(true);
109  proxy->setChecked(m_isItemProxy);
110  menu->addAction(proxy);
111  signalMapper->setMapping(proxy, Proxy);
112  connect(proxy, &QAction::triggered, signalMapper, static_cast<void (QSignalMapper::*)()>(&QSignalMapper::map));
113  }
114 
115  if (!m_calcUtil->IsServerSystem) {
116  const double scale_ratio = m_xsettings->get("scale-factor").toDouble();
117  if (!qFuzzyCompare(1.0, scale_ratio)) {
118  scale = new QAction(tr("Disable display scaling"), menu);
119  scale->setCheckable(true);
120  scale->setChecked(!m_isItemEnableScaling);
121  menu->addAction(scale);
122  signalMapper->setMapping(scale, SwitchScale);
123  connect(scale, &QAction::triggered, signalMapper, static_cast<void (QSignalMapper::*)()>(&QSignalMapper::map));
124  }
125  }
126 
127  dock->setEnabled(m_appKey != "dde-trash");
128  uninstall->setEnabled(m_isRemovable);
129 
130 #ifndef WITHOUT_UNINSTALL_APP
131  menu->addAction(uninstall);
132 #endif
133 
134  connect(open, &QAction::triggered, signalMapper, static_cast<void (QSignalMapper::*)()>(&QSignalMapper::map));
135  connect(desktop, &QAction::triggered, signalMapper, static_cast<void (QSignalMapper::*)()>(&QSignalMapper::map));
136  connect(dock, &QAction::triggered, signalMapper, static_cast<void (QSignalMapper::*)()>(&QSignalMapper::map));
137  connect(startup, &QAction::triggered, signalMapper, static_cast<void (QSignalMapper::*)()>(&QSignalMapper::map));
138 
139  connect(uninstall, &QAction::triggered, signalMapper, static_cast<void (QSignalMapper::*)()>(&QSignalMapper::map));
140 
141  signalMapper->setMapping(open, Open);
142  signalMapper->setMapping(desktop, Desktop);
143  signalMapper->setMapping(dock, Dock);
144  signalMapper->setMapping(startup, Startup);
145  signalMapper->setMapping(uninstall, Uninstall);
146 
147  connect(signalMapper, static_cast<void (QSignalMapper::*)(const int)>(&QSignalMapper::mapped), this, &MenuWorker::handleMenuAction);
148  connect(menu, &QMenu::aboutToHide, this, &MenuWorker::handleMenuClosed);
149  connect(menu, &QMenu::aboutToHide, menu, &QMenu::deleteLater);
150 
151  menu->move(pos);
152  m_menuIsShown = true;
153  m_menuGeometry = menu->geometry();
154  menu->exec();
155 }

180 void MenuWorker::handleMenuAction(int index)
181 {
182  switch (index) {
183  case Open:
184  handleOpen();
185  break;
186  case Desktop:
187  handleToDesktop();
188  break;
189  case Dock:
190  handleToDock();
191  break;
192  case Startup:
193  handleToStartup();
194  break;
195  case Proxy:
196  handleToProxy();
197  break;
198  case SwitchScale:
199  handleSwitchScaling();
200  break;
201  case Uninstall:
202  emit unInstallApp(m_currentModelIndex);
203  break;
204  default:
205  break;
206  }
207 }

209 void MenuWorker::handleToDesktop(){
210  qDebug() << "handleToDesktop" << m_appKey;
211  if (m_isItemOnDesktop){
212  QDBusPendingReply<bool> reply = m_launcherInterface->RequestRemoveFromDesktop(m_appKey);
213  reply.waitForFinished();
214  if (!reply.isError()) {
215  bool ret = reply.argumentAt(0).toBool();
216  qDebug() << "remove from desktop:" << ret;
217  } else {
218  qCritical() << reply.error().name() << reply.error().message();
219  }
220  }else{
221  QDBusPendingReply<bool> reply = m_launcherInterface->RequestSendToDesktop(m_appKey);
222  reply.waitForFinished();
223  if (!reply.isError()) {
224  bool ret = reply.argumentAt(0).toBool();
225  qDebug() << "send to desktop:" << ret;
226  } else {
227  qCritical() << reply.error().name() << reply.error().message();
228  }
229  }
