87 void MusicSearchListview::setSearchText(QString text)
88 {
89  m_CurrentIndex = -1;
90  m_searchText = text;
91  int count = 0;
92  if (m_SearchType == SearchType::SearchMusic) {
93  m_MediaMetas.clear();
94  for (MediaMeta &metaData : DataBaseService::getInstance()->allMusicInfos()) {
95  if (CommonService::getInstance()->containsStr(text, metaData.title)) {
96  m_MediaMetas.append(metaData);
97  count ++;
98  }
99  if (count >= 5) {
100  count = 0;
101  break;
102  }
103  }
104  m_model->clear();
105  for (MediaMeta metaData : m_MediaMetas) {
106  QStandardItem *newItem = new QStandardItem;
107  m_model->appendRow(newItem);
108 
109  auto row = m_model->rowCount() - 1;
110  QModelIndex index = m_model->index(row, 0, QModelIndex());
111 
112  QVariant mediaMeta;
113  mediaMeta.setValue(metaData);
114  m_model->setData(index, mediaMeta, Qt::UserRole + SearchType::SearchMusic);
115  }
116  } else if (m_SearchType == SearchType::SearchSinger) {
117  m_SingerInfos.clear();
118  for (SingerInfo &singer : DataBaseService::getInstance()->allSingerInfos()) {
119  if (CommonService::getInstance()->containsStr(text, singer.singerName)) {
120  m_SingerInfos.append(singer);
121  count ++;
122  }
123  if (count >= 3) {
124  count = 0;
125  break;
126  }
127  }
128  m_model->clear();
129  for (SingerInfo singer : m_SingerInfos) {
130  QStandardItem *newItem = new QStandardItem;
131  m_model->appendRow(newItem);
132 
133  auto row = m_model->rowCount() - 1;
134  QModelIndex index = m_model->index(row, 0, QModelIndex());
135 
136  QVariant singerMeta;
137  singerMeta.setValue(singer);
138  m_model->setData(index, singerMeta, Qt::UserRole + SearchType::SearchSinger);
139 
140  if (singer.musicinfos.values().size() > 0) {
141  QVariant IconMeta;
142  //设置icon
143  bool iconExists = false;
144  for (int i = 0; i < singer.musicinfos.values().size(); i++) {
145  MediaMeta metaBind = singer.musicinfos.values().at(i);
146  QString imagesDirPath = Global::cacheDir() + "/images/" + metaBind.hash + ".jpg";
147  QFileInfo file(imagesDirPath);
148  if (file.exists()) {
149  QPixmap icon(imagesDirPath);
150  IconMeta.setValue(icon);
151  m_model->setData(index, IconMeta, Qt::UserRole + SearchType::SearchIcon);
152  iconExists = true;
153  break;
154  }
155  }
156  if (!iconExists) {
157  IconMeta.setValue(m_defaultIcon);
158  m_model->setData(index, IconMeta, Qt::UserRole + SearchType::SearchIcon);
159  }
160  }
161  }
162  } else if (m_SearchType == SearchType::SearchAlbum) {
163  m_AlbumInfos.clear();
164  for (AlbumInfo &album : DataBaseService::getInstance()->allAlbumInfos()) {
165  if (CommonService::getInstance()->containsStr(text, album.albumName)) {
166  m_AlbumInfos.append(album);
167  count ++;
168  }
169  if (count >= 3) {
170  count = 0;
171  break;
172  }
173  }
174  m_model->clear();
175  for (AlbumInfo album : m_AlbumInfos) {
176  QStandardItem *newItem = new QStandardItem;
177  m_model->appendRow(newItem);
178 
179  auto row = m_model->rowCount() - 1;
180  QModelIndex index = m_model->index(row, 0, QModelIndex());
181 
182  QVariant singerMeta;
183  singerMeta.setValue(album);
184  m_model->setData(index, singerMeta, Qt::UserRole + SearchType::SearchAlbum);
185 
186  if (album.musicinfos.values().size() > 0) {
187  QVariant IconMeta;
188  //设置icon
189  bool iconExists = false;
190  for (int i = 0; i < album.musicinfos.values().size(); i++) {
191  MediaMeta metaBind = album.musicinfos.values().at(i);
192  QString imagesDirPath = Global::cacheDir() + "/images/" + metaBind.hash + ".jpg";
193  QFileInfo file(imagesDirPath);
194  if (file.exists()) {
195  QPixmap icon(imagesDirPath);
196  IconMeta.setValue(icon);
197  m_model->setData(index, IconMeta, Qt::UserRole + SearchType::SearchIcon);
198  iconExists = true;
199  break;
200  }
201  }
202  if (!iconExists) {
203  IconMeta.setValue(m_defaultIcon);
204  m_model->setData(index, IconMeta, Qt::UserRole + SearchType::SearchIcon);
205  }
206  }

304 void MusicSearchListview::slotOnClicked(const QModelIndex &index)
305 {
306  if (m_SearchType == SearchType::SearchMusic) {
307  MediaMeta mediaMeta = index.data(Qt::UserRole + SearchType::SearchMusic).value<MediaMeta>();
308  emit CommonService::getInstance()->signalSwitchToView(SearchMusicResultType, mediaMeta.title);
309  if (m_SearchResultWidget) {
310  m_SearchResultWidget->setLineEditSearchString(mediaMeta.title);
311  }
312  } else if (m_SearchType == SearchType::SearchSinger) {
313  SingerInfo singerInfo = index.data(Qt::UserRole + SearchType::SearchSinger).value<SingerInfo>();
314  emit CommonService::getInstance()->signalSwitchToView(SearchSingerResultType, singerInfo.singerName);
315  if (m_SearchResultWidget) {
316  m_SearchResultWidget->setLineEditSearchString(singerInfo.singerName);
317  }
318  } else if (m_SearchType == SearchType::SearchAlbum) {
319  AlbumInfo albumInfo = index.data(Qt::UserRole + SearchType::SearchAlbum).value<AlbumInfo>();
320  emit CommonService::getInstance()->signalSwitchToView(SearchAlbumResultType, albumInfo.albumName);
321  if (m_SearchResultWidget) {
322  m_SearchResultWidget->setLineEditSearchString(albumInfo.albumName);
323  }
324  }
325 }
