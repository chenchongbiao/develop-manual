83 void FashionTrayWidgetWrapper::paintEvent(QPaintEvent *event)
84 {
85  Q_UNUSED(event);
86 
87  if (m_dragging) {
88  return;
89  }
90 
91  if (rect().height() > PLUGIN_BACKGROUND_MIN_SIZE) {
92  QPainter painter(this);
93  painter.setRenderHint(QPainter::Antialiasing, true);
94 
95  QColor color;
96  if (DGuiApplicationHelper::instance()->themeType() == DGuiApplicationHelper::LightType) {
97  color = Qt::black;
98  painter.setOpacity(0.5);
99 
100  if (m_hover) {
101  painter.setOpacity(0.6);
102  }
103 
104  if (m_pressed) {
105  painter.setOpacity(0.3);
106  }
107  } else {
108  color = Qt::white;
109  painter.setOpacity(0.1);
110 
111  if (m_hover) {
112  painter.setOpacity(0.2);
113  }
114 
115  if (m_pressed) {
116  painter.setOpacity(0.05);
117  }
118  }
119 
120  DStyleHelper dstyle(style());
121  const int radius = dstyle.pixelMetric(DStyle::PM_FrameRadius);
122 
123  QPainterPath path;
124 
125  int minSize = std::min(width(), height());
126  QRect rc(0, 0, minSize, minSize);
127  rc.moveTo(rect().center() - rc.center());
128 
129  path.addRoundedRect(rc, radius, radius);
130  painter.fillPath(path, color);
131  }
132 }

212 void FashionTrayWidgetWrapper::handleMouseMove(QMouseEvent *event)
213 {
214  if(m_absTrayWidget.isNull())
215  return;
216 
217  if (event->buttons() != Qt::MouseButton::LeftButton) {
218  return QWidget::mouseMoveEvent(event);
219  }
220 
221  if ((event->pos() - MousePressPoint).manhattanLength() < TRAY_ITEM_DRAG_THRESHOLD) {
222  return;
223  }
224 
225  // 如果是触屏事件转换而来并且没有收到后端的延时触屏消息，不进行拖拽
226  if (event->source() == Qt::MouseEventSynthesizedByQt && !TouchSignalManager::instance()->isDragIconPress()) {
227  return;
228  }
229 
230  event->accept();
231 
232  QDrag drag(this);
233  QMimeData *mimeData = new QMimeData;
234  mimeData->setData(TRAY_ITEM_DRAG_MIMEDATA, m_itemKey.toLocal8Bit());
235  QPixmap pixmap = grab();
236 
237  drag.setMimeData(mimeData);
238  drag.setPixmap(pixmap);
239  drag.setHotSpot(pixmap.rect().center() / pixmap.devicePixelRatioF());
240 
241  m_absTrayWidget->setVisible(false);
242  m_dragging = true;
243  Q_EMIT dragStart();
244 
245  // start drag
246  drag.exec();
247 
248  m_absTrayWidget->setVisible(true);
249  m_dragging = false;
250  m_hover = false;
251  m_pressed = false;
252  Q_EMIT dragStop();
253 }
