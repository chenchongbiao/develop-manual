56 FootnoteData Footnotes97::footnote( U32 globalCP, bool& ok )
57 {
58 #ifdef WV2_DEBUG_FOOTNOTES
59  wvlog << "Footnotes97::footnote(): globalCP=" << globalCP << std::endl;
60 #endif
61  ok = true; // let's assume we will find it
62  if ( m_footnoteRefIt && m_footnoteRefIt->currentStart() == globalCP &&
63  m_footnoteTxtIt != m_footnoteTxt.end() ) {
64  bool fAuto = m_footnoteRefIt->current()->nAuto;
65  ++( *m_footnoteRefIt ); // yay, but it is hard to make that more elegant
66 
67  U32 start = *m_footnoteTxtIt;
68  ++m_footnoteTxtIt;
69  return FootnoteData( FootnoteData::Footnote, fAuto, start, *m_footnoteTxtIt );
70  }
71 
72  if ( m_endnoteRefIt && m_endnoteRefIt->currentStart() == globalCP &&
73  m_endnoteTxtIt != m_endnoteTxt.end() ) {
74  bool fAuto = m_endnoteRefIt->current()->nAuto;
75  ++( *m_endnoteRefIt ); // yay, but it is hard to make that more elegant
76 
77  U32 start = *m_endnoteTxtIt;
78  ++m_endnoteTxtIt;
79  return FootnoteData( FootnoteData::Endnote, fAuto, start, *m_endnoteTxtIt );
80  }
81 
82  wvlog << "Bug: There is no footnote or endnote with the CP " << globalCP << std::endl;
83  ok = false;
84  return FootnoteData( FootnoteData::Footnote, false, 0, 0 );
85 }

97 void Footnotes97::init( U32 fcRef, U32 lcbRef, U32 fcTxt, U32 lcbTxt, AbstractOLEStreamReader* tableStream,
98  PLCF<Word97::FRD>** ref, PLCFIterator<Word97::FRD>** refIt,
99  std::vector<U32>& txt, std::vector<U32>::const_iterator& txtIt )
100 {
101  if ( lcbRef == 0 )
102  return;
103 
104  tableStream->seek( fcRef, SEEK_SET );
105  *ref = new PLCF<Word97::FRD>( lcbRef, tableStream );
106  *refIt = new PLCFIterator<Word97::FRD>( **ref );
107 
108 #ifdef WV2_DEBUG_FOOTNOTES
109  wvlog << "Footnotes97::init()" << std::endl;
110  ( *ref )->dumpCPs();
111 #endif
112 
113  if ( lcbTxt == 0 )
114  wvlog << "Bug: lcbTxt == 0 but lcbRef != 0" << std::endl;
115  else {
116  if ( static_cast<U32>( tableStream->tell() ) != fcTxt ) {
117  wvlog << "Warning: Found a hole in the table stream" << std::endl;
118  tableStream->seek( fcTxt, SEEK_SET );
119  }
120  for ( U32 i = 0; i < lcbTxt; i += sizeof( U32 ) ) {
121  txt.push_back( tableStream->readU32() );
122 #ifdef WV2_DEBUG_FOOTNOTES
123  wvlog << "read: " << txt.back() << std::endl;
124 #endif
125  }
126  txtIt = txt.begin();
127  }
128 #ifdef WV2_DEBUG_FOOTNOTES
129  wvlog << "Footnotes97::init() done" << std::endl;
130 #endif
131 }
