137 void CPictureItem::setRotation90(bool leftOrRight)
138 {
139 
140  QPointF center = this->boundingRect().center();
141  qreal angle = 0.0;
142 
143  // 左旋转减小 右旋转增大
144  if (leftOrRight == true) {
145  angle = - 90.0;
146  } else {
147  angle = 90.0;
148  }
149 
150  // 矩阵变换
151  QTransform trans;
152  trans.translate(center.x(), center.y());
153  trans.rotate(angle);
154  trans.translate(-center.x(), -center.y());
155 
156  // 设置当前旋转角度
157  setDrawRotatin(angle + drawRotation());
158  setTransform(trans, true);
159 }

161 void CPictureItem::loadGraphicsUnit(const CGraphicsUnit &data)
162 {
163  if (data.data.pPic != nullptr) {
164  CGraphicsRectItem::loadGraphicsRectUnit(data.data.pPic->rect);
165 
166  qDebug() << "-------data.reson = " << data.reson;
167  //仅在 '复制' 或者 '保存/加载到ddf文件时'加载图片原数据
168  if (data.reson == EDuplicate || data.reson == ESaveToDDf) {
169  m_pixmap = QPixmap::fromImage(data.data.pPic->image);
170  _srcByteArry = data.data.pPic->srcByteArry;
171  }
172  this->_flipHorizontal = data.data.pPic->flipHorizontal;
173  this->_flipVertical = data.data.pPic->flipVertical;
174  }
175  loadHeadData(data.head);
176  update();
177 }

179 CGraphicsUnit CPictureItem::getGraphicsUnit(EDataReason reson) const
180 {
181  CGraphicsUnit unit;
182  unit.reson = reson;
183 
184  unit.head.dataType = this->type();
185  unit.head.dataLength = sizeof(SGraphicsPictureUnitData);
186 
187  unit.head.pos = this->pos();
188  unit.head.rotate = this->drawRotation();
189  unit.head.zValue = this->zValue();
190  unit.head.blurCount = _blurInfos.count();
191  unit.head.blurInfos = _blurInfos;
192  unit.head.trans = this->transform();
193 
194  unit.data.pPic = new SGraphicsPictureUnitData();
195  unit.data.pPic->rect.topLeft = this->rect().topLeft();
196  unit.data.pPic->rect.bottomRight = this->rect().bottomRight();
197  unit.data.pPic->flipHorizontal = this->_flipHorizontal;
198  unit.data.pPic->flipVertical = this->_flipVertical;
199 
200  //仅在 '复制' 或者 '保存/加载到ddf文件时'读取图片原数据
201  if (reson == EDuplicate || reson == ESaveToDDf) {
202  unit.data.pPic->image = m_pixmap.toImage();
203 
204  //源数据不为空,那么进行对当前图片
205  if (_srcByteArry.isEmpty()) {
206  QBuffer buferTemp;
207  if (buferTemp.open(QIODevice::ReadWrite)) {
208  m_pixmap.save(&buferTemp, "png", 100);
209  unit.data.pPic->srcByteArry = buferTemp.buffer();
210  buferTemp.close();
211  }
212  } else {
213  unit.data.pPic->srcByteArry = _srcByteArry;
214  }
215  }
216  return unit;
217 }
