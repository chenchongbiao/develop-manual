74 QString DFMFullTextSearchManager::getFileContents(const QString &filePath)
75 {
76  std::string text = "";
77  FormattingStyle options;
78  options.table_style = TABLE_STYLE_TABLE_LOOK;
79  options.list_style.setPrefix(" * ");
80  options.url_style = URL_STYLE_UNDERSCORED;
81  XmlParseMode mode = PARSE_XML;
82 
83  PlainTextExtractor::ParserType parser_type = PlainTextExtractor::PARSER_AUTO;
84  QFileInfo fileInfo(filePath);
85  QString ext = fileInfo.suffix();
86  if (ext == "rtf")
87  parser_type = PlainTextExtractor::PARSER_RTF;
88  else if (ext == "odt" || ext == "ods" || ext == "odp" || ext == "odg" || ext == "docx" || ext == "xlsx" || ext == "pptx" || ext == "ppsx")
89  parser_type = PlainTextExtractor::PARSER_ODF_OOXML;
90  else if (ext == "xls")
91  parser_type = PlainTextExtractor::PARSER_XLS;
92  else if (ext == "xlsb")
93  parser_type = PlainTextExtractor::PARSER_XLSB;
94  else if (ext == "doc" || ext == "dot" || ext == "wps")
95  parser_type = PlainTextExtractor::PARSER_DOC;
96  else if (ext == "ppt" || ext == "pps" || ext == "dps")
97  parser_type = PlainTextExtractor::PARSER_PPT;
98  else if (ext == "pdf")
99  parser_type = PlainTextExtractor::PARSER_PDF;
100  else if (ext == "txt" || ext == "text")
101  parser_type = PlainTextExtractor::PARSER_TXT;
102  else {
103  qDebug() << "Unsupported file extension: " << ext;
104  return "";
105  }
106 
107  //创建文件解析器
108  PlainTextExtractor extractor(parser_type);
109 
110  //设置 xml 格式解析模式
111  if (mode != PARSE_XML) {
112  extractor.setXmlParseMode(mode);
113  }
114 
115  //
116  extractor.setFormattingStyle(options);
117 
118  //解析文件内容,
119  /*
120  * 原因是解析出来的内容和 tests目录下面 *.out 中的内容不相符，
121  * 例如: 解析 tests/1.doc.out 最终结果的内容就是现在下面 printf 所打印输出的内容格式，
122  * 从可以 tests/Makefile 中看到，在测试的过程中是通过 diff 命令比较，也就是通过比较 1.doc
123  * 的解析结果内容和 1.doc.out 内容的是否相同来判断程序是否正常运行的
124  */
125  if (!extractor.processFile(filePath.toStdString(), text)) {
126  qDebug() << "Error processing file " << filePath;
127  return "";
128  }
129  return text.c_str();
130 }

160 void DFMFullTextSearchManager::doPagingSearch(const SearcherPtr &searcher, const QueryPtr &query, int32_t hitsPerPage, bool raw, bool interactive)
161 {
162  // Collect enough docs to show 5 pages
163  TopScoreDocCollectorPtr collector = TopScoreDocCollector::create(5 * hitsPerPage, false);
164  searcher->search(query, collector);
165  Collection<ScoreDocPtr> hits = collector->topDocs()->scoreDocs;
166 
167  int32_t numTotalHits = collector->getTotalHits();
168  qDebug() << numTotalHits << " total matching documents";
169 
170  int32_t start = 0;
171  int32_t end = std::min(numTotalHits, hitsPerPage);
172 
173  while (true) {
174  if (end > hits.size()) {
175  std::wcout << L"Only results 1 - " << hits.size() << L" of " << numTotalHits << L" total matching documents collected.\n";
176  std::wcout << L"Collect more (y/n) ?";
177  String line;
178  std::wcin >> line;
179  boost::trim(line);
180 
181  if (line.empty() || boost::starts_with(line, L"n")) {
182  break;
183  }
184 
185  collector = TopScoreDocCollector::create(numTotalHits, false);
186  searcher->search(query, collector);
187  hits = collector->topDocs()->scoreDocs;
188  }
189 
190  end = std::min(hits.size(), start + hitsPerPage);
191 
192  for (int32_t i = start; i < end; ++i) {
193  if (m_state == JobController::Stoped) {
194  return;
195  }
196 
197  if (raw) { // output raw format
198  std::wcout << L"doc=" << hits[i]->doc << L" score=" << hits[i]->score << L"\n";
199  continue;
200  }
201 
202  DocumentPtr doc = searcher->doc(hits[i]->doc);
203  String path = doc->get(L"path");
204  if (!path.empty()) {
205  /*fix bug 45096 对搜索结果里面的修改时间进行判断，如果时间不一样则表示文件已经更改，搜索结果不出现该文件*/
206  QFileInfo info(QString::fromStdWString(path));
207  QString modifyTime = info.lastModified().toString("yyyyMMddHHmmss");
208  String storeTime = doc->get(L"modified");
209  if (modifyTime.toStdWString() != storeTime) {
210  continue;
211  } else {
212  searchResults.append(StringUtils::toUTF8(path).c_str());
213  qDebug() << QString::number(i + 1) << " " << QString::fromStdWString(path.c_str());
214  }
215 
216  } else {
217  qDebug() << QString::number(i + 1).append(". No path for this document");
218  }
219  }
220 
221  if (!interactive) {
222  break;
223  }
224 
225  if (numTotalHits >= end) {
226  break;
227  end = std::min(numTotalHits, start + hitsPerPage);
228  }
229  }

232 bool DFMFullTextSearchManager::searchByKeyworld(const QString &keyword)
233 {
234  try {
235  int32_t hitsPerPage = SEARCH_RESULT_NUM;
236  // only searching, so read-only=true

238

240  AnalyzerPtr analyzer = newLucene<StandardAnalyzer>(LuceneVersion::LUCENE_CURRENT);

242  parser->setAllowLeadingWildcard(true);//设定第一个* 可以匹配

248  } else {
249  QString word = "*" + keyword + "*";
250  query = parser->parse(word.toStdWString());
251  }
252  qDebug() << "Searching for: " << keyword;
253 
254  doPagingSearch(searcher, query, hitsPerPage, false, true);
255  reader->close();
256  } catch (LuceneException &e) {
257  qDebug() << "Exception: " << QString::fromStdWString(e.getError());
258  return false;
259  }

261  return true;

278 void DFMFullTextSearchManager::readFileName(const char *filePath, QStringList &result)
279 {
280  if (!isCreateIndex && m_state == JobController::Stoped) {
281  return;
282  }
283 
284  QRegExp reg("^/(boot|dev|proc|sys|run|lib|usr).*$");
285  if (reg.exactMatch(QString(filePath)) && !QString(filePath).startsWith("/run/user")) {
286  return;
287  }
288 
289  std::string path(filePath);
290  int pathLevel = 0;
291  std::for_each(std::begin(path), std::end(path), [&](char ch) {
292  if (ch == '/') {
293  pathLevel++;
294  }
295  });
296  /*目录级数限定*/
297  if (pathLevel > 20) {
298  return;
299  }
300 
301  DIR *dir = NULL;
302  if (!(dir = opendir (filePath))) {
303  //trace ("can't open: %s\n", dname);
304  return;
305  }
306  struct dirent *dent = NULL;
307  int len = strlen (filePath);
308  if (len >= FILENAME_MAX - 1) {
309  //trace ("filename too long: %s\n", dname);
310  return;
311  }
312 
313  char fn[FILENAME_MAX] = "";
314  strcpy (fn, filePath);
315  if (strcmp (filePath, "/")) {
316  // TODO: use a more performant fix to handle root directory
317  fn[len++] = '/';
318  }
319  while ((dent = readdir (dir))) {
320  if (!dent->d_name[0] == '.') {
321  // file is dotfile, skip
322  continue;
323  }
324  if ((QString(dent->d_name).at(0) == '.') && !QString(dent->d_name).startsWith(".local")) {
325  continue;
326  }
327  if (!strcmp (dent->d_name, ".") || !strcmp (dent->d_name, "..") || !strcmp (dent->d_name, ".avfs")) {
328  continue;
329  }
330 
331  struct stat st;
332  strncpy (fn + len, dent->d_name, FILENAME_MAX - len);
333  if (lstat (fn, &st) == -1) {
334  //warn("Can't stat %s", fn);
335  continue;
336  }
337  const bool is_dir = S_ISDIR (st.st_mode);
338  if (is_dir) {
339  readFileName(fn, result);
340  } else {
341  QFileInfo fileInfo(fn);
342  QString suffix = fileInfo.suffix();
343  QRegExp reg("(rtf)|(odt)|(ods)|(odp)|(odg)|(docx)|(xlsx)|(pptx)|(ppsx)|"
344  "(xls)|(xlsb)|(doc)|(dot)|(wps)|(ppt)|(pps)|(txt)|(htm)|(html)|(pdf)|(dps)");
345  if (reg.exactMatch(suffix)) {
346  result.append(fn);
347  }
348  }
349  }
350  if (dir) {
351  closedir (dir);
352  }
353 }

355 bool DFMFullTextSearchManager::updateIndex(const QString &filePath)
356 {
357  bool res = false;
358  qDebug() << "Update index";
359  QStringList files;
360  readFileName(filePath.toStdString().c_str(), files);
361  for (QString file : files) {
362  if (m_state == JobController::Stoped) {
363  qDebug() << "full text search stop!";
364  return false;
365  }
366 
367  try {
368  IndexReaderPtr reader = IndexReader::open(FSDirectory::open(indexStorePath.toStdWString()), true);
369  SearcherPtr searcher = newLucene<IndexSearcher>(reader);
370  TermQueryPtr query = newLucene<TermQuery>(newLucene<Term>(L"path", file.toStdWString()));
371 
372  // 文件路径为唯一值，所以搜索一个结果就行了
373  TopDocsPtr topDocs = searcher->search(query, 1);
374  int32_t numTotalHits = topDocs->totalHits;
375  if (numTotalHits == 0) {
376  try {
377  IndexWriterPtr writer = newLucene<IndexWriter>(FSDirectory::open(indexStorePath.toStdWString()),
378  newLucene<StandardAnalyzer>(LuceneVersion::LUCENE_CURRENT),
379  IndexWriter::MaxFieldLengthLIMITED);
380  writer->addDocument(getFileDocument(file));
381  qDebug() << "Add file: [" << file << "]";
382  writer->close();
383  res = true;
384  } catch (LuceneException &ex) {
385  qDebug() << "Add file error:" << QString::fromStdWString(ex.getError());
386  }
387 
388  } else {
389  DocumentPtr doc = searcher->doc(topDocs->scoreDocs[0]->doc);
390  QFileInfo info(file);
391  QString modifyTime = info.lastModified().toString("yyyyMMddHHmmss");
392  String storeTime = doc->get(L"modified");
393  if (modifyTime.toStdWString() == storeTime) {
394  continue;
395  } else {
396  try {
397  IndexWriterPtr writer = newLucene<IndexWriter>(FSDirectory::open(indexStorePath.toStdWString()),
398  newLucene<StandardAnalyzer>(LuceneVersion::LUCENE_CURRENT),
399  IndexWriter::MaxFieldLengthLIMITED);
400  //定义一个更新条件
401  TermPtr term = newLucene<Term>(L"path", file.toStdWString());
402  //更新
403  writer->updateDocument(term, getFileDocument(file));
404  qDebug() << "Update file: [" << file << "]";
405  //关闭
406  writer->close();
407  res = true;
408  } catch (LuceneException &ex) {
409  qDebug() << "Update file error:" << QString::fromStdWString(ex.getError());
410  }
411  }
412  }
413  reader->close();
414  } catch (LuceneException &ex) {
415  qDebug() << QString::fromStdWString(ex.getError());
416  }
417  }
418 
419  return res;
420 }

422 bool DFMFullTextSearchManager::createFileIndex(const QString &sourcePath)
423 {
424  QDir sourceDir(sourcePath);
425  if (!sourceDir.exists()) {
426  qDebug() << "Source directory doesn't exist: " << sourcePath;
427  return false;
428  }
429 
430  QDir indexDir(indexStorePath);
431  if (!indexDir.exists()) {
432  if (!indexDir.mkpath(indexStorePath)) {
433  qDebug() << "Unable to create directory: " << indexStorePath;
434  return false;
435  }
436  }
437 
438  uint64_t beginIndex = MiscUtils::currentTimeMillis();
439 
440  try {
441  IndexWriterPtr writer = newLucene<IndexWriter>(FSDirectory::open(indexStorePath.toStdWString()),
442  newLucene<StandardAnalyzer>(LuceneVersion::LUCENE_CURRENT),
443  true,
444  IndexWriter::MaxFieldLengthLIMITED);
445  qDebug() << "Indexing to directory: " << indexStorePath;
446  writer->deleteAll();
447  indexDocs(writer, sourcePath);
448 
449  uint64_t endIndex = MiscUtils::currentTimeMillis();
450  uint64_t indexDuration = endIndex - beginIndex;
451  writer->optimize();
452 
453  uint64_t optimizeDuration = MiscUtils::currentTimeMillis() - endIndex;
454  writer->close();
455 
456  qDebug() << "\n\nIndex time: " << indexDuration + optimizeDuration << " milliseconds\n\n";
457  } catch (LuceneException &e) {
458  qDebug() << "Exception: " << QString::fromStdWString(e.getError());
459  return false;
460  }
461  return true;
462 }
