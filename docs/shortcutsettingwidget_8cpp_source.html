184 void ShortCutSettingWidget::addShortcut(QList<ShortcutInfo *> list, ShortcutModel::InfoType type)
185 {
186  if ((m_assistiveToolsGroup == nullptr) && (type == ShortcutModel::AssistiveTools)) {
187  m_assistiveToolsIdList.clear();
188  QList<ShortcutInfo *>::iterator it = list.begin();
189  for (; it != list.end(); ++it) {
190  ShortcutInfo *assistiveToolsinfo = *it;
191  m_assistiveToolsIdList << assistiveToolsinfo->id;
192  }
193  return;
194  }
195 
196  if ((m_workspaceGroup == nullptr) && (type == ShortcutModel::Workspace)) {
197  m_workspaceIdList.clear();
198  QList<ShortcutInfo *>::iterator it = list.begin();
199  for (; it != list.end(); ++it) {
200  ShortcutInfo *workspaceIdListlsinfo = *it;
201  m_workspaceIdList << workspaceIdListlsinfo->id;
202  }
203  return;
204  }
205  QMap<ShortcutModel::InfoType, QList<ShortcutItem *>*> InfoMap {
206  {ShortcutModel::System, &m_systemList},
207  {ShortcutModel::Window, &m_windowList},
208  {ShortcutModel::Workspace, &m_workspaceList},
209  {ShortcutModel::AssistiveTools, &m_assistiveToolsList},
210  {ShortcutModel::Custom, &m_customList}
211  };
212 
213  QMap<ShortcutModel::InfoType, SettingsGroup *> GroupMap {
214  {ShortcutModel::System, m_systemGroup},
215  {ShortcutModel::Window, m_windowGroup},
216  {ShortcutModel::Workspace, m_workspaceGroup},
217  {ShortcutModel::AssistiveTools, m_assistiveToolsGroup},
218  {ShortcutModel::Custom, m_customGroup}
219  };
220 
221  QList<ShortcutItem *> *itemList{ InfoMap[type] };
222  auto group = GroupMap[type];
223  for (auto it = itemList->begin(); it != itemList->end();) {
224  ShortcutItem *item = *it;
225 
226  group->removeItem(item);
227  m_allList.removeOne(item);
228  it = itemList->erase(it);
229  item->deleteLater();
230  }
231 
232  QList<ShortcutInfo *>::iterator it = list.begin();
233  for (; it != list.end(); ++it) {
234  ShortcutInfo *info = *it;
235  ShortcutItem *item = new ShortcutItem();
236  item->setAccessibleName(info->name);
237  connect(item, &ShortcutItem::requestUpdateKey, this, &ShortCutSettingWidget::requestUpdateKey);
238  item->setShortcutInfo(info);
239  item->setTitle(info->name);
240  info->item = item;
241  m_searchInfos[info->toString()] = info;
242 
243  m_allList << item;
244  switch (type) {
245  case ShortcutModel::System:
246  m_systemGroup->appendItem(item);
247  m_systemList.append(item);
248  break;
249  case ShortcutModel::Window:
250  m_windowGroup->appendItem(item);
251  m_windowList.append(item);
252  break;
253  case ShortcutModel::Workspace:
254  m_workspaceGroup->appendItem(item);
255  m_workspaceList.append(item);
256 
257  if (m_workspaceGroup->itemCount() > 1)
258  m_workspaceHead->setVisible(true);
259  break;
260  case ShortcutModel::AssistiveTools:
261  m_assistiveToolsGroup->appendItem(item);
262  m_assistiveToolsList.append(item);
263  break;
264  case ShortcutModel::Custom:
265  connect(m_head, &SettingsHead::editChanged, item, &ShortcutItem::onEditMode);
266  m_customGroup->appendItem(item);
267  m_customList.append(item);
268 
269  if (m_customGroup->itemCount() > 1)
270  m_head->setVisible(true);
271 
272  connect(item, &ShortcutItem::requestRemove, this, &ShortCutSettingWidget::onDestroyItem);
273  connect(item, &ShortcutItem::shortcutEditChanged, this, &ShortCutSettingWidget::shortcutEditChanged);
274  break;
275  default:
276  break;
277  }
278  }

407 void ShortCutSettingWidget::onSearchStringFinish(const QList<ShortcutInfo*> searchList)
408 {
409  for (int i = 0; i != m_searchGroup->itemCount(); ++i) {
410  m_searchGroup->getItem(i)->deleteLater();
411  }
412  m_searchGroup->clear();
413  QList<ShortcutInfo *> list = searchList;
414  qDebug() << "searchList count is " << searchList.count();
415  for (int i = 0; i < list.count(); i++) {
416  if (m_assistiveToolsGroup == nullptr && m_assistiveToolsIdList.contains(list[i]->id))
417  continue;
418 
419  if (m_workspaceGroup == nullptr && m_workspaceIdList.contains(list[i]->id))
420  continue;
421 
422  ShortcutItem *item = new ShortcutItem;
423  connect(item, &ShortcutItem::requestUpdateKey, this, &ShortCutSettingWidget::requestUpdateKey);
424  item->setShortcutInfo(list[i]);
425  item->setTitle(list[i]->name);
426  item->setFixedHeight(36);
427  m_searchGroup->appendItem(item);

463 void ShortCutSettingWidget::onKeyEvent(bool press, const QString &shortcut)
464 {
465  ShortcutInfo *current = m_model->currentInfo();
466 
467  if (!current)
468  return;
469 
470  ShortcutInfo *conflict = m_model->getInfo(shortcut);
471 
472  if (conflict == current && conflict->accels == current->accels) {
473  current->item->setShortcut(current->accels);
474  return;
475  }
476 
477  if (!press) {
478  if (shortcut.isEmpty()) {
479  current->item->setShortcut(current->accels);
480  return;
481  }
482 
483  if (shortcut == "BackSpace" || shortcut == "Delete") {
484  current->item->setShortcut("");
485  Q_EMIT requestDisableShortcut(current);
486  } else {
487  if (conflict) {
488  // have conflict
489  Q_EMIT requestShowConflict(current, shortcut);
490  current->item->setShortcut(current->accels);
491  } else {
492  // save
493  current->accels = shortcut;
494  Q_EMIT requestSaveShortcut(current);
495  }
496  }
497  return;
498  }
499 
500  // update shortcut to item
501  current->item->setShortcut(shortcut);
502 }
