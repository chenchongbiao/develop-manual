36 enum EDdfVersion {
37  EDdfUnknowed = -1, //未知的版本(ddf被修改过)
38 
39  EDdf5_8_0_Base = 0, //最原始的ddf版本
40  EDdf5_8_0_9_1, //添加了矩形属性的ddf版本
41  EDdf5_8_0_10_1, //添加了直线起点终点的版本（但这个版本在保存时漏掉保存了pen图元的保存 所以这个版本保存的ddf被加载后pen图元显示会问题）

44

46 
47  EDdf5_8_0_48_LATER, //1030专业版5_8_0_48版本后将多保存图片图元的翻转信息,以解决翻转信息丢失的问题
48 
49  EDdf5_8_0_84_LATER, //1040 sprint-1引入模糊,仅模糊图片,需要记录图元的模糊信息(做成了一种属性,而非模糊图元了)(实际发布班为5.9.0.1)

54 
55  EDdfVersionCount,

60 static EDdfVersion getVersion(QDataStream &stream)
61 {
62  EDdfVersion version = EDdfUnknowed;
63  quint32 headCheckFlag;
64 
65  if (stream.device() != nullptr) {
66 
67 
68  qint64 pos = stream.device()->pos();
69  stream.device()->seek(0);
70 
71  stream >> headCheckFlag;
72 
73  if (headCheckFlag == static_cast<quint32>(0xA0B0C0D0)) {
74  int verVar;
75  stream >> verVar;
76  version = EDdfVersion(verVar);
77 
78  } else {
79  version = EDdf5_8_0_Base;
80  //version = EDdfUnknowed;
81  }
82  //还原
83  stream.device()->seek(pos);
84  }
85 
86  return version;
87 }

653  static void deepCopy(CGraphicsUnit &des, const CGraphicsUnit &source)
654  {
655  des.head = source.head;
656  //qDebug() << "source blurCount = " << source.head.blurCount << "des blurCount = " << des.head.blurCount;
657  switch (source.head.dataType) {
658  case RectType: {
659  if (source.data.pRect == nullptr) {
660  des.release();
661  break;
662  }
663 
664  if (des.data.pRect == nullptr) {
665  des.data.pRect = new SGraphicsRectUnitData;
666  }
667  *des.data.pRect = *source.data.pRect;
668  break;
669  }
670  case EllipseType: {
671  if (source.data.pCircle != nullptr) {
672  if (des.data.pCircle == nullptr) {
673  des.data.pCircle = new SGraphicsCircleUnitData;
674  }
675  *des.data.pCircle = *source.data.pCircle;
676  } else {
677  des.release();
678  }
679  break;
680  }
681  case TriangleType: {
682  if (source.data.pTriangle != nullptr) {
683  if (des.data.pTriangle == nullptr) {
684  des.data.pTriangle = new SGraphicsTriangleUnitData;
685  }
686  *des.data.pTriangle = *source.data.pTriangle;
687  } else {
688  des.release();
689  }
690  break;
