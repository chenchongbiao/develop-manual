190  std::string parseRkRec(std::vector<unsigned char>::const_iterator src, short int xf_index)
191  {
192  double number;
193  if ((*src) & 0x02)
194  number = (double)(getS32LittleEndian(src) >> 2);
195  else {
196  union {
197  unsigned char xls_num[8];
198  double num;
199  } rk_num_conv;
200  std::fill(rk_num_conv.xls_num, rk_num_conv.xls_num + 8, '\0');
201 #ifdef WORDS_BIGENDIAN
202 std:
203  reverse_copy(src, src + 4, dconv.xls_num);
204  rk_num_conv.xls_num[0] &= 0xfc;
205 #else
206  std::copy(src, src + 4, rk_num_conv.xls_num + 4);
207  rk_num_conv.xls_num[3] &= 0xfc;
208 #endif
209  number = rk_num_conv.num;
210  }
211  if ((*src) & 0x01)
212  number *= 0.01;
213  return formatXLSNumber(number, xf_index);
214  }

216  std::string parseXLUnicodeString(std::vector<unsigned char>::const_iterator *src, std::vector<unsigned char>::const_iterator src_end, const std::vector<size_t> &record_sizes, size_t &record_index, size_t &record_pos)
217  {
218  if (record_pos >= record_sizes[record_index]) {
219  size_t diff = record_pos - record_sizes[record_index];
220  if (diff > 0)
221  *m_log_stream << "Warning: XLUnicodeString starts " << diff << " bytes past record boundary.\n";
222  record_pos = diff;
223  record_index++;
224  }
225  /*
226  This part is mostly based on OpenOffice/LibreOffice XLS format documentation and filter source code.
227  record id BIFF -> XF type String type
228  0x0004 2-7 -> 3 byte 8-bit length, byte string
229  0x0004 8 -> 3 byte 16-bit length, unicode string
230  0x0204 2-7 -> 2 byte 16-bit length, byte string
231  0x0204 8 -> 2 byte 16-bit length, unicode string
232  */
233 #warning TODO: Add support for record 0x0004 (in BIFF2).
234  if (src_end - *src < 2) {
235  *m_log_stream << "Unexpected end of buffer.\n";
236  *src = src_end;
237  return "";
238  }
239  int count = getU16LittleEndian(*src);
240  *src += 2;
241  record_pos += 2;
242  if (src_end - *src < 1) {
243  *m_log_stream << "Unexpected end of buffer.\n";
244  *src = src_end;
245  return "";
246  }
247  int flags = 0;
248  if (m_biff_version >= BIFF8) {
249  flags = **src;
250  *src += 1;
251  record_pos += 1;
252  }
253  int char_size = (flags & 0x01) ? 2 : 1;
254  int after_text_block_len = 0;
255  if (flags & 0x08) { // rich text
256  if (m_verbose_logging)
257  *m_log_stream << "Rich text flag enabled.\n";
258  if (src_end - *src < 2) {
259  *m_log_stream << "Unexpected end of buffer.\n";
260  *src = src_end;
261  return "";
262  }
263  after_text_block_len += 4 * getU16LittleEndian(*src);
264  *src += 2;
265  record_pos += 2;
266  }
267  if (flags & 0x04) { // asian
268  if (m_verbose_logging)
269  *m_log_stream << "Asian flag enabled.\n";
270  if (src_end - *src < 4) {
271  *m_log_stream << "Unexpected end of buffer.\n";
272  *src = src_end;
273  return "";
274  }
275  after_text_block_len += getS32LittleEndian(*src);
276  *src += 4;
277  record_pos += 4;
278  }
279  if (m_verbose_logging && after_text_block_len > 0)
280  *m_log_stream << "Additional formatting blocks found, size " << after_text_block_len << " bytes.\n";
281  std::string dest;
282  std::vector<unsigned char>::const_iterator s = *src;
283  int char_count = 0;
284  for (int i = 0; i < count; i++, s += char_size, record_pos += char_size) {
285  if (s >= src_end) {
286  *m_log_stream << "Unexpected end of buffer.\n";
287  *src = src_end;
288  return dest;
289  }
290  if (record_pos > record_sizes[record_index])
291  *m_log_stream << "Warning: record boundary crossed.\n";
292  if (record_pos == record_sizes[record_index]) {
293  if (m_verbose_logging)
294  *m_log_stream << "Record boundary reached.\n";
295  record_index++;
296  record_pos = 0;
297  // At the beginning of each CONTINUE record the option flags byte is repeated.
298  // Only the character size flag will be set in this flags byte, the Rich-Text flag and the Far-East flag are set to zero.
299  // In each CONTINUE record it is possible that the character size changes
300  // from 8‑bit characters to 16‑bit characters and vice versa.
301  if (s >= src_end) {
302  *m_log_stream << "Unexpected end of buffer.\n";
303  *src = src_end;
304  return dest;
305  }
306  if ((*s) != 0 && (*s) != 1)
307  *m_log_stream << "Incorrect XLUnicodeString flag.\n";
308  char_size = ((*s) & 0x01) ? 2 : 1;
309  if (char_size == 2) {
310  s--;
311  record_pos--;
312  }
313  i--;
314  continue;
315  }
316  if (char_size == 2) {
317  if (src_end - *src < 2) {
318  *m_log_stream << "Unexpected end of buffer.\n";
319  *src = src_end;
320  return dest;
321  }
322  unsigned int uc = getU16LittleEndian(s);
323 #warning TODO: Find explanation (documentation) of NULL characters (OO skips them).
324  if (uc == 0)
325  continue;
326  if (utf16_unichar_has_4_bytes(uc)) {
327  record_pos += 2;
328  s += 2;
329  if (src_end - *src < 2) {
330  *m_log_stream << "Unexpected end of buffer.\n";
331  *src = src_end;
332  return dest;
333  }
334  uc = (uc << 16) | getU16LittleEndian(s);
335  }
336  dest += unichar_to_utf8(uc);
337  char_count++;
338  } else {
339  if (s >= src_end) {
340  *m_log_stream << "Unexpected end of buffer.\n";
341  *src = src_end;
342  return dest;
343  }
344  std::string c2(1, *s);
345  if (m_codepage != "ASCII") {
346  TextConverter tc(m_codepage);
347  dest += ustring_to_string(tc.convert(c2));
348  } else
349  dest += c2;
350  char_count++;
351  }
352  }
353  *src = s + after_text_block_len;
354  record_pos += after_text_block_len;
355  return dest;
356  }

358  void parseSharedStringTable(const std::vector<unsigned char> &sst_buf)
359  {
360  if (m_verbose_logging)
361  *m_log_stream << "Parsing shared string table.\n";
362  int sst_size = getS32LittleEndian(sst_buf.begin() + 4);
363  std::vector<unsigned char>::const_iterator src = sst_buf.begin() + 8;
364  size_t record_index = 0;
365  size_t record_pos = 8;
366  while (src < sst_buf.end() && m_shared_string_table.size() <= sst_size)
367  m_shared_string_table.push_back(parseXLUnicodeString(&src, sst_buf.end(), m_shared_string_table_record_sizes, record_index, record_pos));
368  }

370  std::string cellText(int row, int col, const std::string &s)
371  {
372  std::string r;
373  while (row > m_last_row) {
374  r += "\n";
375  ++m_last_row;
376  m_last_col = 0;
377  }
378  if (col > 0 && col <= m_last_col)
379  r += "\t";
380  while (col > m_last_col) {
381  r += "\t";
382  ++m_last_col;
383  }
384  r += s;
385  return r;
386  }

388  bool processRecord(int rec_type, const std::vector<unsigned char> &rec, std::string &text)
389  {
390  if (m_verbose_logging)
391  *m_log_stream << std::hex << "record=0x" << rec_type << std::endl;
392  if (rec_type != XLS_CONTINUE && m_prev_rec_type == XLS_SST)
393  parseSharedStringTable(m_shared_string_table_buf);
394  switch (rec_type) {
395  case XLS_BLANK: {
396  int row = getU16LittleEndian(rec.begin());
397  int col = getU16LittleEndian(rec.begin() + 2);
398  text += cellText(row, col, "");
399  break;
400  }
401  case XLS_BOF: {
402  m_last_row = 0;
403  m_last_col = 0;
404 #warning TODO: Check for stream type, ignore charts, or make it configurable
405 #warning TODO: Mark beginning of sheet (configurable)
406  break;
407  }
408  case XLS_CODEPAGE: {
409  if (rec.size() == 2) {
410  int codepage = getU16LittleEndian(rec.begin());
411  if (codepage == 1200)
412  break;
413  else if (codepage == 367)
414  m_codepage = "ASCII";
415  else
416  m_codepage = "cp" + int2string(codepage);
417  }
418  break;
419  }
420  case XLS_CONTINUE: {
421  if (m_prev_rec_type != XLS_SST)
422  return true; // do not change m_prev_rec_type
423  m_shared_string_table_buf.reserve(m_shared_string_table_buf.size() + rec.size());
424  m_shared_string_table_buf.insert(m_shared_string_table_buf.end(), rec.begin(), rec.begin() + rec.size());
425  m_shared_string_table_record_sizes.push_back(rec.size());
426  if (m_verbose_logging)
427  *m_log_stream << "XLS_CONTINUE record for XLS_SST found. Index: " << m_shared_string_table_record_sizes.size() - 1 << ", size:" << rec.size() << ".\n";
428  return true;
429  }
430  case XLS_DATE_1904:
431  m_date_shift = 24107.0;
432  break;
433  case XLS_EOF: {
434  text += "\n";
435 #warning TODO: Mark end of sheet (configurable)
436  break;
437  }
438  case XLS_FILEPASS: {
439  *m_log_stream << "XLS file is encrypted.\n";
440  U16 encryption_type = getU16LittleEndian(rec.begin());
441  if (encryption_type == 0x0000)
442  *m_log_stream << "XOR obfuscation encryption type detected.\n";
443  else if (encryption_type == 0x0001) {
444  *m_log_stream << "RC4 encryption type detected.\n";
445  U16 header_type = getU16LittleEndian(rec.begin() + 2);
446  if (header_type == 0x0001)
447  *m_log_stream << "RC4 encryption header found.\n";
448  else if (header_type == 0x0002 || header_type == 0x0003)
449  *m_log_stream << "RC4 CryptoAPI encryption header found.\n";
450  else
451  *m_log_stream << "Unknown RC4 encryption header.\n";
452  } else
453  *m_log_stream << "Unknown encryption type.\n";
454  return false;
455  }
456  case XLS_FORMAT: {
457  int num_format_id = getU16LittleEndian(rec.begin());
458  m_defined_num_format_ids.insert(num_format_id);
459  break;
460  }
461  case XLS_FORMULA: {
462  m_last_string_formula_row = -1;
463  int row = getU16LittleEndian(rec.begin());
464  int col = getU16LittleEndian(rec.begin() + 2);
465  if (((unsigned char)rec[12] == 0xFF) && (unsigned char)rec[13] == 0xFF) {
466  if (rec[6] == 0) {
467  m_last_string_formula_row = row;
468  m_last_string_formula_col = col;
469  } else if (rec[6] == 1) {
470 #warning TODO: check and test boolean formulas
471  text += (rec[8] ? "TRUE" : "FALSE");
472  } else if (rec[6] == 2)
473  text += "ERROR";
474  } else {
475  int xf_index = getU16LittleEndian(rec.begin() + 4);
476  text += cellText(row, col, parseXNum(rec.begin() + 6, xf_index));
477  }
478  break;
479  }
480  case XLS_INTEGER_CELL: {
481  int row = getU16LittleEndian(rec.begin());
482  int col = getU16LittleEndian(rec.begin() + 2);
483  text += cellText(row, col, int2string(getU16LittleEndian(rec.begin() + 7)));
484  break;
485  }
486  case XLS_RSTRING:
487  case XLS_LABEL: {
488  m_last_string_formula_row = -1;
489  int row = getU16LittleEndian(rec.begin());
490  int col = getU16LittleEndian(rec.begin() + 2);
491  std::vector<unsigned char>::const_iterator src = rec.begin() + 6;
492  std::vector<size_t> sizes;
493  sizes.push_back(rec.size() - 6);
494  size_t record_index = 0;
495  size_t record_pos = 0;
496  text += cellText(row, col, parseXLUnicodeString(&src, rec.end(), sizes, record_index, record_pos));
497  break;
498  }
499  case XLS_LABEL_SST: {
500  m_last_string_formula_row = -1;
501  int row = getU16LittleEndian(rec.begin());
502  int col = getU16LittleEndian(rec.begin() + 2);
503  int sst_index = getU16LittleEndian(rec.begin() + 6);
504  if (sst_index >= m_shared_string_table.size() || sst_index < 0) {
505  *m_log_stream << "Incorrect SST index.\n";
506  return false;
507  } else
508  text += cellText(row, col, m_shared_string_table[sst_index]);
509  break;
510  }
511  case XLS_MULBLANK: {
512  int row = getU16LittleEndian(rec.begin());
513  int start_col = getU16LittleEndian(rec.begin() + 2);
514  int end_col = getU16LittleEndian(rec.begin() + rec.size() - 2);
515  for (int c = start_col; c <= end_col; c++)
516  text += cellText(row, c, "");
517  break;
518  }
519  case XLS_MULRK: {
520  m_last_string_formula_row = -1;
521  int row = getU16LittleEndian(rec.begin());
522  int start_col = getU16LittleEndian(rec.begin() + 2);
523  int end_col = getU16LittleEndian(rec.begin() + rec.size() - 2);
524  for (int offset = 4, col = start_col; col <= end_col; offset += 6, col++) {
525  int xf_index = getU16LittleEndian(rec.begin() + offset);
526  text += cellText(row, col, parseRkRec(rec.begin() + offset + 2, xf_index));
527  }
528  break;
529  }
530  case XLS_NUMBER:
531  case 0x03:
532  case 0x103:
533  case 0x303: {
534  m_last_string_formula_row = -1;
535  int row = getU16LittleEndian(rec.begin());
536  int col = getU16LittleEndian(rec.begin() + 2);
537  text += cellText(row, col, parseXNum(rec.begin() + 6, getU16LittleEndian(rec.begin() + 4)));
538  break;
539  }
540  case XLS_RK: {
541  m_last_string_formula_row = -1;
542  int row = getU16LittleEndian(rec.begin());
543  int col = getU16LittleEndian(rec.begin() + 2);
544  int xf_index = getU16LittleEndian(rec.begin() + 4);
545  text += cellText(row, col, parseRkRec(rec.begin() + 6, xf_index));
546  break;
547  }
548  case XLS_SST: {
549  m_shared_string_table_buf.clear();
550  m_shared_string_table_record_sizes.clear();
551  m_shared_string_table.clear();
552  m_shared_string_table_buf.reserve(rec.size());
553  m_shared_string_table_buf.insert(m_shared_string_table_buf.end(), rec.begin(), rec.begin() + rec.size());
554  m_shared_string_table_record_sizes.push_back(rec.size());
555  break;
556  }
557  case XLS_STRING: {
558  std::vector<unsigned char>::const_iterator src = rec.begin();
559  if (m_last_string_formula_row < 0) {
560  *m_log_stream << "String record without preceeding string formula.\n";
561  break;
562  }
563  std::vector<size_t> sizes;
564  sizes.push_back(rec.size());
565  size_t record_index = 0;
566  size_t record_pos = 0;
567  text += cellText(m_last_string_formula_row, m_last_string_formula_col, parseXLUnicodeString(&src, rec.end(), sizes, record_index, record_pos));
568  break;
569  }
570  case XLS_XF:
571  case 0x43: {
572  XFRecord xf_record;
573  xf_record.num_format_id = getU16LittleEndian(rec.begin() + 2);
574  m_xf_records.push_back(xf_record);

582  void parseXLS(ThreadSafeOLEStreamReader &reader, std::string &text)
583  {
584  m_xf_records.clear();
585  m_date_shift = 25569.0;
586  m_shared_string_table.clear();
587  m_shared_string_table_buf.clear();
588  m_shared_string_table_record_sizes.clear();
589  m_prev_rec_type = 0;
590  int m_last_string_formula_row = -1;
591  int m_last_string_formula_col = -1;
592  m_defined_num_format_ids.clear();
593  m_last_row = 0;
594  m_last_col = 0;
595 
596  std::vector<unsigned char> rec;
597  bool read_status = true;
598  while (read_status) {
599  if (oleEof(reader)) {
600  *m_log_stream << "No BOF record found.\n";
601  return;
602  }
603  U16 rec_type, rec_len;
604  if (!reader.readU16(rec_type) || !reader.readU16(rec_len)) {
605  *m_log_stream << reader.getLastError() << "\n";
606  return;
607  }
608  enum BofRecordTypes {
609  BOF_BIFF_2 = 0x009,
610  BOF_BIFF_3 = 0x209,
611  BOF_BIFF_4 = 0x0409,
612  BOF_BIFF_5_AND_8 = XLS_BOF
613  };
614  if (rec_type == BOF_BIFF_2 || rec_type == BOF_BIFF_3 || rec_type == BOF_BIFF_4 || rec_type == BOF_BIFF_5_AND_8) {
615  int bof_struct_size;
616 #warning Has all BOF records size 8 or 16?
617  if (rec_len == 8 || rec_len == 16) {
618  switch (rec_type) {
619  case BOF_BIFF_5_AND_8: {
620  U16 biff_ver, data_type;
621  if (!reader.readU16(biff_ver) || !reader.readU16(data_type)) {
622  *m_log_stream << reader.getLastError() << "\n";
623  return;
624  }
625  //On microsoft site there is documentation only for "BIFF8". Documentation from OpenOffice is better:
626  /*
627  BIFF5:
628  Offset Size Contents
629  0 2 BIFF version (always 0500H for BIFF5). ...
630  2 2 Type of the following data: 0005H = Workbook globals
631  0006H = Visual Basic module
632  0010H = Sheet or dialogue
633  0020H = Chart
634  0040H = Macro sheet
635  0100H = Workspace (BIFF5W only)
636  4 2 Build identifier, must not be 0
637  6 2 Build year
638  */
639  /*
640  BIFF8:
641  Offset Size Contents
642  0 2 BIFF version (always 0600H for BIFF8)
643  2 2 Type of the following data: 0005H = Workbook globals
644  0006H = Visual Basic module
645  0010H = Sheet or dialogue
646  0020H = Chart
647  0040H = Macro sheet
648  0100H = Workspace (BIFF8W only)
649  4 2 Build identifier, must not be 0
650  6 2 Build year, must not be 0
651  8 4 File history flags
652  12 4 Lowest Excel version that can read all records in this file
653  */
654  if (biff_ver == 0x600) {
655  if (m_verbose_logging)
656  *m_log_stream << "Detected BIFF8 version\n";
657  rec.resize(8);
658  read_status = reader.read(&*rec.begin(), 8);
659  m_biff_version = BIFF8;
660  bof_struct_size = 16;
661  } else {
662  if (m_verbose_logging)
663  *m_log_stream << "Detected BIFF5 version\n";
664  m_biff_version = BIFF5;
665  bof_struct_size = 8;
666  }
667  break;
668  }
669  case BOF_BIFF_3:
670  if (m_verbose_logging)
671  *m_log_stream << "Detected BIFF3 version\n";
672  m_biff_version = BIFF3;
673  bof_struct_size = 6;
674  break;
675  case BOF_BIFF_4:
676  if (m_verbose_logging)
677  *m_log_stream << "Detected BIFF4 version\n";
678  m_biff_version = BIFF4;
679  bof_struct_size = 6;
680  break;
681  default:
682  if (m_verbose_logging)
683  *m_log_stream << "Detected BIFF2 version\n";
684  m_biff_version = BIFF2;
685  bof_struct_size = 4;
686  }
687  rec.resize(rec_len - (bof_struct_size - 4));
688  read_status = reader.read(&*rec.begin(), rec_len - (bof_struct_size - 4));
689  break;
690  } else {
691  *m_log_stream << "Invalid BOF record.\n";
692  return;
693  }
694  } else {
695  rec.resize(126);
696  read_status = reader.read(&*rec.begin(), 126);
697  }
698  }
699  if (oleEof(reader)) {
700  *m_log_stream << "No BOF record found.\n";
701  return;
702  }
703  bool eof_rec_found = false;
704  while (read_status) {
705  U16 rec_type;
706  if (!reader.readU16(rec_type)) {
707  *m_log_stream << reader.getLastError() << "\n";
708  return;
709  }
710  if (oleEof(reader)) {
711  processRecord(XLS_EOF, std::vector<unsigned char>(), text);
712  return;
713  }
714  U16 rec_len;
715  if (!reader.readU16(rec_len)) {
716  *m_log_stream << reader.getLastError() << "\n";
717  return;
718  }
719  if (rec_len > 0) {
720  rec.resize(rec_len);
721  read_status = reader.read(&*rec.begin(), rec_len);
722  } else
723  rec.clear();
724  if (eof_rec_found) {
725  if (rec_type != XLS_BOF)
726  break;
727  }
728  if (!processRecord(rec_type, rec, text))
729  return;
730  if (rec_type == XLS_EOF)
731  eof_rec_found = true;
732  else
733  eof_rec_found = false;
734  }
735  }
