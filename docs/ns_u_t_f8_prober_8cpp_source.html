40 void nsUTF8Prober::Reset(void)
41 {
42  mCodingSM->Reset();
43  mNumOfMBChar = 0;
44  mState = eDetecting;
45 }
46 
47 nsProbingState nsUTF8Prober::HandleData(const char* aBuf, PRUint32 aLen)
48 {
49  nsSMState codingState;
50 
51  for (PRUint32 i = 0; i < aLen; i++)
52  {
53  codingState = mCodingSM->NextState(aBuf[i]);
54  if (codingState == eItsMe)
55  {
56  mState = eFoundIt;
57  break;
58  }
59  if (codingState == eStart)
60  {
61  if (mCodingSM->GetCurrentCharLen() >= 2)
62  mNumOfMBChar++;
63  }
64  }
65 
66  if (mState == eDetecting)
67  if (GetConfidence() > SHORTCUT_THRESHOLD)
68  mState = eFoundIt;
69  return mState;
70 }
71 
72 #define ONE_CHAR_PROB (float)0.50
73 
74 float nsUTF8Prober::GetConfidence(void)
75 {
76  float unlike = (float)0.99;
77 
78  if (mNumOfMBChar < 6)
79  {
80  for (PRUint32 i = 0; i < mNumOfMBChar; i++)
81  unlike *= ONE_CHAR_PROB;
82  return (float)1.0 - unlike;
83  }
84  else
85  return (float)0.99;
86 }
