299 void SoundWorker::cardsChanged(const QString &cards)
300 {
301  QMap<uint, QStringList> tmpCardIds;
302  QJsonDocument doc = QJsonDocument::fromJson(cards.toUtf8());
303  QJsonArray jCards = doc.array();
304  for (QJsonValue cV : jCards) {
305  QJsonObject jCard = cV.toObject();
306  const uint cardId = static_cast<uint>(jCard["Id"].toInt());
307  const QString cardName = jCard["Name"].toString();
308  QJsonArray jPorts = jCard["Ports"].toArray();
309 
310  QStringList tmpPorts;
311 
312  for (QJsonValue pV : jPorts) {
313  QJsonObject jPort = pV.toObject();
314  const double portAvai = jPort["Available"].toDouble();
315  if (portAvai == 2.0 || portAvai == 0.0) { // 0 Unknown 1 Not available 2 Available
316  const QString portId = jPort["Name"].toString();
317  const QString portName = jPort["Description"].toString();
318 
319  Port *port = m_model->findPort(portId, cardId);
320  const bool include = port != nullptr;
321  if (!include) { port = new Port(m_model); }
322 
323  port->setId(portId);
324  port->setName(portName);
325  port->setDirection(Port::Direction(jPort["Direction"].toDouble()));
326  port->setCardId(cardId);
327  port->setCardName(cardName);
328  port->setIsActive(portId == m_activeSinkPort || portId == m_activeSourcePort);
329 
330  if (!include) { m_model->addPort(port); }
331 
332  tmpPorts << portId;
333  }
334  }
335  if (!jPorts.isEmpty())
336  tmpCardIds.insert(cardId, tmpPorts);
337  }
338 
339  for (Port *port : m_model->ports()) {
340  //if the card is not in the list
341  if (!tmpCardIds.contains(port->cardId())) {
342  m_model->removePort(port->id(), port->cardId());
343  }
344  else if (!tmpCardIds[port->cardId()].contains(port->id())) {
345  m_model->removePort(port->id(), port->cardId());
346  }
347  }
348 }

386 void SoundWorker::getSoundEnabledMapFinished(QDBusPendingCallWatcher *watcher)
387 {
388  if (!watcher->isError()) {
389  QDBusReply<QMap<QString, bool>> value = watcher->reply();
390  auto map = value.value();
391  for (auto it = map.constBegin(); it != map.constEnd(); ++it) {
392  if (!m_model->checkSEExist(it.key())) continue;
393 
394  DDesktopServices::SystemSoundEffect type = m_model->getEffectTypeByGsettingName(it.key());
395  m_model->setEffectData(type, it.value());
396 
397  QDBusPendingCall call = m_soundEffectInter->GetSoundFile(it.key());
398  QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(call, this);
399  watcher->setProperty("Type", type);
400  connect(watcher, &QDBusPendingCallWatcher::finished, this, &SoundWorker::getSoundPathFinished);
401  }
402  } else {
403  qDebug() << "get sound enabled map error." << watcher->error();
404  }
405 
406  watcher->deleteLater();
407 }
