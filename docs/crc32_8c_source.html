115 /* Calculate CRC-32 Checksum for NAACCR Record,
116  skipping area of record containing checksum field.
117 
118  Uses reflected table driven method documented by Ross Williams.
119 
120  PARAMETERS:
121  unsigned char *p NAACCR Record Buffer
122  unsigned long reclen NAACCR Record Length
123  unsigned long checksumpos Position of CHECKSUM (as in Data Dictionary)
124  unsigned long checksumlen Length of checksum Field
125 
126  RETURNS:
127  checksum value
128 
129  Author:
130  Eric Durbin 1998-10-14
131 
132  Status:
133  Public Domain
134 */
135 unsigned long CalcCRC32(unsigned char *p, unsigned long reclen, unsigned long checksumpos, unsigned long checksumlen)
136 {
137  unsigned long j;
138 
139  /* initialize value */
140  unsigned long crc = CRC32_XINIT;
141 
142  /* process each byte prior to checksum field */
143  for (j = 1; j < checksumpos; j++) {
144  crc = crctable[(crc ^ *p++) & 0xFFL] ^ (crc >> 8);
145  }
146 
147  /* skip checksum position */
148  j += checksumlen;
149  p += checksumlen;
150 
151  /* process remaining bytes in record */
152  while (j <= reclen) {
153  crc = crctable[(crc ^ *p++) & 0xFFL] ^ (crc >> 8);
154  j++;
155  }
156 
157  /* return XOR out value */
158  return crc ^ CRC32_XOROT;
159 }

161 /* Assign CRC-32 checksum for NAACCR record passed in buffer p of length
162  reclen with checksum occupying position checksumpos of length checksumlen.
163 
164  PARAMETERS:
165  unsigned char *p NAACCR Record Buffer
166  unsigned long reclen NAACCR Record Length
167  unsigned long checksumpos Position of CHECKSUM (as in Data Dictionary)
168  unsigned long checksumlen Length of checksum Field
169 
170  RETURNS:
171  0 Checksum successfully placed in checksum field
172  -1 Checksum length too short
173  -2 Checksum length too long
174  -3 Checksum position exceeds record length
175 
176  Author:
177  Eric Durbin 1998-10-14
178 
179  Status:
180  Public Domain
181 */
182 int AssignCRC32(unsigned char *p, unsigned long reclen, unsigned long checksumpos, unsigned long checksumlen)
183 {
184  unsigned long CRCValue; /* value of CRC checksum for record */
185  unsigned long Mask = 0x0000000FL; /* Mask for converting hex string */
186  char *HexString ="0123456789ABCDEF"; /* hex string characters */
187  unsigned char *s; /* string pointer */
188 
189  /* validate checksum length */
190  if (checksumlen < MINIMUM_CHECKSUM_LEN)
191  return -1;
192  if (checksumlen > MAXIMUM_CHECKSUM_LEN)
193  return -2;
194  if (checksumpos > reclen)
195  return -3;
196 
197  CRCValue = CalcCRC32(p, reclen, checksumpos, checksumlen);
198 
199  /* convert CRC Value to ASCII Hex String */
200  s = p+checksumpos-1;
201  while (checksumlen--) {
202  *(s+checksumlen) = *(HexString + (CRCValue & Mask));
203  CRCValue >>= 4; /* right shift 4 bits */
204  }
205  return 0;
206 }

208 /* compare CRC-32 checksum in NAACCR record
209 
210  PARAMETERS:
211  unsigned char *p NAACCR Record Buffer
212  unsigned long reclen NAACCR Record Length
213  unsigned long checksumpos Position of CHECKSUM (as in Data Dictionary)
214  unsigned long checksumlen Length of checksum Field
215 
216  RETURNS:
217  0 Checksum matches
218  -1 Checksum mismatch
219 
220  Author:
221  Eric Durbin 1998-10-14
222 
223  Status:
224  Public Domain
225 */
226 int CompareCRC32(unsigned char *p, unsigned long reclen, unsigned long checksumpos, unsigned long checksumlen)
227 {
228  unsigned long RecordCRCValue = 0L; /* initialize record's CRC Value */
229  unsigned long CalcCRCValue;
230  unsigned char *s = p+checksumpos-1;/* incoming checksum string */
231 
232  CalcCRCValue = CalcCRC32(p, reclen, checksumpos, checksumlen);
233 
234  /* extract records hexadecimal checksum value */
235  while (checksumlen--) {
236  RecordCRCValue <<= 4; /* left shift 4 bits */
237  RecordCRCValue += (*s>'9') ? *s++-'A'+10 : *s++-'0';
238  }
239 
240  /* compare values */
241  if (RecordCRCValue != CalcCRCValue)
242  return -1;
243  return 0;
244 }
