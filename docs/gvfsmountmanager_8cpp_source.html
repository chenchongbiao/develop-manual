209 QVolume GvfsMountManager::gVolumeToqVolume(GVolume *volume)
210 {
211  QVolume qVolume;
212  char *name;
213  char **ids;
214  GIcon *icon;
215  GFile *activation_root;
216 
217  name = g_volume_get_name(volume);
218  qVolume.setName(QString(name));
219  g_free(name);
220 
221  ids = g_volume_enumerate_identifiers(volume);
222  if (ids && ids[0] != nullptr) {
223  for (int i = 0; ids[i] != nullptr; i++) {
224  char *id = g_volume_get_identifier(volume, ids[i]);
225  if (QString(ids[i]) == G_VOLUME_IDENTIFIER_KIND_UNIX_DEVICE) {
226  qVolume.setUnix_device(QString(id));
227  } else if (QString(ids[i]) == G_VOLUME_IDENTIFIER_KIND_LABEL) {
228  qVolume.setLable(QString(id));
229  } else if (QString(ids[i]) == G_VOLUME_IDENTIFIER_KIND_UUID) {
230  qVolume.setUuid(QString(id));
231  } else if (QString(ids[i]) == G_VOLUME_IDENTIFIER_KIND_NFS_MOUNT) {
232  qVolume.setNfs_mount(QString(id));
233  }
234 
235  if (qVolume.unix_device().isEmpty()) {
236  qVolume.setUnix_device(qVolume.uuid());
237  }
238 
239  g_free(id);
240  }
241  }
242  g_strfreev(ids);
243 
244  qVolume.setCan_mount(g_volume_can_mount(volume));
245  qVolume.setCan_eject(g_volume_can_eject(volume));
246  qVolume.setShould_automount(g_volume_should_automount(volume));
247 
248 
249  icon = g_volume_get_icon(volume);
250  if (icon) {
251  if (G_IS_THEMED_ICON(icon)) {
252  qVolume.setIcons(getIconNames(G_THEMED_ICON(icon)));
253  }
254  g_object_unref(icon);

256 
257  icon = g_volume_get_symbolic_icon(volume);

259  if (G_IS_THEMED_ICON(icon)) {
260  qVolume.setSymbolic_icons(getIconNames(G_THEMED_ICON(icon)));
261  }
262  g_object_unref(icon);
263  }
264 
265  GMount *mount = g_volume_get_mount(volume);
266  if (mount) {
267  qVolume.setIsMounted(true);
268  GFile *root = g_mount_get_root(mount);
269  char *uri = g_file_get_uri(root);
270  qVolume.setMounted_root_uri(QString(uri));
271 
272  g_object_unref(root);
273  g_free(uri);
274  }
275 
276  activation_root = g_volume_get_activation_root(volume);
277  if (activation_root != nullptr) {
278  char *action_root_uri = g_file_get_uri(activation_root);
279  qVolume.setActivation_root_uri(QString(action_root_uri));
280  g_object_unref(activation_root);
281  g_free(action_root_uri);
282  }
283 
284  GDrive *gDrive = g_volume_get_drive(volume);
285  if (gDrive) {
286  QDrive qDrive = gDriveToqDrive(gDrive);
287  qVolume.setDrive_unix_device(QString(g_drive_get_identifier(gDrive, "unix-device")));
288  qVolume.setDrive(qDrive);
289  }
290 
291  return qVolume;
292 }

541 void GvfsMountManager::monitor_mount_removed(GVolumeMonitor *volume_monitor, GMount *mount)
542 {
543  Q_UNUSED(volume_monitor)
544  qCDebug(mountManager()) << "==============================monitor_mount_removed==============================" ;
545  QMount qMount = gMountToqMount(mount);
546 
547  //fix: 探测光盘推进,弹出和挂载状态机标识
548  if (qMount.name().contains("CD/DVD") || qMount.name().contains("CD") || qMount.icons().contains("media-optical")) { //CD/DVD
549  DFMOpticalMediaWidget::g_mapCdStatusInfo[getVolTag(mount)].bMntFlag = false;
550  }
551 
552  qCDebug(mountManager()) << "===================" << qMount.mounted_root_uri() << "=======================";
553 
554  QVolume volume = getVolumeByMountedRootUri(qMount.mounted_root_uri());
555  qCDebug(mountManager()) << volume.isValid() << volume;
556  if (volume.isValid()) {
557  volume.setIsMounted(false);
558  volume.setMounted_root_uri("");
559  Volumes.insert(volume.unix_device(), volume);
560  } else {
561  NoVolumes_Mounts_Keys.removeOne(qMount.mounted_root_uri());
562  }
563 
564  bool removed = Mounts.remove(qMount.mounted_root_uri());
565  if (removed) {
