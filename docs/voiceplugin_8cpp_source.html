88  QJsonParseError parseJsonErr;
89  QJsonDocument document = QJsonDocument::fromJson(jsonStr.toUtf8(), &parseJsonErr);
90  if (!(parseJsonErr.error == QJsonParseError::NoError)) {
91  qDebug() << tr("解析json文件错误！");
92  qDebug() << __FUNCTION__ << "---1---" << strlist;
93  return strlist;
94  }
95  QJsonObject jsonObject = document.object();
96  // 找到semantic
97  QString function;
98  QString artist;
99  QString song;
100  QString songList;
101  QString source;
102  QString sourceType;
103  QString insType;
104  QString genre;
105  QString number;
106  // 不包含semantic，直接返回空
107  if (!jsonObject.contains(QStringLiteral("intent"))) {
108  return strlist;
109  }
110  QJsonValue intentValue = jsonObject.value(QStringLiteral("intent"));
111  QJsonObject intentObject = intentValue.toObject();
112  // 不包含semantic，直接返回空
113  if (!intentObject.contains(QStringLiteral("semantic"))) {
114  return strlist;
115  }
116  QJsonValue semanticValue = intentObject.value(QStringLiteral("semantic"));
117  // semantic是一个集合,不是集合直接返回空
118  if (!semanticValue.isArray()) {
119  return strlist;
120  }
121  QJsonArray semanticArray = semanticValue.toArray();
122  qDebug() << __FUNCTION__ << "semanticArray.size() = " << semanticArray.size();
123  for (int i = 0; i < semanticArray.size(); i++) {
124  QJsonValue commandValue = semanticArray.at(i);
125  QJsonObject commandObject = commandValue.toObject();
126  // 遍历找到集合中的intent并获取内容
127  if (commandObject.contains(QStringLiteral("intent"))) {
128  function = commandObject["intent"].toString();
129  }
130  if (commandObject.contains(QStringLiteral("slots"))) {
131  QJsonValue slotsValue = commandObject["slots"];
132  QJsonObject slotsObject = slotsValue.toObject();
133  // 找到集合slots,不是集合继续执行
134  if (!slotsValue.isArray()) {
135  continue;
136  }
137  QJsonArray slotsArray = slotsValue.toArray();
138  // 遍历集合中的内容
139  qDebug() << __FUNCTION__ << "slotsArray.size() = " << slotsArray.size();
140  for (int j = 0; j < slotsArray.size(); j++) {
141  QJsonValue slotsArrayValue = slotsArray.at(j);
142  QJsonObject slotsArrayObject = slotsArrayValue.toObject();
143  if (!slotsArrayObject.contains(QStringLiteral("name"))) {
144  continue;
145  }
146  qDebug() << __FUNCTION__ << "name = " << slotsArrayObject["name"].toString();
147  if (slotsArrayObject["name"].toString() == "artist") {
148  if (slotsArrayObject.contains(QStringLiteral("value"))) {
149  artist = slotsArrayObject["value"].toString();
150  }
151  }
152  if (slotsArrayObject["name"].toString() == "song") {
153  if (slotsArrayObject.contains(QStringLiteral("value"))) {
154  song = slotsArrayObject["value"].toString();
155  }
156  }
157  if (slotsArrayObject["name"].toString() == "source") {
158  if (slotsArrayObject.contains(QStringLiteral("value"))) {
159  source = slotsArrayObject["value"].toString();
160  }
161  }
162  if (slotsArrayObject["name"].toString() == "sourceType") {
163  if (slotsArrayObject.contains(QStringLiteral("value"))) {
164  sourceType = slotsArrayObject["value"].toString();
165  }
166  }
167  if (slotsArrayObject["name"].toString() == "insType") {
168  if (slotsArrayObject.contains(QStringLiteral("value"))) {
169  insType = slotsArrayObject["value"].toString();
170  }
171  }
172  if (slotsArrayObject["name"].toString() == "number") {
173  if (slotsArrayObject.contains(QStringLiteral("value"))) {
174  number = slotsArrayObject["value"].toString();
175  }
176  }
177  if (slotsArrayObject["name"].toString() == "songlist") {
178  if (slotsArrayObject.contains(QStringLiteral("value"))) {
179  songList = slotsArrayObject["value"].toString();

191  if (function == "PLAY") {
