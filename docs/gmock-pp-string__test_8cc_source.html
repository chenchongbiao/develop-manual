42 // Matcher to verify that to strings are identical up to whitespace
43 // Not 100% correct, because it treats "AB" as equal to "A B".
44 ::testing::Matcher<const std::string &> SameExceptSpaces(const std::string &s)
45 {
46  auto remove_spaces = [](std::string to_split) {
47  to_split.erase(std::remove(to_split.begin(), to_split.end(), ' '),
48  to_split.end());
49  return to_split;
50  };
51  return ::testing::ResultOf(remove_spaces, remove_spaces(s));
52 }

54 // Verify that a macro expands to a given text. Ignores whitespace difference.
55 // In MSVC, GMOCK_PP_STRINGIZE() returns nothing, rather than "". So concatenate
56 // with an empty string.
57 #define EXPECT_EXPANSION(Result, Macro) \
58  EXPECT_THAT("" GMOCK_PP_STRINGIZE(Macro), SameExceptSpaces(Result))
