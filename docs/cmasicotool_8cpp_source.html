67 void CMasicoTool::toolStart(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *pInfo)
68 {
69  Q_UNUSED(event)
70  Q_UNUSED(pInfo)
71 
72  //光标入栈
73  drawApp->saveCursor();
74 
75  //缓存光标
76  _blurCursor = *dApp->overrideCursor();
77 
78  //获取到这次能进行模糊的图元
79  auto selectItems = event->scene()->selectGroup()->items();
80  auto pCurSelectItem = selectItems.count() == 1 ? selectItems.first() : nullptr;
81  _blurEnableItems = getBlurEnableItems(pCurSelectItem);
82 
83  if (_blurEnableItems.count() > 0) {
84  _pressedPosBlurEnable = true;
85 
86  //置顶前,收集到可能会被模糊的图片图元数据(置顶后再获取到图元z值的话就不是最原始的z值了)
87  event->scene()->recordItemsInfoToCmd(_blurEnableItems,
88  UndoVar, true);
89 
90  //置顶功能:实现进行模糊时,被模糊的图元置顶
91  //置顶功能Step1.保存场景当前图元的z值
92  saveZ(event->scene());
93 
94  //置顶功能Step2.置顶
95  event->scene()->moveBzItemsLayer(selectItems, EUpLayer, -1);
96 
97  } else {
98  _pressedPosBlurEnable = false;
99  }
100 
101  //点下时进行鼠标样式的初始化:鼠标样式是否设置为模糊样式的条件
102  bool cursorBlurEnable = (pCurSelectItem != nullptr) && (pCurSelectItem->sceneBoundingRect().contains(event->pos()));
103  drawApp->setApplicationCursor(cursorBlurEnable ? _blurCursor : QCursor(Qt::ForbiddenCursor));
104 }

118 void CMasicoTool::toolUpdate(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *pInfo)
119 {
120  Q_UNUSED(pInfo)
121 
122  if (!_pressedPosBlurEnable)
123  return;
124 
125  //0.获取到当前最上层的图元是
126  auto currenItem = event->scene()->topBzItem(event->pos(), true,
127  event->eventType() == CDrawToolEvent::ETouchEvent ? drawApp->touchFeelingEnhanceValue() : 0);
128 
129  bool blurActive = isBlurEnable(currenItem);
130 
131  //1.顶层图元发生改变或没变化的情况处理
132  if (_pLastTopItem != currenItem) {
133  //1.1:那么原来的顶层图元_pLastTopItem如果可以模糊那么需要进行模糊路径的结束
134  if (isBlurEnable(_pLastTopItem)) {
135  _pLastTopItem->blurEnd();
136  }
137  //1.2:新的顶层图元currenItem如果可以模糊那么进行模糊开始
138  if (blurActive) {
139  currenItem->blurBegin(currenItem->mapFromScene(event->pos()));
140  _bluringItemsSet.insert(currenItem);
141  }
142  _pLastTopItem = currenItem;
143  } else {
144  //1.3:顶层图元未发生改变:那么直接进行模糊路径刷新即可
145  if (blurActive) {
146  currenItem->blurUpdate(currenItem->mapFromScene(event->pos()), true);
147  }
148  }
149 
150  //2.更新鼠标样式
151  drawApp->setApplicationCursor(blurActive ? _blurCursor : QCursor(Qt::ForbiddenCursor));
152 }

154 void CMasicoTool::toolFinish(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *pInfo)
155 {
156  Q_UNUSED(event)
157  Q_UNUSED(pInfo)
158  if (_pressedPosBlurEnable) {
159 
160  //置顶功能Step3.还原z值
161  restoreZ();
162 
163  _pressedPosBlurEnable = false;
164 
165  enum {EDoBLur = 1};
166  if (pInfo->_opeTpUpdate == EDoBLur) {
167  for (auto pItem : _bluringItemsSet) {
168 
169  if (pItem->isBlurActived()) {
170  pItem->blurEnd();
171  }
172  //pItem->resetCachePixmap();
173 
174  event->scene()->recordItemsInfoToCmd(QList<CGraphicsItem *>() << pItem,
175  RedoVar, false);
176  }
177  event->scene()->finishRecord(false);
178 
179  _bluringItemsSet.clear();
180  } else {
181  //没有移动那么就直接清理命令
182  CUndoRedoCommand::clearCommand();
183  }
184  }
185 
186  _pLastTopItem = nullptr;
187 
188  //恢复光标
189  drawApp->restoreCursor();
190 }

205 bool CMasicoTool::isEnable(CGraphicsView *pView)
206 {
207  if (pView == nullptr)
208  return false;
209 
210  auto items = pView->drawScene()->selectGroup()->items();
211  bool isBlur = false;
212  if (items.count() == 1) {
213  CGraphicsItem *pItem = items[0];
214 
215  if (pItem->isBzGroup()) {
216  QList<CGraphicsItem *> lists = static_cast<CGraphicsItemGroup *>(pItem)->getBzItems(true);
217  foreach (CGraphicsItem *p, lists) {
218  if (p->isBlurEnable()) {
219  isBlur = true;
220  break;
221  }
222  }
223 
224  } else {
225  if (pItem->isBlurEnable())
226  isBlur = true;
227  }
228  }
229  return isBlur;
230 }

237 QList<CGraphicsItem *> CMasicoTool::getBlurEnableItems(const CGraphicsItem *pItem)
238 {
239  QList<CGraphicsItem *> resultItems;
240  if (pItem == nullptr)
241  return QList<CGraphicsItem *>();
242 
243  if (pItem->isBzGroup()) {
244  auto items = static_cast<const CGraphicsItemGroup *>(pItem)->items(true);
245  for (auto p : items) {
246  if (/*p->type() == PictureType*/p->isBlurEnable()) {
247  resultItems.append(const_cast<CGraphicsItem *>(p));
248  }
249  }
250  } else {
251  if (/*pItem->type() == PictureType*/pItem->isBlurEnable()) {
252  resultItems.append(const_cast<CGraphicsItem *>(pItem));
253  }
254  }
255  return resultItems;
256 }
