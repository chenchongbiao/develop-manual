43 void aes_128_encrypt(const byte key[], const byte iv[], const secure_string &ptext, secure_string &ctext)
44 {
45  EVP_CIPHER_CTX_free_ptr ctx(EVP_CIPHER_CTX_new(), ::EVP_CIPHER_CTX_free);
46  int rc = EVP_EncryptInit_ex(ctx.get(), EVP_aes_128_cfb(), NULL, key, iv);
47  if (rc != 1) {
48  throw std::runtime_error("EVP_EncryptInit_ex failed");
49  }
50 
51  // Recovered text expands upto BLOCK_SIZE
52  ctext.resize(ptext.size() + AES_128_BLOCK_SIZE);
53  int out_len1 = (int)ctext.size();
54 
55  rc = EVP_EncryptUpdate(ctx.get(), (byte*)&ctext[0], &out_len1, (const byte*)&ptext[0], (int)ptext.size());
56  if (rc != 1)
57  throw std::runtime_error("EVP_EncryptUpdate failed");
58 
59  int out_len2 = (int)ctext.size() - out_len1;
60  rc = EVP_EncryptFinal_ex(ctx.get(), (byte*)&ctext[0]+out_len1, &out_len2);
61  if (rc != 1)
62  throw std::runtime_error("EVP_EncryptFinal_ex failed");
63 
64  // Set cipher text size now that we know it
65  ctext.resize(out_len1 + out_len2);
66 }

68 void aes_128_decrypt(const byte key[], const byte iv[], const secure_string &ctext, secure_string &rtext)
69 {
70  EVP_CIPHER_CTX_free_ptr ctx(EVP_CIPHER_CTX_new(), ::EVP_CIPHER_CTX_free);
71  int rc = EVP_DecryptInit_ex(ctx.get(), EVP_aes_128_cfb(), NULL, key, iv);
72  if (rc != 1) {
73  throw std::runtime_error("EVP_DecryptInit_ex failed");
74  }
75 
76  // Recovered text contracts upto BLOCK_SIZE
77  rtext.resize(ctext.size());
78  int out_len1 = (int)rtext.size();
79 
80  rc = EVP_DecryptUpdate(ctx.get(), (byte*)&rtext[0], &out_len1, (const byte*)&ctext[0], (int)ctext.size());
81  if (rc != 1) {
82  throw std::runtime_error("EVP_DecryptUpdate failed");
83  }
84 
85  int out_len2 = (int)rtext.size() - out_len1;
86  rc = EVP_DecryptFinal_ex(ctx.get(), (byte*)&rtext[0]+out_len1, &out_len2);
87  if (rc != 1) {
88  throw std::runtime_error("EVP_DecryptFinal_ex failed");
89  }
90 
91  // Set recovered text size now that we know it
92  rtext.resize(out_len1 + out_len2);
93 }

95 secure_string bin_to_hex(const CryptoUtils::byte *bin, const int length)
96 {
97  secure_string str;
98  char twoChar[3];
99  for (int i = 0; i < length; i++) {
100  snprintf(twoChar, 3, "%02X", bin[i]);
101  str += twoChar;
102  }
103  return str;
104 }

106 void hex_to_bin(const secure_string &hexStr, byte *outputBuffer, const size_t length)
107 {
108  const char * hex = hexStr.c_str();
109  memset(outputBuffer, 0, length);
110  for (unsigned int i = 0; i < length; i++) {
111  int d;
112  sscanf(hex + i * 2, "%02X", &d);
113  outputBuffer[i] = (byte)d;
114  }
115 }
