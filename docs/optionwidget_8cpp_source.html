85 void OptionWidget::setItem(const App &item)
86 {
87  m_CategoryItem = item;
88 
89  if (item.Name.isEmpty()) {
90  m_displayName->setText(item.DisplayName);
91  } else {
92  m_displayName->setText(item.Name);
93  }
94 
95  m_userCheck = item.isUser;
96 
97  const QIcon &icon = QIcon::fromTheme(item.Icon, QIcon::fromTheme("application-x-desktop"));
98 
99  const qreal ratio = qApp->devicePixelRatio();
100 
101  QPixmap pixmap = icon.pixmap(16 * ratio, 16 * ratio).scaled(16 * ratio, 16 * ratio,
102  Qt::KeepAspectRatioByExpanding,
103  Qt::FastTransformation);
104 
105  pixmap.setDevicePixelRatio(ratio);
106 
107  m_optionIcon->setPixmap(pixmap);
108 
109  m_execPath->setText(" ("+ item.Exec + ")");
110  setAccessibleName(item.Id);
111 }
