51 void CUndoRedoCommand::finishRecord(bool doRedoCmd)
52 {
53  //所有的CmdInfo组成了这样的一个单元操作
54 
55  if (!s_recordedCmdInfoList.isEmpty()) {
56  CUndoRedoCommandGroup *pGropCmd = new CUndoRedoCommandGroup;
57  for (int i = 0; i < s_recordedCmdInfoList.size();) {
58  SCommandInfoCouple cmdInfo = s_recordedCmdInfoList[i];
59 
60  bool success = pGropCmd->addCommand(cmdInfo);
61 
62  if (success) {
63  ++i;
64  } else {
65  s_recordedCmdInfoList.removeAt(i);
66  }
67  }
68 
69  if (pGropCmd->count() == 0) {
70  delete pGropCmd;
71  pGropCmd = nullptr;
72 
73  qWarning() << "no couple undo redo command!!!!!!!!!!";
74  return;
75  }
76 
77  setBlockRedoWhenPushedToStack(!doRedoCmd);
78  pushStack(pGropCmd);
79  setBlockRedoWhenPushedToStack(false);
80 
81  //执行完后清理操作
82  clearCommand();
83  }
84 }

86 void CUndoRedoCommand::recordUndoCommand(CUndoRedoCommand::EDrawUndoCmdType tp,
87  int expendTp,
88  const QList<QVariant> &datas,
89  bool init,
90  bool NoNeedRedoVars)
91 {
92  if (datas.isEmpty())
93  return;
94 
95  if (tp >= EDrawGroupCmd)
96  return;
97 
98  if (init) {
99  clearCommand();
100  }
101 
102  SCommandInfo undoCmd;
103  undoCmd.tp = tp;
104  undoCmd.expTp = expendTp;
105  undoCmd.urFlag = UndoVar;
106  undoCmd.vars = datas;
107 
108  SCommandInfoCouple cp;
109  cp.undoInfo = undoCmd;
110  cp.noneedRedoVars = NoNeedRedoVars;
111 
112  s_recordedCmdInfoList.append(cp);
113 
114  int index = s_recordedCmdInfoList.size() - 1;
115 
116  s_forFindCoupleMap.insert(CKey(reinterpret_cast<long long>(datas.first().toLongLong()), tp, expendTp), index);
117 }

119 void CUndoRedoCommand::recordRedoCommand(CUndoRedoCommand::EDrawUndoCmdType tp, int expendTp, const QList<QVariant> &datas)
120 {
121  if (datas.isEmpty())
122  return;
123 
124  if (tp >= EDrawGroupCmd)
125  return;
126 
127  CKey key(reinterpret_cast<long long>(datas.first().toLongLong()), tp, expendTp);
128  auto itF = s_forFindCoupleMap.find(key);
129  if (itF != s_forFindCoupleMap.end()) {
130  SCommandInfoCouple &cp = s_recordedCmdInfoList[itF.value()];
131 
132  //类型必须一致
133  if (cp.undoInfo.tp == tp && cp.undoInfo.urFlag == UndoVar) {
134  SCommandInfo reDocmd;
135  reDocmd.tp = tp;
136  reDocmd.expTp = expendTp;
137  reDocmd.urFlag = RedoVar;
138  reDocmd.vars = datas;
139  cp.redoInfo = reDocmd;
140  return;
141  }
142  }
143  qWarning() << "undo redo command not match to a pair !!!!";
144 }

265 void CUndoRedoCommandGroup::noticeUser()
266 {
267  if (_noticeOnfinished) {
268  QList<CGraphicsItem *> bzItems;
269  CDrawScene *pScene = nullptr;
270  for (CUndoRedoCommand *pCmd : _allCmds) {
271  CItemUndoRedoCommand *pItemCmd = dynamic_cast<CItemUndoRedoCommand *>(pCmd);
272  if (pItemCmd != nullptr) {
273  QGraphicsItem *pItem = pItemCmd->item();
274  if (pScene == nullptr) {
275  pScene = qobject_cast<CDrawScene *>(pItem->scene());
276  }
277  CGraphicsItem *pBzItem = dynamic_cast<CGraphicsItem *>(pItem);
278  if (pBzItem != nullptr) {
279  bzItems.append(pBzItem);
280  }
281  } else {
282  CSceneUndoRedoCommand *pSceneCmd = dynamic_cast<CSceneUndoRedoCommand *>(pCmd);
283  if (pScene == nullptr) {
284  pScene = qobject_cast<CDrawScene *>(pSceneCmd->scene());
285  }
286 
287  bool finished = false;
288  switch (pSceneCmd->tp()) {
289  case CSceneUndoRedoCommand::EItemAdded:
290  case CSceneUndoRedoCommand::EItemRemoved: {
291  CSceneItemNumChangedCommand *pNumCmd = dynamic_cast<CSceneItemNumChangedCommand *>(pCmd);
292  if (pNumCmd != nullptr) {
293  QList<CGraphicsItem *> &items = pNumCmd->items();
294  for (CGraphicsItem *pItem : items) {
295  CGraphicsItem *pBzItem = dynamic_cast<CGraphicsItem *>(pItem);
296  if (pBzItem != nullptr && pBzItem->scene() != nullptr) {
297  bzItems.append(pBzItem);
298  }
299  }
300  }
301  break;
302  }
303  case CSceneUndoRedoCommand::EGroupChanged: {
304  bzItems.clear();
305  auto pSnGpCMd = static_cast<CSceneGroupChangedCommand *>(pSceneCmd);
306  qDebug() << "pSnGpCMd->realChangedItems() count = " << pSnGpCMd->realChangedItems().count();
307  bzItems.append(pSnGpCMd->realChangedItems());
308  finished = true;
309  break;
310  }
311  default:
312  break;
313  }
314  if (finished)
315  break;
316  }
317  }
318 
319  if (pScene != nullptr) {
320  pScene->clearSelectGroup();
321  for (CGraphicsItem *pBzItem : bzItems) {
322  pScene->selectItem(pBzItem->thisBzProxyItem(true));
323  }
324 
325 
326  //如果处于非选择工具下(比如模糊工具下),那么需要判断当前的情况是否支持保持当前工具
327  auto view = CManageViewSigleton::GetInstance()->getCurView();
328  auto toolModle = CManageViewSigleton::GetInstance()->getCurScene()->getDrawParam()->getCurrentDrawToolMode();
329  if (toolModle != selection) {
330  if (!drawApp->isViewToolEnable(view, toolModle)) {
331  drawApp->setViewCurrentTool(view, selection);
332  } else {
333  drawApp->setViewCurrentTool(view, toolModle);
334  }
335  }
336  pScene->refreshLook();
337  }
338  }
339 }

651 CCmdBlock::CCmdBlock(CDrawScene *pScene, CSceneUndoRedoCommand::EChangedType EchangedTp,
652  const QList<QGraphicsItem *> list, bool doRedo)
653  : _doRedo(doRedo), _pScene(pScene), _scenChangedType(EchangedTp)
654 {
655  if (_pScene == nullptr)
656  return;
657 
658  //是否需要还原信息(删除图元其实不需要)
659  bool needRedoInfo = true;
660  //记录undo
661  QList<QVariant> vars;
662  vars << reinterpret_cast<long long>(pScene);
663  if (EchangedTp == CSceneUndoRedoCommand::ESizeChanged) {
664  vars << pScene->sceneRect();
665  } else if (EchangedTp == CSceneUndoRedoCommand::EItemAdded || EchangedTp == CSceneUndoRedoCommand::EItemRemoved) {
666  for (QGraphicsItem *pItem : list) {
667  vars << reinterpret_cast<long long>(pItem);
668  }
669  needRedoInfo = false;
670  } else if (EchangedTp == CSceneUndoRedoCommand::EGroupChanged) {
671  QVariant var;
672  var.setValue(_pScene->getGroupTree());
673  vars << var;
674 
675  //真正的变动图元,可以用于撤销还原后的选中操作
676  for (QGraphicsItem *pItem : list) {
677  vars << reinterpret_cast<long long>(pItem);
678  }
679  }
680  CUndoRedoCommand::recordUndoCommand(CUndoRedoCommand::ESceneChangedCmd,
681  EchangedTp, vars, needRedoInfo);
682 }
