155 void AccountsWidget::addUser(User *user, bool t1)
156 {
157  //active
158  m_userList << user;
159  DStandardItem *item = new DStandardItem;
160  item->setData(0, AccountsWidget::ItemDataRole);
161 
162  /* 用户列表显示用户类型 */
163  auto *subTitleAction = new DViewItemAction;
164  if (user->userType() == User::UserType::Administrator) {
165  subTitleAction->setText(tr("Administrator"));
166  } else {
167  subTitleAction->setText(tr("Standard User"));
168  }
169  subTitleAction->setFontSize(DFontSizeManager::T8);
170  subTitleAction->setTextColorRole(DPalette::TextTips);
171  item->setTextActionList({subTitleAction});
172 
173  connect(user, &User::userTypeChanged, this, [ = ](int userType) {
174  if (userType == User::UserType::Administrator) {
175  subTitleAction->setText(tr("Administrator"));
176  } else {
177  subTitleAction->setText(tr("Standard User"));
178  }
179  });
180 
181  DViewItemAction *onlineFlag = new DViewItemAction(Qt::AlignCenter | Qt::AlignRight, QSize(), QSize(), true);
182  OnlineIcon *onlineIcon = new OnlineIcon(m_userlistView->viewport());
183  onlineIcon->setFixedSize(12, 12);
184  onlineFlag->setWidget(onlineIcon);
185  item->setActionList(Qt::Edge::RightEdge, {onlineFlag});
186  onlineFlag->setVisible(user->online());
187  if (onlineFlag->widget()) {
188  onlineFlag->widget()->setVisible(onlineFlag->isVisible());
189  }
190  connect(user, &User::onlineChanged, this, [=](const bool &online) {
191  onlineFlag->setVisible(online);
192  if (onlineFlag->widget()) {
193  onlineFlag->widget()->setVisible(onlineFlag->isVisible());
194  }
195  });
196 
197  m_userItemModel->appendRow(item);
198  connectUserWithItem(user);
199 
200  connect(user, &User::isCurrentUserChanged, this, [ = ](bool isCurrentUser) {
201  if (isCurrentUser) {
202  auto tindex = m_userList.indexOf(user);
203  auto titem = m_userItemModel->takeRow(tindex);
204 
205  m_userItemModel->insertRow(0, titem);
206 
207  m_userList.removeOne(user);
208  m_userList.push_front(user);
209 
210  showDefaultAccountInfo();
211  }
212  });
213 
214  if (t1)
215  return;
216 
217  auto path = user->currentAvatar();
218  path = path.startsWith("file://") ? QUrl(path).toLocalFile() : path;
219  QPixmap pixmap = pixmapToRound(path);
220 
221  item->setIcon(QIcon(pixmap));
222  item->setText(user->displayName());
223 
224  if (user->isCurrentUser()) {
225  //如果是当前用户
226  auto tttitem = m_userItemModel->takeRow(m_userItemModel->rowCount() - 1);
227  Q_ASSERT(tttitem[0] == item);
228  m_userItemModel->insertRow(0, item);
229 
230  m_userList.push_front(user);
231  m_userList.pop_back();
232  m_currentUserAdded = true;
233 
234  QTimer::singleShot(0, this, &AccountsWidget::showDefaultAccountInfo);
235  }
236 }

258 void AccountsWidget::connectUserWithItem(User *user)
259 {
260  connect(user, &User::nameChanged, this, [ = ](const QString &) {
261  int tindex = m_userList.indexOf(user);
262  auto titem = m_userItemModel->item(tindex);
263  if (!titem) {
264  return;
265  }
266  titem->setText(user->displayName());
267  });
268  connect(user, &User::fullnameChanged, this, [ = ](const QString &) {
269  int tindex = m_userList.indexOf(user);
270  auto titem = m_userItemModel->item(tindex);
271  if (titem) {
272  titem->setText(user->displayName());
273  }
274  });
275  connect(user, &User::currentAvatarChanged, this, [ = ](const QString & avatar) {
276  int tindex = m_userList.indexOf(user);
277  auto titem = m_userItemModel->item(tindex);
278  if (!titem) {
279  return;
280  }
281  auto ratio = this->devicePixelRatioF();
282  auto path = avatar;
283  if (ratio > 4.0) {
284  path.replace("icons/", "icons/bigger/");
285  }
286 
287  path = path.startsWith("file://") ? QUrl(path).toLocalFile() : path;
288  QPixmap pixmap = pixmapToRound(path);
289 
290  titem->setIcon(QIcon(pixmap));
291  });
292 }

294 QPixmap AccountsWidget::pixmapToRound(const QPixmap &src)
295 {
296  if (src.isNull()) {
297  return QPixmap();
298  }
299 
300  auto pixmap = QPixmap(src);
301  QSize size = pixmap.size();
302  QPixmap mask(size);
303  mask.fill(Qt::transparent);
304 
305  QPainter painter(&mask);
306  painter.setRenderHint(QPainter::Antialiasing);
307 
308  QPainterPath path;
309  path.addEllipse(0, 0, size.width(), size.height());
310  painter.setClipPath(path);
311  painter.drawPixmap(0, 0, pixmap);
312 
313  return mask;
314 }
