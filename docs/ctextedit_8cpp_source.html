176 void CTextEdit::onTextChanged()
177 {
178  if (m_pItem == nullptr)
179  return;
180 
181  // 如果是两点的状态高度需要自适应
182  if (m_pItem->isAutoAdjustSize()) {
183  QSizeF size = this->document()->size();
184  QRectF rect = m_pItem->rect();
185  rect.setHeight(size.height());
186  rect.setWidth(size.width());
187  m_pItem->setRect(rect);
188  } else {
189  //自动换行时,如果当前的文字字体太大那么输入的文字可能显示不到,我们就设置
190  int ftHeight = QFontMetrics(this->currentFont(false)).lineSpacing();
191  if (this->height() < ftHeight) {
192  auto curRect = m_pItem->rect();
193  curRect.setHeight(ftHeight);
194  QMetaObject::invokeMethod(this, [ = ]() {
195  m_pItem->setRect(curRect);
196  }, Qt::QueuedConnection);
197  }
198  }
199 }

269  event->accept();

308 QVector<QTextLayout::FormatRange> CTextEdit::getCharFormats(int posBegin, int posEnd)
309 {
310  QVector<QTextLayout::FormatRange> fmts;
311  int beginPos = posBegin;
312  int endPos = posEnd;
313  auto beginBlk = document()->findBlock(beginPos);
314  auto endBlk = document()->findBlock(endPos).next();
315  for (auto itBk = beginBlk; itBk != endBlk; itBk = itBk.next()) {
316  auto fmtVector = itBk.textFormats();
317  for (auto fmt : fmtVector) {
318  int spanBeginPosInDoc = itBk.position() + fmt.start;
319  int spanEndPosInDoc = spanBeginPosInDoc + fmt.length - 1;
320 
321  if (spanEndPosInDoc < beginPos || spanBeginPosInDoc > endPos) {
322  continue;
323  }
324  //qDebug() << "s b = " << beginPos << "e b = " << endPos << "span b = " << spanBeginPosInDoc << "span e = " << spanEndPosInDoc;
325  int newStart = qMax(spanBeginPosInDoc, beginPos);
326  int newEnd = qMin(spanEndPosInDoc, endPos);
327  fmt.start = newStart;
328  fmt.length = newEnd - newStart + 1;
329 
330  fmts.append(fmt);
331  }
332  }
333  //qDebug() << "fmts count ======= " << fmts.count();
334  return fmts;
335 }

337 void CTextEdit::updateSelectionFormat()
338 {
339  QTextCursor c = textCursor();
340  int beginPos = qMin(c.selectionStart(), c.selectionEnd());
341  int endPos = qMax(c.selectionStart(), c.selectionEnd()) - 1; //光标的首位(pos == 0)位于第一个字符之前,所以其结尾位置需要-1以和字符在doc中的索引保持同步
342  auto fmts = getCharFormats(beginPos, endPos);
343 
344  //格式数据是否有冲突
345  QTextCharFormat fmt = fmts.first().format;
346  for (int i = 1; i < fmts.count(); ++i) {
347  QTextCharFormat ft = fmts.at(i).format;
348  for (int j = 0; j < senseProertiesCount; ++j) {
349  QTextCharFormat::Property property = senseProerties[j];
350  if (fmt.hasProperty(property)) {
351  if (fmt.property(property) != ft.property(property)) {
352  fmt.clearProperty(property);
353  }
354  }
355  }
356  }
357  _selectionFmt = fmt;
358 }

360 int CTextEdit::toWeight(const QString &styleName)
361 {
362  int weight = 0;
363  if (styleName == "Thin") {
364  weight = 0;
365  } else if (styleName == "ExtraLight") {
366  weight = 12;
367  } else if (styleName == "Light") {
368  weight = 25;
369  } else if (styleName == "Normal" || styleName == "Regular") {
370  weight = 50;
371  } else if (styleName == "Medium") {
372  weight = 57;
373  } else if (styleName == "DemiBold") {
374  weight = 63;
375  } else if (styleName == "Bold") {
376  weight = 75;
377  } else if (styleName == "ExtraBold") {
378  weight = 81;
379  } else if (styleName == "Black") {
380  weight = 87;
381  }
382  return weight;
383 }
