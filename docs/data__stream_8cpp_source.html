216 bool BufferStream::read(void *data, int element_size, size_t elements_num)
217 {
218  size_t len = element_size * elements_num;
219  if (len > impl->m_size - impl->m_pointer)
220  return false;
221  memcpy(data, impl->m_buffer + impl->m_pointer, len);
222  impl->m_pointer += len;
223  return true;

226 bool BufferStream::seek(int offset, int whence)
227 {
228  size_t position;
229  switch (whence) {
230  case SEEK_SET:
231  position = offset;
232  break;
233  case SEEK_CUR:
234  position = impl->m_pointer + offset;
235  break;
236  case SEEK_END:
237  position = impl->m_size + offset;
238  break;
239  default:
240  return false;
241  }
242  if (position > impl->m_size)
243  return false;
244  impl->m_pointer = position;
245  return true;
246 }
