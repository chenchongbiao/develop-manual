46 const QPixmap ThemeAppIcon::getIcon(const QString iconName, const int size, const qreal ratio)
47 {
48  QPixmap pixmap;
49  QString key;
50  QIcon icon;
51  // 把size改为小于size的最大偶数 :)
52  const int s = int(size * ratio) & ~1;
53  const float iconZoom = size / 64.0 * 0.8;
54 
55  if (iconName == "dde-calendar") {
56  QDate const date(QDate::currentDate());
57 
58  auto calendar = new QWidget() ;
59  calendar->setFixedSize(s, s);
60 
61  calendar->setAutoFillBackground(true);
62  QPalette palette = calendar->palette();
63  palette.setBrush(QPalette::Window,
64  QBrush(QPixmap(":/indicator/resources/calendar_bg.svg").scaled(
65  calendar->size(),
66  Qt::IgnoreAspectRatio,
67  Qt::SmoothTransformation)));
68  calendar->setPalette(palette);
69 
70  QVBoxLayout *layout = new QVBoxLayout;
71  layout->setSpacing(0);
72  auto month = new QLabel();
73  auto monthPix = ImageUtil::loadSvg(QString(":/icons/resources/month%1.svg").arg(date.month()), QSize(36, 16)*iconZoom, ratio);
74  month->setPixmap(monthPix.scaled(monthPix.width()*ratio,monthPix.height()*ratio, Qt::KeepAspectRatio, Qt::SmoothTransformation));
75  month->setFixedHeight(month->pixmap()->height());
76  month->setAlignment(Qt::AlignCenter);
77  month->setFixedWidth(s - 5 * iconZoom);
78  layout->addWidget(month, Qt::AlignVCenter);
79 
80  auto day = new QLabel();
81  auto dayPix =ImageUtil::loadSvg(QString(":/icons/resources/day%1.svg").arg(date.day()), QSize(32, 30)*iconZoom, ratio);
82  day->setPixmap(dayPix.scaled(dayPix.width()*ratio,dayPix.height()*ratio, Qt::KeepAspectRatio, Qt::SmoothTransformation));
83  day->setAlignment(Qt::AlignCenter);
84  day->setFixedHeight(day->pixmap()->height()/ratio);
85  day->raise();
86  layout->addWidget(day, Qt::AlignVCenter);
87 
88  auto week = new QLabel();
89  auto weekPix = ImageUtil::loadSvg(QString(":/icons/resources/week%1.svg").arg(date.dayOfWeek()), QSize(26, 13)*iconZoom, ratio);
90  week->setPixmap(weekPix.scaled(weekPix.width()*ratio,weekPix.height()*ratio, Qt::KeepAspectRatio, Qt::SmoothTransformation));
91  week->setFixedHeight(week->pixmap()->height());
92  week->setAlignment(Qt::AlignCenter);
93  week->setFixedWidth(s + 5 * iconZoom);
94  layout->addWidget(week, Qt::AlignVCenter);
95  layout->setSpacing(0);
96  layout->setContentsMargins(0, 10 * iconZoom, 0, 10 * iconZoom);
97  calendar->setLayout(layout);
98  pixmap = calendar->grab(calendar->rect());
99  if (pixmap.size().width() != s) {
100  pixmap = pixmap.scaled(s, s, Qt::KeepAspectRatio, Qt::SmoothTransformation);
101  }
102  return pixmap;
103  }
104 
105  do {
106  // load pixmap from our Cache
107  if (iconName.startsWith("data:image/")) {
108  key = QCryptographicHash::hash(iconName.toUtf8(), QCryptographicHash::Md5).toHex();
109 
110  // FIXME(hualet): The cache can reduce memory usage,
111  // that is ~2M on HiDPI enabled machine with 9 icons loaded,
112  // but I don't know why since QIcon has its own cache and all of the
113  // icons loaded are loaded by QIcon::fromTheme, really strange here.
114  if (QPixmapCache::find(key, &pixmap))
115  break;
116  }
117 
118  // load pixmap from Byte-Data
119  if (iconName.startsWith("data:image/")) {
120  const QStringList strs = iconName.split("base64,");
121  if (strs.size() == 2)
122  pixmap.loadFromData(QByteArray::fromBase64(strs.at(1).toLatin1()));
123 
124  if (!pixmap.isNull())
125  break;
126  }
127 
128  // load pixmap from File
129  if (QFile::exists(iconName)) {
130  pixmap = QPixmap(iconName);
131  if (!pixmap.isNull())
132  break;
133  }
134 
135  icon = QIcon::fromTheme(iconName);
136  if (icon.isNull()) {
137  //手动更新图标缓存
138  system("gtk-update-icon-cache /usr/share/icons/hicolor/");
139 
140  icon = QIcon::fromTheme(iconName);
141  }
142 
143  if(icon.isNull()) {
144  icon = QIcon::fromTheme("application-x-desktop");
145  }
146 
147  // load pixmap from Icon-Theme
148  const int fakeSize = std::max(48, s); // cannot use 16x16, cause 16x16 is label icon
149  pixmap = icon.pixmap(QSize(fakeSize, fakeSize));
150  if (!pixmap.isNull())
151  break;
152 
153  // fallback to a Default pixmap
154  pixmap = QPixmap(":/icons/resources/application-x-desktop.svg");
155  if (!pixmap.isNull())
156  break;
157 
158  Q_UNREACHABLE();
159 
160  } while (false);
161 
162  if (!key.isEmpty()) {
163  QPixmapCache::insert(key, pixmap);
164  }
165 
166  if (pixmap.size().width() != s) {
167  pixmap = pixmap.scaled(s, s, Qt::KeepAspectRatio, Qt::SmoothTransformation);
168  }
169  pixmap.setDevicePixelRatio(ratio);
170  return pixmap;
171 }
