34 void FileManagerLibrary::loadLibrary(const QString &fileName)
35 {
36  QLibrary lib(fileName);
37  bool isLoaded = lib.load();
38  if (isLoaded){
39  m_auto_add_rightmenu = (func_auto_add_rightmenu) lib.resolve("auto_add_rightmenu");
40  m_auto_operation = (func_auto_operation) lib.resolve("auto_operation");
41  m_auto_add_emblem = (func_auto_add_emblem) lib.resolve("auto_add_emblem");
42  m_InitLLSInfo = (func_InitLLSInfo) lib.resolve("InitLLSInfo");
43  m_free_memory = (func_free_memory) lib.resolve("free_memory");
44 
45  if (m_auto_add_rightmenu && m_auto_operation && m_auto_add_emblem && m_InitLLSInfo && m_free_memory){
46  m_isCompletion = true;
47  }
48  }else{
49  qDebug() << "load library " << LibraryName() << "failed: " << lib.errorString();
50  }
51 }
