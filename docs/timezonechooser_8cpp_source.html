250 bool TimeZoneChooser::eventFilter(QObject *watched, QEvent *event)
251 {
252  // Qt make popups' position 2px higher than the bottom of its associated widget,
253  // and the 2px's hard coded, so I need to move the popup to a proper position so
254  // it won't overlap with the SearchInput.
255  if (watched == m_popup && event->type() == QEvent::Move) {
256  const QMoveEvent *move = static_cast<QMoveEvent *>(event);
257  const QPoint destPos = m_searchInput->mapToGlobal(QPoint(0, m_searchInput->height() + 1));
258  // Don't panic, it won't cause dead loop.
259  if (move->pos() != destPos) {
260  m_popup->move(destPos);
261  }
262  }
263 
264  return false;
265 }

306 void TimeZoneChooser::setupSize()
307 {
308  static const double MapPixWidth = 978.0;
309  static const double MapPixHeight = 500.0;
310  static const double MapPictureWidth = 978.0;
311  static const double MapPictureHeight = 500.0;
312 
313  QFont font = m_title->font();
314  font.setPointSizeF(getFontSize());
315  m_title->setFont(font);
316  //获取标题部分根据字体大小计算得到的字体高度
317  double fontHeight = m_title->fontMetrics().height() + 10.0;
318 
319  QSize fitSize = getFitSize();
320  //先根据屏幕大小，计算出大小合适的地图尺寸
321  const double offsetW = 20 * 2.0;
322  //close Button height and margin 6.0 * 2 + 20, layout spacing * 4 , title font height, search and btn height 36
323  const double offsetH = 6.0 * 2 + 20 + 10.0 * 4 + fontHeight + 36 * 2;
324 
325  //比对地图和屏幕大小，取其中最小的大小
326  const float mapWidth = qMin(MapPixWidth, fitSize.width() - offsetW);
327  const float mapHeight = qMin(MapPixHeight, fitSize.height() - offsetH);
328 
329  //再计算地图和界面宽高比，取其中最小最大缩放比
330  const double widthScale = MapPictureWidth / mapWidth;
331  const double heightScale = MapPictureHeight / mapHeight;
332  const double scale = qMax(widthScale, heightScale);
333  //根据宽高最大缩放比设置地图大小
334  m_map->setFixedSize(MapPictureWidth / scale, MapPictureHeight / scale);
335 
336  //再根据地图大小设置界面大小，使界面与地图保持合适位置
337  setFixedSize(m_map->size().width() + offsetW, m_map->size().height() + offsetH);
338 }
