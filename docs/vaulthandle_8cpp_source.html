36 CryFsHandle::CryFsHandle(QObject *parent) : QObject(parent)
37 {
38  m_process = new QProcess(this);
39  m_mutex = new QMutex;
40  m_thread = new QThread;

87 int CryFsHandle::runVaultProcess(QString lockBaseDir, QString unlockFileDir, QString passWord)
88 {
89  QString cryfsBinary = QStandardPaths::findExecutable("cryfs");
90  if (cryfsBinary.isEmpty()) return static_cast<int>(ErrorCode::CryfsNotExist);
91 
92  QStringList arguments;
93  arguments << lockBaseDir << unlockFileDir;
94 
95  m_process->setEnvironment({"CRYFS_FRONTEND=noninteractive"});
96  m_process->start(cryfsBinary, arguments);
97  m_process->waitForStarted();
98  m_process->write(passWord.toUtf8());
99  m_process->waitForBytesWritten();
100  m_process->closeWriteChannel();
101  m_process->waitForFinished();
102  m_process->terminate();
103 
104  if(m_process->exitStatus() == QProcess::NormalExit && m_process->exitCode() == 0)
105  return static_cast<int>(ErrorCode::Success);
106  else
107  return m_process->exitCode();
108 
109 }

111 int CryFsHandle::lockVaultProcess(QString unlockFileDir)
112 {
113  QString fusermountBinary = QStandardPaths::findExecutable("fusermount");
114  if (fusermountBinary.isEmpty()) return static_cast<int>(ErrorCode::FusermountNotExist);
115 
116  m_process->start(fusermountBinary, {"-zu", unlockFileDir});
117  m_process->waitForStarted();
118  m_process->waitForFinished();
119  m_process->terminate();
120 
121  if(m_process->exitStatus() == QProcess::NormalExit && m_process->exitCode() == 0) {
122  // 修复bug-52351
123  // 保险箱上锁成功后,删除挂载目录
124  if(rmdir(unlockFileDir.toStdString().c_str()) == -1) {
125  qDebug() << "Vault Info: remove vault unlock dir failure";
126  } else {
127  qDebug() << "Vault Info: remove vault unlock dir success";
128  }
129  return static_cast<int>(ErrorCode::Success);
130  }
131  else
132  return m_process->exitCode();
133 }

135 void CryFsHandle::slotReadError()
136 {
137  QString error = m_process->readAllStandardError().data();
138  if(m_activeState.contains(1))
139  {
140  if(error.contains("mountpoint is not empty"))
141  m_activeState[1] = static_cast<int>(ErrorCode::MountpointNotEmpty);
142  else if(error.contains("Permission denied"))
143  m_activeState[1] = static_cast<int>(ErrorCode::PermissionDenied);
144  }
145  else if(m_activeState.contains(3))
146  {
147  if(error.contains("mountpoint is not empty"))
148  m_activeState[3] = static_cast<int>(ErrorCode::MountpointNotEmpty);
149  else if(error.contains("Permission denied"))
150  m_activeState[3] = static_cast<int>(ErrorCode::PermissionDenied);
151  }
152  else if(m_activeState.contains(7))
153  {
154  if(error.contains("Device or resource busy"))
155  m_activeState[7] = static_cast<int>(ErrorCode::ResourceBusy);
156  }
157  emit signalReadError(error);
158 }
