120 void cleanPath(QString &path)
121 {

124  {
125  path.remove(0, sizeof("/data") - 1);

127 
128  if (path.size() > 1 && path.at(0) == '/' && path.endsWith("/")) {
129  path.chop(1);

196 void PathManager::loadSystemPaths()
197 {
198  m_systemPathsMap["Home"] = DFMStandardPaths::location(DFMStandardPaths::HomePath);
199  m_systemPathsMap["Desktop"] = DFMStandardPaths::location(DFMStandardPaths::DesktopPath);
200  m_systemPathsMap["Videos"] = DFMStandardPaths::location(DFMStandardPaths::VideosPath);
201  m_systemPathsMap["Music"] = DFMStandardPaths::location(DFMStandardPaths::MusicPath);
202  m_systemPathsMap["Pictures"] = DFMStandardPaths::location(DFMStandardPaths::PicturesPath);
203  m_systemPathsMap["Documents"] = DFMStandardPaths::location(DFMStandardPaths::DocumentsPath);
204  m_systemPathsMap["Downloads"] = DFMStandardPaths::location(DFMStandardPaths::DownloadsPath);
205  m_systemPathsMap["Trash"] = DFMStandardPaths::location(DFMStandardPaths::TrashFilesPath);
206  m_systemPathsMap["System Disk"] = DFMStandardPaths::location(DFMStandardPaths::DiskPath);
207  m_systemPathsMap["Network"] = DFMStandardPaths::location(DFMStandardPaths::NetworkRootPath);
208  m_systemPathsMap["UserShare"] = DFMStandardPaths::location(DFMStandardPaths::UserShareRootPath);
209  m_systemPathsMap["Computer"] = DFMStandardPaths::location(DFMStandardPaths::ComputerRootPath);
210  m_systemPathsMap["Recent"] = DFMStandardPaths::location(DFMStandardPaths::RecentPath);
211  m_systemPathsMap["Vault"] = DFMStandardPaths::location(DFMStandardPaths::Vault); // 保险库路径
212 
213  m_systemPathsSet.reserve(m_systemPathsMap.size());
214 
215  foreach (const QString &key, m_systemPathsMap.keys()) {
216  const QString &path = m_systemPathsMap.value(key);
217 
218  if (key != "Trash")
219  m_systemPathsSet << path;
220 
221  if(key == "Desktop" || key == "Videos" || key == "Music" ||
222  key == "Pictures" || key == "Documents" || key == "Downloads" ||
223  key == "Trash"){
224  mkPath(path);
225  }
226  }
227 }

262 bool PathManager::isVisiblePartitionPath(const DAbstractFileInfoPointer &fi)
263 {
264 #ifdef SP3_UNSTABLE_FEATURE_ENABLE
265  QString suffix = fi->suffix();
266  if (DFMGlobal::isRootUser()) { // root 用户不受限
267  return true;
268  } else if (suffix == SUFFIX_USRDIR || suffix == SUFFIX_GVFSMP) { // 系统路径，MTP等不受限
269  return true;
270  } else { // localdisk
271  DUrl url = fi->redirectedFileUrl();
272  DUrl parentUrl = url.parentUrl();
273  const QString &path = url.toLocalFile();
274  const QString &parentPath = parentUrl.toLocalFile();
275 
276  // [0] 光驱
277  if (isOptical(fi))
278  return true;
279 
280  // [1] 当打开文管时挂载U盘，进入到此函数时，很可能还没挂载完成,获取不到path
281  // 为了避免这种情况下无法显示，因此直接返回true
282  // todo：此处可能会被报bug，但目前没有想到更好的方案，待优化
283  if (path.isEmpty())
284  return true;
285 
286  // [2] 排除系统盘 & 数据盘
287  const QStringList &sysRanges = getSystemDiskAndDataDiskPathGroup();
288  if (sysRanges.contains(path))
289  return true;
290 
291  // [3] 挂载只显示挂载到以下挂载点的内容
292  const QStringList &parentRanges = getMountRangePathGroup();
293  if (parentRanges.contains(parentPath))
294  return true;
295  }
296  qDebug() << "partition ignore path:" << fi->redirectedFileUrl().toLocalFile();
297 
298  return false;
299 #else
300  Q_UNUSED(fi)
301  return true;
302 #endif
303 }
