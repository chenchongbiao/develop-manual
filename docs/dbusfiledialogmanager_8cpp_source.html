99  DBusFileDialogHandle *handle = new DBusFileDialogHandle();
100  Q_UNUSED(new FiledialogAdaptor(handle));
101 
102  const QDBusObjectPath path("/com/deepin/filemanager/filedialog/" + key);
103 
104  if (m_dialogObjectMap.contains(path)) {
105  return path;
106  }
107 
108  if (!QDBusConnection::sessionBus().registerObject(path.path(), handle)) {
109  qWarning("Cannot register to the D-Bus object.\n");
110  handle->deleteLater();
111 
112  return QDBusObjectPath();
113  }
114 
115  m_dialogObjectMap[path] = handle;
116  connect(handle, &DBusFileDialogHandle::destroyed, this, &DBusFileDialogManager::onDialogDestroy);
117 
118  return path;
