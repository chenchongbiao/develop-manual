108 void WindowManager::loadWindowState(DFileManagerWindow *window)
109 {
110  const QVariantMap &state = DFMApplication::appObtuselySetting()->value("WindowManager", "WindowState").toMap();
111 
112  int width = state.value("width").toInt();
113  int height = state.value("height").toInt();
114  NetWmStates windowState = static_cast<NetWmStates>(state.value("state").toInt());
115  // fix bug 30932,获取全屏属性，必须是width全屏和height全屏熟悉都满足，才判断是全屏
116  if ((m_windows.size() == 0) && ((windowState & NetWmStateMaximizedHorz) != 0 && (windowState & NetWmStateMaximizedVert) != 0))
117  {
118  window->showMaximized();
119  }
120  else {
121  window->resize(width, height);
122  }
123 }

126 void WindowManager::saveWindowState(DFileManagerWindow *window)
127 {
129  NetWmStates states = static_cast<NetWmStates>(window->window()->windowHandle()->property("_d_netWmStates").toInt());
130  QVariantMap state;
131  // fix bug 30932,获取全屏属性，必须是width全屏和height全屏熟悉都满足，才判断是全屏
132  if ((states & NetWmStateMaximizedHorz) == 0 || (states & NetWmStateMaximizedVert) == 0) {
133  state["width"] = window->size().width();
134  state["height"] = window->size().height();
135  }
136  else {
137  const QVariantMap &state1 = DFMApplication::appObtuselySetting()->value("WindowManager", "WindowState").toMap();
138  state["width"] = state1.value("width").toInt();
139  state["height"] = state1.value("height").toInt();
140  state["state"] = static_cast<int>(states);
141  }
142  DFMApplication::appObtuselySetting()->setValue("WindowManager", "WindowState", state);
143 }

172 void WindowManager::showNewWindow(const DUrl &url, const bool& isNewWindow)
173 {
174  if (!isNewWindow){
175  for(int i=0; i< m_windows.count(); i++){
176  QWidget* window = const_cast<QWidget *>(m_windows.keys().at(i));
177  DUrl currentUrl = static_cast<DFileManagerWindow *>(window)->currentUrl();
178  if (currentUrl == url){
179  DFileManagerWindow * wd = static_cast<DFileManagerWindow *>(window);
180  qDebug() << currentUrl << wd;
181  if (wd->isMinimized()) {
182  wd->setWindowState(wd->windowState() & ~Qt::WindowMinimized);
183  }
184  qApp->setActiveWindow(wd);
185  return;
186  }
187  }
188  }
189 
190  QX11Info::setAppTime(QX11Info::appUserTime());
191  DFileManagerWindow *window = new DFileManagerWindow(url.isEmpty() ? DFMApplication::instance()->appUrlAttribute(DFMApplication::AA_UrlOfNewWindow) : url);
192  loadWindowState(window);
193  window->setAttribute(Qt::WA_DeleteOnClose);
194  window->show();
195 
196  qDebug() << "new window" << window->winId() << url;
197 
198  connect(window, &DFileManagerWindow::aboutToClose,
199  this, &WindowManager::onWindowClosed);
200 
201  m_windows.insert(window, window->winId());
202  window->requestToSelectUrls(); //###: here, when selected files and then drag a tab to create a new window.
203  // will select these files again in new window.
204 
205  if (m_windows.count() == 1){
206  QPoint pos = QCursor::pos();
207  QRect currentScreenGeometry;
208 
209  for (QScreen *screen : qApp->screens()) {
210  if (screen->geometry().contains(pos)) {
211  currentScreenGeometry = screen->geometry();
212  }
213  }
214 
215  if (currentScreenGeometry.isEmpty()) {
216  currentScreenGeometry = qApp->primaryScreen()->geometry();
217  }
218 
219  window->moveCenter(currentScreenGeometry.center());
220  }
221 
222  // 修复bug-45226 文管弹出一个模态窗后，再次弹出文管框，
223  // 该模态框将失去焦点，无法正常显示
224  if(!VaultHelper::isModel)
225  qApp->setActiveWindow(window);
226 }

229 quint64 WindowManager::getWindowId(const QWidget *window)
230 {
231  int winId = static_cast<int>(m_windows.value(window->topLevelWidget(), 0));
232 
233  if (winId != 0)
234  return static_cast<quint64>(winId);
235 
236  const QWidget *newW = window;
237 
238  while (newW) {
239  if (newW->inherits("DFileManagerWindow")) {
240  return newW->winId();
241  }
242 
243  newW = newW->parentWidget();
244  }
245 
246  return window->window()->internalWinId();
247 }

249 QWidget *WindowManager::getWindowById(quint64 winId)
250 {
251  if (winId <= 0)
252  return Q_NULLPTR;
253 
254  const QWidget *widget = m_windows.key(winId);
255 
256  if (widget)
257  return const_cast<QWidget*>(widget);
258 
259  for (QWidget *widget : qApp->topLevelWidgets()) {
260  if (widget->internalWinId() == winId)
261  return widget;
262  }
263 
264  return Q_NULLPTR;
265 }
