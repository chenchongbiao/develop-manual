85 MediaMeta MediaLibrary::creatMediaMeta(QString path)
86 {
87  MediaMeta mediaMeta;
88  QFileInfo fileinfo(path);
89  while (fileinfo.isSymLink()) { //to find final target
90  fileinfo.setFile(fileinfo.symLinkTarget());
91  }
92  auto hash = DMusic::filepathHash(fileinfo.absoluteFilePath());
93  mediaMeta.hash = hash;
94  mediaMeta = MetaDetector::getInstance()->updateMetaFromLocalfile(mediaMeta, fileinfo);
95 
96  return mediaMeta;
97 }
