351 void MainFrame::autoStartToPlay()
352 {
353  QString strOpenPath = DataBaseService::getInstance()->getFirstSong();
354  qDebug() << "autoStartToPlay:========" << strOpenPath;
355  auto lastplaypage = MusicSettings::value("base.play.last_playlist").toString(); //上一次的页面
356  if (!strOpenPath.isEmpty()) {
357  //通知设置当前页面
358  Player::getInstance()->setCurrentPlayListHash(lastplaypage, true);
359  qDebug() << "lastplaypage:========" << lastplaypage;
360  return ;
361  }
362  auto lastMeta = MusicSettings::value("base.play.last_meta").toString();
363  if (!lastMeta.isEmpty()) {
364  bool bremb = MusicSettings::value("base.play.remember_progress").toBool();
365  bool bautoplay = MusicSettings::value("base.play.auto_play").toBool();
366  //通知设置当前页面&查询数据
367  Player::getInstance()->setCurrentPlayListHash(lastplaypage, true);
368  //获取上一次的歌曲信息
369  MediaMeta medmeta = DataBaseService::getInstance()->getMusicInfoByHash(lastMeta);
370  //上一次进度的赋值
371  medmeta.offset = MusicSettings::value("base.play.last_position").toInt();
372  if (medmeta.localPath.isEmpty())
373  return;
374  if (bremb) {
375  Player::getInstance()->setActiveMeta(medmeta);
376  //加载进度
380  //Player::getInstance()->loadMediaProgress(medmeta.localPath);
381  // 设置进度
382  m_footerWidget->slotSetWaveValue(MusicSettings::value("base.play.last_position").toInt(), medmeta.length);
383  //加载波形图数据
384  m_footerWidget->slotLoadDetector(lastMeta);
385  }
386  //自动播放处理
387  if (bautoplay) {
388  slotAutoPlay(medmeta); //不再延迟处理，直接播放
389  }
390  }
391 }

393 void MainFrame::showPopupMessage(const QString &songListName, int selectCount, int insertCount)
394 {
395  QFontMetrics fm(font());
396  QString name = fm.elidedText(songListName, Qt::ElideMiddle, 300);
397 
398  auto text = tr("Successfully added to \"%1\"").arg(name); //need translation
399  if (selectCount - insertCount > 0) {
400  if (selectCount == 1 || insertCount == 0)
401  text = tr("Already added to the playlist");
402  else {
403  if (insertCount == 1)
404  text = tr("1 song added");
405  else
406  text = tr("%1 songs added").arg(insertCount);
407  }
408  }
409  if (selectCount == 0 && insertCount == 0) //cda仅仅添加cd歌单，暂未导入数据，
410  text = tr("A disc is connected");

415  m_popupMessage->setVisible(true);

421  if (oldMsgList.size() > 0) {
422  oldMsgList.first()->deleteLater(); // auto delete
423  }
424  if (oldMsgList.size() >= 2)
425  return;
426

539 void MainFrame::slotShortCutTriggered()
540 {
541  QShortcut *objCut = dynamic_cast<QShortcut *>(sender()) ;
542  Q_ASSERT(objCut);
543 
544  if (objCut == addmusicfilesShortcut) {
545  m_importWidget->slotAddMusicButtonClicked(); //open filedialog
546  }
547 
548  if (objCut == viewshortcut) {
549  QRect rect = window()->geometry();
550  QPoint pos(rect.x() + rect.width() / 2, rect.y() + rect.height() / 2);
551  Shortcut sc;
552  QStringList shortcutString;
553  QString param1 = "-j=" + sc.toStr();
554  QString param2 = "-p=" + QString::number(pos.x()) + "," + QString::number(pos.y());
555  shortcutString << "-b" << param1 << param2;
556 
557  QProcess *shortcutViewProc = new QProcess(this);
558  shortcutViewProc->startDetached("killall deepin-shortcut-viewer");
559  shortcutViewProc->startDetached("deepin-shortcut-viewer", shortcutString);
560 
561  connect(shortcutViewProc, SIGNAL(finished(int)), shortcutViewProc, SLOT(deleteLater()));
562  }
563 
564  if (objCut == searchShortcut) {
565  m_titlebarwidget->slotSearchEditFoucusIn();
566  }
567 
568  if (objCut == windowShortcut) {
569  if (windowState() == Qt::WindowMaximized) {
570  showNormal();
571  } else {
572  showMaximized();
573  }
574  }
575 }
