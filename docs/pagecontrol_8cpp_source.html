38 void pageControl::setPageCount(int count)
39 {
40  for (int i = m_pageCount ; i < count ; i++)
41  addButton();
42 
43  for (int i = count; i < m_pageCount ; i++) {
44  DIconButton *pageButton = qobject_cast<DIconButton *>(layout()->takeAt(0)->widget());
45  pageButton->deleteLater();
46  }
47 
48  m_pageCount = count;
49 
50  setCurrent(0);
51 }

61 void pageControl::UpdateIconSize(double scaleX, double scaleY)
62 {
63  double scale = (qAbs(1 - scaleX) < qAbs(1 - scaleY)) ? scaleX : scaleY;
64  for (int i = 0; i < m_pageCount ; i++) {
65  DIconButton *pageButton = qobject_cast<DIconButton *>(layout()->itemAt(i)->widget());
66  pageButton->setIconSize(QSize(PAGE_ICON_SIZE * scale, PAGE_ICON_SIZE * scale));
67  pageButton->setFixedSize(QSize(PAGE_BUTTON_SIZE * scale, PAGE_BUTTON_SIZE * scale));
68  }
69  setFixedHeight(PAGE_BUTTON_SIZE * scale);
70 }

72 void pageControl::addButton()
73 {
74  DIconButton *pageButton = new DIconButton(this);
75  pageButton->setIcon(m_iconNormal);
76  pageButton->setAccessibleName("thisPageButton");
77  pageButton->setIconSize(QSize(PAGE_ICON_SIZE, PAGE_ICON_SIZE));
78  pageButton->setFixedSize(QSize(PAGE_BUTTON_SIZE, PAGE_BUTTON_SIZE));
79  pageButton->setBackgroundRole(DPalette::Button);
80  pageButton->setAutoExclusive(true);
81  pageButton->setFocusPolicy(Qt::NoFocus);
82  pageButton->setCheckable(true);
83  pageButton->setFlat(true);
84 
85  layout()->addWidget(pageButton);
86 
87  connect(pageButton, &DIconButton::toggled, this, &pageControl::pageBtnClicked);
88 }

90 void pageControl::pageBtnClicked(bool checked)
91 {
92  DIconButton *pageButton = qobject_cast<DIconButton *>(sender());
93  if (!pageButton)
94  return;
95 
96  pageButton->setAccessibleName("addPageBtn");
97  pageButton->setIcon(checked ? m_iconActive : m_iconNormal);
98 
99  if (checked)
100  emit onPageChanged(layout()->indexOf(pageButton));
101 }
