266 void DateSettings::onProcessComboBox(const int &value)
267 {
268  if (!m_ntpServerList || m_ntpServerList->count() <= value)
269  return;
270 
271  QString itemText = m_ntpServerList->itemText(value);
272 
273  if (m_autoSyncTimeSwitch->checked()) {
274  m_address->setVisible(itemText == tr("Customize"));
275  if (itemText == tr("Customize"))
276  m_addressContent->setText(m_customNtpServer);
277  m_buttonTuple->setVisible(itemText == tr("Customize"));
278  if (m_addressContent->isAlert()) {
279  m_addressContent->setAlert(false);
280  }
281  }
282 
283  if (!m_bIsUserOperate)
284  return;
285 
286  m_bIsUserOperate = false;
287 
288  if (itemText != tr("Customize") && "" != itemText) {
289  Q_EMIT requestNTPServer(itemText);
290  }
291  setButtonShowState(m_autoSyncTimeSwitch->checked());
292 }

390 void DateSettings::updateSettingTime()
391 {
392  QDateTime datetime = QDateTime::currentDateTime();
393  qint64 second = datetime.toSecsSinceEpoch();
394  int nHours = datetime.time().hour();
395 
396  //不论设置的时间比当前时间大或者小，都需要重新设置该页面的时间
397  if (qAbs(second - m_timeSec) > 60) {
398  m_yearWidget->setValue(datetime.date().year());
399  m_monthWidget->setValue(datetime.date().month());
400  m_dayWidget->setValue(datetime.date().day());
401  if (m_Is24HourType) {
402  m_timeHourWidget->setValue(datetime.time().hour());
403  } else {
404  if (datetime.time().hour() == 0) {
405  nHours = 12;
406  } else if (datetime.time().hour() > 12) {
407  nHours -= 12;
408  }
409  m_timeHourWidget->setValue(nHours);
410  }
411  m_timeMinWidget->setValue(datetime.time().minute());
412  m_timeSec = 0;
413  m_syncSettingTimer->stop();
414  }
415 
416  m_timeSec = second;
417 }
