299  inline void Parser9x::realFC( U32& fc, bool& unicode ) const
300  {
301  if ( fc & 0x40000000 ) {
302  fc = ( fc & 0xbfffffff ) >> 1;
303  unicode = false;
304  }
305  else
306  unicode = m_fib.nFib >= Word8nFib;
307  }
