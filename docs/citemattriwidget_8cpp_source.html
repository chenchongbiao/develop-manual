313 SComDefualData CComAttrWidget::getGraphicItemsDefualData(int tp)
314 {
315  SComDefualData data;
316  auto itemList = graphicItems();
317  if (itemList.isEmpty())
318  return data;
319  CGraphicsItem *pFirstBzItem = itemList.first();
320  CGraphicsUnit unitData = pFirstBzItem->getGraphicsUnit(ENormal);
321  data.penColor = unitData.head.pen.color();
322  data.penWidth = unitData.head.pen.width();
323  data.bursh = unitData.head.brush;
324 
325  data.rectRadius = (tp == Rect ? unitData.data.pRect->xRedius : data.rectRadius);
326  data.polySideCount = (tp == Polygon ? unitData.data.pPolygon->pointNum : data.polySideCount);
327  data.starAnCount = (tp == Star ? unitData.data.pPolygonStar->anchorNum : data.polySideCount);
328  data.starInRadiusRadio = (tp == Star ? unitData.data.pPolygonStar->radius : data.polySideCount);
329 
330  if (tp == Pen || (tp == (Pen | Line))) {
331  if (graphicItems().first()->type() == PenType) {
332  data.penStartType = unitData.data.pPen->start_type;
333  data.penEndType = unitData.data.pPen->end_type;
334  } else {
335  data.penStartType = unitData.data.pLine->start_type;
336  data.penEndType = unitData.data.pLine->end_type;
337  }
338  } else if (tp == Line) {
339  data.lineStartType = unitData.data.pLine->start_type;
340  data.lineEndType = unitData.data.pLine->end_type;
341  }
342 
343  data.blurType = (tp == MasicPen ? static_cast<EBlurEffect>(unitData.data.pBlur->effect) : data.blurType);
344  data.blurWidth = (tp == MasicPen ? data.penWidth : data.blurWidth);
345 
346  if (tp == Text) {
347  auto pText = dynamic_cast<CGraphicsTextItem *>(graphicItems().first());
348 
349  data.textColor = pText->textColor();
350  data.textFontSize = pText->fontSize();
351  data.textFontFamily = pText->fontFamily();
352  data.textFontStyle = pText->fontStyle();
353  data.comVaild[TextColor] = data.textColor.isValid() ? true : false;
354  data.comVaild[TextSize] = data.textFontSize > 0 ? true : false;
355  data.comVaild[TextFont] = data.textFontFamily.isEmpty() ? false : true;
356  data.comVaild[TextHeavy] = data.textFontStyle.isEmpty() ? false : true;
357  } else if (tp == Image) {
358  if (graphicItem()->sceneBoundingRect() != graphicItem()->drawScene()->sceneRect()) {
359  data.comVaild[PropertyImageAdjustScence] = true;
360  } else {
361  data.comVaild[PropertyImageAdjustScence] = false;
362  }
363  data.comVaild[PropertyImageFlipType] = true;
364  }
365 
366  unitData.release();
367 
368  QList<CGraphicsItem *> lists = graphicItems();
369 
370  if (lists.count() > 1) {
371  //多选时 判断值不同时，应该设置为无效
372  for (CGraphicsItem *pItem : lists) {
373  if (pItem->pen().color() != data.penColor) {
374  data.comVaild[LineColor] = false;
375  }
376  if (pItem->pen().width() != data.penWidth) {
377  data.comVaild[LineWidth] = false;
378  }
379  if (pItem->brush().color() != data.bursh.color()) {
380  data.comVaild[FillColor] = false;
381  }
382  //都是矩形
383 
384  switch (tp) {
385  case Rect: {
386  CGraphicsRectItem *pRect = dynamic_cast<CGraphicsRectItem *>(pItem);
387  if (pRect->getXRedius() != data.rectRadius) {
388  data.comVaild[RectRadius] = false;
389  }
390  break;
391  }
392  case Star: {
393  CGraphicsPolygonalStarItem *pStar = dynamic_cast<CGraphicsPolygonalStarItem *>(pItem);
394  if (pStar->anchorNum() != data.starAnCount) {
395  data.comVaild[Anchors] = false;
396  }
397  if (pStar->innerRadius() != data.starInRadiusRadio) {
398  data.comVaild[StarRadius] = false;
399  }
400  break;
401  }
402  case Polygon: {
403  CGraphicsPolygonItem *polygon = dynamic_cast<CGraphicsPolygonItem *>(pItem);
404  if (polygon->nPointsCount() != data.polySideCount) {
405  data.comVaild[SideNumber] = false;
406  }
407  break;
408  }
409  case Pen: {
410  CGraphicsPenItem *pPen = dynamic_cast<CGraphicsPenItem *>(pItem);
411  if (pPen->getPenStartType() != data.penStartType) {
412  data.comVaild[PenStartType] = false;
413  }
414  if (pPen->getPenEndType() != data.penEndType) {
415  data.comVaild[PenEndType] = false;
416  }
417  break;
418  }
419  case Line: {
420  CGraphicsLineItem *pLIne = dynamic_cast<CGraphicsLineItem *>(pItem);
421  if (pLIne->getLineStartType() != data.lineStartType) {
422  data.comVaild[LineStartType] = false;
423  }
424  if (pLIne->getLineEndType() != data.lineEndType) {
425  data.comVaild[LineEndType] = false;
426  }
427  break;

429  case Pen|Line: {
430  if (pItem->type() == PenType) {
431  CGraphicsPenItem *pPen = dynamic_cast<CGraphicsPenItem *>(pItem);
432  if (pPen->getPenStartType() != data.penStartType) {
433  data.comVaild[PenStartType] = false;
434  }
435  if (pPen->getPenEndType() != data.penEndType) {
436  data.comVaild[PenEndType] = false;

439  CGraphicsLineItem *pLIne = dynamic_cast<CGraphicsLineItem *>(pItem);

441  data.comVaild[PenStartType] = false;

444  data.comVaild[PenEndType] = false;

447  break;
448  }
449  case Text: {
450  CGraphicsTextItem *pText = dynamic_cast<CGraphicsTextItem *>(pItem);
451  if (int(pText->/*getSelectedFontSize*/fontSize()) != data.textFontSize) {
452  data.comVaild[TextSize] = false;
453  }
454  if (pText->/*getSelectedFontFamily*/fontFamily() != data.textFontFamily) {
455  data.comVaild[TextFont] = false;
456  }

458  data.comVaild[TextHeavy] = false;
459  }
460  if (pText->/*getSelectedTextColor*/textColor() != data.textColor) {
461  data.comVaild[TextColor] = false;
462  }
463  break;
464  }

477  data.comVaild[PropertyImageAdjustScence] = true;
478  } else {
479  data.comVaild[PropertyImageAdjustScence] = false;
