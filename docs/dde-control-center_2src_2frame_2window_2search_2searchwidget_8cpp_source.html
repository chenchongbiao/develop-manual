62 void DCompleterStyledItemDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const
63 {
64  QPalette::ColorGroup cg = (option.state & QStyle::State_Enabled)
65  ? QPalette::Normal : QPalette::Disabled;
66  if (cg == QPalette::Normal && !(option.state & QStyle::State_Active)) {
67  cg = QPalette::Inactive;
68  }
69 
70  if (option.showDecorationSelected && (option.state & (QStyle::State_Selected | QStyle::State_MouseOver))) {
71  painter->fillRect(option.rect, option.palette.brush(cg, QPalette::Highlight));
72  }
73  QIcon itemIcon = QIcon::fromTheme(index.data(Qt::UserRole + 1).toString());
74  QSize iconSize = QSize(option.rect.height() - 2, option.rect.height() - 2);
75  painter->drawPixmap(QRect(0, option.rect.y(), option.rect.height() - 0, option.rect.height() - 2), itemIcon.pixmap(iconSize));
76 
77  // draw text
78  if (option.state & (QStyle::State_Selected | QStyle::State_MouseOver)) {
79  painter->setPen(option.palette.color(cg, QPalette::HighlightedText));
80  }
81  else {
82  painter->setPen(option.palette.color(cg, QPalette::Text));
83  }
84  painter->setFont(option.font);
85  painter->drawText(option.rect.adjusted(option.rect.height() + 8, 0, 0, 0), Qt::AlignVCenter, index.data(Qt::DisplayRole).toString());
86 }

319 void SearchModel::loadxml()
320 {
321  clear(); // It doesn't seem to leak memory
322  m_EnterNewPagelist.clear();
323  m_inputList.clear();
324  m_TxtListAll.clear();
325 
326  //添加一项空数据，为了防止使用setText输入错误数据时直接跳转到list中正确的第一个页面
327  m_EnterNewPagelist.append(std::make_shared<SearchBoxStruct>());
328  m_inputList.append(SearchDataStruct());
329  appendRow(new QStandardItem(""));
330 
331  for (SearchBoxStruct::Ptr searchBoxStrcut : m_originList) {
332  //"蓝牙","数位板"不存在则不加载该模块search数据
333  //目前只用到了模块名，未使用detail信息，之后再添加模块内区分
334  bool bIsLeapfrog = false;
335  auto res = std::any_of(m_unexsitList.begin(), m_unexsitList.end(), [=](const UnexsitStruct &date) {
336  return searchBoxStrcut->actualModuleName == date.module;
337  });
338 
339  if (res) {
340  bIsLeapfrog = true;
341  continue;
342  }
343 
344  if (bIsLeapfrog) continue;
345 
346  //“鼠标”可移除设备 : 指点杆，触控板
347  //“网络”模块可移除设备 : 个人热点，有线网，无线网
348  //“电源”模块可移除设备 : 使用电池
349  //不存在时，不加载数据
350  //是以上模块才会有此判断，其他模块不用此判断(包含在m_defaultRemoveableList的页面才需要“添加/移除”xml信息)
351  if (m_defaultRemoveableList.contains(searchBoxStrcut->fullPagePath.section('/', 2, -1))) {
352  auto result = std::find_if(m_removeableActualExistList.begin(),
353  m_removeableActualExistList.end(),
354  [=](const QPair<QString, QString> &date) {
355  return date.second == searchBoxStrcut->fullPagePath.section('/', 2, -1);
356  });
357 
358  //设备不存在，不加载xml数据
359  if (result == m_removeableActualExistList.end()) {
360  continue;
361  }
362  }
363 
364  if ("" == searchBoxStrcut->actualModuleName || "" == searchBoxStrcut->translateContent) {
365  continue;
366  }
367 
368  //判断是否为服务器,是服务器时,若当前不是服务器就不添加"Server"
369  if (isLoadText(searchBoxStrcut->translateContent)) {
370  continue;
371  }
372 
373  //判断是否为contens服务器,是contens服务器时,若当前不是服务器就不添加"Server"
374  if (isLoadContensText(searchBoxStrcut->translateContent)) {
375  continue;
376  }
377 
378  //判断是否为服务器，如果是服务器状态下搜索不到网络账户相关（所有界面）
379  if (m_bIsServerType && tr("Cloud Account") == searchBoxStrcut->actualModuleName) {
380  continue;
381  }
382 
383  //qDebug()<<"m_deepinwm->compositingAllowSwitch() = "<<m_deepinwm->compositingAllowSwitch();
384  if (!m_bIsServerType && !m_deepinwm->compositingAllowSwitch()) {
385  qDebug() << "search not Window!";
386  if (tr("Window Effect") == searchBoxStrcut->translateContent) {
387  continue;
388  }
389  }
390 
391  m_EnterNewPagelist.append(searchBoxStrcut);
392 
393  if (m_TxtListAll.contains(searchBoxStrcut->translateContent)) {
394  continue;
395  }
396 
397  m_TxtListAll.append(searchBoxStrcut->translateContent);
398 
399  // Add search result content
400  if (!m_bIsChinese) {
401  auto icon = m_iconMap.find(searchBoxStrcut->fullPagePath.section('/', 1, 1));
402  if (icon == m_iconMap.end()) {
403  continue;
404  }
405 
406  if ("" == searchBoxStrcut->childPageName) {
407  appendRow(new QStandardItem(icon.value(), QString("%1 --> %2").arg(searchBoxStrcut->actualModuleName).arg(searchBoxStrcut->translateContent)));
408  }
409  else {
410  appendRow(new QStandardItem(
411  icon.value(), QString("%1 --> %2 / %3").arg(searchBoxStrcut->actualModuleName).arg(searchBoxStrcut->childPageName).arg(searchBoxStrcut->translateContent)));
412  }
413  }
414  else {
415  appendChineseData(searchBoxStrcut);
416  }
417  }
418 }

515 void SearchModel::appendChineseData(SearchBoxStruct::Ptr data)
516 {

525  if ("" == dataBackup->childPageName) {
526  //先添加使用appenRow添加Qt::EditRole数据(用于下拉框显示),然后添加Qt::UserRole数据(用于输入框搜索)
527  //Qt::EditRole数据用于显示搜索到的结果(汉字)
528  //Qt::UserRole数据用于输入框输入的数据(拼音/汉字 均可)

535  QString("%1 --> %2")
536  .arg(dataBackup->actualModuleName)
537  .arg(dataBackup->translateContent),
538  Qt::UserRole);
539  setData(index(rowCount() - 1, 0), icon->name(), Qt::UserRole + 1);
540 
541  QString hanziTxt = QString("%1 --> %2").arg(dataBackup->actualModuleName).arg(dataBackup->translateContent);
