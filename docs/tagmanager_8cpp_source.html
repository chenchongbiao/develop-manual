193 bool TagManager::makeFilesTags(const QList<QString> &tags, const QList<DUrl> &files)
194 {
195  bool result{ true };
196 
197  if (!tags.isEmpty() && !files.isEmpty()) {
198  QMap<QString, QVariant> tag_and_file{};
199 
200  for (const QString &tag_name : tags) {
201  QString color_name;
202 
203  // for default tags
204  for (const QString &color : Tag::ColorName) {
205  if (tag_name == Tag::ActualAndFakerName().value(color)) {
206  color_name = color;
207  break;
208  }
209  }
210 
211  if (color_name.isEmpty()) {
212  color_name = randomColor();
213  }
214 
215  tag_and_file[tag_name] = QVariant{QList<QString>{ color_name }};
216  }
217 
218  QVariant insert_tags_var{ TagManagerDaemonController::instance()->disposeClientData(tag_and_file, Tag::ActionType::BeforeMakeFilesTags) };
219  QMap<QString, QVariant> file_and_tag{};
220 
221  for (const DUrl &url : files) {
222  file_and_tag[url.toLocalFile()] = QVariant{tags};
223  }
224 
225  QVariant tag_files_var{};
226 
227  if (insert_tags_var.toBool()) {
228  tag_files_var = TagManagerDaemonController::instance()->disposeClientData(file_and_tag, Tag::ActionType::MakeFilesTags);
229  }
230 
231  if (insert_tags_var.toBool()) {
232 
233  if (!tag_files_var.toBool()) {
234  qWarning() << "Create tags successfully! But failed to tag files";
235  }
236  result = true;
237  }
238  }
239 
240  return result;
241 }
