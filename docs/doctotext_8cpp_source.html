124 string get_document(string input)
125 {
126  bool extract_metadata = false;
127  string text = "";
128  FormattingStyle options;
129  options.table_style = TABLE_STYLE_TABLE_LOOK;
130  options.list_style.setPrefix(" * ");
131  options.url_style = URL_STYLE_UNDERSCORED;
132  XmlParseMode mode = PARSE_XML;
133  clock_t start, end;
134  start = clock(); /*记录起始时间*/
135 // printf("filename:%s\n",input.c_str());
136  size_t last = input.find_last_of("/");
137  std::string name = input.substr(last + 1);
138  std::string dir = input.substr(0, last);
139  std::string ext = getDefaultExtension(input);
140  PlainTextExtractor::ParserType parser_type = PlainTextExtractor::PARSER_AUTO;
141  try {
142 // if (ext == "docx"|| ext == "dot" )
143 // parser_type = PlainTextExtractor::PARSER_DOC;
144 // else if (ext == "html" || ext == "htm" || ext == "xhtml" || ext == "xht")
145 // parser_type = PlainTextExtractor::PARSER_HTML;
146 // else if (ext == "xml")
147 // parser_type = PlainTextExtractor::PARSER_ODFXML;
148 // else if (ext == "txt" || ext == "md" || ext == "markdown")
149 // extract_metadata = true;
150 // else if (ext == "doc")
151 // parser_type = PlainTextExtractor::PARSER_DOC;
152 // else if (ext == "rtf")
153 // parser_type = PlainTextExtractor::PARSER_RTF;
154 // else if (ext == "xls" || ext == "xlsx")
155 // parser_type = PlainTextExtractor::PARSER_XLS;
156 // else if (ext == "ppt" || ext == "pptx")
157 // parser_type = PlainTextExtractor::PARSER_PPT;
158 // else if (ext == "pdf")
159 // parser_type = PlainTextExtractor::PARSER_PDF;
160 
161  if (ext == "rtf")
162  parser_type = PlainTextExtractor::PARSER_RTF;
163  else if (ext == "odt" || ext == "ods" || ext == "odp" || ext == "odg" || ext == "docx" || ext == "xlsx" || ext == "pptx" || ext == "ppsx")
164  parser_type = PlainTextExtractor::PARSER_ODF_OOXML;
165  else if (ext == "xls")
166  parser_type = PlainTextExtractor::PARSER_XLS;
167  else if (ext == "xlsb")
168  parser_type = PlainTextExtractor::PARSER_XLSB;
169  else if (ext == "doc" || ext == "dot" || ext == "wps" )
170  parser_type = PlainTextExtractor::PARSER_DOC;
171  else if (ext == "ppt" || ext == "pps")
172  parser_type = PlainTextExtractor::PARSER_PPT;
173  else if (ext == "htm" || ext == "html")
174  parser_type = PlainTextExtractor::PARSER_HTML;
175  else if (ext == "pages" || ext == "key" || ext == "numbers")
176  parser_type = PlainTextExtractor::PARSER_IWORK;
177  else if (ext == "pdf")
178  parser_type = PlainTextExtractor::PARSER_PDF;
179  else if (ext == "txt" || ext == "text")
180  parser_type = PlainTextExtractor::PARSER_TXT;
181  else {
182  std::cout << "Unsupported file extension: " << ext << std::endl;
183  return "";
184  }
185  std::cout << "Conversion complete: " << input << std::endl;
186  } catch (...) {
187  std::cerr << "Error: " << input << std::endl;
188  }
189  //创建文件解析器
190  PlainTextExtractor extractor(parser_type);
191 
192  //设置 xml 格式解析模式
193  if (mode != PARSE_XML) {
194  extractor.setXmlParseMode(mode);
195  }
196 
197  //
198  extractor.setFormattingStyle(options);
199 
200  if (extract_metadata) {
201  //解析文件属性
202  Metadata meta;
203  if (!extractor.extractMetadata(input, meta)) {
204  cerr << "Error processing file " << input << ".\n";
205  return "";
206  }
207  cout << "Author: " << meta.author() << (meta.authorType() == Metadata::ESTIMATED ? " (estimated)" : "")
208  << "\nCreation time: " << date_to_string(meta.creationDate()) << (meta.creationDateType() == Metadata::ESTIMATED ? " (estimated)" : "")
209  << "\nLast modified by: " << meta.lastModifiedBy() << (meta.lastModifiedByType() == Metadata::ESTIMATED ? " (estimated)" : "")
210  << "\nLast modification time: " << date_to_string(meta.lastModificationDate()) << (meta.lastModificationDateType() == Metadata::ESTIMATED ? " (estimated)" : "")
211  << "\nPage count: " << meta.pageCount() << (meta.pageCountType() == Metadata::ESTIMATED ? " (estimated)" : "")
212  << "\nWord count: " << meta.wordCount() << (meta.wordCountType() == Metadata::ESTIMATED ? " (estimated)" : "")
213  << "\n";
214  } else {
215  //解析文件内容,
216  /*
217  * 原因是解析出来的内容和 tests目录下面 *.out 中的内容不相符，
218  * 例如: 解析 tests/1.doc.out 最终结果的内容就是现在下面 printf 所打印输出的内容格式，
219  * 从可以 tests/Makefile 中看到，在测试的过程中是通过 diff 命令比较，也就是通过比较 1.doc
220  * 的解析结果内容和 1.doc.out 内容的是否相同来判断程序是否正常运行的
221  */
222  if (!extractor.processFile(input, text)) {
223  cerr << "Error processing file " << input << ".\n";
224  return "";
225  }
226 
227  //打印文件内容
228 // printf("%s\n", text.c_str());
229 
230  /*
231  * 解析 link 信息
232  *
233  * 例如：文件 example.html 内容如下
234  * "text before link <a href="target">link</a> text after link"
235  * 内容中包含一个 link， 我们使用 URL_STYLE_TEXT_ONLY 类型解析后的结果是
236  * "text before link link text after link".
237  * 我们还获取到以下一些值：
238  * 调用 getLinkUrl() 返回: "target".
239  * 调用 getLinkText() 返回:"link".
240  * 调用 getLinkTextSize() 返回: 4 (因为 "link" 是 4 个字节).
241  * 调用 getLinkTextPosition() 返回: 17 (因为 "text before link " 是 17 个字节).
242  */
243  std::vector<Link> links;
244  extractor.getParsedLinks(links);
245  if (links.size() > 0) {
246  printf("parsed links:\n");
247  for (size_t i = 0; i < links.size(); ++i) {
248  printf("%s @ index = %d length = %d\n", links[i].getLinkUrl(), links[i].getLinkTextPosition(), strlen(links[i].getLinkText()));
249  }
250  }
251 
252  //解析附件，这个现在只有 eml 类型的实现
253  std::vector<Attachment> attachments;
254  extractor.getAttachments(attachments);
255  if (attachments.size() > 0) {
256  printf("parsed attachments:\n");
257  for (size_t i = 0; i < attachments.size(); ++i) {
258  printf("\nname: %s\n", attachments[i].filename());
259  std::map<std::string, Variant> variables = attachments[i].getFields();
260  for (std::map<std::string, Variant>::iterator it = variables.begin(); it != variables.end(); ++it) {
261  if (it->first != "Content-ID") {
262  printf("field: %s, value: %s\n", it->first.c_str(), it->second.getString());
263  }
264  }
265  }
266  }
267  }
268  end = clock(); /*记录结束时间*/
269  {
270  double seconds = (double)(end - start) / CLOCKS_PER_SEC;
271  fprintf(stderr, "Use time is: %.8f s\n", seconds);
272  }
273  return text.c_str();
274 }

277 int getdir(const char *pathname)
278 {
279  DIR *path = NULL;
280  path = opendir(pathname);
281 
282  if (path == NULL) {
283  perror("failed");
284  exit(1);
285  }
286  struct dirent *ptr; //目录结构体---属性：目录类型 d_type, 目录名称d_name
287  char buf[1024] = {0};
288  while ((ptr = readdir(path)) != NULL) {
289  if (strcmp(ptr->d_name, ".") == 0 || strcmp(ptr->d_name, "..") == 0) {
290  continue;
291  }
292  //如果是目录
293  if (ptr->d_type == DT_DIR) {
294 
295  sprintf(buf, "%s/%s", pathname, ptr->d_name);
296 // printf("目录:%s\n",buf);
297  getdir(buf);
298  }
299  if (ptr->d_type == DT_REG) {
300  sprintf(buf, "%s/%s", pathname, ptr->d_name); //把pathname和文件名拼接后放进缓冲字符数组
301  printf("文件:%s\n", buf);
302  string contents = get_document(string(buf));
303  printf("========================================================\n%s\n", contents.c_str());
304  }
305  }
306  return 0;
307 }
