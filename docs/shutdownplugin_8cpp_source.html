111 const QString ShutdownPlugin::itemContextMenu(const QString &itemKey)
112 {
113  QList<QVariant> items;
114  items.reserve(6);
115 
116  QMap<QString, QVariant> shutdown;
117  shutdown["itemId"] = "Shutdown";
118  shutdown["itemText"] = tr("Shut down");
119  shutdown["isActive"] = true;
120  items.push_back(shutdown);
121 
122  QMap<QString, QVariant> reboot;
123  reboot["itemId"] = "Restart";
124  reboot["itemText"] = tr("Reboot");
125  reboot["isActive"] = true;
126  items.push_back(reboot);
127 
128 #ifndef DISABLE_POWER_OPTIONS
129 
130  QProcessEnvironment enviromentVar = QProcessEnvironment::systemEnvironment();
131  bool can_sleep = enviromentVar.contains("POWER_CAN_SLEEP") ? QVariant(enviromentVar.value("POWER_CAN_SLEEP")).toBool()
132  : valueByQSettings<bool>("Power", "sleep", true) &&
133  m_powerManagerInter->CanSuspend();
134  ;
135  if (can_sleep) {
136  QMap<QString, QVariant> suspend;
137  suspend["itemId"] = "Suspend";
138  suspend["itemText"] = tr("Suspend");
139  suspend["isActive"] = true;
140  items.push_back(suspend);
141  }
142 
143  bool can_hibernate = enviromentVar.contains("POWER_CAN_HIBERNATE") ? QVariant(enviromentVar.value("POWER_CAN_HIBERNATE")).toBool()
144  : checkSwap() && m_powerManagerInter->CanHibernate();
145 
146  if (can_hibernate) {
147  QMap<QString, QVariant> hibernate;
148  hibernate["itemId"] = "Hibernate";
149  hibernate["itemText"] = tr("Hibernate");
150  hibernate["isActive"] = true;
151  items.push_back(hibernate);
152  }
153 
154 #endif
155 
156  QMap<QString, QVariant> lock;
157  lock["itemId"] = "Lock";
158  lock["itemText"] = tr("Lock");
159  lock["isActive"] = true;
160  items.push_back(lock);
161 
162  QMap<QString, QVariant> logout;
163  logout["itemId"] = "Logout";
164  logout["itemText"] = tr("Log out");
165  logout["isActive"] = true;
166  items.push_back(logout);
167 
168  if (!QFile::exists(ICBC_CONF_FILE)) {
169  if (DBusAccount().userList().count() > 1 || DSysInfo::uosType() == DSysInfo::UosType::UosServer) {
170  QMap<QString, QVariant> switchUser;
171  switchUser["itemId"] = "SwitchUser";
172  switchUser["itemText"] = tr("Switch account");
173  switchUser["isActive"] = true;
174  items.push_back(switchUser);
175  }
176 
177 #ifndef DISABLE_POWER_OPTIONS

179  power["itemId"] = "power";
180  power["itemText"] = tr("Power settings");
181  power["isActive"] = true;
182  items.push_back(power);
183 #endif
184  }
185 
186  QMap<QString, QVariant> menu;
187  menu["items"] = items;
188  menu["checkableMenu"] = false;
189  menu["singleCheck"] = false;
190 
191  return QJsonDocument::fromVariant(menu).toJson();
192 }

194 void ShutdownPlugin::invokedMenuItem(const QString &itemKey, const QString &menuId, const bool checked)
195 {
196  Q_UNUSED(itemKey)
197  Q_UNUSED(checked)
198 
199  // 使得下面逻辑代码延迟200ms执行，保证线程不阻塞
200  QTime dieTime = QTime::currentTime().addMSecs(200);
201  while( QTime::currentTime() < dieTime )
202  QCoreApplication::processEvents(QEventLoop::AllEvents, 200);
203 
204  if (menuId == "power") {
205  DDBusSender()
206  .service("com.deepin.dde.ControlCenter")
207  .interface("com.deepin.dde.ControlCenter")
208  .path("/com/deepin/dde/ControlCenter")
209  .method(QString("ShowModule"))
210  .arg(QString("power"))
211  .call();
212  }
213  else if (menuId == "Lock") {
214  if (QFile::exists(ICBC_CONF_FILE)) {
215  QDBusMessage send = QDBusMessage::createMethodCall("com.deepin.dde.lockFront", "/com/deepin/dde/lockFront", "com.deepin.dde.lockFront", "SwitchTTYAndShow");
216  QDBusConnection conn = QDBusConnection::connectToBus("unix:path=/run/user/1000/bus","unix:path=/run/user/1000/bus");
217  QDBusMessage reply = conn.call(send);
218 #ifdef QT_DEBUG
219  qInfo()<<"----------"<<reply;
220 #endif
221 
222  } else {
223  DDBusSender()
224  .service("com.deepin.dde.lockFront")
225  .interface("com.deepin.dde.lockFront")
226  .path("/com/deepin/dde/lockFront")
227  .method(QString("Show"))
228  .call();
229  }
230  }
231  else
232  DDBusSender()
233  .service("com.deepin.dde.shutdownFront")
234  .interface("com.deepin.dde.shutdownFront")
235  .path("/com/deepin/dde/shutdownFront")
236  .method(QString(menuId))
237  .call();
238 }

311 bool ShutdownPlugin::checkSwap()
312 {
313  if (!valueByQSettings<bool>("Power", "hibernate", false))
314  return false;
315 
316  bool hasSwap = false;
317  QFile file("/proc/swaps");
318  if (file.open(QIODevice::Text | QIODevice::ReadOnly)) {
319  const QString &body = file.readAll();
320  QTextStream stream(body.toUtf8());
321  while (!stream.atEnd()) {
322  const std::pair<bool, qint64> result =
323  checkIsPartitionType(stream.readLine().simplified().split(
324  " ", QString::SplitBehavior::SkipEmptyParts));
325  qint64 image_size{ get_power_image_size() };
326 
327  if (result.first) {
328  hasSwap = image_size < result.second;
329  }
330 
331  if (hasSwap) {
332  break;
333  }
334  }
335 
336  file.close();
337  } else {
338  qDebug() << "open /proc/swaps failed! please check permission!!!";
339  }
340 
341  return hasSwap;
342 }
