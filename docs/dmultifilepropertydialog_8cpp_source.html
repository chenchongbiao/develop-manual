302 std::pair<std::size_t, std::size_t> DMultiFilePropertyDialog::getTheQuantityOfFilesAndFolders(const QList<DUrl> &urlList)noexcept
303 {
304  std::pair<std::size_t, std::size_t> quantityOfFilesAndFolders{0, 0};
305 
306  if(urlList.empty() == true){
307  return quantityOfFilesAndFolders;
308  }
309 
310  for(const auto& url : urlList){
311  QFileInfo info{ url.toLocalFile() };
312 
313  if(info.isFile() == true){
314  ++quantityOfFilesAndFolders.first;
315 
316  }else{
317  ++quantityOfFilesAndFolders.second;
318 
319  }
320 
321  }
322 
323  return quantityOfFilesAndFolders;
324 }
