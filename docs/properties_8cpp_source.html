60 bool Properties::load(const QString &fileName, const QString &group) {
61 
62  // NOTE: This class is used for reading of property files instead of QSettings
63  // class, which considers separator ';' as comment
64 
65  // Try open file
66  QFile file(fileName);
67  if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
68  return false;
69  }
70 
71  // Clear old data
72  data.clear();
73 
74  // Indicator whether group was found or not, if name of group was not
75  // specified, groupFound is always true
76  bool groupFound = group.isEmpty();
77 
78  // Read propeties
79  QTextStream in(&file);
80  while (!in.atEnd()) {
81 
82  // Read new line
83  QString line = in.readLine();
84 
85  // Skip empty line or line with invalid format
86  if (line.trimmed().isEmpty()) {
87  continue;
88  }
89 
90  // Read group
91  // NOTE: symbols '[' and ']' can be found not only in group names, but
92  // only group can start with '['
93  if (!group.isEmpty() && line.trimmed().startsWith("[")) {
94  QString tmp = line.trimmed().replace("[", "").replace("]", "");
95  groupFound = group.trimmed().compare(tmp) == 0;
96  }
97 
98  // If we are in correct group and line contains assignment then read data
99  int first_equal = line.indexOf('=');
100 
101  if (groupFound && first_equal >= 0) {
102  data.insert(line.left(first_equal).trimmed(), line.mid(first_equal + 1).trimmed());
103  }
104  }
105  file.close();
106 
107  return true;
108 }

117 bool Properties::save(const QString &fileName, const QString &group) {
118 
119  // Try open file
120  QFile file(fileName);
121  if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
122  return false;
123  }
124 
125  // Write group
126  QTextStream out(&file);
127  if (!group.isEmpty()) {
128  out << "[" + group + "]\n";
129  }
130 
131  // Write data
132  foreach (QString key, data.keys()) {
133  out << key << "=" << data.value(key).toString() << "\n";
134  }
135 
136  // Exit
137  file.close();
138  return true;
139 }
