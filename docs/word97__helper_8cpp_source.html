319 // The length of the SPRM parameter
320 U16 determineParameterLength( U16 sprm, const U8* in, WordVersion version )
321 {
322  if ( version == Word8 ) {
323  static const char operandSizes[ 8 ] = { 1, 1, 2, 4, 2, 2, 0, 3 };
324 
325  int index = ( sprm & 0xE000 ) >> 13;
326  if ( operandSizes[ index ] != 0 )
327  return operandSizes[ index ];
328  else {
329  // Get length of variable size operand.
330  switch ( sprm ) {
331  case sprmTDefTable10:
332  case sprmTDefTable:
333  return readU16( in ) + 1;
334  break;
335  case sprmPChgTabs:
336  if ( *in == 255 ) {
337  U8 itbdDelMax = in[ 1 ];
338  U8 itbdAddMax = in[ 1 + itbdDelMax * 4 ];
339  return 1 + itbdDelMax * 4 + itbdAddMax * 3;
340  }
341  else
342  return *in + 1;
343  break;
344  default:
345  return *in + 1;
346  break;
347  }
348  }
349  }
350  else { // Word67
351  if ( sprm > 255 )
352  wvlog << "Error: Trying to get the length of a flaky SPRM (" << sprm << ", 0x" << std::hex
353  << sprm << std::dec << ") via the Word 95 method!" << std::endl;
354  return Word95::SPRM::determineParameterLength( static_cast<U8>( sprm ), in );
355  }
356 }

358 // Apply a <T> grpprl of a given size ("count" bytes long)
359 // This template function might be a bit sick, but it helps to
360 // avoid duplicated code, so what ;)
361 template<class T>
362 void apply(T* const t,
363  S16 ( T::* applySPRM ) ( const U8*, const Style*, AbstractOLEStreamReader*, WordVersion ),
364  const U8* grpprl, U16 count, const Style* style, AbstractOLEStreamReader* dataStream, WordVersion version )
365 {
366  if ( !grpprl )
367  return;
368 
369  // We are using an integer here, that we can detect situations where
370  // we read beyond the limit due to a buggy spec/impl.
371  // The plain U16 would overflow and we'd be trapped in a loop.
372  int safeCount = count;
373 
374  // walk through the grpprl, applying one sprm after the other
375  while ( safeCount > 1 ) {
376  S16 result = ( t->*applySPRM )( grpprl, style, dataStream, version );
377  if ( result == -1 ) {
378  U16 sprm;
379  if ( version == Word8 ) {
380  sprm = readU16( grpprl );
381  grpprl += 2;
382 #ifdef WV2_DEBUG_SPRMS
383  wvlog << "Seems like that's a different SPRM (0x" << std::hex << sprm << std::dec << ")... skipping" << std::endl;
384 #endif
385  }
386  else {
387  sprm = *grpprl++;
388 #ifdef WV2_DEBUG_SPRMS
389  wvlog << "Seems like that's a different SPRM (" << sprm << ")... skipping" << std::endl;
390 #endif
391  }
392 
393  U16 len = determineParameterLength( sprm, grpprl, version );
394  grpprl += len;
395  safeCount -= len + ( version == Word8 ? 2 : 1 );
396  }
397  else {
398  grpprl += result;
399  safeCount -= result;
400  }
401  }
402  if ( safeCount < 0 )
403  wvlog << "Warning: We read past the end of the grpprl, buggy spec?" << std::endl;
404 }
