111 void CGraphicsTriangleItem::calcPoints_helper(QVector<QPointF> &outVector, qreal offset) const
112 {
113  QRectF rc = rect();
114  QPointF top = QPointF((rc.x() + rc.width() / 2), rc.y());
115  const QPen pen = this->paintPen();
116 
117  outVector.clear();
118  //获取填充区域路径
119  qreal offsetWidth = offset;
120  QLineF line1(top, rc.bottomLeft());
121  QLineF line2(rc.bottomLeft(), rc.bottomRight());
122  QLineF line3(rc.bottomRight(), top);
123  QVector<QLineF> lines;
124  lines << line3 << line1 << line2;
125  for (int i = 0; i < lines.size(); ++i) {
126  QLineF ln1 = lines.at(i);
127  QLineF ln2 = (i == lines.size() - 1 ? lines[0] : lines[i + 1]);
128  qreal angle = 180 - ln1.angleTo(ln2);
129 
130  qreal offsetLen = offsetWidth / qSin(qDegreesToRadians(angle / 2.0));
131  QLineF tempLine(ln2);
132  tempLine.setAngle(tempLine.angle() + angle / 2.0);
133  tempLine.setLength(offsetLen);
134 
135  outVector.append(tempLine.p2());
136  }
137 }
