64 bool VpnVPNCSection::allInputValid()
65 {
66  bool valid = true;
67 
68  if (m_gateway->text().isEmpty()) {
69  valid = false;
70  m_gateway->setIsErr(true);
71  } else {
72  m_gateway->setIsErr(false);
73  }
74  if (m_userName->text().isEmpty()) {
75  valid = false;
76  m_userName->setIsErr(true);
77  } else {
78  m_userName->setIsErr(false);
79  }
80  if (m_currentPasswordType == NetworkManager::Setting::SecretFlagType::None
81  && m_password->text().isEmpty()) {
82  valid = false;
83  m_password->setIsErr(true);
84  } else {
85  m_password->setIsErr(false);
86  }
87  if (m_groupName->text().isEmpty()) {
88  valid = false;
89  m_groupName->setIsErr(true);
90  } else {
91  m_groupName->setIsErr(false);
92  }
93  if (m_currentGroupPasswordType == NetworkManager::Setting::SecretFlagType::None
94  && m_groupPassword->text().isEmpty()) {
95  valid = false;
96  m_groupPassword->setIsErr(true);
97  } else {
98  m_groupPassword->setIsErr(false);
99  }
100  if (m_userHybrid->checked() && m_caFile->edit()->text().isEmpty()) {
101  valid = false;
102  m_caFile->setIsErr(true);
103  } else {
104  m_caFile->setIsErr(false);
105  }
106 
107  return valid;

110 void VpnVPNCSection::saveSettings()
111 {
112  // retrieve the data map
113  m_dataMap = m_vpnSetting->data();
114  m_secretMap = m_vpnSetting->secrets();
115 
116  // save values about user
117  m_dataMap.insert("IPSec gateway", m_gateway->text());
118  m_dataMap.insert("Xauth username", m_userName->text());
119  m_dataMap.insert("Xauth password-flags", QString::number(m_currentPasswordType));
120  if (m_currentPasswordType == NetworkManager::Setting::SecretFlagType::None) {
121  m_dataMap.insert("xauth-password-type", "save");
122  m_secretMap.insert("Xauth password", m_password->text());
123  } else if (m_currentPasswordType == NetworkManager::Setting::SecretFlagType::NotSaved) {
124  m_dataMap.insert("xauth-password-type", "ask");
125  m_secretMap.remove("Xauth password");
126  } else if (m_currentPasswordType == NetworkManager::Setting::SecretFlagType::NotRequired) {
127  m_dataMap.insert("xauth-password-type", "unused");
128  m_secretMap.remove("Xauth password");
129  }
130  // save values about gourp
131  m_dataMap.insert("IPSec ID", m_groupName->text());
132  m_dataMap.insert("IPSec secret-flags", QString::number(m_currentGroupPasswordType));
133  if (m_currentGroupPasswordType == NetworkManager::Setting::SecretFlagType::None) {
134  m_dataMap.insert("ipsec-secret-type", "save");
135  m_secretMap.insert("IPSec secret", m_groupPassword->text());
136  } else if (m_currentGroupPasswordType == NetworkManager::Setting::SecretFlagType::NotSaved) {
137  m_dataMap.insert("ipsec-secret-type", "ask");
138  m_secretMap.remove("IPSec secret");
139  } else if (m_currentGroupPasswordType == NetworkManager::Setting::SecretFlagType::NotRequired) {
140  m_dataMap.insert("ipsec-secret-type", "unused");
141  m_secretMap.remove("IPSec secret");
142  }
143 
144  if (m_userHybrid->checked()) {
145  m_dataMap.insert("IKE Authmode", "hybrid");
146  m_dataMap.insert("CA-File", m_caFile->edit()->text());
147  } else {
148  m_dataMap.remove("IKE Authmode");
149  }
150 
151  m_vpnSetting->setData(m_dataMap);
152  m_vpnSetting->setSecrets(m_secretMap);
153 
154  m_vpnSetting->setInitialized(true);
155 }

235 void VpnVPNCSection::initConnection()
236 {
237  connect(m_passwordFlagsChooser, &ComboxWidget::onSelectChanged, this, [ = ](const QString &dataSelected) {
238  for (auto it = PasswordFlagsStrMap.cbegin(); it != PasswordFlagsStrMap.cend(); ++it) {
239  if (it->first == dataSelected) {
240  onPasswordFlagsChanged(it->second);
241  break;
242  }
243  }
244  });
245  connect(m_groupPasswordFlagsChooser, &ComboxWidget::onSelectChanged, this, [ = ](const QString &dataSelected) {
246  for (auto it = PasswordFlagsStrMap.cbegin(); it != PasswordFlagsStrMap.cend(); ++it) {
247  if (it->first == dataSelected) {
248  onGroupPasswordFlagsChanged(it->second);
249  break;
250  }
251  }
252  });
253  connect(m_userHybrid, &SwitchWidget::checkedChanged, this, [ = ](const bool checked) {
254  m_caFile->setVisible(checked);
255  allInputValid();
256  });
257 
258  connect(m_caFile, &FileChooseWidget::requestFrameKeepAutoHide, this, &VpnVPNCSection::requestFrameAutoHide);
259 
260  connect(m_passwordFlagsChooser, &ComboxWidget::onIndexChanged, this, &VpnVPNCSection::editClicked);
261  connect(m_groupPasswordFlagsChooser, &ComboxWidget::onIndexChanged, this, &VpnVPNCSection::editClicked);
262  connect(m_userHybrid, &SwitchWidget::checkedChanged, this, &VpnVPNCSection::editClicked);
263  connect(m_caFile->edit()->lineEdit(), &QLineEdit::textChanged, this, &VpnVPNCSection::editClicked);
264 }
