83 void PowerModule::active()
84 {
85  m_widget = new PowerWidget;
86  m_widget->setVisible(false);
87 
88  m_widget->initialize(m_model->haveBettary());
89 
90  m_powerSetting = new QGSettings("com.deepin.dde.control-center", QByteArray(), this);
91  m_isSuspend = m_powerSetting->get(GSETTING_SHOW_SUSPEND).toBool();
92  m_model->setSuspend(m_isSuspend && m_model->canSleep());
93 
94  bool hibernate = m_powerSetting->get(GSETTING_SHOW_HIBERNATE).toBool();
95  m_model->setHibernate(!IsServerSystem && hibernate && m_model->canHibernate());
96  connect(m_model, &PowerModel::canHibernateChanged, this, [=](const bool &value){
97  m_model->setHibernate(!IsServerSystem && hibernate && value);
98  });
99 
100  connect(m_model, &PowerModel::hibernateChanged, this, &PowerModule::showUseElectric);
101 
102  connect(m_model, &PowerModel::haveBettaryChanged, m_widget, &PowerWidget::removeBattery);
103  connect(m_model, &PowerModel::batteryPercentageChanged, this, &PowerModule::onBatteryPercentageChanged);
104  connect(m_widget, &PowerWidget::requestShowGeneral, this, &PowerModule::showGeneral);
105  connect(m_widget, &PowerWidget::requestShowUseBattery, this, &PowerModule::showUseBattery);
106  connect(m_widget, &PowerWidget::requestShowUseElectric, this, &PowerModule::showUseElectric);
107 
108  m_frameProxy->pushWidget(this, m_widget);
109  m_widget->setVisible(true);
110  m_widget->setDefaultWidget();
111 }

113 int PowerModule::load(const QString &path)
114 {
115  if (!m_widget) {
116  active();
117  }
118 
119  QListView *list = m_widget->getListViewPointer();
120  powerType type = DEFAULT;
121  powerServerType serverType = SERVER_DEFAULT;
122 
123  if (!list) {
124  return 0;
125  }
126 
127  if (IsServerSystem) {
128  if (path == "Plugged In") {
129  serverType = SERVER_USE_ELECTRIC;
130  } else if (path == "On Battery") {
131  serverType = SERVER_USE_BATTERY;
132  }
133 
134  if (serverType > SERVER_DEFAULT && serverType < SERVER_COUNT) {
135  QModelIndex index = list->model()->index(serverType, 0);
136  list->setCurrentIndex(index);
137  list->clicked(index);
138  }
139  } else {
140  if (path == "General") {
141  type = GENERAL;
142  } else if (path == "Plugged In") {
143  type = USE_ELECTRIC;
144  } else if (path == "On Battery") {
145  type = USE_BATTERY;
146  }
147 
148  if (type > DEFAULT && type < COUNT) {
149  QModelIndex index = list->model()->index(type, 0);
150  list->setCurrentIndex(index);
151  list->clicked(index);
152  }
153  }
154  return 0;
155 }

241 void PowerModule::onBatteryPercentageChanged(const double value)
242 {
243  if (!m_model->getDoubleCompare(m_nBatteryPercentage, value)) {
244  m_nBatteryPercentage = value;
245 
246  QString remindData = "";
247  if (m_model->getDoubleCompare(value, 20.0)
248  || m_model->getDoubleCompare(value, 15.0)
249  || m_model->getDoubleCompare(value, 10.0)) {
250  remindData = tr("Battery low, please plug in");
251  } else if (m_model->getDoubleCompare(value, 5.0)) {
252  remindData = tr("Battery critically low");
253  }
254 
255  //send system info
256  if ("" != remindData) {
257  Dtk::Core::DUtil::DNotifySender(remindData.toLatin1().data());
258  }
259  }
