159 void CGraphicsPolygonalStarItem::calcPolygon_helper(QPolygonF &outPolygon, int n, qreal offset) const
160 {
161  if (n == 0)return;
162 
163  outPolygon.clear();
164 
165 // //根据垂直于边线的方向偏移offset得到外圆半径的偏移值finalOffset
166  qreal angle = qDegreesToRadians(90.);
167  QPointF pointCenter = this->rect().center();
168 
169  qreal outerEllipseXDiameter = this->rect().width();
170  qreal outerEllipseYDiameter = this->rect().height();
171 
172  qreal outer_w = outerEllipseXDiameter / 2.0;
173  qreal outer_h = outerEllipseYDiameter / 2.0;
174 
175  //根据外圆和内圆的关系计算内圆的半径
176  qreal inner_w = outer_w * innerRadius() / 100.0;
177  qreal inner_h = outer_h * innerRadius() / 100.0;
178 
179  if (n > 0) {
180  qreal preAngle = qDegreesToRadians(360. / n);
181  qreal innerAngle = angle + preAngle / 2;
182  for (int i = 0; i < n; i++) {
183 
184  qreal outer_Angle = angle + preAngle * i;
185  qreal inner_Angle = innerAngle + preAngle * i;
186 
187  qreal outer_x = pointCenter.x() + outer_w * qCos(outer_Angle);
188  qreal outer_y = pointCenter.y() - outer_h * qSin(outer_Angle);
189 
190 
191  //偶数是外圈点
192  outPolygon.push_back(QPointF(outer_x, outer_y));
193 
194  qreal inner_x = pointCenter.x() + inner_w * qCos(inner_Angle);
195  qreal inner_y = pointCenter.y() - inner_h * qSin(inner_Angle);
196 
197  //奇数是内圈点
198  outPolygon.push_back(QPointF(inner_x, inner_y));
199 
200  }
201  }
202 
203  if (!qFuzzyIsNull(offset)) {
204  QList<QLineF> outlines;
205  auto fGetLines = [ = ](const QPolygonF & outPolygon, QList<QLineF> &resultLines) {
206  resultLines.clear();
207  for (int i = 0; i < outPolygon.size(); ++i) {
208  if (i != 0) {
209  resultLines.append(QLineF(outPolygon.at(i - 1), outPolygon.at(i)));
210  }
211  }
212  resultLines.push_front(QLineF(outPolygon.last(), outPolygon.first()));
213  };
214 
215  fGetLines(outPolygon, outlines);
216 
217  auto fGetOffsetPos = [ = ](const QList<QLineF> &lines, QVector<QPointF> &result) {
218  result.clear();
219  for (int i = 0; i < lines.size(); ++i) {
220  QLineF curLine = lines[i];
221  QLineF nextLine = (i == lines.size() - 1 ? lines[0] : lines[i + 1]);
222 
223  bool isInter = /*(i%2!=0)*/curLine.angleTo(nextLine) > 180; //是否这个线条是和内圈点相关
224  qreal inc = (isInter ? -1 : 1); //内圈点是相反的角度
225 
226  qreal finalDegree = (180 - curLine.angleTo(nextLine)) * inc; //两条线相交的交角*/
227 
228  qreal sinValue = qSin(qDegreesToRadians(finalDegree / 2.));
229 
230  qreal offLen = qFuzzyIsNull(sinValue) ? 0.0 : offset / sinValue;
231  //qDebug()<<"i === "<<i<<"offLen ================ "<<offLen<<"finalDegree = "<<finalDegree;
232 
233  QLineF tempLine(nextLine);
234  qreal newAngle = tempLine.angle() + finalDegree / 2.0 + (isInter ? (360 - curLine.angleTo(nextLine)) : 0);
235  tempLine.setAngle(newAngle);
236  qreal finallenth = qAbs(offLen)/*>tempLine.length()?tempLine.length():qAbs(offLen)*/;
237 
238  if (finallenth > nextLine.length()) {
239  finallenth = nextLine.length();
240  }
241  tempLine.setLength(finallenth);
242  result.append(tempLine.p2());
243  }
244  };
245 
246  QVector<QPointF> outecliPos;
247 
248  fGetOffsetPos(outlines, outecliPos);
249 
250  outPolygon.clear();
251  for (int i = 0; i < outecliPos.size(); ++i) {
252  outPolygon.append(outecliPos[i]);
253  }
254  }
255 }
