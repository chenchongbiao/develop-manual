120 QVariantMap QObjectHelper::qobject2qvariant( const QObject* object,
121  const QStringList& ignoredProperties)
122 {
123  QVariantMap result;
124  const QMetaObject *metaobject = object->metaObject();
125  int count = metaobject->propertyCount();
126  for (int i=0; i<count; ++i) {
127  QMetaProperty metaproperty = metaobject->property(i);
128  const char *name = metaproperty.name();
129 
130  if (ignoredProperties.contains(QLatin1String(name)) || (!metaproperty.isReadable()))
131  continue;
132 
133  QVariant value = object->property(name);
134  result[QLatin1String(name)] = value;
135  }
136  return result;
137 }

162 void QObjectHelper::qvariant2qobject(const QVariantMap& variant, QObject* object)
163 {
164  const QMetaObject *metaobject = object->metaObject();
165 
166  QVariantMap::const_iterator iter;
167  for (iter = variant.constBegin(); iter != variant.constEnd(); ++iter) {
168  int pIdx = metaobject->indexOfProperty(iter.key().toLatin1());
169 
170  if (pIdx < 0) {
171  continue;
172  }
173 
174  QMetaProperty metaproperty = metaobject->property(pIdx);
175  QVariant::Type type = metaproperty.type();
176  QVariant v(iter.value());
177  if (v.canConvert(type)) {
178  v.convert(type);
179  metaproperty.write(object, v);
180  }else if (QString(QLatin1String("QVariant")).compare(QLatin1String(metaproperty.typeName())) == 0) {
181  metaproperty.write(object, v);
182  }
183  }
184 }
