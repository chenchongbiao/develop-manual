97 int SystemInfoModule::load(const QString &path)
98 {
99  if (!m_sysinfoWidget) {
100  active();
101  }
102 
103  QListView *list = m_sysinfoWidget->getSystemListViewPointer();
104  SystemType type = Default;
105 
106  if (!list) {
107  return 0;
108  }
109 
110  if (path == "About This PC") {
111  type = AboutThisPC;
112  } else if (path == "Edition License") {
113  type = EditionLicense;
114  } else if (path == "End User License Agreement") {
115  type = EndUserLicenseAgreement;
116  } else if (path == "Privacy Policy") {
117  type = PrivacyPolicy;
118  }
119 
120  QModelIndex index = list->model()->index(type, 0);
121  if (type > Default && type < MaxType) {
122  index = list->model()->index(type, 0);
123  list->setCurrentIndex(index);
124  list->clicked(index);
125  }
126 
127  return 0;
128 }
