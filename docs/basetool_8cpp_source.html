11 static float codecConfidenceForData(const QTextCodec *codec, const QByteArray &data, const QLocale::Country &country)
12 {
13  qreal hep_count = 0;
14  int non_base_latin_count = 0;
15  qreal unidentification_count = 0;
16  int replacement_count = 0;
17 
18  QTextDecoder decoder(codec);
19  const QString &unicode_data = decoder.toUnicode(data);
20 
21  for (int i = 0; i < unicode_data.size(); ++i) {
22  const QChar &ch = unicode_data.at(i);
23 
24  if (ch.unicode() > 0x7f)
25  ++non_base_latin_count;
26 
27  switch (ch.script()) {
28  case QChar::Script_Hiragana:
29  case QChar::Script_Katakana:
30  hep_count += (country == QLocale::Japan) ? 1.2 : 0.5;
31  unidentification_count += (country == QLocale::Japan) ? 0 : 0.3;
32  break;
33  case QChar::Script_Han:
34  hep_count += (country == QLocale::China) ? 1.2 : 0.5;
35  unidentification_count += (country == QLocale::China) ? 0 : 0.3;
36  break;
37  case QChar::Script_Hangul:
38  hep_count += (country == QLocale::NorthKorea) || (country == QLocale::SouthKorea) ? 1.2 : 0.5;
39  unidentification_count += (country == QLocale::NorthKorea) || (country == QLocale::SouthKorea) ? 0 : 0.3;
40  break;
41  case QChar::Script_Cyrillic:
42  hep_count += (country == QLocale::Russia) ? 1.2 : 0.5;
43  unidentification_count += (country == QLocale::Russia) ? 0 : 0.3;
44  break;
45  case QChar::Script_Devanagari:
46  hep_count += (country == QLocale::Nepal || country == QLocale::India) ? 1.2 : 0.5;
47  unidentification_count += (country == QLocale::Nepal || country == QLocale::India) ? 0 : 0.3;
48  break;
49  default:
50  // full-width character, emoji, 常用标点, 拉丁文补充1，天城文补充，CJK符号和标点符号（如：【】）
51  if ((ch.unicode() >= 0xff00 && ch <= 0xffef)
52  || (ch.unicode() >= 0x2600 && ch.unicode() <= 0x27ff)
53  || (ch.unicode() >= 0x2000 && ch.unicode() <= 0x206f)
54  || (ch.unicode() >= 0x80 && ch.unicode() <= 0xff)
55  || (ch.unicode() >= 0xa8e0 && ch.unicode() <= 0xa8ff)
56  || (ch.unicode() >= 0x0900 && ch.unicode() <= 0x097f)
57  || (ch.unicode() >= 0x3000 && ch.unicode() <= 0x303f)) {
58  ++hep_count;
59  } else if (ch.isSurrogate() && ch.isHighSurrogate()) {
60  ++i;
61 
62  if (i < unicode_data.size()) {
63  const QChar &next_ch = unicode_data.at(i);
64 
65  if (!next_ch.isLowSurrogate()) {
66  --i;
67  break;
68  }
69 
70  uint unicode = QChar::surrogateToUcs4(ch, next_ch);
71 
72  // emoji
73  if (unicode >= 0x1f000 && unicode <= 0x1f6ff) {
74  hep_count += 2;
75  }
76  }
77  } else if (ch.unicode() == QChar::ReplacementCharacter) {
78  ++replacement_count;
79  } else if (ch.unicode() > 0x7f) {
80  // 因为UTF-8编码的容错性很低，所以未识别的编码只需要判断是否为 QChar::ReplacementCharacter 就能排除
81  if (codec->name() != "UTF-8")
82  ++unidentification_count;
83  }
84  break;
85  }
86  }
87 
88  float c = static_cast<float>(qreal(hep_count) / non_base_latin_count / 1.2);
89 
90  c -= static_cast<float>(qreal(replacement_count) / non_base_latin_count);
91  c -= static_cast<float>(qreal(unidentification_count) / non_base_latin_count);
92 
93  return qMax(0.0f, c);
94 }
