204 bool FileDialogStatusBar::eventFilter(QObject *watched, QEvent *event)
205 {
206  if (watched != m_fileNameEdit)
207  return false;
208 
209  if (event->type() == QEvent::FocusIn) {
210  TIMER_SINGLESHOT_OBJECT(this, 10, {
211  QMimeDatabase db;
212  const QString &name = m_fileNameEdit->text();
213  const QString &suffix = db.suffixForFileName(name);
214 
215  if (suffix.isEmpty())
216  m_fileNameEdit->selectAll();
217  else
218  m_fileNameEdit->setSelection(0, name.length() - suffix.length() - 1);
219  }, this)
220  } else if (event->type() == QEvent::Show) {
221  TIMER_SINGLESHOT_OBJECT(this, 500, m_fileNameEdit->setFocus(), this);
222  }
223 
224  return false;
225 }
