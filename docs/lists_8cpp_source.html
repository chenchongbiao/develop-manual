60  unsigned int createFakeLSID( const Word97::ANLD& anld )
61  {
62  // As our structure contains padding bytes we have to use
63  // an array which is guaranteed not to have "holes" for the
64  // CRC32 sum...
65  unsigned char buffer[102];
66  buffer[0] = anld.nfc;
67  buffer[1] = anld.cxchTextBefore;
68  buffer[2] = anld.cxchTextAfter;
69  buffer[3] = anld.jc;
70  buffer[4] = anld.fPrev;
71  buffer[5] = anld.fHang;
72  buffer[6] = anld.fSetBold;
73  buffer[7] = anld.fSetItalic;
74  buffer[8] = anld.fSetSmallCaps;
75  buffer[9] = anld.fSetCaps;
76  buffer[10] = anld.fSetStrike;
77  buffer[11] = anld.fSetKul;
78  buffer[12] = anld.fPrevSpace;
79  buffer[13] = anld.fBold;
80  buffer[14] = anld.fItalic;
81  buffer[15] = anld.fSmallCaps;
82  buffer[16] = anld.fCaps;
83  buffer[17] = anld.fStrike;
84  buffer[18] = anld.kul;
85  buffer[19] = anld.ico;
86  unsigned short* tmp = reinterpret_cast<unsigned short*>( buffer ) + 10;
87  *tmp++ = anld.ftc;
88  *tmp++ = anld.hps;
89  *tmp++ = anld.iStartAt;
90  *tmp++ = anld.dxaIndent;
91  *tmp++ = anld.dxaSpace;
92  buffer[30] = anld.fNumber1;
93  buffer[31] = anld.fNumberAcross;
94  buffer[32] = anld.fRestartHdn;
95  buffer[33] = anld.fSpareX;
96  tmp += 2;
97  for ( int i = 0; i < 32; ++i )
98  *tmp++ = anld.rgxch[i];
99  // 98 bytes up to this point + 4 for the CRC32 sum
100  buffer[98] = 0;
101  buffer[99] = 0;
102  buffer[100] = 0;
103  buffer[101] = 0;
104  return CalcCRC32( buffer, 102, 98, 4 );
105  }

228 ListLevel::ListLevel( AbstractOLEStreamReader* tableStream ) :
229  m_lvlf( tableStream, false ), m_grpprlPapx( 0 ), m_grpprlChpx( 0 )
230 {
231 #ifdef WV2_DEBUG_LIST_READING
232  wvlog << " ListLevel::ListLevel() ######" << std::endl
233  << " iStartAt=" << static_cast<int>( m_lvlf.iStartAt ) << " nfc=" << static_cast<int>( m_lvlf.nfc )
234  << " jc=" << static_cast<int>( m_lvlf.jc ) << std::endl << " fLegal=" << static_cast<int>( m_lvlf.fLegal )
235  << " fNoRestart=" << static_cast<int>( m_lvlf.fNoRestart ) << " fPrev=" << static_cast<int>( m_lvlf.fPrev )
236  << std::endl << " fPrevSpace=" << static_cast<int>( m_lvlf.fPrevSpace ) << " fWord6="
237  << static_cast<int>( m_lvlf.fWord6 ) << std::endl << " cbGrpprlPapx=" << static_cast<int>( m_lvlf.cbGrpprlPapx )
238  << " cbGrpprlChpx=" << static_cast<int>( m_lvlf.cbGrpprlChpx ) << std::endl;
239 #endif
240 
241  if ( m_lvlf.cbGrpprlPapx != 0 ) {
242  m_grpprlPapx = new U8[ m_lvlf.cbGrpprlPapx ];
243  tableStream->read( m_grpprlPapx, m_lvlf.cbGrpprlPapx );
244  }
245  if ( m_lvlf.cbGrpprlChpx != 0 ) {
246  m_grpprlChpx = new U8[ m_lvlf.cbGrpprlChpx ];
247  tableStream->read( m_grpprlChpx, m_lvlf.cbGrpprlChpx );
248  }
249 
250  U16 len = tableStream->readU16();
251  if ( len != 0 ) {
252  XCHAR* string = new XCHAR[ len ];
253  for ( int i = 0; i < len; ++i )
254  string[ i ] = tableStream->readU16();
255  m_numberText = UString( reinterpret_cast<UChar*>( string ), len );
256  delete[] string;
257  }
258 }

260 ListLevel::ListLevel( const Word97::ANLD& anld ) : m_grpprlPapx( 0 ), m_grpprlChpx( 0 )
261 {
262  m_lvlf.iStartAt = anld.iStartAt;
263  m_lvlf.nfc = anld.nfc;
264  m_lvlf.jc = anld.jc;
265  m_lvlf.fPrev = anld.fPrev;
266  m_lvlf.fPrevSpace = anld.fPrevSpace;
267  m_lvlf.fWord6 = true;
268  m_lvlf.dxaSpace = anld.dxaSpace;
269  m_lvlf.dxaIndent = anld.dxaIndent;
270 
271  // number text
272 #ifdef WV2_DEBUG_LIST_PROCESSING
273  wvlog << "cxchTextBefore=" << static_cast<int>( anld.cxchTextBefore ) << " cxchTextAfter="
274  << static_cast<int>( anld.cxchTextAfter ) << std::endl;
275 #endif
276  if ( anld.cxchTextBefore > 0 && anld.cxchTextBefore <= 32 ) {
277  m_numberText = UString( reinterpret_cast<const UChar*>( &anld.rgxch[ 0 ] ), anld.cxchTextBefore );
278 #ifdef WV2_DEBUG_LIST_PROCESSING
279  wvlog << "String (before): '" << m_numberText.ascii() << "'" << std::endl;
280 #endif
281  }
282  m_numberText += UString( static_cast<char>( 0 ) ); // we are faking list level 0, so we have
283  // to insert the "variable"
284  if ( anld.cxchTextAfter > 0 && anld.cxchTextAfter <= 32 ) {
285  int start = anld.cxchTextAfter > anld.cxchTextBefore ? anld.cxchTextBefore : 0;
286  m_numberText += UString( reinterpret_cast<const UChar*>( &anld.rgxch[ start ] ), anld.cxchTextAfter - start );
287 #ifdef WV2_DEBUG_LIST_PROCESSING
288  wvlog << "String (after): '" << UString( reinterpret_cast<const UChar*>( &anld.rgxch[ start ] ), anld.cxchTextAfter - start ).ascii() << "'" << std::endl;
289 #endif
290  }
291 
292  // grpprls
293  // ###### TODO: PAPX
294  m_lvlf.cbGrpprlPapx = 0;
295 
296  // Allocate enough for all sprms and set the cb as limit
297  m_grpprlChpx = new U8[ 29 ];
298  U8* grpprl = m_grpprlChpx;
299  m_lvlf.cbGrpprlChpx = 0;
300 
301  if ( anld.fSetBold )
302  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCFBold, anld.fBold, &grpprl );
303  if ( anld.fSetItalic )
304  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCFItalic, anld.fItalic, &grpprl );
305  if ( anld.fSetSmallCaps )
306  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCFSmallCaps, anld.fSmallCaps, &grpprl );
307  if ( anld.fSetCaps )
308  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCFCaps, anld.fCaps, &grpprl );
309  if ( anld.fSetStrike )
310  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCFStrike, anld.fStrike, &grpprl );
311  if ( anld.fSetKul )
312  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCKul, anld.kul, &grpprl );
313 
314  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCIco, anld.ico, &grpprl );
315  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCRgFtc0, static_cast<U16>( anld.ftc ), &grpprl );
316  m_lvlf.cbGrpprlChpx += writeCharProperty( sprmCHps, anld.hps, &grpprl );
317 
318 #ifdef WV2_DEBUG_LIST_PROCESSING
319  wvlog << "The CHPX is " << static_cast<int>( m_lvlf.cbGrpprlChpx ) << " bytes long" << std::endl;
320 #endif
321 }

594 ListInfo::ListInfo( Word97::PAP& pap, ListInfoProvider& listInfoProvider ) :
595  m_linkedIstd( istdNil ), m_restartingCounter( false ), m_numberFormat( 0 ),
596  m_alignment( 0 ), m_isLegal( false ), m_notRestarted( false ), m_prev( false ),
597  m_prevSpace( false ), m_isWord6( false ), m_followingChar( 0 ), m_lsid( 0 )
598 {
599  if ( !listInfoProvider.setPAP( &pap ) )
600  return;
601  const ListLevel* const level = listInfoProvider.formattingListLevel();
602  const ListData* const listData = listInfoProvider.m_currentLst;
603 
604  if ( listData ) {
605  m_linkedIstd = listData->istdForLevel( pap.ilvl );
606  m_restartingCounter = listData->restartingCounter();
607  m_lsid = listData->lsid();
608  }
609  else
610  wvlog << "Bug: The ListData is 0!!" << std::endl;
611 
612  m_startAt = listInfoProvider.startAt();
613 
614  if ( level ) {
615  m_numberFormat = level->numberFormat();
616  m_alignment = level->alignment();
617  m_isLegal = level->isLegal();
618  m_notRestarted = level->notRestarted();

848 void ListInfoProvider::convertCompatANLD()
849 {
850  // We are creating a fake "unique" list ID via a CRC32 sum.
851  // It should be rather unlikely that two different ANLDs
852  // have the same lsid. In the case that the lsid is already
853  // there we assume that we already hit that ANLD before and
854  // just use its formatting information right away.
855  // There's one problem with that approach: If you have a Word 97
856  // document using some compat-lists and the lsid the CRC algorithm
857  // creates is already there, we end up using the wrong formatting
858  // information. I'm sure it's more likely to win in the lottery
859  // than facing that situation, though :-)
860  // Note that compat lists are always simple lists. (ilvl == 0)
861  const S32 lsid = createFakeLSID( m_pap->anld );
862  m_pap->ilvl = 0;
863 #ifdef WV2_DEBUG_LIST_PROCESSING
864  wvlog << " ListInfoProvider::convertCompatANLD" << std::endl
865  << " generated a unique lsid: " << lsid << std::endl;
866 #endif
867 
868  // Does it already exist?
869  std::vector<ListData*>::const_iterator it = m_listData.begin();
870  std::vector<ListData*>::const_iterator end = m_listData.end();
871  for ( int i = 1; it != end; ++it, ++i ) // 1-based index!
872  if ( ( *it )->lsid() == lsid ) {
873  m_pap->ilfo = i;
874  return;
875  }
876 
877  // It's not there yet...
878  // Fake a LFO structure (with 0 overridden levels and the lsid we generated)
879  m_listFormatOverride.push_back( new ListFormatOverride( lsid ) );
880  m_pap->ilfo = m_listFormatOverride.size();
881 
882  // And another fake, the LSTF
883  ListData* listData = new ListData( lsid, m_pap->anld.fRestartHdn );
884  listData->appendListLevel( new ListLevel( m_pap->anld ) );
885  m_listData.push_back( listData );
886 }
