70 void CSelectTool::toolStart(IDrawTool::CDrawToolEvent *event, ITERecordInfo *pInfo)
71 {
72  _hightLight = QPainterPath();
73 
74  CGraphicsItem *pStartPosTopBzItem = pInfo->startPosTopBzItem;
75 
76  QGraphicsItem *pFirstItem = pInfo->startPosItems.isEmpty() ? nullptr : pInfo->startPosItems.first();
77  bool isMrNodeItem = event->scene()->isBussizeHandleNodeItem(pFirstItem) && (event->scene()->getAssociatedBzItem(pFirstItem)->type() == MgrType);
78 
79  bool clearBeforeSelect = true;
80 
81  if (event->keyboardModifiers() == Qt::ShiftModifier) {
82  // 点住shift那么不用清除
83  clearBeforeSelect = false;
84 
85  //首先应该要满足当前鼠标点击下的图元不是节点
86  if (!isMrNodeItem) {
87  //如果点击的图元当前是选中的,那么在按住shift的情况下应该是清除它的选中
88  if (pStartPosTopBzItem != nullptr) {
89  CGraphicsItem *pProxItem = pStartPosTopBzItem->thisBzProxyItem(true);
90  if (pProxItem->isSelected()) {
91  if (event->scene()->selectGroup()->count() > 1) {
92  event->scene()->notSelectItem(pProxItem);
93  return;
94  }
95  }
96  }
97  }
98  } else {
99  if (isMrNodeItem) {
100  clearBeforeSelect = false;
101  } else if (pStartPosTopBzItem != nullptr) {
102  CGraphicsItem *pProxItem = pStartPosTopBzItem->thisBzProxyItem(true);
103  if (pProxItem->isSelected()) {
104  //其代理图元(自身或者其顶层组合图元)被选中那么不用清除
105  clearBeforeSelect = false;
106  }
107  }
108  }
109  //清理当前选中
110  if (clearBeforeSelect) {
111  event->scene()->clearSelectGroup();
112  }
113 
114  if (pStartPosTopBzItem != nullptr) {
115  if (!isMrNodeItem) {
116  event->scene()->selectItem(pStartPosTopBzItem->thisBzProxyItem());
117  }
118  } else {
119  if (!isMrNodeItem) {
120  //点击处是空白的那么清理所有选中
121  event->scene()->clearSelectGroup();
122  }
123  }
124 }

126 void CSelectTool::toolUpdate(IDrawTool::CDrawToolEvent *event, ITERecordInfo *pInfo)
127 {
128  event->setAccepted(false);
129 
130  //根据要做的类型去执行相应的操作
131  switch (pInfo->_opeTpUpdate) {
132  case ERectSelect: {
133  event->scene()->update();
134  break;
135  }
136  case EDragSceneMove: {
137  //拖拽画布移动
138  QPointF pos0 = pInfo->_startPos;
139  QPointF pos1 = event->pos();
140  QPointF mov = pos1 - pos0;
141 
142  QScrollBar *horBar = event->view()->horizontalScrollBar();
143  int horValue = horBar->value() - qRound(mov.x());
144  horBar->setValue(qMin(qMax(horBar->minimum(), horValue), horBar->maximum()));
145 
146  QScrollBar *verBar = event->view()->verticalScrollBar();
147  int verValue = verBar->value() - qRound(mov.y());
148  verBar->setValue(qMin(qMax(verBar->minimum(), verValue), verBar->maximum()));
149 
150  break;
151  }
152  case EDragMove:
153  case ECopyMove: {
154  //执行移动操作
155  processItemsMove(event, pInfo, EChangedUpdate);
156  break;
157  }
158  case EResizeMove: {
159  //交给图元去完成
160  processItemsScal(event, pInfo, EChangedUpdate);
161  break;
162  }
163  case ERotateMove: {
164  if (!pInfo->etcItems.isEmpty()) {
165  QGraphicsItem *pItem = !pInfo->etcItems.isEmpty() ? pInfo->etcItems.first() : nullptr;
166  CGraphicsItem *pMrItem = dynamic_cast<CGraphicsItem *>(pItem);
167  processItemsRot(event, pInfo, EChangedUpdate);
168  drawApp->setApplicationCursor(pMrItem->handleNode()->getCursor());
169  }
170  break;
171  }
172  default:
173  break;
174  }
175 }

177 void CSelectTool::toolFinish(IDrawTool::CDrawToolEvent *event, ITERecordInfo *pInfo)
178 {
179  bool doUndoFinish = true;
180  switch (pInfo->_opeTpUpdate) {
181  case ERectSelect: {
182  QPointF pos0 = pInfo->_startPos;
183  QPointF pos1 = event->pos();
184 
185  QPointF topLeft(qMin(pos0.x(), pos1.x()), qMin(pos0.y(), pos1.y()));
186  QPointF bomRight(qMax(pos0.x(), pos1.x()), qMax(pos0.y(), pos1.y()));
187 
188  event->scene()->selectItemsByRect(QRectF(topLeft, bomRight));
189 
190  event->scene()->update();
191 
192  doUndoFinish = false;
193  break;
194  }
195  case EDragMove: {
196  event->scene()->recordItemsInfoToCmd(event->scene()->selectGroup()->items(true), RedoVar);
197  m_isItemMoving = false;
198  break;
199  }
200  case EResizeMove: {
201  //记录Redo点
202  doUndoFinish = (event->view()->activeProxWidget() == nullptr);
203  if (doUndoFinish)
204  event->scene()->recordItemsInfoToCmd(event->scene()->selectGroup()->items(true), RedoVar);
205 
206  break;
207  }
208  case ECopyMove: {
209  event->scene()->recordSecenInfoToCmd(CSceneUndoRedoCommand::EGroupChanged, RedoVar, event->scene()->selectGroup()->items());
210  m_isItemMoving = false;
211  break;
212  }
213  case ERotateMove: {
214  event->scene()->recordItemsInfoToCmd(event->scene()->selectGroup()->items(true), RedoVar);
215  break;
216  }
217  default:
218  return;
219  }
220 
221  QList<CGraphicsItem *> items = event->scene()->selectGroup()->getBzItems(true);
222  if (!items.isEmpty()) {
223  QGraphicsItem *pItem = items.first();
224  if (pItem != nullptr && pItem->type() == PictureType) {
225  event->scene()->selectGroup()->updateAttributes();
226  }
227  }
228 
229  //入栈
230  if (doUndoFinish)
231  CUndoRedoCommand::finishRecord();
232 
233  IDrawTool::toolFinish(event, pInfo);
234 }

253 int CSelectTool::decideUpdate(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *pInfo)
254 {
255  int tpye = ENothingDo;
256  if (isActived()) {
257  QGraphicsItem *pStartPosTopQtItem = event->scene()->firstItem(pInfo->_startPos,
258  pInfo->startPosItems,
259  true, true);
260  if (pStartPosTopQtItem == nullptr) {
261  if (event->eventType() == CDrawToolEvent::ETouchEvent) {
262  if (pInfo->elapsedFromStartToUpdate() > 200) {
263  tpye = ERectSelect;
264  } else {
265  tpye = EDragSceneMove;
266  }
267  } else {
268  tpye = ERectSelect;
269  }
270  } else {
271  if (event->scene()->isBussizeItem(pStartPosTopQtItem)) {
272  // 业务图元可执行移动可执行复制
273  if (event->keyboardModifiers() == Qt::AltModifier) {
274  //复制移动
275  tpye = ECopyMove;
276  if (event->scene()->selectGroup()->count() > 0) {
277 
278  //0.复制前得到当前场景的组合快照(用于Undo撤销)
279  event->scene()->recordSecenInfoToCmd(CSceneUndoRedoCommand::EGroupChanged, UndoVar);
280 
281  //1.复制出一样的一个组合图元(这里复制的是选择组合图元,之后应该删掉它)
282  CGraphicsItemGroup *pNewGroup = CDrawScene::copyCreatGroup(event->scene()->selectGroup());
283 
284  QList<CGraphicsItem *> needSelectItems;
285  if (pNewGroup != nullptr) {
286  //2.获取到拷贝出来的图元
287  QList<CGraphicsItem *> outItems = pNewGroup->items();
288  QList<CGraphicsItem *> newBzItems = pNewGroup->getBzItems(true);
289 
290  //3.删掉复制出来的临时用于选择管理的组合,即pNewGroup
291  event->scene()->destoryGroup(pNewGroup, true);
292  pNewGroup = nullptr;
293 
294  //4.记录增加的基本业务图元(用于撤销还原)
295  QList<QVariant> vars;
296  vars << reinterpret_cast<long long>(event->scene());
297  for (auto p : newBzItems) {
298  vars << reinterpret_cast<long long>(p);
299  }
300  CUndoRedoCommand::recordUndoCommand(CUndoRedoCommand::ESceneChangedCmd,
301  CSceneUndoRedoCommand::EItemAdded, vars, false, true);
302 
303  needSelectItems = outItems;
304  }
305 
306  //5.复制完成后记录场景当前组合的快照(用于还原)这要在toolFinished再调用
307  //event->scene()->recordSecenInfoToCmd(CSceneUndoRedoCommand::EGroupChanged, RedoVar);
308 
309  //6.选中被复制出来的图元
310  event->scene()->clearSelectGroup();
311  for (auto p : needSelectItems) {
312  event->scene()->selectItem(p, true, false, false);
313  }
314  event->scene()->selectGroup()->updateBoundingRect();
315  event->scene()->selectGroup()->updateAttributes();
316  pInfo->etcItems.append(event->scene()->selectGroup());
317  }
318  } else {
319  tpye = EDragMove;
320  pInfo->etcItems.append(event->scene()->selectGroup());
321  QList<CGraphicsItem *> lists = event->scene()->selectGroup()->items(true);
322  event->scene()->recordItemsInfoToCmd(lists, UndoVar);
323  }
324  m_isItemMoving = true;
325  } else if (event->scene()->isBussizeHandleNodeItem(pStartPosTopQtItem)) {
326  CSizeHandleRect *pHandle = dynamic_cast<CSizeHandleRect *>(pStartPosTopQtItem);
327  pInfo->_etcopeTpUpdate = pHandle->dir();
328  pInfo->etcItems.clear();
329 
330  pInfo->etcItems.append(event->scene()->selectGroup());
331 
332  //记录undo点
333  if (event->view()->activeProxWidget() == nullptr)
334  event->scene()->recordItemsInfoToCmd(event->scene()->selectGroup()->items(true), UndoVar, true);
335 
336  tpye = (pHandle->dir() != CSizeHandleRect::Rotation ? EResizeMove : ERotateMove);
337  }
338  }
339  }
340  return tpye;
341 }

349 void CSelectTool::drawMore(QPainter *painter,
350  const QRectF &rect,
351  CDrawScene *scene)
352 {
353  //注意painter是在scene的坐标系
354 
355  Q_UNUSED(rect)
356  painter->save();
357  for (auto it = _allITERecordInfo.begin();
358  it != _allITERecordInfo.end(); ++it) {
359  ITERecordInfo &info = it.value();
360  if (info._opeTpUpdate == ERectSelect) {
361  QPointF pos0 = info._startPos;
362  QPointF pos1 = info._curEvent.pos();
363 
364  QPointF topLeft(qMin(pos0.x(), pos1.x()), qMin(pos0.y(), pos1.y()));
365  QPointF bomRight(qMax(pos0.x(), pos1.x()), qMax(pos0.y(), pos1.y()));
366 
367  QPen pen;
368  pen.setWidthF(0.5);
369  QBrush selectBrush = drawApp->systemThemeColor();
370  QColor selectColor = selectBrush.color();
371  selectColor.setAlpha(20);
372  selectBrush.setColor(selectColor);
373  selectColor.setAlpha(100);
374  pen.setColor(selectColor);
375 
376  painter->setPen(pen);
377  painter->setBrush(selectBrush);
378  painter->drawRect(QRectF(topLeft, bomRight));
379  } else if (info._opeTpUpdate == ERotateMove) {
380 
381  painter->setClipping(false);
382  QPoint posInView = scene->drawView()->viewport()->mapFromGlobal(QCursor::pos());
383  QPointF posInScene = scene->drawView()->mapToScene(posInView);
384 
385  qreal scled = scene->drawView()->getScale();
386  QPointF paintPos = posInScene + QPointF(50 / scled, 0);
387 
388  qreal rote = scene->selectGroup()->count() == 0 ? 0 : scene->selectGroup()->items().first()->drawRotation();
389  QString angle = QString("%1°").arg(QString::number(rote, 'f', 1));
390  QFont f;
391  f.setPointSizeF(11 / scled);
392 
393  QFontMetrics fontMetrics(f);
394  int width = fontMetrics.width(angle);
395  QRectF rotateRect(paintPos, paintPos + QPointF(width, fontMetrics.height()));
396 
397  painter->setPen(Qt::NoPen);
398  painter->setBrush(QColor("#E5E5E5"));
399  painter->drawRoundRect(rotateRect);
400  painter->setFont(f);
401  painter->setPen(Qt::black);
402  painter->drawText(rotateRect, Qt::AlignCenter, angle);
403  painter->setClipping(true);
404  }
405  }
406  painter->restore();
407 }

455 void CSelectTool::processItemsScal(IDrawTool::CDrawToolEvent *event, IDrawTool::ITERecordInfo *info, EChangedPhase phase)
456 {
457  CGraphItemScalEvent scal(CGraphItemEvent::EScal);
458  scal.setEventPhase(phase);
459  scal.setToolEventType(info->_opeTpUpdate);
460  scal.setPressedDirection(info->_etcopeTpUpdate);
461  scal._scenePos = event->pos();
462  scal._oldScenePos = info->_prePos;
463  scal._sceneBeginPos = info->_startPos;
464 
465  bool xBlock = false;
466  bool yBlock = false;
467  CSizeHandleRect::EDirection dir = CSizeHandleRect::EDirection(info->_etcopeTpUpdate);
468  CSizeHandleRect::getTransBlockFlag(dir, xBlock, yBlock);
469  scal.setXTransBlocked(xBlock);
470  scal.setYTransBlocked(yBlock);
471 
472  bool xNegitiveOffset = false;
473  bool yNegitiveOffset = false;
474  CSizeHandleRect::getTransNegtiveFlag(dir, xNegitiveOffset, yNegitiveOffset);
475  scal.setXNegtiveOffset(xNegitiveOffset);
476  scal.setYNegtiveOffset(yNegitiveOffset);
477  scal.setKeepOrgRadio(event->keyboardModifiers() & Qt::ShiftModifier);
478 
479  //分发事件
480  for (auto item : info->etcItems) {
481  if (CDrawScene::isDrawItem(item) || item == event->scene()->selectGroup()) {
482  CGraphicsItem *pBzItem = static_cast<CGraphicsItem *>(item);
483 
484  scal.setPos(pBzItem->mapFromScene(event->pos()));
485  scal.setOldPos(pBzItem->mapFromScene(info->_prePos));
486  scal.setOrgSize(pBzItem->rect().size());
487  scal.setCenterPos((event->keyboardModifiers() & Qt::AltModifier) ? pBzItem->rect().center() :
488  CSizeHandleRect::transCenter(dir, pBzItem));
489  scal._sceneCenterPos = pBzItem->mapToScene(scal.centerPos());
490 
491  if (phase == EChangedBegin) {
492  pBzItem->operatingBegin(&scal);
493  } else if (phase == EChangedUpdate) {
494  pBzItem->operating(&scal);
495  } else if (phase == EChangedFinished) {
496  pBzItem->operatingEnd(&scal);
497  }
498  }
499  }
500  event->view()->viewport()->update();
501 }
