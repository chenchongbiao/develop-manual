141 bool CSizeHandleRect::isFatherDragging()
142 {
143  CGraphicsItem *pParentItem = dynamic_cast<CGraphicsItem *>(parentItem());
144  if (pParentItem != nullptr && pParentItem->scene() != nullptr) {
145  EDrawToolMode model = pParentItem->drawScene()->getDrawParam()->getCurrentDrawToolMode();
146  int operatingTp = pParentItem->operatingType();
147  if ((operatingTp == CSelectTool::EDragMove || operatingTp == CSelectTool::ECopyMove) && model == selection) {
148  return true;
149  }
150  }
151  return false;
152 }

230 QCursor CSizeHandleRect::getCursor()
231 {
232  static QPixmap m_RotateCursor(QPixmap(":/theme/light/images/mouse_style/rotate_mouse.svg").scaled(24, 24, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
233 
234  static QPixmap m_LeftTopCursor(QPixmap(":/theme/light/images/mouse_style/icon_drag_leftup.svg").scaled(24, 24, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
235 
236  static QPixmap m_RightTopCursor(QPixmap(":/theme/light/images/mouse_style/icon_drag_rightup.svg").scaled(24, 24, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
237 
238  static QPixmap m_LeftRightCursor(QPixmap(":/theme/light/images/mouse_style/icon_drag_left.svg").scaled(24, 24, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
239  static QPixmap m_UpDownCursor(QPixmap(":/theme/light/images/mouse_style/icon_drag_up.svg").scaled(24, 24, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
240 
241  QCursor cursorResult(Qt::ArrowCursor);
242  QMatrix matrix;
243  CGraphicsItem *parent = /*parentItem()*/dynamic_cast<CGraphicsItem *>(parentItem());
244  qreal rotaAngle = (parent == nullptr ? 0 : parent->drawRotation());
245  matrix.rotate(rotaAngle);
246 
247  switch (m_dir) {
248  case Right:
249  case Left:
250  cursorResult = QCursor(m_LeftRightCursor.transformed(matrix, Qt::SmoothTransformation));
251  break;
252  case Top:
253  case Bottom:
254  cursorResult = QCursor(m_UpDownCursor.transformed(matrix, Qt::SmoothTransformation));
255  break;
256  case RightTop:
257  case LeftBottom:
258  cursorResult = QCursor(m_RightTopCursor.transformed(matrix, Qt::SmoothTransformation));
259  break;
260  case LeftTop:
261  case RightBottom:
262  cursorResult = QCursor(m_LeftTopCursor.transformed(matrix, Qt::SmoothTransformation));
263  break;
264 
265  case Rotation: {
266  cursorResult = m_RotateCursor;
267  //matrix.rotate(this->rotation());
268  QPixmap pixmap = cursorResult.pixmap().transformed(matrix, Qt::SmoothTransformation);
269  cursorResult = QCursor(pixmap);
270  break;
271  }
272  default:
273  break;
274  }
275 
276  return cursorResult;
277 }
