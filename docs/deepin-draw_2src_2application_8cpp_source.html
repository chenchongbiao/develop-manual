198 QStringList Application::getRightFiles(const QStringList &files, bool notice)
199 {
200  //过滤文件
201  //判断文件是否是可读或者可写的(图片判断是否可读 ddf判断是否可读可写)
202  Application::QFileClassedMap outFiles;
203  QStringList paths = this->doFileClassification(files, outFiles);
204 
205  outFiles.remove(Application::EDrawAppSup);
206 
207  for (auto it = outFiles.begin(); it != outFiles.end(); ++it) {
208 
209  Application::EFileClassEnum classTp = it.key();
210 
211  QStringList &problemFiles = it.value();
212 
213  if (!problemFiles.isEmpty() && notice) {
214 
215  //提示有文件的有问题的文件集(为什么检查是否关闭程序的标记为是否等于第一个key而不是最后一个key呢？因为最上层弹窗是最后一次noticeFileRightProblem)
216  //所以用户点击关闭的最后一个弹窗是最先执行的noticeFileRightProblem，所以要在用户关闭完所有的弹窗后再判断是否需要推出程序。
217  QMetaObject::invokeMethod(this, "noticeFileRightProblem", Qt::QueuedConnection,
218  Q_ARG(const QStringList &, problemFiles),
219  Q_ARG(Application::EFileClassEnum, classTp),
220  Q_ARG(bool, classTp == outFiles.firstKey()));
221 
222  }
223  }
224  return paths;
225 }

227 QStringList Application::doFileClassification(const QStringList &inFilesPath, Application::QFileClassedMap &out)
228 {
229  out.clear();
230 
231  QStringList supFiles;
232  for (int i = 0; i < inFilesPath.size(); ++i) {
233  QString path = inFilesPath.at(i);
234 
235  if (!isFileExist(path)) {
236  out[ENotExist].append(path);
237  continue;
238  }
239 
240  QFileInfo info(path);
241  if (info.isFile()) {
242  const QString suffix = info.suffix().toLower();
243  if (supPictureSuffix().contains(suffix) || supDdfStuffix().contains(suffix)) {
244  out[EDrawAppSup].append(path);
245  if (!info.isReadable()) {
246  out[EDrawAppSupButNotReadable].append(path);
247  } else {
248  out[EDrawAppSupAndReadable].append(path);
249  supFiles.append(path);
250  }
251  } else {
252  out[EDrawAppNotSup].append(path);
253  }
254  } else {
255  out[ENotFile].append(path);
256  }
257  }
258  return supFiles;
259 }

314 bool Application::isFileNameLegal(const QString &path, int *outErrorReson)
315 {
316  if (path.isEmpty()) {
317  if (outErrorReson != nullptr) {
318  *outErrorReson = 0;
319  }
320  return false;
321  }
322 
323  QRegExp regExp("[:\\*\\?\"<>\\|]");
324 
325  if (path.contains(regExp)) {
326 
327  if (outErrorReson != nullptr) {
328  *outErrorReson = 1;
329  }
330  return false;
331  }
332 
333  QRegExp splitExp("[/\\\\]");
334 
335 
336  int pos = splitExp.indexIn(path, 0);
337 
338  while (pos != -1) {
339  QString dirStr = path.left(pos + 1);
340  if (dirStr.count() > 1) {
341  QDir dir(dirStr);
342  if (!dir.exists()) {
343  if (outErrorReson != nullptr) {
344  *outErrorReson = 2;
345  }
346  return false;
347  }
348  }
349  pos = splitExp.indexIn(path, pos + 1);
350  }
351 
352  bool isdir = (path.endsWith('/') || path.endsWith('\\'));
353  return !isdir;
354 }

432 void Application::noticeFileRightProblem(const QStringList &problemfile, Application::EFileClassEnum classTp, bool checkQuit)
433 {
434  if (problemfile.isEmpty())
435  return;
436 
437  QWidget *pParent = actWin;
438 
439  DDialog dia(pParent);
440  dia.setFixedSize(404, 163);
441  dia.setModal(true);
442  QString shortenFileName = QFontMetrics(dia.font()).elidedText(QFileInfo(problemfile.first()).fileName(), Qt::ElideMiddle, dia.width() / 2);
443 
444  QString message;
445 
446  switch (classTp) {
447  case ENotExist: {
448  message = tr("The file does not exist");
449  break;
450  }
451  case ENotFile:
452  case EDrawAppNotSup:
453  message = (problemfile.size() == 1 ?
454  tr("Unable to open \"%1\", unsupported file format").arg(shortenFileName) :
455  tr("Unsupported file format"));
456  break;
457  case EDrawAppSupButNotReadable:
458  message = (problemfile.size() == 1 ?
459  tr("Unable to open the write-only file \"%1\"").arg(shortenFileName) :
460  tr("Unable to open the write-only files"));
461  break;
462  default:
463  return;
464  }
465 
466  dia.setMessage(message);
467  dia.setIcon(QPixmap(":/theme/common/images/deepin-draw-64.svg"));
468  dia.addButton(tr("OK"), true, DDialog::ButtonNormal);
469  //保持弹窗在主窗口中心
470  QPoint centerPos = drawApp->topMainWindow()->mapToGlobal(drawApp->topMainWindow()->rect().center() - dia.rect().center());
471  dia.move(centerPos);
472  dia.exec();
473 
474  if (checkQuit) {
475  //如果没有任何文件加载成功(没有view就表示没有任何文件加载成功)
476  if (pParent == nullptr || CManageViewSigleton::GetInstance()->isEmpty()) {
477  dApp->quit();
478  }
479  }
480 }

522  ColorPanel *pColorPanel = pColor->colorPanel();

530  return true;
