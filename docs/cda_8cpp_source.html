146 void CdaThread::run()
147 {
148  while (m_needRun) {
149  QString strcdrom = GetCdRomString();
153  if (strcdrom.isEmpty()) {
154  setCdaState(CDROM_INVALID);
155  continue;
156  }
157 
158  QSharedPointer<DBlockDevice> blk(DDiskManager::createBlockDevice(strcdrom));
159  DBlockDevice *pblk = blk.data();
160  if (!pblk) {
161  continue;
162  }
163 
164  qulonglong blocksize = pblk->size();
168  if (blocksize == 0
169  || pblk->fsType() == DBlockDevice::iso9660
170  || queryIdTypeFormDbus().toLower() == "udf") {
171  setCdaState(CDROM_MOUNT_WITHOUT_CD);
172  continue;
173  }
174 
178  if (m_cdaStat != CDROM_MOUNT_WITH_CD) {
182  input_item_node_t *p_items = getInputNode();
186  if (!p_items) {
187  setCdaState(CDROM_INVALID);
188  qCritical() << __FUNCTION__ << "read input_item_node_t failed,maybe caused by rejecting CD";
189  continue;
190  }
191 
192  m_mediaList.clear();
193  for (int i = 0; i < p_items->i_children; i++) {
194  input_item_node_t *child = p_items->pp_children[i];
195  if (child != nullptr) {
196  qDebug() << __FUNCTION__ << "thread id:" << QThread::currentThread() \
197  << "name:" << child->p_item->psz_name << "duration:" << child->p_item->i_duration;
198  //空光盘有可能节点不为空，检测持续时间
199  if (child->p_item->i_duration == 0)
200  continue;
201  MediaMeta meta;
202  meta.hash = DMusic::filepathHash(QString(child->p_item->psz_name));
203  meta.title = child->p_item->psz_name;
204  meta.localPath = child->p_item->psz_uri;
205  meta.length = child->p_item->i_duration / 1000; //microseconds to milliseconds
206  meta.track = i + 1; //cd的歌曲索引值，从1开始，用于歌曲切歌
207  meta.filetype = "cdda";
208  meta.mmType = MIMETYPE_CDA; //mimetype用于识别cd
209  m_mediaList << meta;
210  }
211  }
212 
213  input_item_node_Delete_func input_item_node_Delete_fc = (input_item_node_Delete_func)VlcDynamicInstance::VlcFunctionInstance()->resolveSymbol("input_item_node_Delete");
217  input_item_node_Delete_fc(p_items);
221  if (m_mediaList.size() > 0) {
222  setCdaState(CDROM_MOUNT_WITH_CD);
223  }
224  }
225  sleep(1);
226  }
227 }
