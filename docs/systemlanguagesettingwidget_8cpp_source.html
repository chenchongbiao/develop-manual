111 void SystemLanguageSettingWidget::onSearch(const QString &text)
112 {
113  if (text.length() == 0) {
114  m_searchStatus = false;
115  m_view->setModel(m_model);
116  } else {
117  m_searchStatus = true;
118  m_searchModel = new QStandardItemModel(this);
119 
120  for (auto md : m_datas) {
121  if (md.text().contains(text, Qt::CaseInsensitive)) {
122  auto item = new DStandardItem(md.text());
123  item->setText(md.text());
124  item->setData(md.key(),KeyRole);
125  item->setData(md.pinyin(),PingYinRole);
126  m_searchModel->appendRow(item);
127  }
128  }
129  m_view->setModel(m_searchModel);
130  }
131 }

166 void SystemLanguageSettingWidget::setModelData(const QList<MetaData> &datas)
167 {
168  m_datas = datas;
169  QStringList removeLangList = m_keyboardModel->localLang();
170  for (QList<MetaData>::iterator iter = m_datas.begin(); iter != m_datas.end();) {
171  if (removeLangList.contains(iter->text())) {
172  m_datas.erase(iter);
173  }
174  ++iter;
175  }
176 
177  for (auto md : m_datas) {
178  auto item = new DStandardItem();
179  item->setText(md.text());
180  item->setData(md.key(),KeyRole);
181  item->setData(md.pinyin(),PingYinRole);
182  m_model->appendRow(item);
183  }
184  m_view->setModel(m_model);
185 }

187 bool SystemLanguageSettingWidget::eventFilter(QObject *watched, QEvent *event)
188 {
189  Q_UNUSED(watched)
190 
191  if (event->type() != QEvent::Move && event->type() != QEvent::Resize)
192  return false;
193 
194  QRect rect = m_contentWidget->rect();
195 
196  rect.moveTopLeft(-m_contentWidget->pos());
197  rect.setHeight(m_contentWidget->window()->height() - m_contentWidget->mapTo(window(), rect.topLeft()).y());
198 
199  QPainterPath path;
200  path.addRoundedRect(rect, 5, 5);
201  m_clipEffectWidget->setClipPath(path);
202 
203  return false;
204 }
