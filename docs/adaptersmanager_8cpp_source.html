85 void AdaptersManager::setAdapterPowered(const Adapter *adapter, const bool &powered)
86 {
87  if (!adapter)
88  return;
89 
90  QDBusObjectPath path(adapter->id());
91  QDBusPendingCall call = m_bluetoothInter->SetAdapterPowered(path, powered);
92 
93  if (powered) {
94  QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(call, this);
95  connect(watcher, &QDBusPendingCallWatcher::finished, [this, call, adapter] {
96  if (!call.isError()) {
97  QDBusObjectPath dPath(adapter->id());
98  m_bluetoothInter->SetAdapterDiscoverableTimeout(dPath, 60 * 5);
99  m_bluetoothInter->SetAdapterDiscoverable(dPath, true);
100  m_bluetoothInter->RequestDiscovery(dPath);
101  } else {
102  qWarning() << call.error().message();
103  }
104  });
105  } else {
106  QDBusPendingCall call = m_bluetoothInter->ClearUnpairedDevice();
107  QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(call, this);
108  connect(watcher, &QDBusPendingCallWatcher::finished, [ = ] {
109  if (call.isError())
110  qWarning() << call.error().message();
111  });
112  }
113 }

116 {

126  break;
127  case Device::StateConnected: {
128  m_bluetoothInter->DisconnectDevice(path);

240 void AdaptersManager::adapterAdd(Adapter *adapter, const QJsonObject &adpterObj)
241 {
242  if (!adapter)
243  return;
244 
245  inflateAdapter(adapter, adpterObj);
246  QDBusObjectPath dPath(adpterObj["Path"].toString());
247  QDBusPendingCall call = m_bluetoothInter->GetDevices(dPath);
248  QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(call, this);
249  connect(watcher, &QDBusPendingCallWatcher::finished, [this, adapter, call, watcher] {
250  if (adapter) {
251  if (!call.isError()) {
252  QDBusReply<QString> reply = call.reply();
253  const QString replyStr = reply.value();
254  QJsonDocument doc = QJsonDocument::fromJson(replyStr.toUtf8());
255  adapter->initDevicesList(doc);
256  emit this->adapterIncreased(adapter);
257  } else {
258  qWarning() << call.error().message();
259  }
260  }
261  delete watcher;
262  });
263 
264  QString id = adapter->id();
265  if (!id.isEmpty()) {
266  if (!m_adapters.contains(id)) {
267  m_adapters[id] = adapter;
268  } else if (m_adapters[id] == nullptr) {
269  m_adapters[id] = adapter;
270  }
271  }
272 }
