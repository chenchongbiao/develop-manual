145 void SinkInputWidget::setMuteIcon()
146 {
147  if (m_inputInter->mute()) {
148  const auto ratio = devicePixelRatioF();
149  QString iconString = "audio-volume-muted-symbolic";
150  if (DGuiApplicationHelper::instance()->themeType() == DGuiApplicationHelper::LightType) {
151  iconString.append("-dark");
152  }
153  QPixmap muteIcon = QIcon::fromTheme(iconString).pixmap(ICON_SIZE * ratio, ICON_SIZE * ratio);
154  muteIcon.setDevicePixelRatio(ratio);
155  QPixmap appIconSource(getIconFromTheme(m_inputInter->icon(), QSize(ICON_SIZE, ICON_SIZE), devicePixelRatioF()));
156 
157  QPixmap temp(appIconSource.size());
158  temp.fill(Qt::transparent);
159  temp.setDevicePixelRatio(ratio);
160  QPainter p1(&temp);
161  p1.drawPixmap(0, 0, appIconSource);
162  p1.setCompositionMode(QPainter::CompositionMode_DestinationIn);
163  p1.fillRect(temp.rect(), QColor(0, 0, 0, 40));
164  p1.end();
165  appIconSource = temp;
166 
167  QPainter p(&appIconSource);
168  p.drawPixmap(0, 0, muteIcon);
169 
170  appIconSource.setDevicePixelRatio(ratio);
171  m_volumeBtnMin->setIcon(appIconSource);
172  } else {
173  m_volumeBtnMin->setIcon(getIconFromTheme(m_inputInter->icon(), QSize(ICON_SIZE, ICON_SIZE), devicePixelRatioF()));
174  }
175 
176  refreshIcon();
177 }

185 void SinkInputWidget::refreshIcon()
186 {
187  if (!m_inputInter)
188  return;
189 
190  QString iconLeft = QString(m_inputInter->mute() ? "audio-volume-muted-symbolic" : "audio-volume-low-symbolic");
191  QString iconRight = QString("audio-volume-high-symbolic");
192 
193  if (DGuiApplicationHelper::instance()->themeType() == DGuiApplicationHelper::LightType) {
194  iconLeft.append("-dark");
195  iconRight.append("-dark");
196  }
197 
198  const auto ratio = devicePixelRatioF();
199  QPixmap ret = ImageUtil::loadSvg(iconRight, ":/", ICON_SIZE, ratio);
200  m_volumeIconMax->setPixmap(ret);
201 
202  ret = ImageUtil::loadSvg(iconLeft, ":/", ICON_SIZE, ratio);
203  m_volumeBtnMin->setIcon(ret);
204 }
