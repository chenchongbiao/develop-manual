114  template<class T, bool shortCount>
115  PLF<T, shortCount>::PLF( AbstractOLEStreamReader* reader, bool preservePos )
116  {
117  if ( preservePos )
118  reader->push();
119 
120  U32 count = 0;
121  if ( shortCount ) // work around a broken spec, e.g. for LSTF
122  count = reader->readU16();
123  else
124  count = reader->readU32();
125 
126  for ( U32 i = 0; i < count; ++i )
127  m_items.push_back( new T( reader, false ) );
128  if ( preservePos )
129  reader->pop();
130  it = m_items.begin();
131  }

133  template<class T, bool shortCount>
134  PLF<T, shortCount>::PLF( const U8* ptr )
135  {
136  U32 count = 0;
137  if ( shortCount ) { // work around a broken spec, e.g. for LSTF
138  count = readU16( ptr );
139  ptr += 2;
140  }
141  else {
142  count = readU32( ptr );
143  ptr += 4;
144  }
145 
146  for ( U32 i = 0; i < count; ++i, ptr += T::sizeOf )
147  m_items.push_back( new T( ptr ) );
148  it = m_items.begin();
149  }

218  PLCF<T>::PLCF( U32 length, AbstractOLEStreamReader* reader, bool preservePos )
219  {

221  reader->push();

224  m_indices.push_back( reader->readU32() );
225  for ( U32 i = 0; i < count; ++i ) // n "T"s

291  template<class T>
292  U32 PLCF<T>::calculateCount( U32 length )
293  {
294  if ( ( length - 4 ) % ( T::sizeOf + 4 ) ) {
295  wvlog << "Warning: PLCF size seems to be screwed" << std::endl;
296  wvlog << "Warning: length: " << length << ", size: " << T::sizeOf << ", mod: " << ( length - 4 ) % ( T::sizeOf + 4 ) << std::endl;
297  return 0;
298  }
299  return ( length - 4 ) / ( T::sizeOf + 4 );
300  }

303  // A method to "upgrade" the type of a PLCF from Word 6/7 to Word 8
304  // data structures, using the generated conversion code. Tricky :-)
305  // The ownership of the new PLCF is transferred to you!
306  template<typename OldT, typename NewT> PLCF<NewT>* convertPLCF( const PLCF<OldT>& old )
307  {
308  PLCF<NewT>* ret( new PLCF<NewT> );
309  ret->m_indices = old.m_indices; // the indices remain the same
310 
311  typename std::vector<OldT*>::const_iterator oldIt( old.m_items.begin() );
312  typename std::vector<OldT*>::const_iterator oldEnd( old.m_items.end() );
313  for ( ; oldIt != oldEnd; ++oldIt )
314  ret->m_items.push_back( new NewT( Word95::toWord97( **oldIt ) ) );
315  return ret;
316  }

440  template<class Offset>
441  FKP<Offset>::FKP( AbstractOLEStreamReader* reader, bool preservePos )
442  {
443  if ( preservePos )
444  reader->push();
445  reader->push();
446  reader->seek( 511, SEEK_CUR );
447  m_crun = reader->readU8();
448  reader->pop();
449 
450  m_rgfc = new U32[ m_crun + 1 ];
451  for ( U8 i = 0; i <= m_crun; ++i ) // <= crun, because of crun+1 FCs!
452  m_rgfc[ i ] = reader->readU32();
453 
454  m_rgb = new Offset[ m_crun ];
455  for ( U8 i = 0; i < m_crun; ++i )
456  m_rgb[ i ].read( reader, false );
457 
458  m_internalOffset = ( static_cast<U16>( m_crun ) + 1 ) * 4 + static_cast<U16>( m_crun ) * Offset::sizeOf;
459 
460  // store the rest of the FKP in an internal array for later use
461  const U16 length = 511 - m_internalOffset;
462  m_fkp = new U8[ length ]; // 511, because we don't need crun
463  for ( U16 i = 0; i < length; ++i )
464  m_fkp[ i ] = reader->readU8();
465 
466  if ( preservePos )
467  reader->pop();
468  }

470  template<class Offset>
471  FKP<Offset>::FKP( const U8* ptr )
472  {
473  m_crun = ptr[ 511 ];
474 
475  m_rgfc = new U32[ m_crun + 1 ];
476  for ( U8 i = 0; i <= m_crun; ++i, ptr += 4 ) // <= crun, because of crun+1 FCs!
477  m_rgfc[ i ] = readU32( ptr );
478 
479  m_rgb = new Offset[ m_crun ];
480  for ( U8 i = 0; i < m_crun; ++i, ptr += Offset::sizeOf )
481  m_rgb[ i ].readPtr( ptr );

486  U16 length = 511 - m_internalOffset;
487  m_fkp = new U8[ length ]; // 511, because we don't need crun

545  template<class Offset>
546  const U8* FKPIterator<Offset>::current() const
547  {
548  if ( m_index < m_fkp.m_crun ) {
549  // Note: The first byte of the "offset" types (BX or U8) is always
550  // the word offset into the array (or 0!).
551  const U8 tmp = m_fkp.m_rgb[ m_index ].offset;
552  // Now we have to calculate the real offset and then locate it
553  // within our cached array...
554  if ( tmp != 0 ) {
555  const int pos = tmp * 2 - m_fkp.m_internalOffset;
556  if ( pos < 0 ) {
557  wvlog << "ERROR: FKP internalOffset (" << m_fkp.m_internalOffset << ") is bigger than " <<
558  "2*" << (int)tmp << ", FKP array index would be negative!" << std::endl;
559  return 0;
560  } else if ( pos >= 511 - m_fkp.m_internalOffset ) {
561  wvlog << "ERROR: FKP array index (" << pos << " is bigger than allocated size ("
562  << 511 - m_fkp.m_internalOffset << ")" << std::endl;
563  return 0;
564  } else {
565  return &m_fkp.m_fkp[ pos ];
566  }
567  }
568  }
569  return 0;
570  }
