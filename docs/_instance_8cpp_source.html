14 void logCallback(void *data,
15  int level,
16  const libvlc_log_t *ctx,
17  const char *fmt,
18  va_list args)
19 {
20  Q_UNUSED(ctx)
21 
22  VlcInstance *instance = static_cast<VlcInstance *>(data);
23  if (instance->logLevel() > level) {
24  return;
25  }
26 
27  char *result;
28  if (vasprintf(&result, fmt, args) < 0) {
29  return; // LCOV_EXCL_LINE
30  }
31 
32  QString message(result);
33  free(result);
34 
35  message.prepend("VlcInstance libvlc: ");
36  switch (level) {
37  case Vlc::ErrorLevel: {
38  qCritical(message.toUtf8().data(), NULL);
39  if (message.contains("cannot write")) {
40  /*****************************************
41  *vlc write error. we need to stop
42  * player then start it
43  * **************************************/
44  instance->catchPulseError(0);
45  }
46  break;
47  }
48  case Vlc::WarningLevel:
49  qWarning(message.toUtf8().data(), NULL);
50  break;
51  case Vlc::NoticeLevel:
52  case Vlc::DebugLevel:
53  default:
54  qDebug(message.toUtf8().data(), NULL);
55  break;
56  }
57 }
