56 void VpnTLSSection::saveSettings()
57 {
58  // retrieve the data map
59  m_dataMap = m_vpnSetting->data();
60 
61  m_dataMap.insert("tls-remote", m_remote->text());
62  if (m_currentRemoteCertType == "default") {
63  m_dataMap.remove("remote-cert-tls");
64  } else {
65  m_dataMap.insert("remote-cert-tls", m_currentRemoteCertType);
66  }
67  m_dataMap.insert("ta", m_caCert->edit()->text());
68  if (m_customKeyDirection->checked()) {
69  m_dataMap.insert("ta-dir", m_currentKeyDirection);
70  } else {
71  m_dataMap.remove("ta-dir");
72  }
73 
74  m_vpnSetting->setData(m_dataMap);
75 
76  m_vpnSetting->setInitialized(true);
77 }

88 void VpnTLSSection::initUI()
89 {
90  m_remote->setTitle(tr("Subject Match"));
91  m_remote->setText(m_dataMap.value("tls-remote"));
92 
93  m_remoteCertTypeChooser->setTitle(tr("Remote Cert Type"));
94  m_currentRemoteCertType = "default";
95  QString curRemoteCertOption = RemoteCertTypeStrMap.at(0).first;
96  for (auto it = RemoteCertTypeStrMap.cbegin(); it != RemoteCertTypeStrMap.cend(); ++it) {
97  m_remoteCertTypeChooser->comboBox()->addItem(it->first, it->second);
98  if (it->second == m_dataMap.value("remote-cert-tls")) {
99  m_currentRemoteCertType = it->second;
100  curRemoteCertOption = it->first;
101  }
102  }
103  m_remoteCertTypeChooser->setCurrentText(curRemoteCertOption);
104 
105  m_caCert->setTitle(tr("Key File"));
106  m_caCert->edit()->setText(m_dataMap.value("ta"));
107 
108  m_customKeyDirection->setTitle(tr("Customize Key Direction"));
109  m_customKeyDirection->setChecked(m_dataMap.keys().contains("ta-dir"));
110 
111  m_keyDirectionChooser->setTitle(tr("Key Direction"));
112  m_keyDirectionChooser->comboBox()->addItem("0", "0");
113  m_keyDirectionChooser->comboBox()->addItem("1", "1");
114  m_currentKeyDirection = m_dataMap.value("ta-dir", "0");
115  m_keyDirectionChooser->setCurrentText(m_currentKeyDirection);
116  m_keyDirectionChooser->setVisible(m_customKeyDirection->checked());
117 
118  appendItem(m_remote);
119  appendItem(m_remoteCertTypeChooser);
120  appendItem(m_caCert);
121  appendItem(m_customKeyDirection);
122  appendItem(m_keyDirectionChooser);
123 
124  m_remote->textEdit()->installEventFilter(this);
125  m_caCert->edit()->lineEdit()->installEventFilter(this);
126 }

128 void VpnTLSSection::initConnection()
129 {
130  connect(m_remoteCertTypeChooser, &ComboxWidget::onSelectChanged, this, [ = ](const QString &dataSelected) {
131  for (auto it = RemoteCertTypeStrMap.cbegin(); it != RemoteCertTypeStrMap.cend(); ++it) {
132  if (it->first == dataSelected) {
133  m_currentRemoteCertType = it->second;
134  break;
135  }
136  }
137  });
138 
139  connect(m_customKeyDirection, &SwitchWidget::checkedChanged, m_keyDirectionChooser, &ComboxWidget::setVisible);
140 
141  connect(m_keyDirectionChooser, &ComboxWidget::onSelectChanged, this, [ = ](const QString &dataSelected) {
142  m_currentKeyDirection = dataSelected;
143  });
144 
145  connect(m_caCert, &FileChooseWidget::requestFrameKeepAutoHide, this, &VpnTLSSection::requestFrameAutoHide);
146 
147  connect(m_remoteCertTypeChooser, &ComboxWidget::onIndexChanged, this, &VpnTLSSection::editClicked);
148  connect(m_customKeyDirection, &SwitchWidget::checkedChanged, this, &VpnTLSSection::editClicked);
149  connect(m_keyDirectionChooser, &ComboxWidget::onIndexChanged, this, &VpnTLSSection::editClicked);
150 }
