28 FontCollection::FontCollection( AbstractOLEStreamReader* reader, const Word97::FIB& fib )
29 {
30  m_fallbackFont = new Word97::FFN();
31  m_fallbackFont->xszFfn = "Helvetica";
32 
33  reader->push();
34  reader->seek( fib.fcSttbfffn );
35 
36  if ( fib.nFib < Word8nFib ) { // older than Word97
37  int bytesLeft = reader->readU16() - 2;
38  while ( bytesLeft > 0 ) {
39  Word97::FFN* ffn = new Word97::FFN( reader, Word97::FFN::Word95, false );
40  m_fonts.push_back( ffn );
41  bytesLeft -= ffn->cbFfnM1 + 1;
42  }
43  }
44  else { // Word97 or newer
45  const U16 count = reader->readU16();
46  const U16 extraData = reader->readU16();
47  if ( extraData != 0 )
48  wvlog << "Huh?? Found STTBF extra data within the STTBF of FFNs" << std::endl;
49 
50  for ( int i = 0; i < count; ++i )
51  m_fonts.push_back( new Word97::FFN( reader, Word97::FFN::Word97, false ) );
52  }
53 
54  if ( reader->tell() - fib.fcSttbfffn != fib.lcbSttbfffn )
55  wvlog << "Warning: Didn't read lcbSttbfffn bytes: read=" << reader->tell() - fib.fcSttbfffn
56  << " lcbSttbfffn=" << fib.lcbSttbfffn << std::endl;
57  reader->pop();
58 }

60 FontCollection::~FontCollection()
61 {
62  std::for_each( m_fonts.begin(), m_fonts.end(), Delete<Word97::FFN>() );
63  delete m_fallbackFont;

66 const Word97::FFN& FontCollection::font( S16 ftc ) const
67 {
68  if ( ftc >= 0 && static_cast<U16>( ftc ) < m_fonts.size() )
69  return *m_fonts[ ftc ];
70  return *m_fallbackFont;
71 }
