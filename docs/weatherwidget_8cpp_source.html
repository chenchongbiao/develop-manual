113  QRect rect(0,0,width(), 50);

115 
116  QPen pen(Qt::white);

211  painter.setFont(curFont);

213  QFontMetrics fm(curFont);
214

216  text = QString("%1-%2°F")
217  .arg((int)(32 + item.temperature().first * 1.8))
218  .arg((int)(32 + item.temperature().second * 1.8));

221  .arg((int)(item.temperature().first))
222  .arg((int)(item.temperature().second));
223

226

228  painter.drawText(weekArea, Qt::AlignVCenter | Qt::AlignRight, item.dayName());
229  painter.restore();
230  }
231  y+=rect.height();
232  }

284 QPixmap WeatherWidget::loadPixmap(const QString &path)
285 {
286  qreal ratio = 1.0;
287  QPixmap pixmap;
288 
289  const qreal devicePixelRatio = qApp->devicePixelRatio();
290 
291  if (!qFuzzyCompare(ratio, devicePixelRatio)) {
292  QImageReader reader;
293  reader.setFileName(qt_findAtNxFile(path, devicePixelRatio, &ratio));
294  if (reader.canRead()) {
295  reader.setScaledSize(reader.size() * (devicePixelRatio / ratio));
296  pixmap = QPixmap::fromImage(reader.read());
297  pixmap.setDevicePixelRatio(devicePixelRatio);
298  }
299  } else {
300  pixmap.load(path);
301  }
302 
303  return pixmap;
304 }
