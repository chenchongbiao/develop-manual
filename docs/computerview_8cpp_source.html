80 class ViewReturnEater : public QObject
81 {
82  Q_OBJECT
83 public:
84  explicit ViewReturnEater(QListView *parent) : QObject(parent) {}
85 
86 protected:
87  bool eventFilter(QObject *, QEvent *e)
88  {
89  if (e->type() == QEvent::Type::KeyPress) {
90  QKeyEvent *ke = static_cast<QKeyEvent *>(e);
91  if (ke->key() == Qt::Key::Key_Return || ke->key() == Qt::Key::Key_Enter) {
92  QListView *v = qobject_cast<QListView *>(parent());
93  if (v) {
94  auto model = v->model();
95  const QModelIndex &curIdx = v->selectionModel()->currentIndex();
96  if (!curIdx.isValid()) {
97  return true;
98  }
99  // 目的是可移动设备的item重命名时，按enter键时，不应该进入对应的目录，因此直接返回false
100  if (model->flags(curIdx) & Qt::ItemFlag::ItemIsEditable) {
101  return false;
102  }
103  Q_EMIT entered(curIdx);
104  return true;
105  }
106  }
107  }
108  return false;
109  }
110 
111 Q_SIGNALS:
112  void entered(const QModelIndex &idx);
113 };

292 void ComputerView::contextMenu(const QPoint &pos)
293 {
294  const QModelIndex &idx = m_view->indexAt(pos);
295  if (!idx.isValid()) {
296  return;
297  }
298  const QVector<MenuAction> &av = idx.data(ComputerModel::DataRoles::ActionVectorRole).value<QVector<MenuAction>>();
299  QSet<MenuAction> disabled;
300  if (!WindowManager::tabAddableByWinId(WindowManager::getWindowId(this))) {
301  disabled.insert(MenuAction::OpenInNewTab);
302  disabled.insert(MenuAction::OpenDiskInNewTab);
303  }
304 
305  const QString &strVolTag = idx.data(ComputerModel::DataRoles::VolumeTagRole).value<QString>();
306  if (strVolTag.startsWith("sr") // fix bug#25921 仅针对光驱设备实行禁用操作
307  && idx.data(ComputerModel::DataRoles::DiscUUIDRole).value<QString>().isEmpty()
308  && !idx.data(ComputerModel::DataRoles::DiscOpticalRole).value<bool>()
309  && (!idx.data(ComputerModel::DataRoles::OpenUrlRole).value<DUrl>().isValid()
310  || idx.data(ComputerModel::DataRoles::SizeTotalRole).value<int>() == 0)) {
311  //fix:光驱还没有加载成功前，右键点击光驱“挂载”，光驱自动弹出。
312  disabled.insert(MenuAction::OpenDiskInNewWindow);
313  disabled.insert(MenuAction::OpenDiskInNewTab);
314  disabled.insert(MenuAction::Mount);
315  //fix:不插光盘，打开文件管理器，光盘的弹出按钮不能置灰
316  //disabled.insert(MenuAction::Eject);
317  disabled.insert(MenuAction::SafelyRemoveDrive);
318 
319  disabled.insert(MenuAction::Property);
320  }
321  if (strVolTag.startsWith("sr") && DFMOpticalMediaWidget::g_mapCdStatusInfo[strVolTag].bBurningOrErasing) { // 如果当前光驱设备正在执行擦除/刻录，则禁用所有右键菜单选项
322  for (MenuAction act : av)
323  disabled.insert(act);
324  }
325 
326  DFileMenu *menu = nullptr;
328  if (idx.data(ComputerModel::DataRoles::SchemeRole) == DFMVAULT_SCHEME) {
329  quint64 wndId = WindowManager::getWindowId(this);
330  menu = DFileMenuManager::createVaultMenu(WindowManager::getWindowById(wndId));
331  } else {
332  menu = DFileMenuManager::genereteMenuByKeys(av, disabled);
333  }
334 
335  menu->setEventData(DUrl(), {idx.data(ComputerModel::DataRoles::DFMRootUrlRole).value<DUrl>()}, WindowManager::getWindowId(this), this);
336  //fix bug 33305 在用右键菜单复制大量文件时，在复制过程中，关闭窗口这时this释放了，在关闭拷贝menu的exec退出，menu的deleteLater崩溃
337  QPointer<ComputerView> me = this;
338  menu->exec(this->mapToGlobal(pos));
339  menu->deleteLater(me);
340 }

362 bool ComputerView::eventFilter(QObject *obj, QEvent *event)
363 {
364  if (event->type() == QEvent::MouseButtonRelease && obj == m_view->viewport()) {
365  QMouseEvent *e = static_cast<QMouseEvent *>(event);
366  const QModelIndex &idx = m_view->indexAt(e->pos());
367  if (e->button() == Qt::MouseButton::LeftButton && (!idx.isValid() || !(idx.flags() & Qt::ItemFlag::ItemIsEnabled))) {
368  m_view->selectionModel()->clearSelection();
369  }
370  return false;
371  } else if (event->type() == QEvent::KeyPress && obj == m_view) {
372  QKeyEvent *ke = static_cast<QKeyEvent *>(event);
373  if (ke->modifiers() == Qt::Modifier::ALT) {
374  this->event(event);
375  return true;
376  }
377  return false;
378  } else {
379  return QObject::eventFilter(obj, event);
380  }
381 }
