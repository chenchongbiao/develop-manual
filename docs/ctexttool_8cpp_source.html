61 CGraphicsItem *CTextTool::creatItem(IDrawTool::CDrawToolEvent *event, ITERecordInfo *pInfo)
62 {
63  Q_UNUSED(pInfo)
64  if ((event->eventType() == CDrawToolEvent::EMouseEvent && event->mouseButtons() == Qt::LeftButton)
65  || event->eventType() == CDrawToolEvent::ETouchEvent) {
66  CGraphicsTextItem *pItem = new CGraphicsTextItem(QObject::tr("Input text here"));
67  pItem->setPos(event->pos().x(), event->pos().y());
68  pItem->textEditor()->setAlignment(Qt::AlignLeft);
69  pItem->textEditor()->selectAll();
70 
71  CGraphicsView *pView = event->scene()->drawView();
72 
73  QFontMetrics fm(pView->getDrawParam()->getTextFont());
74  QSizeF size = pItem->textEditor()->document()->size();
75 
76  // 设置默认的高度会显示不全,需要设置为字体高度的1.4倍
77  pItem->setRect(QRectF(m_sPointPress.x(), m_sPointPress.y(), size.width(), fm.height() * 1.4));
78 
79  // 设置新建图元属性
80  pItem->setFontSize(pView->getDrawParam()->getTextFont().pointSize());
81  pItem->setFontFamily(pView->getDrawParam()->getTextFont().family());
82  pItem->setFontStyle(pView->getDrawParam()->getTextFontStyle());
83  pItem->setTextColor(pView->getDrawParam()->getTextColor());
84 
85  pItem->textEditor()->document()->clearUndoRedoStacks();
86  event->scene()->addCItem(pItem);
87 
88  return pItem;
89  }
90  return nullptr;
91 }
