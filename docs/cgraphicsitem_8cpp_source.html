53  const QPen &pen,
54  bool penStrokerShape,
55  const qreal incW, bool doSimplified)
56 {
57  // We unfortunately need this hack as QPainterPathStroker will set a width of 1.0
58  // if we pass a value of 0.0 to QPainterPathStroker::setWidth()
59  const qreal penWidthZero = qreal(0.00000001);
60 
61  if (orgPath == QPainterPath() || pen == Qt::NoPen)
62  return orgPath;
63  QPainterPathStroker ps(pen);
64 
65  if (pen.widthF() <= 0.0)
66  ps.setWidth(penWidthZero);
67  else
68  ps.setWidth(pen.widthF() + incW);
69 
70  QPainterPath p;
71 
72  //获取线条的填充区域路径(线条的宽度会形成一个填充区域，这个区域是实际显示是用的pen颜色的区域)
73  if (penStrokerShape) {
74  p = ps.createStroke(orgPath);
75  } else {
76  p = ps.createStroke(orgPath) + orgPath;
77  }
78  if (doSimplified) {
79  p = p.simplified();
80  }
81 
82  return p;
83 }
