108 DFMCrumbInterface *DFMCrumbManager::createControllerByUrl(const DUrl &fileUrl, DFMCrumbBar *crumbBar) const
109 {
110  Q_D(const DFMCrumbManager);
111 
112  KeyType theType = fileUrl.scheme();
113 
114  const QList<CrumbCreaterType> creatorList = d->controllerCreatorHash.values(theType);
115 
116  Q_CHECK_PTR(crumbBar);
117 
118  if (!creatorList.isEmpty()) {
119  DFMCrumbInterface *i = (creatorList.first().second)();
120  i->setCrumbBar(crumbBar);
121  return i;
122  }
123 
124  return nullptr;
125 }
