162  gpointer
163 load_database (FsearchApplication*app)
164 {
165  if (!app->db) {
166  // create new database
167  timer_start ();
168  app->db = db_new ();
169 
170  bool loaded = false;
171  bool build_new = false;
172  for (GList *l = app->config->locations; l != NULL; l = l->next) {
173  if (app->config->update_database_on_launch) {
174  if (db_location_add (app->db, l->data, app->config, build_location_callback)) {
175  loaded = true;
176  build_new = true;
177  }
178  }
179  else {
180  if (!db_location_load (app->db, l->data)) {
181  if (db_location_add (app->db, l->data, app->config, build_location_callback)) {
182  loaded = true;
183  build_new = true;
184  }
185  }
186  else {
187  loaded = true;
188  }
189  }
190  }
191  if (loaded) {
192  if (build_new) {
193  db_build_initial_entries_list (app->db);
194  }
195  else {
196  db_update_entries_list (app->db);
197  }
198  }
199  timer_stop ();
200  }
201  else {
202  timer_start ();
203  if (app->config->locations) {
204  for (GList *l = app->config->locations; l != NULL; l = l->next) {
205  db_location_add (app->db, l->data, app->config, build_location_callback);
206  }
207  db_build_initial_entries_list (app->db);
208  }
209  timer_stop ();
210  }
211  return NULL;
212 }
