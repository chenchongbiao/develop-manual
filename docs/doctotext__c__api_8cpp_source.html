84 int StreamBuf::overflow(int ch)
85 {
86  if (ch != EOF)
87  {
88  *pptr() = ch;
89  pbump(1);
90  }
91  if (flush() == -1)
92  return EOF;
93  return traits_type::to_int_type(ch);
94 }
95 
96 int StreamBuf::sync()
97 {
98  return flush();
99 }
100 
101 int StreamBuf::flush()
102 {
103  int to_write = pptr() - pbase();
104  if (to_write > 0)
105  {
106  if (fwrite(pbase(), 1, to_write, m_stream) != to_write)
107  return -1;
108  pbump(-to_write);
109  }
110  return 0;
111 }

383 char** DOCTOTEXT_CALL doctotext_attachment_get_keys(DocToTextAttachment* attachment)
384 {
385  if (attachment->m_metadata_field_keys)
386  return attachment->m_metadata_field_keys;
387  attachment->m_metadata_fields_count = doctotext_attachment_fields_count(attachment);
388  if (attachment->m_metadata_fields_count == 0)
389  return NULL;
390  try
391  {
392  const std::map<std::string, Variant>& fields = attachment->m_attachment->getFields();
393  std::map<std::string, Variant>::const_iterator it = fields.begin();
394  attachment->m_metadata_field_keys = new char*[attachment->m_metadata_fields_count];
395  memset(attachment->m_metadata_field_keys, 0, sizeof(char*) * attachment->m_metadata_fields_count);
396  size_t i = 0;
397  while (it != fields.end())
398  {
399  attachment->m_metadata_field_keys[i] = strdup(it->first.c_str());
400  if (!attachment->m_metadata_field_keys[i])
401  return NULL;
402  ++it;
403  ++i;
404  }
405  return attachment->m_metadata_field_keys;
406  }
407  catch (std::bad_alloc& ba)
408  {
409  return NULL;
410  }
