341 void PlayQueueWidget::autoHidden(QWidget *old, QWidget *now)
342 {
343  if (old && now) {
344  if (!(now->objectName() == "infoDialog" || now->objectName() == "InfoTitle" ||
345  now->objectName() == "MainFrame" || now->objectName() == "MessageBox")) {
346 
347  PlayListView *playListQueue = dynamic_cast<PlayListView *>(now);
348 
349  if (playListQueue && playListQueue->objectName() == "PlayListView") {
350  if (!playListQueue->getIsPlayQueue()) {
351  if (!this->isHidden()) {
352  QWidget *parent = static_cast<QWidget *>(this->parent());
353  if (parent) {
354  this->playAnimation(parent->size());
355  }
356  }
357  }
358  } else {
359  if (now->objectName() != "FooterWidget") {
360  if (!this->isHidden()) {
361  QWidget *parent = static_cast<QWidget *>(this->parent());
362  if (parent) {
363  this->playAnimation(parent->size());
364  }
365  }
366  }
367  }
368 
369  }
370  }
371 }
