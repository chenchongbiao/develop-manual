49 QDBusVariant TagManagerDaemon::disposeClientData(const QMap<QString, QVariant> &filesAndTags, const unsigned long long &type)
50 {
51  QDBusVariant dbusVar{};
52 
53  if (!filesAndTags.isEmpty()) {
54  QMap<QString, QVariant>::const_iterator cbeg{ filesAndTags.cbegin() };
55  QMap<QString, QVariant>::const_iterator cend{ filesAndTags.cend() };
56  QMap<QString, QList<QString>> filesAndTagsName{};
57 
58  for (; cbeg != cend; ++cbeg) {
59  QString key{ Tag::escaping_en_skim(cbeg.key()) };
60  QList<QString> values{};
61 
62  for (const QString &qstr : cbeg.value().toStringList()) {
63  values.push_back(Tag::escaping_en_skim(qstr));
64  }
65 
66  filesAndTagsName[key] = values;
67  }
68 
69  QVariant var{ DSqliteHandle::instance()->disposeClientData(filesAndTagsName, type) };
70  dbusVar.setVariant(var);
71  }
72 
73  return dbusVar;
74 }
