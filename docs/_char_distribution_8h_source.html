45 class CharDistributionAnalysis
46 {
47 public:
48  CharDistributionAnalysis()
49  {
50  Reset();
51  }
52 
53  //feed a block of data and do distribution analysis
54  void HandleData(const char *aBuf, PRUint32 aLen) {}
55 
56  //Feed a character with known length
57  void HandleOneChar(const char *aStr, PRUint32 aCharLen)
58  {
59  PRInt32 order;
60 
61  //we only care about 2-bytes character in our distribution analysis
62  order = (aCharLen == 2) ? GetOrder(aStr) : -1;
63 
64  if (order >= 0) {
65  mTotalChars++;
66  //order is valid
67  if ((PRUint32)order < mTableSize) {
68  if (512 > mCharToFreqOrder[order])
69  mFreqChars++;
70  }
71  }
72  }
73 
74  //return confidence base on existing data
75  float GetConfidence();
76 
77  //Reset analyser, clear any state
78  void Reset(void)
79  {
80  mDone = PR_FALSE;
81  mTotalChars = 0;
82  mFreqChars = 0;
83  }
84 
85  //This function is for future extension. Caller can use this function to control
86  //analyser's behavior
87  void SetOpion() {}
88 
89  //It is not necessary to receive all data to draw conclusion. For charset detection,
90  // certain amount of data is enough
91  PRBool GotEnoughData()
92  {
93  return mTotalChars > ENOUGH_DATA_THRESHOLD;
94  }
95 
96 protected:
97  //we do not handle character base on its original encoding string, but
98  //convert this encoding string to a number, here called order.
99  //This allow multiple encoding of a language to share one frequency table
100  virtual PRInt32 GetOrder(const char *str)
101  {
102  return -1;
103  }
104 
105  //If this flag is set to PR_TRUE, detection is done and conclusion has been made
106  PRBool mDone;
107 
108  //The number of characters whose frequency order is less than 512
109  PRUint32 mFreqChars;
110 
111  //Total character encounted.
112  PRUint32 mTotalChars;
113 
114  //Mapping table to get frequency order from char order (get from GetOrder())
115  const PRInt16 *mCharToFreqOrder;
116 
117  //Size of above table
118  PRUint32 mTableSize;
119 
120  //This is a constant value varies from language to language, it is used in
121  //calculating confidence. See my paper for further detail.
122  float mTypicalDistributionRatio;
123 };

126 class EUCTWDistributionAnalysis: public CharDistributionAnalysis
127 {
128 public:
129  EUCTWDistributionAnalysis();
130 protected:
131 
132  //for euc-TW encoding, we are interested
133  // first byte range: 0xc4 -- 0xfe
134  // second byte range: 0xa1 -- 0xfe
135  //no validation needed here. State machine has done that
136  PRInt32 GetOrder(const char *str)
137  {
138  if ((unsigned char)*str >= (unsigned char)0xc4)
139  return 94 * ((unsigned char)str[0] - (unsigned char)0xc4) + (unsigned char)str[1] - (unsigned char)0xa1;
140  else
141  return -1;
142  }
143 };

183 class Big5DistributionAnalysis : public CharDistributionAnalysis
184 {
185 public:
186  Big5DistributionAnalysis();
187 protected:
188  //for big5 encoding, we are interested
189  // first byte range: 0xa4 -- 0xfe
190  // second byte range: 0x40 -- 0x7e , 0xa1 -- 0xfe
191  //no validation needed here. State machine has done that
192  PRInt32 GetOrder(const char *str)
193  {
194  if ((unsigned char)*str >= (unsigned char)0xa4)
195  if ((unsigned char)str[1] >= (unsigned char)0xa1)
196  return 157 * ((unsigned char)str[0] - (unsigned char)0xa4) + (unsigned char)str[1] - (unsigned char)0xa1 + 63;
197  else
198  return 157 * ((unsigned char)str[0] - (unsigned char)0xa4) + (unsigned char)str[1] - (unsigned char)0x40;
199  else
200  return -1;
201  }

204 class SJISDistributionAnalysis : public CharDistributionAnalysis
205 {
206 public:
207  SJISDistributionAnalysis();
208 protected:
209  //for sjis encoding, we are interested
210  // first byte range: 0x81 -- 0x9f , 0xe0 -- 0xfe
211  // second byte range: 0x40 -- 0x7e, 0x81 -- oxfe
212  //no validation needed here. State machine has done that
213  PRInt32 GetOrder(const char *str)
214  {
215  PRInt32 order;
216  if ((unsigned char)*str >= (unsigned char)0x81 && (unsigned char)*str <= (unsigned char)0x9f)
217  order = 188 * ((unsigned char)str[0] - (unsigned char)0x81);
218  else if ((unsigned char)*str >= (unsigned char)0xe0 && (unsigned char)*str <= (unsigned char)0xef)
219  order = 188 * ((unsigned char)str[0] - (unsigned char)0xe0 + 31);
220  else
221  return -1;
222  order += (unsigned char) * (str + 1) - 0x40;
223  if ((unsigned char)str[1] > (unsigned char)0x7f)
224  order--;
225  return order;
226  }
227 };
